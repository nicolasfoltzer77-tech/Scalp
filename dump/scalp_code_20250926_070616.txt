#!/usr/bin/env python3
import json, time, math
from urllib.request import urlopen, Request
from urllib.parse import urlencode
from urllib.error import URLError, HTTPError

BASE="https://api.bitget.com"

def _http_json(path, params=None, tries=3, timeout=10):
    if params: path += ("?" + urlencode(params))
    for i in range(tries):
        try:
            with urlopen(Request(BASE+path, headers={"User-Agent":"scalp/1"}), timeout=timeout) as r:
                raw=r.read().decode("utf-8")
                return json.loads(raw)
        except (URLError, HTTPError):
            if i+1>=tries: raise
            time.sleep(0.6*(2**i))

def ticker_last(symbol:str)->float|None:
    # v2
    j=_http_json("/api/v2/mix/market/ticker", {"symbol":symbol})
    data=j.get("data") or {}
    p=data.get("lastPr") or data.get("last") or data.get("close")
    try: return float(p)
    except: return None

def klines(symbol:str, granularity:int, limit:int=300):
    # try v2 first
    j=_http_json("/api/v2/mix/market/candles", {"symbol":symbol,"granularity":granularity,"limit":limit})
    data=j.get("data")
    if not data:
        j=_http_json("/api/mix/v1/market/candles", {"symbol":symbol,"granularity":granularity,"limit":limit})
        data=j.get("data")
    rows=[]
    # Bitget renvoie récents→anciens sous forme ["ts","open","high","low","close","volume"]
    for r in (data or []):
        try:
            ts=int(r[0])
            o,h,l,c=map(float,(r[1],r[2],r[3],r[4]))
            rows.append((ts,o,h,l,c))
        except: continue
    rows.sort(key=lambda x:x[0])  # ancien→récent
    return rows

# util
def tf_to_seconds(tf:str)->int:
    m={"1m":60,"5m":300,"15m":900,"30m":1800,"1h":3600}
    return m.get(tf,60)

import os, json, time, ccxt, hashlib
from pathlib import Path

DATA=Path("/opt/scalp/data")
F_OK=DATA/"coins_90d.json"
F_PEN=DATA/"coins_pending.json"
F_TOP=DATA/"top.json"

    return ccxt.bitget({
        "apiKey": os.getenv("BITGET_API_KEY",""),
        "secret": os.getenv("BITGET_API_SECRET",""),
        "password": os.getenv("BITGET_PASSPHRASE",""),

def load_list(p):
    try:
        j=json.loads(Path(p).read_text() or "{}")
        return j if isinstance(j,list) else (j.get("symbols") or [])
    except: return []

def load_fixed():
    try:
        j=json.loads(F_FIX.read_text()); lst=j.get("symbols",[])
        return [s for s in lst if isinstance(s,str)]
    except: return []

def universe():
    seen=set(); out=[]
    for s in load_fixed()+load_list(F_OK)+load_list(F_TOP)+load_list(F_PEN):
        if s and s not in seen:

def cache_path(sym:str)->Path:

import time, math
TF_SEC = {"5m":300,"15m":900,"30m":1800}

def last_closed_epoch(tf:str)->int:
    step = TF_SEC[tf]
    now  = int(time.time())
    return (now//step - 1)*step  # dernière bougie FERMÉE

def ttl_left(tf:str)->int:
    step = TF_SEC[tf]
    now  = int(time.time())
    next_close = (now//step + 0)*step  # fin de la bougie courante
    return max(0, next_close - now)

def since_for(tf:str, bars:int)->int:
    end = last_closed_epoch(tf)
    return (end - (bars-1)*TF_SEC[tf])

            a = json.loads(p.read_text())

    for tf in ("5m","15m","30m"):
        r = worst(tf, SYMS)
        out["checks"].append(r)
        if r["stale"]>0: ok = False

#!/usr/bin/env python3
import argparse, json, time
from pathlib import Path

DATA = Path("/opt/scalp/data")
CAND = DATA/"candles"
NOW  = int(time.time())

TF_REQ = [("5m",300), ("15m",900), ("30m",1800)]
STALE = 2*3600  # on considère "frais" si dernière bougie < 2h

def load_universe(p):
    p = Path(p)
    if not p.exists(): return []
    try: j = json.loads(p.read_text())
    except Exception: return []
    syms = j.get("symbols") or j.get("Symbols") or []
    return [s for s in syms if isinstance(s,str)]

def score_symbol(sym:str)->tuple:
    """score = nb de TF présents et frais, puis inverse de staleness."""
    ok = 0
    worst_left = -10**9
    for tf,period in TF_REQ:
        f = CAND/tf/f"{sym}.json"
        if not f.exists(): continue
        try:
            a = json.loads(f.read_text())
            ts = int(a[-1][0])//1000 if a else 0
        except Exception:
            ts = 0
        left = max(0, (ts + period) - NOW)  # validité restante
        age = period - left
        if age <= STALE: ok += 1
        worst_left = max(worst_left, left)
    return (ok, worst_left)  # tri décroissant

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--universe", required=True)
    ap.add_argument("--keep", type=int, default=30)
    ap.add_argument("--fixed", default="BTCUSDT,ETHUSDT,XRPUSDT,SOLUSDT,BNBUSDT")
    ap.add_argument("--out", required=True)
    args = ap.parse_args()

    uni = load_universe(args.universe)
    fixed = [s.strip() for s in (args.fixed or "").split(",") if s.strip()]
    fixed = [s for s in fixed if s in uni]

    rest = [s for s in uni if s not in fixed]
    scored = sorted(rest, key=score_symbol, reverse=True)

    pick = list(dict.fromkeys(fixed + scored))[:args.keep]

    out = {
        "symbols": pick,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "source": "coins_90d->local_candles",
        "keep": args.keep,
        "fixed": fixed,
    }
    Path(args.out).write_text(json.dumps(out, indent=2))
    print(f"[top] n={len(pick)} fixed={len(fixed)} -> {args.out}")

if __name__ == "__main__":
    main()

#!/usr/bin/env python3
import json, time, math, statistics as st
from pathlib import Path

DATA = Path("/opt/scalp/data")
CAND = DATA/"candles"
TOPF = DATA/"top.json"
OUTF = DATA/"analyse_A_out.json"
HIST_DIR = DATA/"history/A"
HIST_DIR.mkdir(parents=True, exist_ok=True)

TF_STEPS = {"30m":1800, "15m":900, "5m":300}
TF_W = {"30m":0.5, "15m":0.3, "5m":0.2}
TAU = {"30m":0.35, "15m":0.30, "5m":0.25}
ATR_GATE = {"30m":0.0025, "15m":0.0018, "5m":0.0012}  # en ratio

P_ENTER, P_EXIT = 0.60, 0.45

def jload(p, d=None):
    try: return json.loads(Path(p).read_text())
    except Exception: return d

def ema(arr, n):
    k = 2/(n+1)
    out=[]
    s=None
    for x in arr:
        s = x if s is None else (x*k + s*(1-k))
        out.append(s)
    return out

def atr(h,l,c,n=14):
    tr=[]
    prev=c[0]
    for i in range(len(c)):
        if i==0: tr.append(h[i]-l[i])
        else: tr.append(max(h[i]-l[i], abs(h[i]-prev), abs(l[i]-prev)))
        prev=c[i]
    return ema(tr,n)

def macd_hist(c, f=12, s=26, sig=9):
    e1=ema(c,f); e2=ema(c,s)
    macd=[e1[i]-e2[i] for i in range(len(c))]
    es=ema(macd, sig)
    return [macd[i]-es[i] for i in range(len(c))]

def rsi(c, n=14):
    gains=[]; losses=[]
    for i in range(1,len(c)):
        d=c[i]-c[i-1]
        gains.append(max(d,0)); losses.append(max(-d,0))
    def rma(x, n):
        if not x: return []
        a=[sum(x[:n])/n]+[0]*(len(x)-n)
        for i in range(n,len(x)):
            a[i-n+1] = (a[i-n]*(n-1)+x[i])/n
        return [0]* (len(x)-len(a)) + a
    rg=rma(gains,n); rl=rma(losses,n)
    out=[50]*len(c)
    for i in range(1,len(c)):
        avg_g=rg[i-1]; avg_l=rl[i-1]
        if avg_l==0 and avg_g==0: out[i]=50
        elif avg_l==0: out[i]=100
        else:
            rs=avg_g/avg_l
            out[i]=100-(100/(1+rs))
    return out

def obv_slope(c, v, n=20):
    obv=[0]
    for i in range(1,len(c)):
        if c[i]>c[i-1]: obv.append(obv[-1]+v[i])
        elif c[i]<c[i-1]: obv.append(obv[-1]-v[i])
        else: obv.append(obv[-1])
    out=[0]*len(c)
    for i in range(len(c)):
        out[i] = 0 if i<n else obv[i]-obv[i-n]
    return out

def tanh(x,cap=3.0):
    x=max(-cap,min(cap,x))
    return math.tanh(x)

def subscores_for_tf(ts,c,h,l,v,tf):
    if len(c)<60: return 0.0, {}
    atr14=atr(h,l,c,14)
    at = atr14[-1]
    atp = 0.0 if c[-1]==0 else at/c[-1]

    e12=ema(c,12); e26=ema(c,26)
    s_ema = tanh((e12[-1]-e26[-1])/(1.5*at + 1e-12))

    hist = macd_hist(c)
    scale = st.median([abs(x) for x in hist[-100:]]) or 1.0
    s_macd = tanh(hist[-1]/scale)

    rsi14 = rsi(c,14)
    s_rsi = max(-1.0, min(1.0,(rsi14[-1]-50)/50))

    # ADX light (proxy via variation ATR/price)
    s_adx = max(-1.0, min(1.0, (atp-0.002)/0.01))

    obvs=obv_slope(c,v,20)
    med = st.median([abs(x) for x in obvs[-200:]]) or 1.0
    s_obv = tanh(obvs[-1]/med)

    w = {"ema":0.35,"macd":0.25,"rsi":0.20,"adx":0.10,"obv":0.10}
    gate = ATR_GATE[tf]
    if atp < gate:
        w["ema"]*=0.5; w["macd"]*=0.5
    S = (w["ema"]*s_ema + w["macd"]*s_macd + w["rsi"]*s_rsi + w["adx"]*s_adx + w["obv"]*s_obv) / sum(w.values())
    return S, {"S":S,"atp":atp}

def softmax_probs(S, tau):
    epos = math.exp(S/tau)
    eneg = math.exp(-S/tau)
    Z = epos + 1 + eneg
    return {"pb":epos/Z, "ph":1/Z, "ps":eneg/Z}

def aggregate_probs(per_tf):
    wsum=sum(TF_W.values())
    pb = sum(TF_W[tf]*per_tf[tf]["pb"] for tf in TF_W if tf in per_tf)/wsum
    ph = sum(TF_W[tf]*per_tf[tf]["ph"] for tf in TF_W if tf in per_tf)/wsum
    ps = sum(TF_W[tf]*per_tf[tf]["ps"] for tf in TF_W if tf in per_tf)/wsum
    Sw = sum(TF_W[tf]*per_tf[tf]["S"]  for tf in TF_W if tf in per_tf)/wsum
    tot=pb+ph+ps
    if tot<=0: pb=ph=ps=1/3
    else: pb,ph,ps = pb/tot, ph/tot, ps/tot
    out={"pb":pb,"ph":ph,"ps":ps,"Sw":Sw}
    for tf in ("30m","15m","5m"):
        if tf in per_tf:
            out[f"pb_{tf}"]=per_tf[tf]["pb"]
            out[f"ps_{tf}"]=per_tf[tf]["ps"]
            out[f"S_{tf}"]= per_tf[tf]["S"]
    return out

def apply_veto_m5(agg):
    pb,ps = agg["pb"], agg["ps"]
    pb30, pb15 = agg.get("pb_30m",0), agg.get("pb_15m",0)
    ps30, ps15 = agg.get("ps_30m",0), agg.get("ps_15m",0)
    S5 = agg.get("S_5m",0.0)
    bull_ok = (pb>=P_ENTER) and (pb30>=0.55) and (pb15>=0.55) and (S5>=0)
    bear_ok = (ps>=P_ENTER) and (ps30>=0.55) and (ps15>=0.55) and (S5<=0)
    if not bull_ok and pb>=P_ENTER: agg["pb"] = max(agg["pb"], P_EXIT-1e-6)
    if not bear_ok and ps>=P_ENTER: agg["ps"] = max(agg["ps"], P_EXIT-1e-6)
    agg["veto5"] = (not bull_ok and pb>=P_ENTER) or (not bear_ok and ps>=P_ENTER)
    return agg

def decide_ctx(pb,ph,ps, prev=None):
    # hystérésis
    if prev=="bull":
        if pb>=P_EXIT: return "bull"
        if ps>=P_ENTER: return "bear"
        return "none" if 0.40<=ph<=0.60 else "none"
    if prev=="bear":
        if ps>=P_EXIT: return "bear"
        if pb>=P_ENTER: return "bull"
        return "none" if 0.40<=ph<=0.60 else "none"
    # neutre
    if pb>=P_ENTER: return "bull"
    if ps>=P_ENTER: return "bear"
    if 0.40<=ph<=0.60: return "range"
    return "none"

def apply_veto_m5(agg,reasons):
    pb,ps=agg["pb"],agg["ps"]
    pb30,pb15=agg.get("pb_30m",0.0),agg.get("pb_15m",0.0)
    ps30,ps15=agg.get("ps_30m",0.0),agg.get("ps_15m",0.0)

def decide_next_ctx(prob,prev_ctx,reasons):

    reasons.append("under_thresholds"); return "none"
def update_context(state,prob,reasons,bar_closed=True):
    target=decide_next_ctx(prob,state.ctx,reasons)
    if target==state.ctx:
        if state.streak>=PERSIST_BARS: reasons.append("persistence_ok")
        return CtxState(ctx=state.ctx,streak_ctx=target,streak=min(state.streak+1,PERSIST_BARS),bars_in_ctx=state.bars_in_ctx+(1 if bar_closed else 0))

def sign(message: str, secret: str) -> str:

def auth_headers(path: str, method="GET", body="") -> dict:
    api_key     = os.getenv("BITGET_API_KEY")

def ema(x,n):
    if not x: return []
    k=2/(n+1); o=[x[0]]
    for v in x[1:]: o.append(o[-1]+k*(v-o[-1]))
    return o
def stdev(x):
    if len(x)<2: return 0.0
    mu=sum(x)/len(x)
    return (sum((v-mu)**2 for v in x)/len(x))**0.5
def atr(h,l,c,n=14):
    trs=[]; prev=c[0]
    for i in range(len(c)):
        trs.append(max(h[i]-l[i], abs(h[i]-prev), abs(l[i]-prev))); prev=c[i]
    return ema(trs,n)
def macd_hist(c, fast=12, slow=26, sig=9):
    m=[a-b for a,b in zip(ema(c,fast), ema(c,slow))]
    s=ema(m,sig); n=min(len(m),len(s))
    return [m[-n+i]-s[-n+i] for i in range(n)]
def rsi(c,n=14):
    if len(c)<n+1: return [50.0]*len(c)
    g=[0.0]; d=[0.0]
    for i in range(1,len(c)):
        ch=c[i]-c[i-1]; g.append(max(ch,0.0)); d.append(max(-ch,0.0))
    ag=ema(g,n); ad=ema(d,n)
    out=[]
    for x,y in zip(ag,ad):
        rs=x/(y+1e-12); out.append(100-100/(1+rs))
    return out
def adx(h,l,c,n=14):
    if len(c)<n+1: return [20.0]*len(c)
    pd=[0.0]; md=[0.0]; TR=[0.0]
    for i in range(1,len(c)):
        up=h[i]-h[i-1]; dn=l[i-1]-l[i]
        pd.append(up if (up>dn and up>0) else 0.0)
        md.append(dn if (dn>up and dn>0) else 0.0)
        TR.append(max(h[i]-l[i], abs(h[i]-c[i-1]), abs(l[i]-c[i-1])))
    pdi=[100*(x/(y+1e-12)) for x,y in zip(ema(pd,n), ema(TR,n))]
    mdi=[100*(x/(y+1e-12)) for x,y in zip(ema(md,n), ema(TR,n))]

def obv(c,v):
    o=[0.0]

def subscores_tf(o,h,l,c,v, tf):
    # EMA spread / ATR gate
    atr14 = atr(h,l,c,14)[-1]
    base = max(1.5*atr14,1e-9)
    s_ema = math.tanh( (ema(c,12)[-1] - ema(c,26)[-1]) / base )

    # MACD hist / sigma100
    hist = macd_hist(c)
    sigma = stdev(hist[-100:]) if len(hist)>=100 else stdev(hist)
    s_macd = math.tanh( (hist[-1])/(sigma+1e-12) )

    s_rsi = max(-1.0,min(1.0,(rsi(c,14)[-1]-50.0)/50.0))

    # ADX
    s_adx = max(-1.0,min(1.0,(adx(h,l,c,14)[-1]-20.0)/20.0))

    # OBV slope(20)/median(|OBV|_200)
    ob = obv(c,v)
    slope = ob[-1]-ob[-21] if len(ob)>=21 else 0.0
    denom = stats.median([abs(x) for x in (ob[-200:] if len(ob)>=200 else ob)]) or 1.0
    s_obv = math.tanh(slope/denom)

    # ATR% gate
    gate = (atr14 / c[-1]) < ATR_GATE[tf]
    w_ema = W_EMA*0.5 if gate else W_EMA
    w_macd= W_MACD*0.5 if gate else W_MACD

    den = (w_ema + w_macd + W_RSI + W_ADX + W_OBV)

def ema(series, n):
    if not series: return []
    k = 2/(n+1)
    out=[series[0]]
    for x in series[1:]:
        out.append(out[-1] + k*(x-out[-1]))
    return out
def stdev(xs):
    if len(xs)<2: return 0.0
    mu = sum(xs)/len(xs)
    return math.sqrt(sum((x-mu)**2 for x in xs)/len(xs))

def atr_ohlc(h,l,c,n=14):
    trs=[]; prev=c[0]
    for i in range(len(c)):
        hi,lo,cl=h[i],l[i],c[i]
        tr=max(hi-lo, abs(hi-prev), abs(lo-prev))
        trs.append(tr); prev=cl
    return ema(trs, n)

def macd_hist(cl, fast=12, slow=26, sig=9):
    macd_line=[a-b for a,b in zip(ema(cl,fast), ema(cl,slow))]
    signal = ema(macd_line, sig)
    # aligne
    n=min(len(macd_line), len(signal))
    return [macd_line[-n+i]-signal[-n+i] for i in range(n)]

def rsi14(cl, n=14):
    if len(cl)<n+1: return [50.0]*len(cl)
    gains=[0.0]; losses=[0.0]
    for i in range(1,len(cl)):
        ch=cl[i]-cl[i-1]
        gains.append(max(ch,0.0)); losses.append(max(-ch,0.0))
    avgG=ema(gains,n); avgL=ema(losses,n)
    out=[]
    for g,l in zip(avgG,avgL):
        rs = g/(l+1e-12)
        out.append(100.0 - 100.0/(1.0+rs))
    return out

def adx14(h,l,c,n=14):
    # DM+/DM-, TR; DX=100*|+DI - -DI|/(+DI + -DI); ADX=EMA(DX,n)
    if len(c)<n+1: return [20.0]*len(c)
    plusDM=[0.0]; minusDM=[0.0]; TR=[0.0]
    for i in range(1,len(c)):
        up=h[i]-h[i-1]; dn=l[i-1]-l[i]
        plusDM.append( up if (up>dn and up>0) else 0.0 )
        minusDM.append( dn if (dn>up and dn>0) else 0.0 )
        TR.append( max(h[i]-l[i], abs(h[i]-c[i-1]), abs(l[i]-c[i-1])) )
    plusDI   = [100*(x/(y+1e-12)) for x,y in zip(ema(plusDM,n), ema(TR,n))]
    minusDI  = [100*(x/(y+1e-12)) for x,y in zip(ema(minusDM,n), ema(TR,n))]
    DX = [100*abs(p-m)/(p+m+1e-12) for p,m in zip(plusDI, minusDI)]
    return ema(DX, n)

def obv(cl, vol):

    for i in range(1,len(cl)):
        if cl[i]>cl[i-1]: out.append(out[-1]+vol[i])
        elif cl[i]<cl[i-1]: out.append(out[-1]-vol[i])
        else: out.append(out[-1])
    return out

def subscores_tf(ohlcv, tf):
    # colonnes
    o=[r[1] for r in ohlcv]; h=[r[2] for r in ohlcv]; l=[r[3] for r in ohlcv]
    c=[r[4] for r in ohlcv]; v=[r[5] for r in ohlcv]
    if len(c)<500: return None

    # EMA spread / ATR gate
    atr = atr_ohlc(h,l,c,14)
    ema12 = ema(c,12); ema26 = ema(c,26)
    n=min(len(atr),len(ema12),len(ema26))
    atr14=atr[-1]; base=max(1.5*atr14,1e-9)
    s_ema = math.tanh( (ema12[-1]-ema26[-1]) / base )

    # MACD hist normalisé par sigma_100
    hist = macd_hist(c)
    sigma = stdev(hist[-100:]) if len(hist)>=100 else stdev(hist)
    s_macd = math.tanh( (hist[-1])/(sigma+1e-12) )

    # RSI
    r = rsi14(c)
    r_last = r[-1] if r else 50.0
    s_rsi = max(-1.0, min(1.0, (r_last-50.0)/50.0 ))

    # ADX
    adx = adx14(h,l,c,14)
    a_last = adx[-1] if adx else 20.0
    s_adx = max(-1.0, min(1.0, (a_last-20.0)/20.0 ))

    # OBV slope(20) / median(|OBV|_200)
    ob = obv(c,v)
    if len(ob)>=220:
        slope = ob[-1]-ob[-21]
        denom = stats.median([abs(x) for x in ob[-200:]]) or 1.0
    else:
        slope = ob[-1]-ob[-min(21,len(ob)-1)]
        denom = stats.median([abs(x) for x in ob]) or 1.0
    s_obv = math.tanh(slope/denom)

    # ATR% gate: si trop faible → demi-poids sur EMA+MACD
    gate = (atr14 / c[-1]) < ATR_GATE[tf]
    w_ema = W_EMA*0.5 if gate else W_EMA
    w_macd= W_MACD*0.5 if gate else W_MACD

    # Score TF ∈ [-1,1]
    num = w_ema*s_ema + w_macd*s_macd + W_RSI*s_rsi + W_ADX*s_adx + W_OBV*s_obv
    den = (w_ema + w_macd + W_RSI + W_ADX + W_OBV)
    S = num/den if den>0 else 0.0

    # Softmax tri-state
    t = TAU[tf]
    epos = math.exp(S/t); eneg = math.exp(-S/t); den3 = epos + 1.0 + eneg
    pb = epos/den3; ph = 1.0/den3; ps = eneg/den3

    return {
        "S": S, "pb": pb, "ph": ph, "ps": ps,
        "gate": gate, "atrpct": atr14/c[-1],
        "last_ts": ohlcv[-1][0]
    }

def aggregate(tf_res):
    # pondération et renormalisation
    S = sum(ALPHA[k]*tf_res[k]["S"] for k in tf_res)
    pb= sum(ALPHA[k]*tf_res[k]["pb"] for k in tf_res)
    ph= sum(ALPHA[k]*tf_res[k]["ph"] for k in tf_res)
    ps= sum(ALPHA[k]*tf_res[k]["ps"] for k in tf_res)
    s = pb+ph+ps
    if s>0: pb,ph,ps = pb/s, ph/s, ps/s
    return S, pb, ph, ps

def veto_context(tf_res, S5):
    # util pour veto 5m
    return (S5<0), (S5>0)

def decide_context(pb,ph,ps, tf_res):

    bull = pb>=0.60 and (tf_res["30m"]["pb"]>=0.55) and (tf_res["15m"]["pb"]>=0.55)
    bear = ps>=0.60 and (tf_res["30m"]["ps"]>=0.55) and (tf_res["15m"]["ps"]>=0.55)
    veto_buy, veto_sell = veto_context(tf_res, tf_res["5m"]["S"])
    if bull and not veto_buy: return "buy"
    if bear and not veto_sell: return "sell"
    if 0.40 <= ph <= 0.60: return "hold"
    return "none"

    if len(c)<n+1: return [50.0]*len(c)

    for i in range(1,len(c)):
        ch=c[i]-c[i-1]; g.append(max(ch,0.0)); d.append(max(-ch,0.0))

        rng=hh-ll
        return 25.0 if rng>0 and (ch[-1]-ch[-6])>0 else 15.0

                "why":["BB2sigma-touch","RSI>40","reversal"]}
    return None

    max_notional=balance*MAX_EXPO*LEV
    if qty*entry>max_notional: qty=max_notional/entry
    step=LOT_STEP
    qty=math.floor(qty/step)*step
    return qty

def ema(vals, n):

    for v in vals:
        e = v if e is None else (v - e)*k + e
        out.append(e)
    return out

def rsi(vals, n=14):
    if len(vals)<n+1: return [50.0]*len(vals)
    gains=[0.0]; losses=[0.0]
    for i in range(1,len(vals)):
        d=vals[i]-vals[i-1]
        gains.append(max(d,0)); losses.append(max(-d,0))
    avg_gain=sum(gains[1:n+1])/n; avg_loss=sum(losses[1:n+1])/n
    rsis=[50.0]*(n)
    for i in range(n+1,len(vals)+1):
        avg_gain=(avg_gain*(n-1)+gains[i-1])/n
        avg_loss=(avg_loss*(n-1)+losses[i-1])/n
        rs= (avg_gain/avg_loss) if avg_loss>0 else 0
        r=100-(100/(1+rs)) if avg_loss>0 else 100
        rsis.append(r)
    return rsis

def true_range(h,l,c_prev): return max(h-l, abs(h-c_prev), abs(l-c_prev))
def atr(hs, ls, cs, n=14):
    trs=[]
    for i in range(len(cs)):
        if i==0: trs.append(hs[i]-ls[i])
        else: trs.append(true_range(hs[i], ls[i], cs[i-1]))
    out=[]; a=None; k=2/(n+1)
    for tr in trs:
        a = tr if a is None else (tr-a)*k + a
        out.append(a)
    return out

def adx(hs, ls, cs, n=14):
    # simple Wilder
    if len(cs)<n+1: return [20.0]*len(cs)
    pdm=[0.0]; ndm=[0.0]
    for i in range(1,len(cs)):
        up=hs[i]-hs[i-1]; dn=ls[i-1]-ls[i]
        pdm.append(up if (up>dn and up>0) else 0.0)
        ndm.append(dn if (dn>up and dn>0) else 0.0)
    tr = []
    for i in range(len(cs)):
        if i==0: tr.append(hs[i]-ls[i])
        else: tr.append(true_range(hs[i], ls[i], cs[i-1]))
    def rma(x,n):
        k=1/n; out=[]; a=None
        for v in x:
            a = v if a is None else a + k*(v-a)
            out.append(a)
        return out
    atrv=rma(tr,n); pdi=[0.0]*len(cs); ndi=[0.0]*len(cs)
    p=rma(pdm,n); n_=rma(ndm,n)
    for i in range(len(cs)):
        if atrv[i]>0:
            pdi[i]=100*p[i]/atrv[i]; ndi[i]=100*n_[i]/atrv[i]
    dx=[0.0]*len(cs)
    for i in range(len(cs)):
        s=pdi[i]+ndi[i]; dx[i]= (100*abs(pdi[i]-ndi[i])/s) if s>0 else 0.0
    return rma(dx,n)

def macd_hist(vals, f=12, s=26, sig=9):
    e1=ema(vals,f); e2=ema(vals,s)
    macd=[e1[i]-e2[i] for i in range(len(vals))]
    sigl=ema(macd,sig)
    return [macd[i]-sigl[i] for i in range(len(vals))]

def bb(vals, n=20, k=2.0):
    if len(vals)<n: return [None]*len(vals), [None]*len(vals)
    ma=[]; dev=[]
    for i in range(len(vals)):
        if i<n-1:
            ma.append(None); dev.append(None)
        else:
            w=vals[i-n+1:i+1]
            m=sum(w)/n
            v=(sum((x-m)**2 for x in w)/n)**0.5
            ma.append(m); dev.append(v)
    upper=[(ma[i]+k*dev[i]) if ma[i] is not None else None for i in range(len(vals))]
    lower=[(ma[i]-k*dev[i]) if ma[i] is not None else None for i in range(len(vals))]
    return upper, lower, ma

def keltner(hs, ls, cs, n=20, k=1.5):
    ema_c=ema(cs,n)
    atrv=atr(hs,ls,cs,n)
    upper=[ema_c[i]+k*atrv[i] for i in range(len(cs))]
    lower=[ema_c[i]-k*atrv[i] for i in range(len(cs))]
    return upper, lower, ema_c

def obv(cl, vo):
    o=[0.0]
    for i in range(1,len(cl)):
        if cl[i]>cl[i-1]: o.append(o[-1]+vo[i])
        elif cl[i]<cl[i-1]: o.append(o[-1]-vo[i])
        else: o.append(o[-1])
    return o

def vwap(arr, n=20):
    tpv=tp(arr); v=vol(arr)
    out=[]; num=0.0; den=0.0
    for i in range(len(arr)):
        num += tpv[i]*v[i]; den += v[i]
        out.append(num/den if den>0 else tpv[i])

def entry_score(m1, m3, m5, ctx):

# ---------- Sizing ----------
def round_step(q, step):
    if step<=0: return q
    return math.floor(q/step)*step

def size_order(price, sl, balance, risk_pct, leverage, min_notional, max_pair_expo_pct, lot_step):
    D = abs(price - sl)
    if D<=0: return 0.0, 0.0, 0.0
    R = balance * risk_pct
    qty = R / D
    notional = qty * price
    # constraints
    if notional < min_notional:
        qty = min_notional / price
        notional = min_notional
    max_notional = balance * max_pair_expo_pct * leverage
    if notional > max_notional:
        qty = max_notional / price
        notional = max_notional
    qty = round_step(qty, lot_step)
    margin = notional / leverage
    return qty, notional, margin
