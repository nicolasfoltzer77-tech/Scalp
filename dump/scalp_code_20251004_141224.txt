#!/usr/bin/env python3
import ccxt, json, time
from pathlib import Path

DATA = Path("/opt/scalp/data")
CACHE90 = DATA / "top90_active.json"
INACTIVE = DATA / "coins_90d.json"

EXCLUDE_LEV = ("UP","DOWN","BULL","BEAR","3L","3S","5L","5S")
EXCLUDE_STABLES = {"USDT","USDC","FDUSD","TUSD","DAI","EUR","TRY","USD"}

def load_usdt_swaps():
    ex = ccxt.bitget({'options': {'defaultType':'swap'}, 'enableRateLimit': True})
    ex.load_markets()
    mapping = {}
    for sym, m in ex.markets.items():
        if not m.get('swap'): continue
        if m.get('quote') != 'USDT': continue
        base = (m.get('base') or '').upper()
        if base in EXCLUDE_STABLES: continue
        if any(tag in base for tag in EXCLUDE_LEV): continue
        mapping[base] = sym
    return ex, mapping

def has_2_daily_candles_around(ex, market, days=90, tol=3):
    now_ms = int(time.time()*1000)
    day = 24*3600*1000
    since = now_ms - (days + tol)*day
    rows = ex.fetch_ohlcv(market, '1d', since=since, limit=10)
    return len(rows) >= 2

def main():
    DATA.mkdir(parents=True, exist_ok=True)
    ex, mapping = load_usdt_swaps()
    actives, inact = [], []
    for base, mkt in mapping.items():
        ok = False
        try:
            ok = has_2_daily_candles_around(ex, mkt, 90, 3)
        except Exception:
            ok = False
        if ok:
            actives.append(base)
        else:
            inact.append({"symbol": base, "reason": "no_1d_90d_window"})
    actives = sorted(dict.fromkeys(actives))
    CACHE90.write_text(json.dumps(actives, indent=2))
    INACTIVE.write_text(json.dumps(inact, indent=2))
    print(f"ACTIVE={len(actives)} INACTIVE={len(inact)}")
if __name__ == "__main__":
    main()

# --- OHLCV UPSERT (SQLite) ---
def upsert_ohlcv(conn, table, rows):
    """
    Upsert de bougies OHLCV dans 'table' (schema ts,o,h,l,c,v).
    - conn : sqlite3.Connection
    - table: str (ex: 'ohlcv_m1', 'ohlcv_m5', ...)
    - rows : iterable de tuples (ts, o, h, l, c, v) (ts en secondes ou ms selon ton schéma)
    Crée la table si nécessaire, clé primaire = ts, et met à jour en cas de conflit.
    """
    cur = conn.cursor()
    cur.execute(f"""
        CREATE TABLE IF NOT EXISTS {table} (
            ts INTEGER PRIMARY KEY,
            o  REAL NOT NULL,
            h  REAL NOT NULL,
            l  REAL NOT NULL,
            c  REAL NOT NULL,
            v  REAL NOT NULL
        )
    """)
    cur.execute(f"CREATE UNIQUE INDEX IF NOT EXISTS idx_{table}_ts ON {table}(ts)")
    cur.executemany(
        f"""INSERT INTO {table} (ts,o,h,l,c,v) VALUES (?,?,?,?,?,?)
            ON CONFLICT(ts) DO UPDATE SET
              o=excluded.o, h=excluded.h, l=excluded.l, c=excluded.c, v=excluded.v
        """,
        list(rows)  # on matérialise au cas où rows est un générateur consommable une fois
    )
    conn.commit()

def _ohlcv_cols(conn, table):
    cur=conn.cursor()
    cur.execute(f"PRAGMA table_info({table})")
    cols=[r[1] for r in cur.fetchall()]  # r[1] = name
    cand=[("open","high","low","close","volume"),
          ("o","h","l","c","v"),
          ("op","hi","lo","cl","vol")]

    if len(rows)<n+1: return None
    trs=[]; pc=rows[0][4]
    for i in range(1,len(rows)):
        h,l,c=rows[i][2],rows[i][3],rows[i][4]
        trs.append(true_range(h,l,pc)); pc=c
    return sum(trs[-n:])/n if trs else None
def macd_hist(vals, f=12, s=26, sig=9):
    if len(vals)<s+sig+5: return None
    ef,es=ema(vals,f),ema(vals,s)
    macd=[ef[i]-es[i] for i in range(len(vals))]
    sigl=ema(macd,sig)
    return macd[-1]-sigl[-1], (macd[-2]-sigl[-2] if len(macd)>1 else 0.0)
def bb(vals, n=20, k=2.0):
    if len(vals)<n: return None
    w=vals[-n:]; m=sum(w)/n; var=sum((x-m)**2 for x in w)/n; sd=max(1e-12,var**0.5)
    return m, m+k*sd, m-k*sd, sd
def vwap(rows, n=20):
    if len(rows)<1: return None
    def tp(r): return (r[2]+r[3]+r[4])/3
    num=den=0.0; out=[]
    for i,r in enumerate(rows):
        num += tp(r)*r[5]; den += r[5]
        out.append(num/den if den>0 else tp(r))
        if i>=n:
            j=rows[i-n]; num -= tp(j)*j[5]; den -= j[5]
    return out

def prefilters(sym, m1):
    # ATR% 1m >= 0.10%
    c1=[x[4] for x in m1]
    if not c1 or len(m1)<PARAM["atr_per"]+2: return False,"not_enough_1m"
    a1=atr(m1, int(PARAM["atr_per"]))
    atrpct = (a1/(c1[-1] or 1e-12))*100 if a1 else 0.0
    if atrpct < 0.10: return False,"atr1%<0.10"
    # cooldown

    if int(time.time()) < cd: return False,"cooldown"
    # spread approx
    rng=(m1[-1][2]-m1[-1][3])/(c1[-1] or 1e-12)
    if rng > 0.010: return False,"spread>1%"
    return True,"ok"

# ---------- Evidence ----------
def evidence_pullback(ctx, m1):
    c=[x[4] for x in m1]; v=[x[5] for x in m1]
    bbv=bb(c,20,2.0); vw=vwap(m1,20); e20=ema(c,20)
    if bbv is None or vw is None: return 0.0
    mid,up,lo,sd = bbv
    r7=rsi(c,7)
    v20 = sum(v[-20:])/max(1,len(v[-20:]))
    if ctx=="buy":
        touch = (c[-1]<=lo) or (c[-1]<=e20[-1]) or (c[-1]<=vw[-1]); side = r7<=45
    elif ctx=="sell":
        touch = (c[-1]>=up) or (c[-1]>=e20[-1]) or (c[-1]>=vw[-1]); side = r7>=55
    else:
        return 0.0
    volok = v[-1] >= PARAM["vol_mult_pullback"]*v20
    return 1.0 if (touch and side and volok) else 0.0

def macd_accel(vals):
    h=macd_hist(vals)
    if not h: return 0.0
    mh,_=h
    # acceleration simple: 3 dernières croissantes/descendantes
    # (approx via diff)
    if len(vals)<5: return 0.0
    e12,e26=ema(vals,12),ema(vals,26)
    M=[e12[i]-e26[i] for i in range(len(vals))]
    d1=M[-1]-M[-2]; d2=M[-2]-M[-3]
    return 1.0 if (d1>0 and d2>0) or (d1<0 and d2<0) else 0.0

def squeeze_ok(h,l,c):
    # BB width < Keltner width
    bbv=bb(c,20,2.0)
    if bbv is None: return 0.0
    mid,up,lo,sd=bbv
    # Keltner (approx EMA20 +/- 1.5*ATR)
    a = sum(true_range(h[i],l[i],c[i-1] if i else c[0]) for i in range(len(c)))/max(1,len(c))
    width_bb = up - lo
    width_k  = 3.0*a
    return 1.0 if width_bb < width_k else 0.0

def evidence_breakout(ctx, m3, m5):
    c3=[x[4] for x in m3]; h3=[x[2] for x in m3]; l3=[x[3] for x in m3]
    c5=[x[4] for x in m5]; h5=[x[2] for x in m5]; l5=[x[3] for x in m5]
    N=int(PARAM["N_breakout"])
    if len(c3)<N+2: return 0.0
    hh=max(c3[-N-1:-1]); ll=min(c3[-N-1:-1])
    accel = macd_accel(c3)
    sq = squeeze_ok(h3,l3,c3)
    # ADX proxy via ATR% 5m
    a5=atr(m5,int(PARAM["atr_per"])) or 0.0
    atrp5 = (a5/(c5[-1] or 1e-12))*100
    adx_ok = atrp5>=0.20
    if ctx=="buy" and c3[-1]>hh and adx_ok and accel and sq: return 1.0
    if ctx=="sell" and c3[-1]<ll and adx_ok and accel and sq: return 1.0
    return 0.0

def candle_reversal(c):
    if len(c)<3: return False
    b2,b1,c0 = c[-3],c[-2],c[-1]
    return (c0<b1 and b1>b2) or (c0>b1 and b1<b2)

def evidence_meanrev(ctx, m1):
    if ctx!="hold": return 0.0
    c=[x[4] for x in m1]; bbv=bb(c,20,2.0)
    if bbv is None: return 0.0
    mid,up,lo,sd=bbv; r7=rsi(c,7)
    if c[-1]>=up and r7>=60 and candle_reversal(c): return 1.0
    if c[-1]<=lo and r7<=40 and candle_reversal(c): return 1.0
    return 0.0

def tick_confirm(sym, ctx, lvl):
    """Confirme une casse au tick: prix dépasse le niveau +/- 0.02%"""
    px,ts = load_tick(sym)
    if px is None: return False
    tol=0.0002
    if ctx=="buy"  and px >= lvl*(1+tol): return True
    if ctx=="sell" and px <= lvl*(1-tol): return True
    return False

# ---------- Sizing ----------
def round_step(q, step):
    if step<=0: return q
    return math.floor(q/step)*step

def size_order(price, sl, balance, risk_pct, leverage, min_notional, max_pair_expo_pct, lot_step):
    D = abs(price-sl) or 1e-12
    R = balance * risk_pct
    qty = R / D
    notional = qty * price
    if notional < min_notional:
        qty = min_notional / price
        notional = min_notional
    max_notional = balance * max_pair_expo_pct * leverage
    if notional > max_notional:
        qty = max_notional / price
        notional = max_notional
    qty = round_step(qty, lot_step)
    margin = notional / leverage
    return qty, notional, margin

def analyze_symbol(sym, ctx_row, cx, balance=1000.0):
    # 1) données
    m1 = fetch_tf(cx,"ohlcv_m1", sym, 900)
    if len(m1)<120: return {"symbol":sym,"ctx":"none","why":"no_1m"}
    m3 = fetch_tf(cx,"ohlcv_m3", sym, 700)
    if len(m3)<40:  m3 = resample_from(m1, 3*60*1000)
    m5 = fetch_tf(cx,"ohlcv_m5", sym, 500)
    if len(m5)<40:  m5 = resample_from(m1, 5*60*1000)

    ok,why = prefilters(sym, m1)
    if not ok: return {"symbol":sym,"ctx":"none","why":why}

    ctx = ctx_row.get("ctx","none")
    # 2) score
    e_pb = evidence_pullback(ctx, m1)
    e_br = evidence_breakout(ctx, m3, m5)
    e_mr = evidence_meanrev(ctx, m1)
    score = 0.5*e_pb + 0.3*e_br + 0.2*e_mr

    veto = veto_5m(ctx, m5)
    if veto!=1.0:
        score = min(score, veto)

    # persistance
    pst = jread(STATE/"persist_B.json", {})
    st  = pst.get(sym, {"ge":0})
    if score >= PARAM["score_enter"]:
        st["ge"] = st.get("ge",0)+1
    else:
        st["ge"] = 0
    pst[sym]=st
    (STATE/"persist_B.json").write_text(json.dumps(pst, indent=2))

    signal=None; note="score={:.2f} pb={} br={} mr={} veto={}".format(score,e_pb,e_br,e_mr,veto)
    if st["ge"] >= int(PARAM["persist_bars"]):
        # 3) SL/TP + sizing
        c1=[x[4] for x in m1]; h1=[x[2] for x in m1]; l1=[x[3] for x in m1]
        atr1 = atr(m1, int(PARAM["atr_per"])) or 0.0
        entry = c1[-1]
        if ctx=="buy":
            sl = entry - PARAM["atr_sl_k"]*atr1; tp = entry + PARAM["atr_tp_k"]*atr1
        elif ctx=="sell":
            sl = entry + PARAM["atr_sl_k"]*atr1; tp = entry - PARAM["atr_tp_k"]*atr1
        else:
            # mean-rev en hold -> TP vers EMA20
            mid = ema(c1,20)[-1]
            if c1[-1] >= (bb(c1,20,2.0)[1] if bb(c1,20,2.0) else c1[-1]):
                ctx="sell"; sl=entry + 1.0*atr1; tp=mid
            else:
                ctx="buy";  sl=entry - 1.0*atr1; tp=mid

        # tick confirm (si breakout actionné, on confirme la casse au tick à ~0.02%)
        if e_br==1.0:
            lvl = entry
            if not tick_confirm(sym, ctx, lvl):
                note += " | tick_not_confirmed"
                # on ne bloque pas, mais on annote

        qty, notion, margin = size_order(entry, sl, balance,
                                         PARAM["risk_pct"], PARAM["leverage"],
                                         PARAM["min_notional"], PARAM["max_pair_expo_pct"],
                                         PARAM["lot_step"])
        if qty>0:
            signal = {
                "ts": int(time.time()),
                "symbol": sym,
                "side": "buy" if ctx=="buy" else "sell",
                "entry": round(entry,6),
                "sl": round(sl,6),
                "tp": round(tp,6),
                "qty": qty,
                "notional": round(notion,2),
                "margin": round(margin,2),
                "leverage": PARAM["leverage"],
                "trail_atr": round(atr1*PARAM["trail_k"],6),
                "timeout_s": 25*60,
                "why": note,
            }
            # cooldown
            cd=jread(STATE/"cooldown.json",{})
            cd[sym]= int(time.time()) + int(PARAM["cooldown_1m"])*60
            (STATE/"cooldown.json").write_text(json.dumps(cd,indent=2))
            # reset persistance
            st["ge"]=0; pst[sym]=st
            (STATE/"persist_B.json").write_text(json.dumps(pst, indent=2))

    return {"symbol":sym,"ctx":ctx,"score":round(score,3),"signal":signal,"why":note}

def softmax_tri(S, tau):

def get_tick(con, base):
    r=q(con,"SELECT last, bid, ask, ts_local FROM tick_last_mat WHERE inst_id=? ORDER BY ts_local DESC LIMIT 1",(base,)).fetchone()
    if not r: return None
    last,bid,ask,ts=r; return float(last),float(bid or last),float(ask or last),int(ts)

def close_trade(con, live, reason, px):
    ts=now_ms()
    # PnL
    side=live["side"]; qty=live["qty"]; entry=live["entry_px"]
    pnl = (px-entry)*(qty if side=="buy" else -qty)
    pnl_pct = (px-entry)/(entry or 1.0) * (1 if side=="buy" else -1)
    q(con,"INSERT INTO paper_closes(lid,close_ts,close_px,reason,pnl_abs,pnl_pct) VALUES(?,?,?,?,?,?)",
      (live["lid"], ts, px, reason, pnl, pnl_pct))
    q(con,"UPDATE paper_live SET status='CLOSED' WHERE lid=?", (live["lid"],))

    delta_qty = -qty if side=="buy" else qty
    r=q(con,"SELECT qty,avg_px FROM paper_positions WHERE symbol=?", (live["symbol"],)).fetchone()
    if r:
      new_qty=r[0]+delta_qty
      new_avg=None if new_qty==0 else r[1]
      q(con,"UPDATE paper_positions SET qty=?, avg_px=?, last_ts=? WHERE symbol=?", (new_qty,new_avg,ts,live["symbol"]))
    con.commit()

def maybe_trail(live, px, atr):
    if live["trail_atr"] is None or atr is None: return live["sl"]
    moved = (px - live["entry_px"]) if live["side"]=="buy" else (live["entry_px"] - px)
    thresh = live["trail_atr"] * atr
    if moved >= thresh:
        if live["side"]=="buy":
            new_sl = max(live["sl"] or -1e18, px - thresh)
        else:
            new_sl = min(live["sl"] or  1e18, px + thresh)
        return new_sl
    return live["sl"]

      WHERE symbol=? AND tf='1m'
      ORDER BY ts DESC LIMIT 30
    """,(base,)).fetchall()

    rows=list(reversed(rows))
    import math
    atr=0.0
    prev_close=rows[0][2]
    trs=[]
    for h,l,c in rows[1:]:
        tr=max(h-l, abs(h-prev_close), abs(l-prev_close))
        trs.append(tr)
        prev_close=c
    n=14; atr = sum(trs[:n])/n

        atr = (atr*(n-1)+t)/n

"""
Logic B — score/signal + sizing & TP/SL.
Sortie decide(): {score,pb,br,mr,veto,signal,why, size, leverage, entry, tp, sl, rr}
"""

import os, math

# ---------------- Params (env ou défauts) ----------------
PARAM = {
  "ema_fast": 9,
  "ema_slow": 21,
  "rsi_per": 14,
  "atr_per": 14,
  "atr_min_1m": float(os.environ.get("ATR_MIN_1M","0.05")),   # % mini d'ATR 1m
  "risk_level": int(os.environ.get("RISK_LEVEL","5")),        # 0..10 -> seuil adaptatif
  "score_gate": 0.60,                                         # base

  # --- Risk & sizing ---
  "equity_usdt": float(os.environ.get("EQUITY_USDT","1000")), # capital de référence
  "risk_pct":   float(os.environ.get("RISK_PCT","0.003")),     # % équité risqué par trade (ex 0.003=0.3%)
  "leverage":   float(os.environ.get("LEVERAGE","5")),         # levier cible
  "tp_atr":     float(os.environ.get("TP_ATR","1.5")),         # TP = entry ± tp_atr * ATR(1m)
  "sl_atr":     float(os.environ.get("SL_ATR","0.8")),         # SL = entry ∓ sl_atr * ATR(1m)
  "min_notional": float(os.environ.get("MIN_NOTIONAL","5")),   # min notionnel (anti under-min)
}

def _ema(arr, p):
    if not arr or p<=0: return []
    k=2/(p+1); s=None; out=[]
    for x in arr: s = x if s is None else (x - s)*k + s; out.append(s)
    return out

def _rsi(closes, p=14):
    if len(closes)<p+2: return None
    gains=[]; losses=[]
    for i in range(1,len(closes)):
        d=closes[i]-closes[i-1]
        gains.append(max(0.0,d)); losses.append(max(0.0,-d))
    ag=sum(gains[-p:])/p; al=sum(losses[-p:])/p
    if al==0: return 100.0
    rs = ag/al
    return 100 - 100/(1+rs)

def _atr(ohlc, p=14):
    if len(ohlc)<p+1: return None
    trs=[]
    for i in range(1,len(ohlc)):
        _,o,h,l,c,_ = ohlc[i]
        pc = ohlc[i-1][4]
        trs.append(max(h-l, abs(h-pc), abs(l-pc)))
    return sum(trs[-p:])/p if trs else None

def _closes(m): return [x[4] for x in m] if m else []
def _softmax(a,b,c):
    m=max(a,b,c); ea,eb,ec = math.exp(a-m), math.exp(b-m), math.exp(c-m); s=ea+eb+ec
    return (ea/s, eb/s, ec/s)

# ---------------- Prefilters ----------------
def prefilters(sym, m1, m3, m5):
    closes1 = _closes(m1)
    if not closes1 or len(m1)<PARAM["atr_per"]+1:
        return False, "not_enough_m1", 1.0, None, None
    atr1 = _atr(m1, PARAM["atr_per"])
    lastc = closes1[-1]
    if not atr1 or lastc<=0:
        return False, "atr_unavail", 1.0, None, None
    atr_pct = (atr1/lastc)*100.0
    th = PARAM["atr_min_1m"]
    if atr_pct < th:
        return False, f"atr1%<{th:.2f}", 1.0, atr1, lastc
    return True, "ok", 0.0, atr1, lastc

# ---------------- Sizing ----------------
def _sizing(entry, atr1, side):
    """Retourne (size, lev, tp, sl, rr)"""
    eq   = PARAM["equity_usdt"]
    rpct = PARAM["risk_pct"]         # fraction d'équité risquée
    lev  = PARAM["leverage"]

    risk_usd = max(1e-9, eq * rpct)  # $ à risque
    sl_dist  = PARAM["sl_atr"] * atr1
    tp_dist  = PARAM["tp_atr"] * atr1
    if sl_dist <= 0: return 0.0, lev, None, None, 0.0

    # quantité nominale (en coin) pour que la perte à SL ~= risk_usd / lev
    qty = (risk_usd * lev) / sl_dist
    # min notionnel
    notionnel = entry * qty
    if notionnel < PARAM["min_notional"]:
        qty = PARAM["min_notional"] / max(1e-9, entry)

    if side=="long":
        tp = entry + tp_dist
        sl = entry - sl_dist
    else:
        tp = entry - tp_dist
        sl = entry + sl_dist

    rr = tp_dist / sl_dist if sl_dist>0 else 0.0
    return float(qty), float(lev), float(tp), float(sl), float(rr)

# ---------------- Decide ----------------
def decide(sym, m1, m3, m5, ctx):
    ok, why_pre, veto, atr1, last1 = prefilters(sym, m1, m3, m5)
    if not ok:
        return {"score":0.0,"pb":0.0,"br":0.0,"mr":0.0,"veto":veto,"signal":None,"why":why_pre}

    if ctx=="none":
        return {"score":0.0,"pb":0.0,"br":0.0,"mr":0.0,"veto":0.0,"signal":None,"why":"ctx_none"}

    c5=_closes(m5); c3=_closes(m3)
    if len(c5)<max(30, PARAM["ema_slow"]+1) or len(c3)<PARAM["rsi_per"]+2:
        return {"score":0.0,"pb":0.0,"br":0.0,"mr":0.0,"veto":0.0,"signal":None,"why":"not_enough_data"}

    eF=_ema(c5, PARAM["ema_fast"]); eS=_ema(c5, PARAM["ema_slow"])
    r3=_rsi(c3, PARAM["rsi_per"])
    if r3 is None:
        return {"score":0.0,"pb":0.0,"br":0.0,"mr":0.0,"veto":0.0,"signal":None,"why":"rsi_unavail"}

    trend = 1.0 if eF[-1] > eS[-1] else -1.0
    mom5 = (eF[-1]-eF[-5]) if len(eF)>5 else 0.0
    mom5_n = mom5/max(1e-9,abs(c5[-1]))
    rsi_n = (r3-50.0)/50.0

    S = 0.55*trend + 0.30*mom5_n*100.0 + 0.15*rsi_n
    pb, ph, ps = _softmax(S, 0.0, -S)

    align = pb if ctx=="buy" else ps
    score = float(max(0.0, align))

    risk = max(0,min(10, PARAM["risk_level"]))
    gate = PARAM["score_gate"] - (risk-5)*0.03
    gate = max(0.40, min(0.80, gate))

    signal=None; reason=f"S={S:.3f} ctx={ctx} pb={pb:.3f} ph={ph:.3f} ps={ps:.3f} gate={gate:.2f}"
    size=0.0; lev=PARAM["leverage"]; tp=None; sl=None; rr=0.0; entry=last1 if last1 else (c5[-1] if c5 else None)

    if score >= gate:
        side = "long" if ctx=="buy" else "short"
        signal = side
        if atr1 and entry:
            size, lev, tp, sl, rr = _sizing(entry, atr1, side)
        reason = "PASS " + reason + (f" atr1={atr1:.6f}" if atr1 else "")
    else:
        reason = "HOLD " + reason

    return {
        "score": round(score,3),
        "pb": round(pb,3), "br": round(ph,3), "mr": round(ps,3),
        "veto": veto, "signal": signal, "why": reason,
        "size": round(size,8), "leverage": lev, "entry": entry,
        "tp": tp, "sl": sl, "rr": round(rr,2),
    }

if __name__=="__main__":
    print("b_logic ready + sizing; PARAM:", PARAM)
