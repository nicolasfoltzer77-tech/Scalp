#!/usr/bin/env python3
import json, time, math
from urllib.request import urlopen, Request
from urllib.parse import urlencode
from urllib.error import URLError, HTTPError

BASE="https://api.bitget.com"

def _http_json(path, params=None, tries=3, timeout=10):
    if params: path += ("?" + urlencode(params))
    for i in range(tries):
        try:
            with urlopen(Request(BASE+path, headers={"User-Agent":"scalp/1"}), timeout=timeout) as r:
                raw=r.read().decode("utf-8")
                return json.loads(raw)
        except (URLError, HTTPError):
            if i+1>=tries: raise
            time.sleep(0.6*(2**i))

def ticker_last(symbol:str)->float|None:
    # v2
    j=_http_json("/api/v2/mix/market/ticker", {"symbol":symbol})
    data=j.get("data") or {}
    p=data.get("lastPr") or data.get("last") or data.get("close")
    try: return float(p)
    except: return None

def klines(symbol:str, granularity:int, limit:int=300):
    # try v2 first
    j=_http_json("/api/v2/mix/market/candles", {"symbol":symbol,"granularity":granularity,"limit":limit})
    data=j.get("data")
    if not data:
        j=_http_json("/api/mix/v1/market/candles", {"symbol":symbol,"granularity":granularity,"limit":limit})
        data=j.get("data")
    rows=[]
    # Bitget renvoie récents→anciens sous forme ["ts","open","high","low","close","volume"]
    for r in (data or []):
        try:
            ts=int(r[0])
            o,h,l,c=map(float,(r[1],r[2],r[3],r[4]))
            rows.append((ts,o,h,l,c))
        except: continue
    rows.sort(key=lambda x:x[0])  # ancien→récent
    return rows

# util
def tf_to_seconds(tf:str)->int:
    m={"1m":60,"5m":300,"15m":900,"30m":1800,"1h":3600}
    return m.get(tf,60)

import os, json, time, ccxt, hashlib
from pathlib import Path
from concurrent.futures import ThreadPoolExecutor, as_completed

DATA=Path("/opt/scalp/data")
F_OK=DATA/"coins_90d.json"

F_TOP=DATA/"top.json"

    return ccxt.bitget({
        "apiKey": os.getenv("BITGET_API_KEY",""),
        "secret": os.getenv("BITGET_API_SECRET",""),

    try:
        j=json.loads(Path(p).read_text() or "{}")

    except: return []

def load_fixed():
    try:
        j=json.loads(F_FIX.read_text()); lst=j.get("symbols",[])
        return [s for s in lst if isinstance(s,str)]
    except: return []

def universe():
    seen=set(); out=[]
    for s in load_fixed()+load_list(F_OK)+load_list(F_TOP)+load_list(F_PEN):
        if s and s not in seen:
            out.append(s); seen.add(s)

    return bitget_id.replace("_UMCBL","/USDT:USDT")

    h=hashlib.sha1(sym.encode()).hexdigest()[:16]
    return CACHE_DIR/f"{h}.json"

        ohlcv=EX.fetch_ohlcv(ccxt_symbol(sym), timeframe="1h", limit=LIMIT)

def parse_iso(ts):
    try:

        return 0

def ttl_left(gen_iso, ttl_s):
    ts = parse_iso(gen_iso)

#!/usr/bin/env python3
import os, json, time, math
from pathlib import Path
import ccxt

DATA = Path("/opt/scalp/data")
COINS = DATA/"coins_90d.json"
OUT   = DATA/"top.json"

def read_tuning():
    keep = 30
    fixed = ["BTCUSDT","ETHUSDT","XRPUSDT","SOLUSDT","BNBUSDT"]
    p = Path("/etc/scalp.tuning")
    if p.exists():
        for ln in p.read_text().splitlines():
            ln = ln.strip()
            if not ln or ln.startswith("#"): continue
            if ln.startswith("KEEP="):
                try: keep = int(ln.split("=",1)[1])
                except: pass
            elif ln.startswith("FIXED_TOP="):
                fixed = [x.strip() for x in ln.split("=",1)[1].split(",") if x.strip()]
    return keep, fixed

def ex_swap():
    return ccxt.bitget({
        "options":{"defaultType":"swap","defaultSubType":"linear"},
        "enableRateLimit": True,
    })

def ex_spot():
    return ccxt.bitget({
        "options":{"defaultType":"spot"},
        "enableRateLimit": True,
    })

def to_ccxt_symbol(base, mkts_swap, mkts_spot):
    b = base.upper()
    if b in mkts_swap:   return b
    if b in mkts_spot:   return b
    if "/" in b:         # déjà ccxt
        if b in mkts_swap or b in mkts_spot: return b
    # normalisations
    if b.endswith("_UMCBL"):
        root = b.split("_")[0]
    else:
        root = b
    if root.endswith("USDT"):
        coin = root[:-4]
        swap = f"{coin}/USDT:USDT"
        spot = f"{coin}/USDT"
        if swap in mkts_swap: return swap
        if spot in mkts_spot: return spot
    return None

def drop_incomplete(rows, tf_s):
    if not rows: return rows
    now=int(time.time()); last=int(rows[-1][0]//1000)
    if now-last < int(0.9*tf_s): rows = rows[:-1]
    return rows

def score_symbol(ohlcv):
    if len(ohlcv) < 48: return 0.0
    # vol = std(log-returns) 48h
    import math, statistics
    c = [r[4] for r in ohlcv]
    v = [r[5] for r in ohlcv]
    rets = []
    for i in range(1,len(c)):
        if c[i-1] and c[i]:
            rets.append(math.log(c[i]/c[i-1]))
    if len(rets)<2: return 0.0
    vol = statistics.pstdev(rets[-48:])
    vavg = sum(v[-48:])/48.0
    # simple min-max local sur le lot, on re-normalisera après
    return vol * (vavg**0.5)

def minmax(xs):
    lo=min(xs); hi=max(xs)
    if hi<=lo: return [0.0]*len(xs)
    return [(x-lo)/(hi-lo) for x in xs]

def main():
    keep, fixed = read_tuning()
    exS = ex_swap(); exP = ex_spot()
    mkS = exS.load_markets()
    mkP = exP.load_markets()

    try:
        uni = json.loads(COINS.read_text()).get("symbols",[])
    except: 
        uni = []

    # map -> ccxt symbols
    mapped = []
    for b in uni:
        s = to_ccxt_symbol(b, mkS, mkP)
        if s: mapped.append((b,s))
    # fetch 1h ohlcv  (limité pour rester rapide)
    rows = []
    for b,s in mapped:
        try:
            src = exS if s in mkS else exP
            o = src.fetch_ohlcv(s, timeframe="1h", limit=80)
            o = drop_incomplete(o, 3600)
            sc = score_symbol(o)
            rows.append((b, sc))
        except Exception:
            continue

    if not rows:
        OUT.write_text(json.dumps({"symbols":[], "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
                                   "source":"coins_90d"}))
        print("[top] empty")
        return

    # normalise et trie décroissant
    scores = [sc for _,sc in rows]
    nz = minmax(scores)
    ranked = sorted([(b, z) for (b,_), z in zip(rows, nz)], key=lambda x: x[1], reverse=True)

    # applique FIXED_TOP et KEEP
    fix = [b for b in fixed if b in [r[0] for r in ranked]]
    rest = [b for b,_ in ranked if b not in fix]
    final = fix + rest
    final = final[:keep]

    OUT.write_text(json.dumps({
        "symbols": final,
        "generated_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "source": "coins_90d->score1h",
        "keep": keep,
        "fixed": fix
    }, indent=2))
    print(f"[top] in={len(uni)} mapped={len(mapped)} kept={len(final)} fixed={len(fix)}")
if __name__=="__main__": main()

#!/usr/bin/env python3
def ema(vals,n):
    if not vals: return []
    k=2/(n+1); out=[vals[0]]
    for x in vals[1:]: out.append(out[-1]+k*(x-out[-1]))
    return out
def minmax_pos(vals,win):
    out=[]
    for i in range(len(vals)):
        a=max(0,i-win+1); w=vals[a:i+1]
        lo=min(w); hi=max(w); rng=(hi-lo) or 1e-9
        out.append((vals[i]-lo)/rng)
    return out
def slope(vals,step=5):
    out=[0]*len(vals)
    for i in range(step,len(vals)):
        prev=vals[i-step] or 1e-9
        out[i]=(vals[i]-prev)/abs(prev)
    return out

    return "hold"

def analyze_symbol(sym):
    rows=[]; end_ms=int(time.time()*1000); need=600
    while len(rows)<need:
        got=history_candles(sym,"1m",limit=min(200,need-len(rows)),end_ms=end_ms)
        if not got: break
        rows=got+rows; end_ms=got[0]["ts"]-1

#!/usr/bin/env python3
import time, json
from urllib.parse import urlencode
from urllib.request import Request, urlopen
from urllib.error import URLError, HTTPError
BASE="https://api.bitget.com"
def _get(path, params=None, tries=4, timeout=12):
    q=f"{BASE}{path}"+(f"?{urlencode(params)}" if params else "")
    for i in range(tries):
        try:
            req=Request(q, headers={"Accept":"application/json"})
            with urlopen(req, timeout=timeout) as r:
                return json.loads(r.read().decode("utf-8"))
        except (URLError,HTTPError):
            if i+1>=tries: raise
            time.sleep(0.6*(2**i))
def normalize_symbol(sym):

    return (s.replace("_UMCBL",""), "usdt-futures")
def history_candles(sym, granularity="1m", limit=200, end_ms=None):
    s, prod=normalize_symbol(sym)
    p={"symbol":s,"granularity":granularity,"limit":int(limit),"productType":prod}
    if end_ms: p["endTime"]=int(end_ms)
    j=_get("/api/v2/mix/market/history-candles", p)
    raw=j.get("data",[]) if isinstance(j,dict) else j
    out=[]
    for k in raw:
        try:
            ts=int(k[0]); o,h,l,c=map(float,k[1:5]); v=float(k[5])
            out.append({"ts":ts,"o":o,"h":h,"l":l,"c":c,"v":v})
        except: pass
    out.sort(key=lambda x:x["ts"])
    return out
def ticker_price(sym):

def subscores_tf(o,h,l,c,v, tf):
    # EMA spread / ATR gate
    atr14 = atr(h,l,c,14)[-1]
    base = max(1.5*atr14,1e-9)
    s_ema = math.tanh( (ema(c,12)[-1] - ema(c,26)[-1]) / base )

    # MACD hist / sigma100
    hist = macd_hist(c)
    sigma = stdev(hist[-100:]) if len(hist)>=100 else stdev(hist)
    s_macd = math.tanh( (hist[-1])/(sigma+1e-12) )

    # RSI
    s_rsi = max(-1.0,min(1.0,(rsi(c,14)[-1]-50.0)/50.0))

    # ADX
    s_adx = max(-1.0,min(1.0,(adx(h,l,c,14)[-1]-20.0)/20.0))

    # OBV slope(20)/median(|OBV|_200)
    ob = obv(c,v)
    slope = ob[-1]-ob[-21] if len(ob)>=21 else 0.0
    denom = stats.median([abs(x) for x in (ob[-200:] if len(ob)>=200 else ob)]) or 1.0
    s_obv = math.tanh(slope/denom)

    # ATR% gate
    gate = (atr14 / c[-1]) < ATR_GATE[tf]
    w_ema = W_EMA*0.5 if gate else W_EMA
    w_macd= W_MACD*0.5 if gate else W_MACD
    num = w_ema*s_ema + w_macd*s_macd + W_RSI*s_rsi + W_ADX*s_adx + W_OBV*s_obv
    den = (w_ema + w_macd + W_RSI + W_ADX + W_OBV)
    S = num/den if den>0 else 0.0

    # Softmax tri-state
    t=TAU[tf]
    epos=math.exp(S/t); eneg=math.exp(-S/t); den3=epos+1+eneg
    pb=epos/den3; ph=1/den3; ps=eneg/den3

    return {"S":S,"pb":pb,"ph":ph,"ps":ps,"atrpct":atr14/c[-1]}

def ema(series, n):
    if not series: return []
    k = 2/(n+1)
    out=[series[0]]
    for x in series[1:]:
        out.append(out[-1] + k*(x-out[-1]))
    return out
def stdev(xs):
    if len(xs)<2: return 0.0
    mu = sum(xs)/len(xs)
    return math.sqrt(sum((x-mu)**2 for x in xs)/len(xs))

def atr_ohlc(h,l,c,n=14):
    trs=[]; prev=c[0]
    for i in range(len(c)):
        hi,lo,cl=h[i],l[i],c[i]
        tr=max(hi-lo, abs(hi-prev), abs(lo-prev))
        trs.append(tr); prev=cl
    return ema(trs, n)

def macd_hist(cl, fast=12, slow=26, sig=9):
    macd_line=[a-b for a,b in zip(ema(cl,fast), ema(cl,slow))]
    signal = ema(macd_line, sig)
    # aligne
    n=min(len(macd_line), len(signal))
    return [macd_line[-n+i]-signal[-n+i] for i in range(n)]

def rsi14(cl, n=14):
    if len(cl)<n+1: return [50.0]*len(cl)
    gains=[0.0]; losses=[0.0]
    for i in range(1,len(cl)):
        ch=cl[i]-cl[i-1]
        gains.append(max(ch,0.0)); losses.append(max(-ch,0.0))
    avgG=ema(gains,n); avgL=ema(losses,n)
    out=[]
    for g,l in zip(avgG,avgL):
        rs = g/(l+1e-12)
        out.append(100.0 - 100.0/(1.0+rs))
    return out

def adx14(h,l,c,n=14):
    # DM+/DM-, TR; DX=100*|+DI - -DI|/(+DI + -DI); ADX=EMA(DX,n)
    if len(c)<n+1: return [20.0]*len(c)
    plusDM=[0.0]; minusDM=[0.0]; TR=[0.0]
    for i in range(1,len(c)):
        up=h[i]-h[i-1]; dn=l[i-1]-l[i]
        plusDM.append( up if (up>dn and up>0) else 0.0 )
        minusDM.append( dn if (dn>up and dn>0) else 0.0 )
        TR.append( max(h[i]-l[i], abs(h[i]-c[i-1]), abs(l[i]-c[i-1])) )
    plusDI   = [100*(x/(y+1e-12)) for x,y in zip(ema(plusDM,n), ema(TR,n))]
    minusDI  = [100*(x/(y+1e-12)) for x,y in zip(ema(minusDM,n), ema(TR,n))]
    DX = [100*abs(p-m)/(p+m+1e-12) for p,m in zip(plusDI, minusDI)]
    return ema(DX, n)

def obv(cl, vol):
    out=[0.0]
    for i in range(1,len(cl)):
        if cl[i]>cl[i-1]: out.append(out[-1]+vol[i])
        elif cl[i]<cl[i-1]: out.append(out[-1]-vol[i])
        else: out.append(out[-1])
    return out

def subscores_tf(ohlcv, tf):
    # colonnes
    o=[r[1] for r in ohlcv]; h=[r[2] for r in ohlcv]; l=[r[3] for r in ohlcv]
    c=[r[4] for r in ohlcv]; v=[r[5] for r in ohlcv]
    if len(c)<500: return None

    # EMA spread / ATR gate
    atr = atr_ohlc(h,l,c,14)
    ema12 = ema(c,12); ema26 = ema(c,26)
    n=min(len(atr),len(ema12),len(ema26))
    atr14=atr[-1]; base=max(1.5*atr14,1e-9)
    s_ema = math.tanh( (ema12[-1]-ema26[-1]) / base )

    # MACD hist normalisé par sigma_100
    hist = macd_hist(c)
    sigma = stdev(hist[-100:]) if len(hist)>=100 else stdev(hist)
    s_macd = math.tanh( (hist[-1])/(sigma+1e-12) )

    # RSI
    r = rsi14(c)
    r_last = r[-1] if r else 50.0
    s_rsi = max(-1.0, min(1.0, (r_last-50.0)/50.0 ))

    # ADX
    adx = adx14(h,l,c,14)
    a_last = adx[-1] if adx else 20.0
    s_adx = max(-1.0, min(1.0, (a_last-20.0)/20.0 ))

    # OBV slope(20) / median(|OBV|_200)
    ob = obv(c,v)
    if len(ob)>=220:
        slope = ob[-1]-ob[-21]
        denom = stats.median([abs(x) for x in ob[-200:]]) or 1.0
    else:
        slope = ob[-1]-ob[-min(21,len(ob)-1)]
        denom = stats.median([abs(x) for x in ob]) or 1.0
    s_obv = math.tanh(slope/denom)

    # ATR% gate: si trop faible → demi-poids sur EMA+MACD
    gate = (atr14 / c[-1]) < ATR_GATE[tf]
    w_ema = W_EMA*0.5 if gate else W_EMA
    w_macd= W_MACD*0.5 if gate else W_MACD

    # Score TF ∈ [-1,1]
    num = w_ema*s_ema + w_macd*s_macd + W_RSI*s_rsi + W_ADX*s_adx + W_OBV*s_obv
    den = (w_ema + w_macd + W_RSI + W_ADX + W_OBV)
    S = num/den if den>0 else 0.0

    # Softmax tri-state
    t = TAU[tf]
    epos = math.exp(S/t); eneg = math.exp(-S/t); den3 = epos + 1.0 + eneg
    pb = epos/den3; ph = 1.0/den3; ps = eneg/den3

    return {
        "S": S, "pb": pb, "ph": ph, "ps": ps,
        "gate": gate, "atrpct": atr14/c[-1],
        "last_ts": ohlcv[-1][0]
    }

def aggregate(tf_res):

    S = sum(ALPHA[k]*tf_res[k]["S"] for k in tf_res)

    ph= sum(ALPHA[k]*tf_res[k]["ph"] for k in tf_res)
    ps= sum(ALPHA[k]*tf_res[k]["ps"] for k in tf_res)
    s = pb+ph+ps
    if s>0: pb,ph,ps = pb/s, ph/s, ps/s
    return S, pb, ph, ps

def decide_context(pb,ph,ps, tf_res):

    veto_buy, veto_sell = veto_context(tf_res, tf_res["5m"]["S"])
    if bull and not veto_buy: return "buy"
    if bear and not veto_sell: return "sell"
    if 0.40 <= ph <= 0.60: return "hold"
    return "none"

def ema(x,n):
    if not x: return []
    k=2/(n+1); o=[x[0]]
    for v in x[1:]: o.append(o[-1]+k*(v-o[-1]))
    return o
def atr(h,l,c,n=14):
    trs=[]; prev=c[0]
    for i in range(len(c)):
        trs.append(max(h[i]-l[i], abs(h[i]-prev), abs(l[i]-prev))); prev=c[i]
    return ema(trs,n)
def macd_hist(c, fast=12, slow=26, sig=9):
    m=[a-b for a,b in zip(ema(c,fast), ema(c,slow))]
    s=ema(m,sig); n=min(len(m),len(s))
    return [m[-n+i]-s[-n+i] for i in range(n)]
def rsi(c,n=7):
    if len(c)<n+1: return [50.0]*len(c)
    g=[0.0]; d=[0.0]
    for i in range(1,len(c)):
        ch=c[i]-c[i-1]; g.append(max(ch,0.0)); d.append(max(-ch,0.0))
    ag=ema(g,n); ad=ema(d,n)
    out=[]
    for x,y in zip(ag,ad):
        rs=x/(y+1e-12); out.append(100-100/(1+rs))
    return out

def load_tf(sym, tf):
    p=CBASE/tf/f"{sym}.json"
    if not p.exists(): return None
    j=json.loads(p.read_text()); k=j["data"]
    if len(k)<100: return None
    ms={"1m":60,"3m":180,"5m":300}[tf]*1000
    if int(time.time()*1000)-k[-1][0] < 0.9*ms: k=k[:-1]
    o=[r[1] for r in k]; h=[r[2] for r in k]; l=[r[3] for r in k]; c=[r[4] for r in k]; v=[r[5] for r in k]
    return {"o":o,"h":h,"l":l,"c":c,"v":v,"ts":k[-1][0]}

def prefilters(sym, ctx, d1):
    reasons=[]
    if not d1:
        return False,["no-1m"]
    c,h,l=d1["c"],d1["h"],d1["l"]
    atr14=atr(h,l,c,14)[-1]; atrpct=atr14/c[-1]
    if atrpct<0.001: reasons.append("atr%<0.10%")
    # cooldown
    try: st=json.loads(STATE.read_text() or "{}")
    except: st={}
    last=st.get(sym,0)
    if time.time()-last<5*60: reasons.append("cooldown")
    # spread
    try:
        t=EX.fetch_ticker(sym.replace("_UMCBL","/USDT:USDT"))
        if (t["ask"]-t["bid"])/t["bid"]>0.002: reasons.append("spread>0.2%")
    except: pass
    return (len(reasons)==0, reasons if reasons else ["ok"])

def detect_pullback(sym, ctx, d1):
    c,h,l,v=d1["c"],d1["h"],d1["l"],d1["v"]
    ema20=ema(c,20)[-1]; last=c[-1]; rsi7=rsi(c,7)[-1]
    ma20=sum(v[-20:])/20; vol=v[-1]
    if ctx=="buy" and last<=ema20 and rsi7>=60 and vol>1.3*ma20:
        atr14=atr(h,l,c,14)[-1]; sl=last-1.2*atr14; tp=last+1.8*atr14
        return {"trigger":"pullback","side":"buy","entry":last,"sl":sl,"tp":tp,
                "why":["ema20-touch","rsi>=60","vol>1.3x20"]}
    if ctx=="sell" and last>=ema20 and rsi7<=40 and vol>1.3*ma20:
        atr14=atr(h,l,c,14)[-1]; sl=last+1.2*atr14; tp=last-1.8*atr14
        return {"trigger":"pullback","side":"sell","entry":last,"sl":sl,"tp":tp,
                "why":["ema20-touch","rsi<=40","vol>1.3x20"]}
    return None

def detect_breakout(sym, ctx, d1,d5):
    if not d5: return None
    c,h,l=d1["c"],d1["h"],d1["l"]
    # ADX 5m simple
    def _adx5(d5):
        ch=d5["c"]; hh=max(d5["h"][-15:]); ll=min(d5["l"][-15:])
        rng=hh-ll
        return 25.0 if rng>0 and (ch[-1]-ch[-6])>0 else 15.0
    adx5=_adx5(d5)
    if adx5<20: return None
    hh=max(h[-20:]); ll=min(l[-20:])
    last=c[-1]; macd_acc=macd_hist(c)[-1]>0
    if ctx=="buy" and last>hh and macd_acc:
        sl=ll; tp=last+(hh-ll)*1.5
        return {"trigger":"breakout","side":"buy","entry":last,"sl":sl,"tp":tp,
                "why":["HH20-break","macd-accel","ADX5>=20"]}
    if ctx=="sell" and last<ll and macd_acc:
        sl=hh; tp=last-(hh-ll)*1.5
        return {"trigger":"breakout","side":"sell","entry":last,"sl":sl,"tp":tp,
                "why":["LL20-break","macd-accel","ADX5>=20"]}
    return None

def detect_mr(sym, ctx, d1):
    if ctx!="hold": return None
    c=d1["c"]; rsi7=rsi(c,7)[-1]; last=c[-1]
    mid=sum(c[-20:])/20; std=stats.pstdev(c[-20:])
    up=mid+2*std; dn=mid-2*std
    if last>up and rsi7<60:
        return {"trigger":"meanrev","side":"sell","entry":last,"sl":last-std,"tp":mid,
                "why":["BB2sigma-touch","RSI<60","reversal"]}
    if last<dn and rsi7>40:
        return {"trigger":"meanrev","side":"buy","entry":last,"sl":last+std,"tp":mid,
                "why":["BB2sigma-touch","RSI>40","reversal"]}
    return None

def size_trade(balance, entry, sl):
    D=abs(entry-sl)
    R=balance*RISK_PCT
    if D<=0: return 0
    qty=R/D
    notional=qty*entry
    if notional<MIN_NOTIONAL: qty=MIN_NOTIONAL/entry
    max_notional=balance*MAX_EXPO*LEV
    if qty*entry>max_notional: qty=max_notional/entry
    step=LOT_STEP
    qty=math.floor(qty/step)*step
    return qty

def main():
    try: arrA=json.loads(A_OUT.read_text())
    except: arrA=[]
    try: sigs=json.loads(SIG.read_text())
    except: sigs=[]
    balance=float(os.getenv("B_BALANCE","100"))
    rejects=[]
    added=[]
    for r in arrA:
        sym,ctx=r.get("symbol"),r.get("context")
        if ctx=="none": 
            rejects.append({"symbol":sym,"ctx":"none","reasons":["ctx-none"]}); 
            continue
        d1=load_tf(sym,"1m"); d5=load_tf(sym,"5m")
        ok,pre=prefilters(sym,ctx,d1)
        if not ok:
            rejects.append({"symbol":sym,"ctx":ctx,"reasons":pre}); 
            continue
        cand=detect_pullback(sym,ctx,d1) or detect_breakout(sym,ctx,d1,d5) or detect_mr(sym,ctx,d1)
        if not cand:
            rejects.append({"symbol":sym,"ctx":ctx,"reasons":["no-trigger"]}); 
            continue
        entry=cand["entry"]; sl=cand["sl"]; qty=size_trade(balance, entry, sl)
        if qty<=0:
            rejects.append({"symbol":sym,"ctx":ctx,"reasons":["qty<=0"]}); 
            continue
        sig={"symbol":sym,"side":cand["side"],"type":cand["trigger"],
             "entry":entry,"sl":sl,"tp":cand["tp"],"qty":qty,"leverage":LEV,
             "timestamp":int(time.time()),"expires":int(time.time()+1500),
             "reasons":cand.get("why",[])}
        sigs.append(sig); added.append(sig)
        # cooldown
        try: st=json.loads(STATE.read_text() or "{}")
        except: st={}
        st[sym]=time.time(); STATE.write_text(json.dumps(st))
    SIG.write_text(json.dumps(sigs, indent=2))
    DBG_REJ.write_text(json.dumps({"ts":time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
                                   "added":len(added),"rejects":rejects}, indent=2))
    print(f"[B] added={len(added)} total={len(sigs)}")
if __name__=="__main__":
    main()
