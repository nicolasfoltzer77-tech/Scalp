#!/usr/bin/env python3
import asyncio, time, json, requests, websockets

REST = "https://api.bitget.com"
WS   = "wss://ws.bitget.com/v2/ws/public"

def rest_contract_check(symbol="BTCUSDT"):
    # Confirme que BTCUSDT existe bien pour usdt-futures
    r = requests.get(f"{REST}/api/v2/mix/market/contracts",
                     params={"productType":"usdt-futures","symbol":symbol}, timeout=8)
    r.raise_for_status()
    j = r.json()
    ok = isinstance(j,dict) and j.get("code") in ("00000","0000",0,None) and j.get("data")
    print("[REST contracts]", "OK" if ok else j)
    if ok:
        d = j["data"][0]
        print(f"  symbol={d.get('symbol')} base={d.get('baseCoin')} quote={d.get('quoteCoin')} pricePrec={d.get('pricePlace')}")

async def sub(ws, tag, instType, instId):
    msg = {"op":"subscribe","args":[{"instType":instType,"channel":"ticker","instId":instId}]}
    await ws.send(json.dumps(msg))
    print(f"[{tag}] ->", msg)
    start=time.time()
    while time.time()-start<12:
        try:
            raw = await asyncio.wait_for(ws.recv(), timeout=6)
        except asyncio.TimeoutError:
            print(f"[{tag}] timeout"); continue
        if isinstance(raw,(bytes,bytearray)):
            try: raw = raw.decode()
            except: pass
        try:
            data = json.loads(raw)
        except Exception:
            continue
        if "event" in data:                    # acks / errors
            print(f"[{tag}] EVENT", data)
        if "data" in data:                     # first payload
            print(f"[{tag}] DATA", str(data)[:220].replace("\n"," "))
            return True
    return False

async def main():
    rest_contract_check("BTCUSDT")
    async with websockets.connect(WS, ping_interval=20, ping_timeout=20, close_timeout=5, max_size=2**20) as ws:
        # 1) Mauvais: _UMCBL interdit en instId (doit être BTCUSDT)
        await sub(ws, "BAD_instId", "USDT-FUTURES", "BTCUSDT_UMCBL")
        # 2) Mauvais: instType incorrect
        await sub(ws, "BAD_instType", "mc", "BTCUSDT")
        # 3) Bon format
        ok = await sub(ws, "GOOD", "USDT-FUTURES", "BTCUSDT")
        print("RESULT:", "OK" if ok else "NO DATA")

if __name__ == "__main__":
    asyncio.run(main())

-- ===== Univers TOP30 + 5 fixes =====
DROP VIEW IF EXISTS v_universe_35;
CREATE VIEW v_universe_35 AS
WITH latest AS (SELECT MAX(ts) AS ts FROM top_scores WHERE name='TOP30'),
base AS (
  SELECT symbol FROM top_scores
  WHERE name='TOP30' AND ts=(SELECT ts FROM latest)
  ORDER BY score DESC LIMIT 30
),
fix AS (
  SELECT 'BTCUSDT' AS symbol UNION ALL
  SELECT 'ETHUSDT' UNION ALL
  SELECT 'BNBUSDT' UNION ALL
  SELECT 'SOLUSDT' UNION ALL
  SELECT 'XRPUSDT'
)
SELECT DISTINCT symbol FROM (SELECT symbol FROM base UNION ALL SELECT symbol FROM fix)
LIMIT 35;

-- 5m
DROP VIEW IF EXISTS v_feat_5m;
CREATE VIEW v_feat_5m AS
WITH b AS (
  SELECT o.symbol,o.ts,o.open,o.high,o.low,o.close,o.volume,
         LAG(o.close) OVER (PARTITION BY o.symbol ORDER BY o.ts) AS pc,
         o.close - LAG(o.close) OVER (PARTITION BY o.symbol ORDER BY o.ts) AS dclose
  FROM ohlcv_5m o
  WHERE o.symbol IN (SELECT symbol FROM v_universe_35)
),
c AS (
  SELECT *,
         CASE WHEN dclose>0 THEN dclose ELSE 0 END AS gain,

  regime TEXT NOT NULL,
  D      REAL    NOT NULL,

         COALESCE(a.ts,b.ts,c.ts) AS ts,
         -- régime priorité trend_up/down puis range sinon squeeze/none
         CASE
           WHEN (a.regime='trend_up' OR b.regime='trend_up' OR c.regime='trend_up') THEN 'trend_up'
           WHEN (a.regime='trend_down' OR b.regime='trend_down' OR c.regime='trend_down') THEN 'trend_down'
           WHEN (a.regime='range'     OR b.regime='range'     OR c.regime='range')     THEN 'range'
           ELSE COALESCE(a.regime,b.regime,c.regime,'none')
         END AS regime,
         -- D global simple
         COALESCE(a.D,0)*0.40 + COALESCE(b.D,0)*0.35 + COALESCE(c.D,0)*0.25 AS D_raw,
         -- moyennes pondérées des probas (valeur neutre 0.25 si manquante)
         (0.40*COALESCE(a.pB5,0.25)+0.35*COALESCE(b.pB15,0.25)+0.25*COALESCE(c.pB30,0.25)) AS Bw,
         (0.40*COALESCE(a.pS5,0.25)+0.35*COALESCE(b.pS15,0.25)+0.25*COALESCE(c.pS30,0.25)) AS Sw,
         (0.40*COALESCE(a.pH5,0.25)+0.35*COALESCE(b.pH15,0.25)+0.25*COALESCE(c.pH30,0.25)) AS Hw,
         a.pB5,a.pS5,a.pH5, b.pB15,b.pS15,b.pH15, c.pB30,c.pS30,c.pH30
  FROM u
  LEFT JOIN a ON a.symbol=u.symbol
  LEFT JOIN b ON b.symbol=u.symbol
  LEFT JOIN c ON c.symbol=u.symbol
),
n AS (
  SELECT symbol, ts, regime,
         D_raw AS D,
         -- normalisation
         (Bw/(Bw+Sw+Hw)) AS pB,
         (Sw/(Bw+Sw+Hw)) AS pS,
         (Hw/(Bw+Sw+Hw)) AS pH,
         MAX(0.0, 1.0 - ( (Bw/(Bw+Sw+Hw)) + (Sw/(Bw+Sw+Hw)) + (Hw/(Bw+Sw+Hw)) )) AS pN,
         a_pB5, a_pS5, a_pH5, b_pB15, b_pS15, b_pH15, c_pB30, c_pS30, c_pH30
  FROM (
    SELECT symbol, ts, regime, D_raw, Bw, Sw, Hw,
           a_pB5, a_pS5, a_pH5, b_pB15, b_pS15, b_pH15, c_pB30, c_pS30, c_pH30
    FROM m
  )
)
INSERT INTO ctx_A(symbol,ts,regime,D,pB,pS,pH,pN,
                  pB5,pS5,pH5,pB15,pS15,pH15,pB30,pS30,pH30)
SELECT symbol, (SELECT now_ts FROM v_now_unix), regime, D, pB, pS, pH, pN,
       a_pB5, a_pS5, a_pH5, b_pB15, b_pS15, b_pH15, c_pB30, c_pS30, c_pH30
FROM n;

DROP VIEW IF EXISTS v_ctx_A;
CREATE VIEW v_ctx_A AS
SELECT symbol, ts, regime, D, pB, pS, pH, pN,
       CASE
         WHEN pB>=0.55 AND pS<=0.35 THEN 'BUY'
         WHEN pS>=0.55 AND pB<=0.35 THEN 'SELL'
         WHEN pH>=pB AND pH>=pS AND pH>=pN THEN 'HOLD'
         ELSE 'NONE'
       END AS decision
FROM ctx_A
ORDER BY symbol;

DROP VIEW IF EXISTS v_ctx_tradeable;
CREATE VIEW v_ctx_tradeable AS
SELECT * FROM v_ctx_A WHERE decision!='NONE';

#!/usr/bin/env python3
import os, time, sqlite3, math
DB="data/scalp.db"
TP_BPS     = float(os.getenv("TP_BPS","15"))       # take profit en bps
SL_BPS     = float(os.getenv("SL_BPS","20"))       # stop loss   en bps
MAX_HOLD   = int(os.getenv("MAX_HOLD_SEC","900"))  # 15 min
FEES_BPS   = float(os.getenv("FEES_BPS","1"))      # aller+retour
POLL       = 0.5

def connect():
    con=sqlite3.connect(DB, timeout=20, isolation_level=None)
    con.execute("PRAGMA journal_mode=WAL")
    con.execute("PRAGMA busy_timeout=20000")
    con.execute("PRAGMA synchronous=NORMAL")
    return con

def last_price(con, symbol, default):
    row=con.execute("SELECT last FROM last_ticks WHERE symbol=?", (symbol,)).fetchone()
    return (row[0] if row else default) or default

def ensure_qty(q): return 1.0 if (q is None or q==0.0) else float(q)

def open_new_positions(con, now):
    rows=con.execute("""
      SELECT id,symbol,side,COALESCE(qty,0),COALESCE(price,0),created_ts
      FROM orders_sim
      WHERE status='SIMULATED' AND id NOT IN (SELECT order_id FROM positions_sim)
      ORDER BY id ASC LIMIT 500
    """).fetchall()
    if not rows: return 0
    con.execute("BEGIN IMMEDIATE")
    for oid,sym,side,qty,px,ts in rows:
        px = px or last_price(con, sym, None) or 0.0
        q  = ensure_qty(qty)
        con.execute("""INSERT OR IGNORE INTO positions_sim
            (order_id,symbol,side,qty,entry_price,entry_ts,status)
            VALUES(?,?,?,?,?,?, 'OPEN')""", (oid,sym,side,q,px,ts or now))
    con.execute("COMMIT")
    return len(rows)

def close_if_needed(con, now):
    rows=con.execute("""
      SELECT id, order_id, symbol, side, qty, entry_price, entry_ts
      FROM positions_sim WHERE status='OPEN'
      ORDER BY id ASC LIMIT 1000
    """).fetchall()
    if not rows: return 0
    n_closed=0
    con.execute("BEGIN IMMEDIATE")
    for pid,oid,sym,side,qty,entry,ets in rows:
        last=last_price(con, sym, entry)
        hold= now - (ets or now)
        # Targets
        tp = entry*(1 + (TP_BPS/10000.0)) if side.upper()=='BUY'  else entry*(1 - (TP_BPS/10000.0))
        sl = entry*(1 - (SL_BPS/10000.0)) if side.upper()=='BUY'  else entry*(1 + (SL_BPS/10000.0))
        reason=None; exit_px=None
        if side.upper()=='BUY':
            if last>=tp: exit_px=last; reason='TP'
            elif last<=sl: exit_px=last; reason='SL'
        else: # SELL
            if last<=tp: exit_px=last; reason='TP'
            elif last>=sl: exit_px=last; reason='SL'
        if reason is None and hold>=MAX_HOLD:
            exit_px=last; reason='TIME'
        if reason is None:
            # Reverse signal ferme la position si ordre plus récent sens opposé
            rev = con.execute("""
              SELECT 1 FROM orders_sim
              WHERE symbol=? AND created_ts>? AND
                    ((side='BUY'  AND ?='SELL') OR (side='SELL' AND ?='BUY'))
              LIMIT 1
            """,(sym, ets or 0, side.upper(), side.upper())).fetchone()
            if rev: exit_px=last; reason='REV'
        if exit_px is not None:
            # PnL: BUY => (exit-entry)*qty ; SELL => (entry-exit)*qty ; moins fees
            gross = (exit_px-entry)*qty if side.upper()=='BUY' else (entry-exit_px)*qty
            fees  = (entry+exit_px)*qty*(FEES_BPS/10000.0)
            pnl   = gross - fees
            pnl_bps = (pnl/ (entry*qty)) * 10000.0 if entry>0 and qty>0 else 0.0
            con.execute("""UPDATE positions_sim
                           SET exit_price=?, exit_ts=?, pnl_abs=?, pnl_bps=?,
                               fees_abs=?, status='CLOSED', close_reason=?
                           WHERE id=?""",
                           (exit_px, now, pnl, pnl_bps, fees, reason, pid))
            n_closed+=1
    con.execute("COMMIT")
    return n_closed

def main():
    con=connect()
    while True:
        now=int(time.time())
        opened=open_new_positions(con, now)
        closed=close_if_needed(con, now)
        if opened==0 and closed==0:
            time.sleep(POLL)

   +(SELECT WEIGHT_BREAKOUT FROM params)*e.e_breakout_up
   +(SELECT WEIGHT_MEANREV  FROM params)*e.e_mean_buy)  AS s_buy_raw,
  ((SELECT WEIGHT_PULLBACK FROM params)*e.e_pullback_dn
   +(SELECT WEIGHT_BREAKOUT FROM params)*e.e_breakout_dn
   +(SELECT WEIGHT_MEANREV  FROM params)*e.e_mean_sell) AS s_sell_raw
FROM t_ev e;

-- Softmax
DROP TABLE IF EXISTS t_prob;
CREATE TEMP TABLE t_prob AS
SELECT
  s.*,
  CASE WHEN s.s_buy_raw  <0 THEN 0 WHEN s.s_buy_raw  >1 THEN 1 ELSE s.s_buy_raw  END AS b0,
  CASE WHEN s.s_sell_raw <0 THEN 0 WHEN s.s_sell_raw >1 THEN 1 ELSE s.s_sell_raw END AS s0
FROM t_score s;

DROP TABLE IF EXISTS t_mix;
CREATE TEMP TABLE t_mix AS
SELECT
  p.*,
  exp(b0/(SELECT TAU_LOW FROM params)) AS eb,
  exp(s0/(SELECT TAU_LOW FROM params)) AS es,
  1.0                                  AS eh
FROM t_prob p;

DROP TABLE IF EXISTS t_class;
CREATE TEMP TABLE t_class AS
SELECT
  m.*,
  (m.eb+m.es+m.eh) AS denom,
  (m.eb/(m.eb+m.es+m.eh)) AS pb,
  (m.es/(m.eb+m.es+m.eh)) AS ps,
  (m.eh/(m.eb+m.es+m.eh)) AS ph
FROM t_mix m;

CREATE TEMP TABLE t_veto AS

  c.*,
  vf.sma9  AS sma9_5,
  vf.sma21 AS sma21_5
FROM t_class c
LEFT JOIN v_indic5m_fast vf
  ON vf.symbol=c.symbol
 AND vf.ts=(SELECT MAX(ts) FROM v_indic5m_fast v2 WHERE v2.symbol=c.symbol);

CREATE TEMP TABLE t_decide AS

  v.*,
  CASE
    WHEN v.pb>=v.ps AND v.pb>=(SELECT SCORE_TRIGGER FROM params) THEN 'buy'
    WHEN v.ps> v.pb AND v.ps>=(SELECT SCORE_TRIGGER FROM params) THEN 'sell'
    ELSE 'none'
  END AS side_raw,
  CASE
    WHEN (SELECT VETO_5M_EMA FROM params)=1
         AND v.pb>=v.ps AND v.sma9_5 IS NOT NULL AND v.sma21_5 IS NOT NULL
         AND v.sma9_5 < v.sma21_5
    THEN 0 ELSE 1 END AS veto_buy,
  CASE
    WHEN (SELECT VETO_5M_EMA FROM params)=1
         AND v.ps>v.pb AND v.sma9_5 IS NOT NULL AND v.sma21_5 IS NOT NULL
         AND v.sma9_5 > v.sma21_5
    THEN 0 ELSE 1 END AS veto_sell
FROM t_veto v;

CREATE TEMP TABLE t_cool AS

FROM t_decide d;

  CASE WHEN c.side_raw='buy'  THEN (c.veto_buy=1)
       WHEN c.side_raw='sell' THEN (c.veto_sell=1)
       ELSE 0 END AS ok_veto
FROM t_cool c;

    * (SELECT RISK_PCT FROM params)
    * (SELECT COALESCE(leverage,50) FROM account_sim LIMIT 1)) AS notional
FROM t_cands c

def jget(url, params=None, retries=3, tout=12):
    for _ in range(retries+1):
        try:
            r = requests.get(url, params=params or {}, timeout=tout)
            if r.status_code == 200:
                return r.json()
            time.sleep(0.4)
        except Exception:
            time.sleep(0.4)
    return None

def fetch_top_by_volume():
    # 1) v2
    j = jget(f"{BASE}/api/v2/mix/market/tickers", {"productType":"umcbl"})
    if j and j.get("code") == "00000" and isinstance(j.get("data"), list):
        rows=[]
        for d in j["data"]:
            s = (d.get("symbol") or "").replace("_UMCBL","")
            vol = float(d.get("usdtVolume") or d.get("quoteVolume") or 0.0)
            if s.endswith("USDT") and vol>0: rows.append((s,vol))
        rows.sort(key=lambda x: x[1], reverse=True)
        return [s for s,_ in rows[:30]]

    # 2) v1 (secours)
    j = jget(f"{BASE}/api/mix/v1/market/tickers", {"productType":"umcbl"})
    if j and j.get("code") == "00000" and isinstance(j.get("data"), list):
        rows=[]
        for d in j["data"]:
            s = (d.get("symbol") or "").replace("_UMCBL","")
            vol = float(d.get("usdtVolume") or d.get("quoteVolume") or 0.0)
            if s.endswith("USDT") and vol>0: rows.append((s,vol))

        return [s for s,_ in rows[:30]]

    # 3) statique
    return FALLBACK[:30]

def ensure_schema():
    con = sqlite3.connect(DB); c = con.cursor()

INC_VAC=${INC_VAC:-200}          # pages à libérer

sqlite3 "$TDB" <<SQL
PRAGMA journal_mode=WAL;
PRAGMA synchronous=NORMAL;
PRAGMA busy_timeout=8000;

-- 1) purge temporelle en lot
WITH c AS (SELECT strftime('%s','now')-$KEEP_SECS AS cut)
DELETE FROM ticks
 WHERE ts < (SELECT cut FROM c)
 LIMIT $BATCH;

-- 2) cap par symbole: supprime l'excès le plus ancien (sans window)
CREATE TEMP TABLE IF NOT EXISTS _sym(symbol TEXT PRIMARY KEY);
DELETE FROM _sym;
INSERT OR IGNORE INTO _sym
SELECT symbol FROM ticks GROUP BY symbol
HAVING COUNT(*) > $KEEP_ROWS;

-- pour chaque symbole en excès, retire l'ancien surplus par lot
SQL

# boucle symboles excédentaires (minimise le temps bloquant)
sqlite3 -newline $'\n' -list "$TDB" "SELECT symbol FROM _sym;" | while read -r s; do
  [ -z "$s" ] && continue
  sqlite3 "$TDB" "
    DELETE FROM ticks
     WHERE symbol='$s'
       AND ts < (SELECT MIN(ts) FROM (
                   SELECT ts FROM ticks WHERE symbol='$s' ORDER BY ts DESC LIMIT $KEEP_ROWS
                 ))
     LIMIT $BATCH;"
done

# 3) mini libération d'espace, non bloquante
sqlite3 "$TDB" "PRAGMA incremental_vacuum=$INC_VAC; PRAGMA wal_checkpoint(PASSIVE);"

def instid(s):  # "BTC/USDT:USDT" -> "BTCUSDT"

def has_90d_history(ex, sym):
    """
    Test léger: demande des bougies 1D autour de J-90.
    Critère: on a au moins une bougie ≤ (now-88j).
    """
    now_ms = int(time.time()*1000)
    since = now_ms - 95*24*3600*1000
    try:
        rows = ex.fetch_ohlcv(sym, timeframe='1d', since=since, limit=10)
    except Exception:
        return False
    if not rows:
        return False
    # une bougie assez ancienne?
    threshold = now_ms - 88*24*3600*1000
    return any(r[0] <= threshold for r in rows)

def quote_volume(ex, sym):
    """
    Récupère le volume quote du ticker si dispo.
    Fallback sur baseVolume sinon.
    """
    try:
        t = ex.fetch_ticker(sym)
    except Exception:
        return 0.0
    for k in ("quoteVolume","quoteVolume24h","info"):
        if k in t and t[k]:
            if k=="info":
                # Bitget renvoie souvent 'baseVolume' ou 'quoteVolume' dans info
                v = t["info"].get("quoteVolume") or t["info"].get("baseVolume")
            else:
                v = t[k]
            try:
                return float(v)
            except Exception:
                pass
    # fallback très faible
    try:
        return float(t.get("baseVolume") or 0.0)
    except Exception:
        return 0.0

def main():

def qrow(cur, q, args=()):
    cur.execute(q, args); r=cur.fetchone(); return r

def qval(cur, q, args=(), default=None):
    r=qrow(cur,q,args); return r[0] if r and r[0] is not None else default

def mark_last_id(cur, last_id):
    cur.execute("UPDATE paper_state SET last_ord_id=? WHERE id=1", (last_id,))

def load_last_id(cur):
    return qval(cur,"SELECT last_ord_id FROM paper_state WHERE id=1",(),0)

def up_equity(cur):
    bal = qval(cur,"SELECT balance FROM account_sim WHERE id=1",(),0.0)
    upnl = qval(cur,"SELECT COALESCE(SUM(u_pnl),0) FROM positions_open",(),0.0)
    now = int(time.time())
    cur.execute("UPDATE account_sim SET equity=?, updated_ts=? WHERE id=1",
                (bal+upnl, now))

def realize(cur, sym, qty_to_close, price, now):
    """Ferme contre la position existante (FIFO simple)."""
    # Une seule ligne par symbole (netting), donc qty_signée suffit
    row = qrow(cur, "SELECT id, qty, entry_price, opened_ts FROM positions_open WHERE symbol=?", (sym,))
    if not row: return 0.0
    pid, pos_qty, entry, opened_ts = row
    # sens inverse pour fermer
    remain = pos_qty + qty_to_close  # qty_to_close est signée (opposée)
    # PnL réalisé sur la portion fermée = (prix_sortie - prix_entree) * qty_signée_initiale_fermée
    closed_qty = -qty_to_close if pos_qty*qty_to_close < 0 else 0.0
    if closed_qty == 0.0:  # rien à fermer
        return 0.0
    pnl = (price - entry) * (closed_qty if pos_qty>0 else -closed_qty)
    # balance++
    bal = qval(cur,"SELECT balance FROM account_sim WHERE id=1",(),0.0)
    cur.execute("UPDATE account_sim SET balance=? WHERE id=1", (bal+pnl,))
    # trade
    cur.execute("""INSERT INTO trades_closed(symbol,qty,entry_price,exit_price,pnl,opened_ts,closed_ts)
                   VALUES(?,?,?,?,?,?,?)""",
                (sym, closed_qty if pos_qty>0 else -closed_qty, entry, price, pnl, opened_ts, now))
    # position restante (si 0 => on efface)
    if abs(remain) < 1e-12:
        cur.execute("DELETE FROM positions_open WHERE id=?", (pid,))
    else:
        # Si on a inversé, on ouvre à ce nouveau prix
        if pos_qty*remain < 0:
            cur.execute("""UPDATE positions_open
                           SET qty=?, entry_price=?, last=?, u_pnl=0, opened_ts=?
                           WHERE id=?""",
                        (remain, price, price, now, pid))
        else:
            cur.execute("""UPDATE positions_open
                           SET qty=?, last=?, u_pnl=(? - entry_price)*CASE WHEN qty>0 THEN qty ELSE -qty END
                           WHERE id=?""",
                        (remain, price, price, pid))
    return pnl

def apply_order(cur, sym, side, qty, price, ts):
    if qty is None: qty = 0.0
    if qty == 0.0:  # rien à faire
        return
    now = int(time.time())
    signed = qty if side=="buy" else -qty
    # existe ?
    row = qrow(cur,"SELECT id, qty, entry_price FROM positions_open WHERE symbol=?", (sym,))
    if not row:
        # ouverture directe si même sens, sinon c'est un short/long selon signed
        cur.execute("""INSERT INTO positions_open(symbol,qty,entry_price,last,u_pnl,opened_ts)
                       VALUES(?,?,?,?,0,?)""",
                    (sym, signed, price, price, now))
        return
    # position existe : fermer partiellement si sens inverse
    pid, pos_qty, entry = row
    if pos_qty * signed < 0:
        # on ferme jusqu'à min(|pos|, |signed|)
        close_part = -math.copysign(min(abs(pos_qty),abs(signed)), signed)  # opposé à la position
        realize(cur, sym, close_part, price, now)
        signed_after = pos_qty + signed
        # si il reste du sens de l'ordre -> se retrouve dans apply suivant
        row2 = qrow(cur,"SELECT id, qty, entry_price FROM positions_open WHERE symbol=?", (sym,))
        if row2:
            pid2, qty2, entry2 = row2
            cur.execute("""UPDATE positions_open

def acquire_lock(path):
    os.makedirs(os.path.dirname(path), exist_ok=True)
    fd=open(path,"w"); fcntl.flock(fd, fcntl.LOCK_EX); return fd
def release_lock(fd):
    fcntl.flock(fd, fcntl.LOCK_UN); fd.close()

def ensure_table(conn, tf):
    conn.execute(f"""CREATE TABLE IF NOT EXISTS ohlcv_{tf}(
      symbol TEXT, ts INTEGER,
      open REAL, high REAL, low REAL, close REAL, volume REAL,
      PRIMARY KEY(symbol,ts))""")

def last_ts(conn, table, symbol):
    r=conn.execute(f"SELECT MAX(ts) FROM {table} WHERE symbol=?",(symbol,)).fetchone()
    return r[0] if r and r[0] else None

def trim_old(conn, table, symbol):
    conn.execute(f"""DELETE FROM {table}
      WHERE symbol=? AND ts NOT IN (
        SELECT ts FROM {table} WHERE symbol=? ORDER BY ts DESC LIMIT {KEEP}
      );""",(symbol,symbol))

def safe_insert(conn, table, rows, symbol, tf):
    # retry exponentiel sur verrou
    for i in range(6):  # ~1.9s max d'attente
        try:
            conn.execute("BEGIN IMMEDIATE;")
            conn.executemany(
              f"INSERT OR REPLACE INTO {table}(symbol,ts,open,high,low,close,volume) VALUES(?,?,?,?,?,?,?)",
              rows)
            trim_old(conn, table, symbol)
            conn.commit()
            return True
        except sqlite3.OperationalError as e:
            try: conn.rollback()
            except: pass
            if "locked" in str(e).lower():
                time.sleep(0.05*(2**i))
                continue
            raise
    print(f"[WARN] {symbol} {tf}: abandon après retries (locked)")
    return False

def fetch_incremental(ex, conn, symbol, tf):
    table=f"ohlcv_{tf}"
    since=last_ts(conn, table, symbol)
    since_ms=(since+1)*1000 if since else None
    data=ex.fetch_ohlcv(symbol,timeframe=tf,limit=KEEP,since=since_ms) or []
    # retire bougie en formation
    if data:
        now=int(time.time()); frame=ex.parse_timeframe(tf)
        if now - int(data[-1][0]/1000) < frame: data=data[:-1]
    if not data: return 0
    rows=[(symbol,int(ts/1000),o,h,l,c,v) for ts,o,h,l,c,v in data]
    ok=safe_insert(conn, table, rows, symbol, tf)
    if ok: print(f"{symbol} {tf}: +{len(rows)}"); return len(rows)
    return 0

def atr(H,L,C,n=14):
    if len(C)<n+1: return 0.0
    s=0.0
    for i in range(-n,0):
        tr=max(H[i]-L[i], abs(H[i]-C[i-1]), abs(L[i]-C[i-1]))
        s+=tr
    return (s/n)/C[-1] if C[-1] else 0.0
def softmax3(b,h,s,tau):
    e=(math.exp(b/tau),math.exp(h/tau),math.exp(s/tau)); z=sum(e)
    return e[0]/z, e[1]/z, e[2]/z

async def ws_once(sub):
    async with websockets.connect(URI, ping_interval=20, close_timeout=2) as ws:
        await ws.send(json.dumps(sub))
        t0=time.time()
        while time.time()-t0<25:  # 25s par branche
            msg=json.loads(await asyncio.wait_for(ws.recv(), timeout=8))
            # OLD = {"action":"snapshot"/"update", "data":[...]}
            # NEW = {"event":"subscribe"} puis "data"
            d=None
            if "data" in msg:
                d=msg["data"][0]
            elif msg.get("action") in ("snapshot","update") and msg.get("data"):
                d=msg["data"][0]
            if not d: 
                continue
            last=float(d.get("last") or d.get("lastPr") or d.get("pr") or 0.0)
            bid=float(d.get("bestBid") or d.get("bidPx") or last)
            ask=float(d.get("bestAsk") or d.get("askPx") or last)
            ts=int((d.get("systemTime") or d.get("ts") or time.time()*1000)//1000)
            if last>0:

async def rest_fallback():

async def main():
    # 3 tentatives avec backoff
    for attempt in range(3):
        # course NEW + OLD, on garde le premier qui écrit
        done, pending = await asyncio.wait(
            [asyncio.create_task(ws_once(SUB_NEW)), asyncio.create_task(ws_once(SUB_OLD))],
            return_when=asyncio.FIRST_COMPLETED)
        ok = any(t.result() for t in done if not t.cancelled())
        for p in pending: p.cancel()
        if ok:
            return
        # fallback REST
        if await rest_fallback():
            return
        await asyncio.sleep(1.5*(attempt+1)+random.random())
    print("TIMEOUT_ALL")
asyncio.run(main())

def parse_msg(msg):
    # Retourne liste [(symbol, ts, last, bid, ask)]
    out=[]
    # Format "OLD": {"action":"snapshot"/"update", "arg":{instType:"mc"...}, "data":[{instId,last,bestBid,bestAsk,systemTime}]}
    act=msg.get("action")
    if act in ("snapshot","update") and isinstance(msg.get("data"),list):
        for d in msg["data"]:
            sym=d.get("instId"); last=d.get("last"); bid=d.get("bestBid"); ask=d.get("bestAsk")
            ts=int((d.get("systemTime") or time.time()*1000)//1000)
            if sym and last:
                out.append((sym, ts, float(last), float(bid or last), float(ask or last)))
        return out
    # Format "NEW mix": {"arg":{instType:"USDT-FUTURES",instId:"BTCUSDT_UMCBL"}, "data":[{lastPr,bestBid,bestAsk,ts}]}
    if isinstance(msg.get("data"),list) and isinstance(msg.get("arg"),dict):
        sym=msg["arg"].get("instId","").replace("_UMCBL","")
        for d in msg["data"]:
            last=d.get("lastPr") or d.get("pr")
            bid=d.get("bestBid") or d.get("bidPx") or last
            ask=d.get("bestAsk") or d.get("askPx") or last
            ts=int((d.get("ts") or time.time()*1000)//1000)
            if sym and last:
                out.append((sym, ts, float(last), float(bid), float(ask)))
        return out
    return out

TF_USED=["5m","15m","30m"]
W_TF={"5m":0.20,"15m":0.30,"30m":0.50}
TAU={"5m":0.25,"15m":0.30,"30m":0.35}
ATR_GATE={"5m":0.12,"15m":0.18,"30m":0.25}  # en %

def smx(pos,neg,t):
    import math
    b=math.exp(max(pos,0)/max(t,1e-6))
    s=math.exp(max(neg,0)/max(t,1e-6))
    h=1.0
    z=b+s+h
    return b/z, s/z, h/z

def con():
    c=sqlite3.connect(DB, timeout=10, isolation_level=None)
    c.execute("PRAGMA journal_mode=WAL"); c.execute("PRAGMA synchronous=NORMAL")
    c.execute("""CREATE TABLE IF NOT EXISTS contexts_A(
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      symbol TEXT, ctx TEXT, pb REAL, ps REAL, ph REAL, score REAL, ts INTEGER)""")
    c.execute("""CREATE TABLE IF NOT EXISTS contexts_A_state(
      symbol TEXT PRIMARY KEY, last_ctx TEXT, persist INT, since_ts INT)""")
    c.execute("CREATE INDEX IF NOT EXISTS idx_ctxA_symbol_ts ON contexts_A(symbol,ts)")
    return c

def top30(c):
    return [r[0] for r in c.execute("""
      SELECT symbol FROM v_universe ORDER BY symbol LIMIT 35
    """)]

def last_row(c, tf, sym):
    view={"5m":"indic_5m","15m":"indic_15m","30m":"indic_30m"}[tf]
    q=f"""SELECT close, sma20, sma50, macd, macd_sig, macd_hist, rsi14, atr_pct, obv
          FROM {view}
          WHERE symbol=?
          ORDER BY ts DESC LIMIT 1"""
    r=c.execute(q,(sym,)).fetchone()
    return r

def main():
    c=con(); now=int(time.time())
    syms=top30(c)
    ins=0; with_ctx=0
    for s in syms:
        agg_score=agg_pb=agg_ps=agg_ph=0.0; wsum=0.0; seen=False
        for tf in TF_USED:
            r=last_row(c,tf,s)
            if not r: continue
            seen=True
            close,sma20,sma50,macd,sig,hist,rsi,atr_pct,obv=r
            # scores élémentaires sur signaux SQL
            s_ema = 0.0 if sma20 is None or sma50 is None or close is None else ( (sma20 - sma50) / (close*0.01 + 1e-12) )
            s_macd = 0.0 if macd is None or sig is None else ( (macd - sig) / (abs(macd)+1e-9) )
            s_rsi = 0.0 if rsi is None else ((rsi-50.0)/50.0)
            s_obv = 0.0 if obv is None else max(-1.0, min(1.0, obv / 1e9))
            # simple proxy ADX via amplitude MACD hist
            s_adx = 0.0 if hist is None else max(-1.0, min(1.0, abs(hist)/ (abs(macd)+1e-9) ))
            score_raw = 0.35*s_ema + 0.25*s_macd + 0.20*s_rsi + 0.10*s_adx + 0.10*s_obv
            score_raw = max(-1.0, min(1.0, score_raw))
            gate = 1.0 if (atr_pct or 0.0) >= ATR_GATE[tf] else 0.0
            score = gate * score_raw
            pb,ps,ph = smx(max(score,0), max(-score,0), TAU[tf])
            w=W_TF[tf]; wsum+=w
            agg_score += w*score; agg_pb += w*pb; agg_ps += w*ps; agg_ph += w*ph
        if not seen or wsum==0:
            c.execute("INSERT INTO contexts_A(symbol,ctx,pb,ps,ph,score,ts) VALUES(?,?,?,?,?,?,?)",
                      (s,"none",0.0,0.0,1.0,0.0,now)); ins+=1; continue
        agg_score/=wsum; agg_pb/=wsum; agg_ps/=wsum; agg_ph/=wsum
        # hysteresis minimale
        row=c.execute("SELECT last_ctx,persist FROM contexts_A_state WHERE symbol=?",(s,)).fetchone()
        prev,pers=(row[0],row[1]) if row else ("none",0)
        ctx = "buy" if agg_pb>=0.60 and agg_pb>agg_ps else ("sell" if agg_ps>=0.60 and agg_ps>agg_pb else ("hold" if agg_ph>=max(agg_pb,agg_ps) else "hold"))
        if prev in("buy","sell") and ctx!=prev and max(agg_pb,agg_ps)<0.45: ctx=prev
        pers = (pers+1) if ctx==prev else 1
        if ctx!=prev and pers<2: ctx=prev
        c.execute("INSERT INTO contexts_A(symbol,ctx,pb,ps,ph,score,ts) VALUES(?,?,?,?,?,?,?)",
                  (s,ctx,agg_pb,agg_ps,agg_ph,agg_score,now))
        c.execute("INSERT INTO contexts_A_state(symbol,last_ctx,persist,since_ts) VALUES(?,?,?,?) \
                   ON CONFLICT(symbol) DO UPDATE SET last_ctx=excluded.last_ctx,persist=excluded.persist,since_ts=excluded.since_ts",
                  (s,ctx,pers,now))
        ins+=1; with_ctx+= int(ctx!='none')
    print(f"[AnalyseA] inserted={ins} with_ctx={with_ctx} ts={now}")

def fetch_v2(sym, gran):
    url=f"{S_BASE}/api/v2/mix/market/candles"
    p={"productType":"umcbl","symbol":sym,"granularity":gran,"limit":LIMIT}
    r=requests.get(url, params=p, timeout=10)
    if r.status_code!=200: raise RuntimeError(f"v2 {r.status_code}")
    j=r.json()
    if j.get("code") not in ("00000","0000",0,None): raise RuntimeError(f"v2 code {j.get('code')}")
    return j.get("data",[])

def fetch_v1(sym_umcbl, gran_str):
    url=f"{S_BASE}/api/mix/v1/market/candles"
    p={"symbol":sym_umcbl,"granularity":gran_str,"limit":LIMIT}
    r=requests.get(url, params=p, timeout=10)
    if r.status_code!=200: raise RuntimeError(f"v1 {r.status_code}")
    j=r.json()
    # v1 répond sans 'code' uniforme parfois; on vérifie présence 'data'
    if "data" not in j: raise RuntimeError("v1 no data")
    return j["data"]

def to_rows(sym, data):
    out=[]
    for k in data:
        # Format: [ts, open, high, low, close, volume, ...] ts en ms
        ts=int(int(k[0])//1000)
        o,h,l,c,vol=map(float,[k[1],k[2],k[3],k[4],k[5]])
        out.append((sym,ts,o,h,l,c,vol))
    return out

def upsert(tbl, rows):
    con=sqlite3.connect(ADB, timeout=10, isolation_level=None)
    con.execute("PRAGMA busy_timeout=8000"); con.execute("PRAGMA journal_mode=WAL")
    con.execute(f"""CREATE TABLE IF NOT EXISTS {tbl}(
        symbol TEXT, ts INTEGER,
        open REAL, high REAL, low REAL, close REAL, volume REAL,
        PRIMARY KEY(symbol,ts))""")
    con.execute(f"CREATE INDEX IF NOT EXISTS idx_{tbl}_sym_ts ON {tbl}(symbol,ts)")
    con.executemany(f"INSERT OR IGNORE INTO {tbl}(symbol,ts,open,high,low,close,volume) VALUES (?,?,?,?,?,?,?)", rows)
    con.close()

def main():
    ok=err=0
    for gran,(tbl,gran_str) in TF_MAP.items():

                ok+=1
            time.sleep(0.25)
        time.sleep(0.5)

#!/usr/bin/env python3
import sqlite3, json, math
DB="/opt/scalp/project/data/scalp.db"
OUT="/opt/scalp/project/data/analyse_A_out.json"

def series(con, sym, tf, n):
    return con.execute(f"SELECT close FROM ohlcv_{tf} WHERE symbol=? ORDER BY ts DESC LIMIT ?",(sym,n)).fetchall()

def mom(v): return (v[0]-v[-1])/v[-1] if len(v)>1 and v[-1] else 0.0
def vola(v): 
    if len(v)<2: return 0.0
    m=sum(v)/len(v); return math.sqrt(sum((x-m)**2 for x in v)/len(v))/m if m else 0.0

con=sqlite3.connect(DB)
syms=[r[0] for r in con.execute("SELECT symbol FROM tops WHERE ts=(SELECT MAX(ts) FROM tops) AND name='TOP35'")]
# normaliser les ids vers 'BASEUSDT'
norm=lambda s: s if s.endswith("USDT") else s.replace("/USDT","USDT").split(':')[0].replace('/','')
ctx={}
for s in syms:
    base = norm(s)
    v5  = [x[0] for x in series(con, base,  "5m",  60)]
    v15 = [x[0] for x in series(con, base, "15m", 60)]
    v30 = [x[0] for x in series(con, base, "30m", 60)]
    if not v5 or not v15 or not v30: 
        continue
    m5,m15,m30 = mom(v5), mom(v15), mom(v30)
    vol = 0.5*vola(v5)+0.3*vola(v15)+0.2*vola(v30)
    score = 0.5*m5 + 0.3*m15 + 0.2*m30
    trend = "bull" if score>0 else "bear"
    strength = round((abs(score)* (1+vol)),6)
    ctx[base]={"symbol":base+"USDT","trend":trend,"strength":strength,"volatility":vol}
with open(OUT,"w") as f: json.dump(ctx,f,indent=2)
print(f"[AnalyseA/OHLCV] {len(ctx)} contextes -> {OUT}")
con.close()

#!/usr/bin/env python3
import sqlite3, json, time, statistics as st

DB="/opt/scalp/project/data/scalp.db"
OUT_JSON="/opt/scalp/project/data/analyse_A_out.json"

NOW=int(time.time())
REQ_1M=200; REQ_5M=200; REQ_15M=100; REQ_30M=60

con=sqlite3.connect(DB, timeout=10, isolation_level=None)
c=con.cursor()
c.execute("PRAGMA journal_mode=WAL"); c.execute("PRAGMA synchronous=NORMAL")

c.execute("""CREATE TABLE IF NOT EXISTS contexts_A(
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  symbol TEXT, has_1m INT, has_5m INT, has_15m INT, has_30m INT,
  ctx TEXT, ts INTEGER
)""")

# pull TOP30
syms=[r[0] for r in c.execute("""
  SELECT symbol FROM top_scores
  WHERE ts=(SELECT MAX(ts) FROM top_scores) AND name='TOP30'
  ORDER BY score DESC, symbol ASC LIMIT 30
""").fetchall()]

def enough(tbl, sym, need):
  row=c.execute(f"SELECT COUNT(*) FROM {tbl} WHERE symbol=?;", (sym,)).fetchone()
  return (row and row[0] >= need)

contexts=[]
ts=NOW
for s in syms:
  h1=enough("ohlcv_1m", s, REQ_1M)
  h5=enough("ohlcv_5m", s, REQ_5M)
  h15=enough("ohlcv_15m", s, REQ_15M)
  h30=enough("ohlcv_30m", s, REQ_30M)

  ctx=None
  if h1:
    # mini-contexte: slope sur 1m via 50 dernières closes
    rows=c.execute("SELECT close FROM ohlcv_1m WHERE symbol=? ORDER BY ts DESC LIMIT 50;", (s,)).fetchall()
    closes=[r[0] for r in rows][::-1]
    if len(closes)>=20:
      sma20=sum(closes[-20:])/20.0
      sma50=sum(closes)/len(closes)
      mom=closes[-1]-closes[-5] if len(closes)>=5 else 0.0
      vol=st.pstdev(closes[-20:]) if len(closes)>=20 else 0.0
      trend="up" if sma20>sma50 else "down"
      ctx=json.dumps({"trend":trend,"mom":mom,"vol":vol}, separators=(",",":"))

  c.execute("""INSERT INTO contexts_A(symbol,has_1m,has_5m,has_15m,has_30m,ctx,ts)
               VALUES(?,?,?,?,?,?,?)""",(s,int(h1),int(h5),int(h15),int(h30),ctx,ts))
  contexts.append({"symbol":s,"has_1m":h1,"has_5m":h5,"has_15m":h15,"has_30m":h30,"ctx":json.loads(ctx) if ctx else None})

# export json
with open(OUT_JSON,"w") as f: json.dump({"ts":ts,"contexts":contexts}, f)

# print summary
total=len(contexts)
with_ctx=sum(1 for x in contexts if x["ctx"] is not None)
print(f"[AnalyseA] contexts={total} with_ctx={with_ctx} ts={ts}")

import asyncio, time, json, websockets

URI = "wss://ws.bitget.com/v2/ws/public"
SUBS = [
  ("NEW", {"op":"subscribe","args":[{"instType":"USDT-FUTURES","channel":"ticker","instId":"BTCUSDT_UMCBL"}]}),
  ("OLD", {"op":"subscribe","args":[{"instType":"mc","channel":"ticker","instId":"BTCUSDT"}]}),
]

async def probe(tag, sub):
    print(f"--- PROBE {tag} ---")
    try:
        async with websockets.connect(URI, ping_interval=20, ping_timeout=20, close_timeout=5, max_size=2**20) as ws:
            await asyncio.sleep(0.5)
            await ws.send(json.dumps(sub))
            start=time.time(); frames=0
            while time.time()-start < 20:
                try:
                    msg = await asyncio.wait_for(ws.recv(), timeout=10)
                except asyncio.TimeoutError:
                    print(f"[{tag}] timeout"); continue
                if isinstance(msg,(bytes,bytearray)):
                    try: msg=msg.decode()
                    except: continue
                try:
                    data=json.loads(msg)
                except Exception:
                    continue
                frames+=1
                if "event" in data and data["event"] in ("error","subscribe","subscribed"):
                    print(f"[{tag}] {data}")
                if "data" in data:
                    # tronque l'affichage
                    ds=str(data)[:220].replace("\n"," ")
                    print(f"[{tag}] DATA {ds}")
                    return True
            print(f"--- END {tag} frames={frames} ---")
            return False
    except Exception as e:
        print(f"[{tag}] FAIL {type(e).__name__}: {e}")
        return False

    ok_any=False
    for tag,sub in SUBS:
        ok = await probe(tag, sub)
        ok_any = ok_any or ok
    print("RESULT:", "OK" if ok_any else "NO DATA")
