import time, sqlite3, math, json
ADB="data/a.db"
# pondérations TF
W5, W15, W30 = 0.2, 0.3, 0.5
TAU=1.5 # température softmax

def ema(vals, n):
    k=2/(n+1)
    e=None
    for v in vals:
        e = v if e is None else (v*k + e*(1-k))
    return e if e is not None else 0.0

def load_close(conn, table, instId, n=50):
    rows=list(conn.execute(f"SELECT c FROM {table} WHERE instId=? ORDER BY ts DESC LIMIT ?", (instId,n)))
    return [r[0] for r in rows][::-1]

def score_ctx(cs):
    # signal scalaire par TF via pentes EMA
    if len(cs)<20: return 0.0
    s_fast = ema(cs[-10:], 5)
    s_slow = ema(cs[-20:],10)
    d = (s_fast - s_slow)/max(1e-9, s_slow)
    return d

def softmax3(x_buy, x_hold, x_sell, tau=TAU):
    exps=[math.exp(x/tau) for x in (x_buy, x_hold, x_sell)]
    s=sum(exps); 
    return [e/s for e in exps]

def decide(inst, c5, c15, c30):
    s5  = score_ctx(c5)
    s15 = score_ctx(c15)
    s30 = score_ctx(c30)
    x = W5*s5 + W15*s15 + W30*s30
    # map vers 3 classes: buy (x), hold (-|x|), sell (-x)
    p_buy, p_hold, p_sell = softmax3(x, -abs(x)*0.5, -x)
    ctx = "buy" if p_buy>max(p_hold,p_sell) else ("sell" if p_sell>max(p_buy,p_hold) else ("hold" if max(p_buy,p_sell)<0.45 else "none"))
    score_A = (p_buy - p_sell) # [-1..1] centré
    return ctx, p_buy, p_hold, p_sell, score_A

def run():
    with sqlite3.connect(ADB) as c:
        c.execute("PRAGMA busy_timeout=8000")
        syms=[r[0] for r in c.execute("SELECT DISTINCT instId FROM ohlcv_5m")]
        now=int(time.time())
        for s in syms:
            c5  = load_close(c,"ohlcv_5m", s, 50)
            c15 = load_close(c,"ohlcv_15m",s, 50)
            c30 = load_close(c,"ohlcv_30m",s, 50)
            if not c5 or not c15 or not c30: continue
            ctx,pb,ph,ps,scoreA = decide(s,c5,c15,c30)
            det=json.dumps({"s5":c5[-1],"s15":c15[-1],"s30":c30[-1]})
            ts = int(time.time()//300*300) # align 5m
            c.execute("""INSERT OR REPLACE INTO contexts_A(instId,ts,ctx,p_buy,p_hold,p_sell,score_A,details,updated_ts)
                         VALUES(?,?,?,?,?,?,?,?,?)""",(s,ts,ctx,pb,ph,ps,scoreA,det,now))
        c.commit()

if __name__=="__main__":

def aggregate_tf(conn, tf_src: str, tf_dst: str, mult: int):
    src = f"ohlcv_{tf_src}"
    dst = f"ohlcv_{tf_dst}"
    cur = conn.cursor()
    cur.execute(f"DELETE FROM {dst};")

    rows = cur.execute(f"""
        SELECT instId, ts, o, h, l, c, v FROM {src}
        ORDER BY instId, ts
    """).fetchall()

    out = []
    bucket = {}
    for instId, ts, o, h, l, c, v in rows:
        grp = int(ts / (mult * 60))  # tranches de mult minutes
        key = (instId, grp)
        if key not in bucket:
            bucket[key] = [ts, o, h, l, c, v]
        else:
            b = bucket[key]
            b[2] = max(b[2], h)
            b[3] = min(b[3], l)
            b[4] = c
            b[5] += v

    for (instId, grp), (ts, o, h, l, c, v) in bucket.items():
        out.append((instId, grp * mult * 60, o, h, l, c, v))

    cur.executemany(f"""
        INSERT OR REPLACE INTO {dst} (instId, ts, o, h, l, c, v)
        VALUES (?, ?, ?, ?, ?, ?, ?)
    """, out)
    conn.commit()
    print(f"[OB] Aggregated {len(out)} candles {tf_src}->{tf_dst}")

# ============================================================

#!/usr/bin/env python3
import sqlite3, time

ROOT = "/opt/scalp/project"
BDB = f"{ROOT}/data/b.db"
XDB = f"{ROOT}/data/x.db"
TDB = f"{ROOT}/data/t.db"

def get_ready_signals(conn):
    return conn.execute("""
        SELECT instId,side,entry,sl,tp,qty,leverage,ts_ms
        FROM signals_B
        WHERE status='ready'
        ORDER BY ts_ms ASC
    """).fetchall()

def last_price(connT, inst):
    r = connT.execute("""
        SELECT lastPr FROM ticks
        WHERE instId=?
        ORDER BY ts_ms DESC LIMIT 1
    """,(inst,)).fetchone()
    return r[0] if r else None

def open_position(connX, inst, side, entry, sl, tp, qty, lev, ts):
    connX.execute("""
        INSERT INTO positions_open(instId,side,entry,sl,tp,qty,leverage,opened_ts)
        VALUES(?,?,?,?,?,?,?,?)
    """,(inst,side,entry,sl,tp,qty,lev,ts))

def close_position(connX, inst, price, ts, reason):
    row = connX.execute("""
        SELECT id,side,entry,sl,tp,qty,leverage
        FROM positions_open
        WHERE instId=?
        ORDER BY opened_ts DESC LIMIT 1
    """,(inst,)).fetchone()
    if not row: return
    pid, side, entry, sl, tp, qty, lev = row
    pnl = (price-entry)*qty*(lev if side=='buy' else -lev)
    connX.execute("""
        INSERT INTO positions_closed(instId,side,entry,exit,qty,leverage,pnl,closed_ts,reason)
        VALUES(?,?,?,?,?,?,?,?,?)
    """,(inst,side,entry,price,qty,lev,pnl,ts,reason))
    connX.execute("DELETE FROM positions_open WHERE id=?",(pid,))

def process_signals():
    connB = sqlite3.connect(BDB)
    connX = sqlite3.connect(XDB)
    connT = sqlite3.connect(TDB)

    signals = get_ready_signals(connB)

    for inst,side,entry,sl,tp,qty,lev,ts in signals:
        connX.execute("""
            CREATE TABLE IF NOT EXISTS positions_open(
              id INTEGER PRIMARY KEY AUTOINCREMENT,
              instId TEXT, side TEXT, entry REAL, sl REAL, tp REAL,
              qty REAL, leverage REAL, opened_ts INTEGER
            );
        """)
        connX.execute("""
            CREATE TABLE IF NOT EXISTS positions_closed(
              id INTEGER PRIMARY KEY AUTOINCREMENT,
              instId TEXT, side TEXT, entry REAL, exit REAL,
              qty REAL, leverage REAL, pnl REAL, closed_ts INTEGER, reason TEXT
            );
        """)

        # --- OPEN POSITION ---
        open_position(connX,inst,side,entry,sl,tp,qty,lev,ts)
        connB.execute("UPDATE signals_B SET status='filled' WHERE instId=? AND ts_ms=?",(inst,ts))

    connX.commit()
    connB.commit()

    # --- PROCESS CLOSES ---
    rows = connX.execute("SELECT instId,side,entry,sl,tp FROM positions_open").fetchall()
    for inst,side,entry,sl,tp in rows:
        price = last_price(connT,inst)
        now   = int(time.time()*1000)
        if not price: continue

        if side=="buy":
            if price<=sl: close_position(connX,inst,price,now,"SL")
            elif price>=tp: close_position(connX,inst,price,now,"TP")
        else:
            if price>=sl: close_position(connX,inst,price,now,"SL")
            elif price<=tp: close_position(connX,inst,price,now,"TP")

    connX.commit()
    connX.close()
    connB.close()
    connT.close()

if __name__=="__main__":
    process_signals()

#!/usr/bin/env python3
import sqlite3, math, statistics

DB_OB = "/opt/scalp/project/data/ob.db"
DB_B  = "/opt/scalp/project/data/b.db"

def ema(vals, n):
    k=2/(n+1); e=None
    for v in vals:
        e=v if e is None else (v*k + e*(1-k))
    return e or 0.0

def rsi(vals, n):
    if len(vals)<n+1: return 50
    gains=[]; losses=[]
    for i in range(1,len(vals)):
        diff=vals[i]-vals[i-1]
        (gains if diff>0 else losses).append(abs(diff))
    avg_gain=sum(gains)/len(gains) if gains else 0
    avg_loss=sum(losses)/len(losses) if losses else 1
    rs=avg_gain/avg_loss
    return 100-(100/(1+rs))

def atr(highs,lows,closes,n=14):
    trs=[highs[i]-lows[i] for i in range(len(highs))]
    if len(trs)<n: return trs[-1]
    return sum(trs[-n:])/n

def bb(vals,n=20):
    if len(vals)<n: return (vals[-1],vals[-1],vals[-1])
    chunk=vals[-n:]
    mid=sum(chunk)/n
    sd=statistics.pstdev(chunk)
    return (mid,mid+2*sd,mid-2*sd)

def obv(closes,vols):
    total=0
    for i in range(1,len(closes)):
        if closes[i]>closes[i-1]: total+=vols[i]
        elif closes[i]<closes[i-1]: total-=vols[i]
    return total

def process_tf(tf):
    table=f"ohlcv_{tf}"
    feat=f"feat_{tf}"
    with sqlite3.connect(DB_OB) as co, sqlite3.connect(DB_B) as cb:
        rows=list(co.execute(f"SELECT instId,ts,o,h,l,c,v FROM {table} ORDER BY ts"))
        symbols=sorted(set(r[0] for r in rows))
        cb.execute(f"DELETE FROM {feat}")
        for sym in symbols:
            data=[r for r in rows if r[0]==sym]
            closes=[d[4] for d in data]
            highs =[d[3] for d in data]
            lows  =[d[4] for d in data]
            vols  =[d[5] for d in data]
            ts    =[d[1] for d in data]
            if len(closes)<30: continue

            ema12v=ema(closes[-26:],12)
            ema26v=ema(closes[-26:],26)
            macd_val=ema12v-ema26v
            macd_sig=ema([ema(closes[-26:],12)-ema(closes[-26:],26)],9)
            macd_hist=macd_val-macd_sig

            rsi7v=rsi(closes,7)
            rsi14v=rsi(closes,14)
            atrv=atr(highs,lows,closes,14)
            bbm,bbu,bbl=bb(closes,20)
            obvv=obv(closes,vols)
            vwap=sum(closes[-20:])/20

            cb.execute(f"""
            INSERT INTO {feat}(instId,ts,c,ema12,ema26,macd,macd_signal,macd_hist,
                                rsi7,rsi14,atr14,obv20,bb_mid,bb_up,bb_lo,vwap)
            VALUES(?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)
            """,(sym,ts[-1],closes[-1],ema12v,ema26v,macd_val,macd_sig,macd_hist,
                 rsi7v,rsi14v,atrv,obvv,bbm,bbu,bbl,vwap))
        cb.commit()

def main():
    for tf in ("1m","3m","5m"):
        process_tf(tf)

if __name__=="__main__":
    main()

import sqlite3

XDB = "/opt/scalp/project/data/x.db"

def update_score_H():
    """Recalcule les scores historiques H pour chaque coin/side/raison."""
    with sqlite3.connect(XDB) as conn:
        conn.execute("""
        INSERT INTO history_H(instId, side, reason, n_trades, win_ratio, score_H)
        SELECT
            instId,
            side,
            reason,
            COUNT(*) AS n,
            AVG(CASE WHEN pnl_usdt>0 THEN 1.0 ELSE 0.0 END) AS win_ratio,
            ROUND(AVG(CASE WHEN pnl_usdt>0 THEN 1.0 ELSE -1.0 END),3) AS score_H
        FROM positions_closed
        GROUP BY instId, side, reason
        ON CONFLICT(instId, side, reason)
        DO UPDATE SET

    try:
        r = conn.execute("SELECT atr14 FROM feat_1m WHERE instId=? ORDER BY ts DESC LIMIT 1", (sym,)).fetchone()
        return float(r[0]) if r and r[0] is not None else None
    except:
        return None

def record_close(conn, instId, side, entry, exit_p, qty, lev, ts_open, ts_close, reason):
    pnl_usdt = (exit_p - entry) * qty if side=="buy" else (entry - exit_p) * qty
    notional = abs(entry * qty) if entry!=0 else 1.0
    pnl_pct = (pnl_usdt / notional) * 100.0
    conn.execute("""
        INSERT INTO positions_closed(instId,side,entry,exit,qty,leverage,ts_open,ts_close,pnl_usdt,pnl_pct,reason_exit)
        VALUES(?,?,?,?,?,?,?,?,?,?,?)
    """,(instId,side,entry,exit_p,qty,lev,ts_open,ts_close,pnl_usdt,pnl_pct,reason))
    conn.commit()

        # 1) SL
        if (side=="buy" and px<=sl) or (side=="sell" and px>=sl):

class EMA:
    def __init__(self): self.vals=[]
    def step(self, v):
        try: self.vals.append(float(v))
        except: pass
    def finalize(self):
        if not self.vals: return 0.0
        n=len(self.vals)
        k=2/(n+1)
        e=None
        for v in self.vals:
            e=v if e is None else (v*k + e*(1-k))
        return e or 0.0

import sqlite3, math, os

DB_PATH = "/opt/scalp/project/data/b.db"

class EMA:
    def __init__(self): self.vals=[]
    def step(self, v):
        try: self.vals.append(float(v))
        except: pass
    def finalize(self):
        if not self.vals: return 0.0
        n=len(self.vals)
        k=2/(n+1)
        e=None
        for v in self.vals:
            e=v if e is None else (v*k + e*(1-k))
        return e or 0.0

def register_udf():
    if not os.path.exists(DB_PATH):
        raise FileNotFoundError(f"Database not found: {DB_PATH}")
    conn = sqlite3.connect(DB_PATH)
    conn.create_aggregate("ema", 1, EMA)
    conn.create_function("tanh", 1, math.tanh)
    conn.commit()
    conn.close()
    print("✅ ema() et tanh() enregistrés dans", DB_PATH)

# === utilitaires ===
def ema(series, n):
    if not series:
        return 0.0
    k = 2 / (n + 1)
    e = None
    for v in series:
        e = v if e is None else (v * k + e * (1 - k))
    return e or 0.0

def rsi(vals, n=14):
    if len(vals) < n + 1:
        return 50.0
    gains, losses = [], []
    for i in range(1, len(vals)):
        diff = vals[i] - vals[i - 1]
        gains.append(max(diff, 0))
        losses.append(abs(min(diff, 0)))
    avg_gain = sum(gains[-n:]) / n
    avg_loss = sum(losses[-n:]) / n
    if avg_loss == 0:
        return 100.0
    rs = avg_gain / avg_loss
    return 100 - (100 / (1 + rs))

def atr(vals, n=14):
    if len(vals) < n + 1:
        return 0.0
    diffs = [abs(vals[i] - vals[i - 1]) for i in range(1, len(vals))]
    return sum(diffs[-n:]) / n

def ema(values, period):
    if len(values)<period: return [None]*len(values)
    k=2/(period+1); out=[values[0]]
    for v in values[1:]: out.append(v*k+out[-1]*(1-k))
    return out

def rsi(values, period):
    if len(values)<period+1: return [None]*len(values)
    gains=[0]; losses=[0]
    for i in range(1,len(values)):
        diff=values[i]-values[i-1]
        gains.append(max(0,diff)); losses.append(abs(min(0,diff)))
    out=[]
    for i in range(len(values)):
        if i<period: out.append(None)
        else:
            avg_gain=mean(gains[i-period+1:i+1])
            avg_loss=mean(losses[i-period+1:i+1])
            if avg_loss==0: out.append(100)
            else:
                rs=avg_gain/avg_loss
                out.append(100-(100/(1+rs)))
    return out

    mid=[]; up=[]; lw=[]
    for i in range(len(values)):
        if i<period: mid.append(None); up.append(None); lw.append(None)
        else:
            w=values[i-period+1:i+1]; m=mean(w); s=pstdev(w)
            mid.append(m); up.append(m+mult*s); lw.append(m-mult*s)
    return mid,up,lw

        else: out.append(mean(trs[i-period+1:i+1]))

# =========================================================
#  Sélection des coins U + vérification WS Bitget

import sqlite3, time, os, asyncio, websockets, json, math

# ---------------------------------------------------------
# Vérification individuelle WS
# ---------------------------------------------------------
async def ws_check_one(sym: str, timeout: float = 1.5) -> bool:
    payload = {"op": "subscribe",
               "args": [{"instType": "mc", "channel": "ticker", "instId": sym}]}
    try:
        async with websockets.connect(
            WS_URI,
            ping_interval=20,
            ping_timeout=20,
            close_timeout=1
        ) as ws:
            await ws.send(json.dumps(payload))
            msg = await asyncio.wait_for(ws.recv(), timeout=timeout)
            if '"event":"error"' in msg or ('"code":' in msg and '"code":0' not in msg):
                return False
            return True
    except Exception:
        return False

# ---------------------------------------------------------
# Vérification par lots (async)
# ---------------------------------------------------------
async def ws_check_batch(symbols, max_conc: int = 8) -> dict:
    sem = asyncio.Semaphore(max_conc)
    async def wrapped(sym):
        async with sem:
            return sym, await ws_check_one(sym)
    tasks = [wrapped(s) for s in symbols]
    out = {}
    for coro in asyncio.as_completed(tasks):
        sym, ok = await coro
        out[sym] = ok
    return out

def update_scores(c: sqlite3.Connection):
    c.execute("""
        WITH bounds AS (
          SELECT MAX(LOG(rank + 2)) AS max_rank FROM u_syms
        )
        UPDATE u_syms
        SET
          score_vol = 1 - (LOG(rank + 2) / (SELECT max_rank FROM bounds)),
          score_trend = (score - (SELECT AVG(score) FROM u_syms)) / (SELECT AVG(score) FROM u_syms),
          score_consistency = 1 - ABS(score_trend),
          score_U = 0.5*score_vol + 0.3*score_trend + 0.2*score_consistency;

#!/usr/bin/env python3
import sqlite3, time, math

ROOT = "/opt/scalp/project"
ADB  = f"{ROOT}/data/a.db"
OADB = f"{ROOT}/data/oa.db"

# pondérations multi-TF
W5, W15, W30 = 0.2, 0.3, 0.5
TAU = 0.35   # température softmax (A1 tendanciel)

def softmax3(xb, xh, xs, tau):
    exb = math.exp(xb/tau)
    exh = math.exp(xh/tau)
    exs = math.exp(xs/tau)
    s = exb+exh+exs
    return exb/s, exh/s, exs/s

def ema(vals, n):
    k=2/(n+1); e=None
    for v in vals:
        e = v if e is None else (v*k + e*(1-k))
    return e or 0.0

def load_close(conn, table, inst, n=60):
    rows=[r[0] for r in conn.execute(
        f"SELECT close FROM {table} WHERE instId=? ORDER BY ts DESC LIMIT ?",
        (inst,n)
    )]
    return rows[::-1]

def score_tf(cs):
    if len(cs)<30: return 0.0
    ef=ema(cs[-10:],5)
    es=ema(cs[-20:],10)
    return math.tanh((ef-es)/(abs(es)+1e-9))

def main():
    conA=sqlite3.connect(ADB)
    conO=sqlite3.connect(OADB)
    syms=[r[0] for r in conO.execute("SELECT DISTINCT instId FROM ohlcv_5m")]
    now=int(time.time())

    for s in syms:
        c5 = load_close(conO,"ohlcv_5m", s)
        c15= load_close(conO,"ohlcv_15m",s)
        c30= load_close(conO,"ohlcv_30m",s)
        if not c5 or not c15 or not c30: continue

        s5 = score_tf(c5)
        s15= score_tf(c15)
        s30= score_tf(c30)
        X  = W5*s5 + W15*s15 + W30*s30

        p_buy, p_hold, p_sell = softmax3(X, -abs(X)*0.3, -X, TAU)
        if p_buy>=0.60: ctx="buy"
        elif p_sell>=0.60: ctx="sell"
        elif 0.40<=p_hold<=0.60: ctx="hold"
        else: ctx="range"

        score_A = p_buy - p_sell

        conA.execute("""
            INSERT OR REPLACE INTO ctx_A(instId,ctx,p_buy,p_hold,p_sell,score_A,updated_ts)
            VALUES(?,?,?,?,?,?,?)
        """,(s,ctx,p_buy,p_hold,p_sell,score_A,now))

    conA.commit()
    conA.close()
    conO.close()

if __name__=="__main__":
    main()
