#!/usr/bin/env sh
set -eu
BIN="/opt/scalp/project/bin"
MAN="/opt/scalp/project/etc/b.manifest"
[ -r "$MAN" ] || { echo "absent: $MAN"; exit 2; }
# filtrer les fichiers manquants et mismatches
BAD=0
while IFS= read -r line; do
  sum="$(echo "$line" | awk '{print $1}')"
  file="$(echo "$line" | awk '{print $2}')"
  # compat chemin absolu/relatif
  base="$(basename "$file")"
  path="$BIN/$base"
  if [ ! -f "$path" ]; then
    echo "missing: $base"; BAD=1; continue
  fi
  cur="$(sha256sum "$path" | awk '{print $1}')"
  [ "$cur" = "$sum" ] || { echo "mismatch: $base"; BAD=1; }
done < "$MAN"
[ $BAD -eq 0 ] && { echo "integrity: ok"; exit 0; } || { echo "integrity: FAIL"; exit 1; }

CREATE TABLE IF NOT EXISTS signals_B_plan(
 id INTEGER PRIMARY KEY AUTOINCREMENT,
 symbol TEXT, ctx TEXT, score REAL, side TEXT,
 entry REAL, sl REAL, tp REAL, qty REAL, margin REAL, notional REAL, created_ts INTEGER);
CREATE TABLE IF NOT EXISTS signals_B(
 id INTEGER PRIMARY KEY AUTOINCREMENT,
 symbol TEXT, side TEXT, price REAL, reason TEXT, created_ts INTEGER);
CREATE INDEX IF NOT EXISTS idx_sigB_ts ON signals_B(created_ts);
CREATE INDEX IF NOT EXISTS idx_sigB_sym_ts ON signals_B(symbol,created_ts);
CREATE TABLE IF NOT EXISTS b_state(symbol TEXT PRIMARY KEY, last_side TEXT, consec INTEGER, last_ts INTEGER);

DROP TABLE IF EXISTS _params;
CREATE TEMP TABLE _params AS
SELECT $MIN_SCORE AS min_score,$PLAN_MAX_AGE AS plan_age,$COOLDOWN_SEC AS cooldown,
       $MAX_CONSEC AS max_consec,$REQUIRE_TREND AS require_trend,$ATR_PCT_1M_MIN AS atr_min_pct,

SELECT i.*, f.vwap20,f.bb_mid20,f.bb_up20,f.bb_dn20,f.macd,f.macd_signal9,f.macd_hist,f.obv
FROM v_indic_1m i

WHERE (i.symbol,i.ts) IN (SELECT symbol, MAX(ts) FROM v_indic_1m GROUP BY symbol);

  AND (c.symbol IS NULL OR (p.side='BUY' AND c.pb>=0.60) OR (p.side='SELL' AND c.ps>=0.60))

SELECT m.symbol,
       CASE WHEN c.ctx='bullish' AND m.close>m.hh_20_excl AND m.close>=COALESCE(m.vwap20,m.close) THEN 'BUY'
            WHEN c.ctx='bearish' AND m.close<m.ll_20_excl AND m.close<=COALESCE(m.vwap20,m.close) THEN 'SELL' END AS side,
       m.close AS entry,
       CASE WHEN c.ctx='bullish' THEN m.close - ((SELECT k_sl FROM _params)*m.atr14)
            WHEN c.ctx='bearish' THEN m.close + ((SELECT k_sl FROM _params)*m.atr14) END AS sl,
       CASE WHEN c.ctx='bullish' THEN m.close + ((SELECT k_tp FROM _params)*m.atr14)

       'breakoutN+VWAP+BB' AS reason, m.ts AS bar_ts, m.atr_pct

WHERE m.atr_pct >= (SELECT atr_min_pct FROM _params)
  AND ( (c.ctx='bullish' AND m.close>m.hh_20_excl AND (m.bb_up20 IS NULL OR m.close>=m.bb_mid20))

WITH obv_step AS (
  SELECT m.symbol,m.ts, m.obv - LAG(m.obv) OVER (PARTITION BY m.symbol ORDER BY m.ts) AS d_obv FROM _m1 m

SELECT m.symbol,
       CASE WHEN m.macd IS NOT NULL AND m.macd_signal9 IS NOT NULL AND m.macd>m.macd_signal9 AND COALESCE(o.d_obv,0)>=0 THEN 'BUY'

  last_side=excluded.last_side,
  consec=CASE WHEN b_state.last_side=excluded.last_side THEN b_state.consec+1 ELSE 1 END,
  last_ts=excluded.last_ts;

  now=$(date +%s)
  next=$(( now - (now % STEP) + STEP + OFFSET ))
  sleep $(( next - now ))

  # timestamp de la bougie 5m qui vient de se clore
  n=$(date +%s)

  UNION ALL
  SELECT CASE WHEN MAX(ts)=$exp THEN 1 ELSE 0 END FROM ohlcv_15m
  UNION ALL
  SELECT CASE WHEN MAX(ts)=$exp THEN 1 ELSE 0 END FROM ohlcv_30m
);" 2>/dev/null
}

  "$SCRIPTS/fetch_ohlcv_once.sh"  >> "$LOG" 2>&1 || true
  "$SCRIPTS/harden_o_db.sh"       >> "$LOG" 2>&1 || true
  "$SCRIPTS/purge_non_current.sh" >> "$LOG" 2>&1 || true

  run_cycle

    run_cycle

#!/usr/bin/env bash
set -euo pipefail
ADB="/opt/scalp/project/data/a.db"

sqlite3 "$ADB" <<'SQL'
PRAGMA busy_timeout=8000;
BEGIN IMMEDIATE;

DROP TABLE IF EXISTS _src;
CREATE TEMP TABLE _src AS
SELECT symbol,'5m' tf,ts,o,h,l,c,v FROM ohlcv_5m
UNION ALL SELECT symbol,'15m',ts,o,h,l,c,v FROM ohlcv_15m
UNION ALL SELECT symbol,'30m',ts,o,h,l,c,v FROM ohlcv_30m;

DROP TABLE IF EXISTS _base;
CREATE TEMP TABLE _base AS
SELECT symbol,tf,ts,o,h,l,c,v,
       LAG(c) OVER (PARTITION BY symbol,tf ORDER BY ts) AS cp
FROM _src;

DROP TABLE IF EXISTS _rsi;
CREATE TEMP TABLE _rsi AS
WITH g AS (
  SELECT symbol,tf,ts,
         MAX(c-COALESCE(cp,c),0) AS g,
         MAX(COALESCE(cp,c)-c,0) AS l
  FROM _base
)
SELECT symbol,tf,ts,
       100-100.0/(1+AVG(g) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 13 PRECEDING)
                    /NULLIF(AVG(l) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 13 PRECEDING),0)) AS rsi
FROM g;

DROP TABLE IF EXISTS _ema;
CREATE TEMP TABLE _ema AS
SELECT symbol,tf,ts,
       AVG(c) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 9 PRECEDING)  AS ema_f,
       AVG(c) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 21 PRECEDING) AS ema_s
FROM _base;

DROP TABLE IF EXISTS _macd;
CREATE TEMP TABLE _macd AS
SELECT symbol,tf,ts,
       (ema_f-ema_s) AS macd,
       AVG(ema_f-ema_s) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 8 PRECEDING) AS macd_sig
FROM _ema;

DROP TABLE IF EXISTS _adx;
CREATE TEMP TABLE _adx AS
WITH x AS (
  SELECT symbol,tf,ts,h,l,
         LAG(h) OVER (PARTITION BY symbol,tf ORDER BY ts) AS hp,
         LAG(l) OVER (PARTITION BY symbol,tf ORDER BY ts) AS lp
  FROM _base
),
dm AS (
  SELECT symbol,tf,ts,
         MAX(h-COALESCE(hp,h),0) AS dmp,
         MAX(COALESCE(lp,l)-l,0) AS dmm
  FROM x
)
SELECT symbol,tf,ts,
       100*ABS(AVG(dmp) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 13 PRECEDING)
             -AVG(dmm) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 13 PRECEDING))
       /(AVG(dmp) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 13 PRECEDING)
        +AVG(dmm) OVER (PARTITION BY symbol,tf ORDER BY ts ROWS 13 PRECEDING)+0.001) AS adx
FROM dm;

DROP TABLE IF EXISTS _obv;
CREATE TEMP TABLE _obv AS
WITH x AS (
  SELECT symbol,tf,ts,c,v,
         LAG(c) OVER (PARTITION BY symbol,tf ORDER BY ts) AS cp
  FROM _base
),
o AS (
  SELECT symbol,tf,ts,
         SUM(CASE WHEN cp IS NULL THEN 0
                  WHEN c>cp THEN v
                  WHEN c<cp THEN -v
                  ELSE 0 END) OVER (PARTITION BY symbol,tf ORDER BY ts) AS obv
  FROM x
)
SELECT symbol,tf,ts,obv,
       obv-LAG(obv,20) OVER (PARTITION BY symbol,tf ORDER BY ts) AS obv_slope
FROM o;

DROP TABLE IF EXISTS _merge;
CREATE TEMP TABLE _merge AS
SELECT b.symbol,b.tf,b.ts,b.c,b.v,
       e.ema_f,e.ema_s,m.macd,m.macd_sig,
       r.rsi,a.adx,o.obv_slope
FROM _base b
LEFT JOIN _ema e USING(symbol,tf,ts)
LEFT JOIN _macd m USING(symbol,tf,ts)
LEFT JOIN _rsi r USING(symbol,tf,ts)
LEFT JOIN _adx a USING(symbol,tf,ts)
LEFT JOIN _obv o USING(symbol,tf,ts);

DROP TABLE IF EXISTS _s_tf;
CREATE TEMP TABLE _s_tf AS
SELECT symbol,tf,ts,
       ((ema_f-ema_s)/NULLIF(ABS(ema_f)+ABS(ema_s),0))*0.4 +
       ((macd-macd_sig)/NULLIF(ABS(macd)+ABS(macd_sig),0))*0.3 +
       ((rsi-50)/50.0)*0.2 +
       ((adx-20)/20.0)*0.1 +
       (obv_slope/NULLIF(ABS(obv_slope)+1,0))*0.1 AS s
FROM _merge;

DROP TABLE IF EXISTS _S;
CREATE TEMP TABLE _S AS
WITH w AS (SELECT 0.5 AS w30,0.3 AS w15,0.2 AS w5)
SELECT s.symbol,
       SUM(CASE 
              WHEN tf='30m' THEN s.s*(SELECT w30 FROM w)
              WHEN tf='15m' THEN s.s*(SELECT w15 FROM w)
              WHEN tf='5m'  THEN s.s*(SELECT w5  FROM w)
              ELSE 0 END) AS S
FROM _s_tf s GROUP BY s.symbol;

-- Normalisation des probas: somme = 1
DROP TABLE IF EXISTS _probs;
CREATE TEMP TABLE _probs AS
WITH raw AS (
  SELECT symbol,S,
         CASE WHEN S>0 THEN S ELSE 0 END AS pb_raw,
         CASE WHEN S<0 THEN -S ELSE 0 END AS ps_raw,
         CASE WHEN ABS(S)<=1 THEN 1-ABS(S) ELSE 0 END AS ph_raw
  FROM _S
),
z AS (
  SELECT symbol,S,pb_raw,ph_raw,ps_raw,(pb_raw+ph_raw+ps_raw) AS z FROM raw
)
SELECT symbol,S,
       CASE WHEN z=0 THEN 0.0 ELSE pb_raw/z END AS pb,
       CASE WHEN z=0 THEN 1.0 ELSE ph_raw/z END AS ph,
       CASE WHEN z=0 THEN 0.0 ELSE ps_raw/z END AS ps
FROM z;

DROP TABLE IF EXISTS _ctx;
CREATE TEMP TABLE _ctx AS
SELECT symbol,S,pb,ph,ps,
       CASE WHEN S>0.25 THEN 'bullish'
            WHEN S<-0.25 THEN 'bearish'
            ELSE 'range' END AS ctx
FROM _probs;

INSERT OR REPLACE INTO a_context_state(symbol,ctx,pb,ph,ps,s,updated_at)
SELECT symbol,ctx,pb,ph,ps,S,datetime('now') FROM _ctx;

INSERT INTO a_signals(ts,symbol,s,pb,ph,ps,ctx)
SELECT datetime('now'),symbol,S,pb,ph,ps,ctx FROM _ctx;

COMMIT;
SQL

WHERE s.id > _st.last_id
  AND s.symbol IN (SELECT symbol FROM _tradable);

SELECT p.*
FROM bdb.signals_B_plan p
JOIN (
  SELECT symbol, side, MAX(created_ts) mx FROM bdb.signals_B_plan GROUP BY symbol, side
) mx ON mx.symbol=p.symbol AND mx.side=p.side AND mx.mx=p.created_ts;

SELECT s.signal_id,s.symbol,s.side,s.price,s.reason,s.created_ts,

def load_symbols_from_A():
    conn = sqlite3.connect(A_DB)
    cur = conn.cursor()
    try:
        rows = cur.execute("SELECT DISTINCT symbol FROM contexts_A").fetchall()
        symbols = [row[0] for row in rows]
    except Exception as e:
        print("⚠️ Erreur lecture A:", e)
        symbols = []
    finally:
        conn.close()
    return symbols

# --- Préparation de la table ticks ---
def init_ticks_table():
    conn = sqlite3.connect(DB_PATH)
    cur = conn.cursor()
    cur.execute("""
    CREATE TABLE IF NOT EXISTS ticks (
        instId TEXT NOT NULL,
        lastPr REAL,
        bidPr REAL,
        askPr REAL,
        bidSz REAL,
        askSz REAL,
        ts_ms INTEGER NOT NULL,
        PRIMARY KEY(instId, ts_ms)
    );
    """)
    conn.commit()
    conn.close()

# --- Enregistrement d’un tick ---
def save_tick(tick):
    instId = tick.get("instId")
    try:
        ts_ms = int(tick.get("ts"))
        lastPr = float(tick.get("lastPr"))
        bidPr = float(tick.get("bidPr"))
        askPr = float(tick.get("askPr"))
        bidSz = float(tick.get("bidSz"))
        askSz = float(tick.get("askSz"))
    except Exception as e:
        print("⚠️ Erreur parsing tick:", e, tick)
        return

    conn = sqlite3.connect(DB_PATH)
    cur = conn.cursor()
    try:
        cur.execute("""
        INSERT OR IGNORE INTO ticks
        (instId, lastPr, bidPr, askPr, bidSz, askSz, ts_ms)
        VALUES (?, ?, ?, ?, ?, ?, ?)
        """, (instId, lastPr, bidPr, askPr, bidSz, askSz, ts_ms))
        conn.commit()
    except Exception as e:
        print("⚠️ Erreur insertion tick:", e)
    finally:
        conn.close()

# --- WebSocket loop ---
async def run_ws(symbols):
    async with websockets.connect(WS_URL, ping_interval=20, ping_timeout=10) as ws:
        print("✅ WS connecté, abonnement à", len(symbols), "symboles")
        args = [{"instType": "USDT-FUTURES", "channel": "ticker", "instId": sym} for sym in symbols]

                    print(f"💾 Tick : {tick.get('instId')} @ {tick.get('lastPr')} (ts {tick.get('ts')})")
            except Exception as e:
                print("⚠️ WS erreur:", e)
                # attendre un peu avant de relancer la boucle
                await asyncio.sleep(1)
