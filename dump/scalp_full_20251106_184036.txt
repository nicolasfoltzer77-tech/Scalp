#!/usr/bin/env bash
set -euo pipefail

# ---------- CONFIG ----------
ROOT="/opt/scalp"
DUMP_DIR="${ROOT}/dump"
mkdir -p "$DUMP_DIR"
TS="$(date +'%Y%m%d_%H%M%S')"
OUT="${DUMP_DIR}/scalp_full_${TS}.txt"
MAX_FILE_SIZE_KB="${MAX_FILE_SIZE_KB:-512}"

# ---------- GIT ENV ----------
[ -f /etc/scalp.env ] && . /etc/scalp.env
: "${GIT_USERNAME:?}"
: "${GIT_TOKEN:?}"
GIT_BRANCH="${GIT_BRANCH:-main}"
GIT_EMAIL_USE="${GIT_EMAIL:-${GIT_USERNAME}@users.noreply.github.com}"
REMOTE_URL="https://${GIT_HOST:-github.com}/${GIT_OWNER:-$GIT_USERNAME}/${GIT_REPO:-Scalp}.git"

# ---------- HEADER ----------
{
  echo "# ===== SCALP PROJECT DUMP ====="
  echo "# $(date -u +'%Y-%m-%d %H:%M:%S UTC')"
  echo "# Root: $ROOT"
  echo
  echo "========== TREE =========="
} > "$OUT"

find "$ROOT" -type f \
  -not -path '*/.git/*' -not -path '*/venv/*' \
  -not -path '*/dump/*' -not -path '*/logs/*' \
  -size -"${MAX_FILE_SIZE_KB}"k \
  \( -name '*.py' -o -name '*.sh' -o -name '*.conf' -o -name '*.sql' -o -name '*.txt' \) |
  sort | while IFS= read -r f; do
    sz=$(stat -c%s "$f" 2>/dev/null || echo 0)
    mt=$(stat -c%y "$f" 2>/dev/null || echo unknown)
    printf "%-80s %12s %s\n" "$f" "$sz" "$mt" >> "$OUT"
done

{
  echo
  echo "========== FILE CONTENT =========="
} >> "$OUT"

find "$ROOT" -type f \
  -not -path '*/.git/*' -not -path '*/venv/*' \
  -not -path '*/dump/*' -not -path '*/logs/*' \
  -size -"${MAX_FILE_SIZE_KB}"k \
  \( -name '*.py' -o -name '*.sh' -o -name '*.conf' -o -name '*.sql' -o -name '*.txt' \) |
  sort | while IFS= read -r f; do
    echo -e "\n----- FILE: $f -----" >> "$OUT"
    cat "$f" 2>/dev/null >> "$OUT" || true
done

# ---------- DATABASES ----------
{
  echo
  echo "========== DATABASE DUMPS =========="
} >> "$OUT"

for DB_PATH in /opt/scalp/project/data/*.db; do
  [ -s "$DB_PATH" ] || continue
  echo -e "\n----- DATABASE: $DB_PATH -----" >> "$OUT"
  {
    echo "-- SCHEMA + DATA DUMP"
    sqlite3 -readonly "$DB_PATH" "PRAGMA busy_timeout=5000; .dump" 2>/dev/null || \
      echo "-- [WARN] locked or unreadable"
  } >> "$OUT"
done

# ---------- GIT PUSH ----------
cd "$ROOT"
git rev-parse --is-inside-work-tree >/dev/null 2>&1 || git init -q .
git config user.name "$GIT_USERNAME"
git config user.email "$GIT_EMAIL_USE"
git remote get-url origin >/dev/null 2>&1 || git remote add origin "$REMOTE_URL"

git add -f "$OUT"
git commit -m "dump ${TS}" || true
AUTH_URL="https://${GIT_USERNAME}:${GIT_TOKEN}@${REMOTE_URL#https://}"
git push -f "$AUTH_URL" HEAD:"$GIT_BRANCH"

#!/usr/bin/env bash
set -euo pipefail
DB_B=/opt/scalp/project/data/b.db
DB_A=/opt/scalp/project/data/a.db
DB_U=/opt/scalp/project/data/u.db

echo "===== B GATE COUNTER ($(date '+%F %T')) ====="
sqlite3 -readonly "$DB_B" <<SQL
ATTACH DATABASE '$DB_A' AS A;
ATTACH DATABASE '$DB_U' AS U;

.headers on
.mode column

-- Gate#1: candidats (universe ∩ feat_1m frais)
WITH uni AS (
  SELECT REPLACE(instId,'/','') AS instId FROM U.universe
),
feat_latest AS (
  SELECT REPLACE(instId,'/','') AS instId, MAX(ts) AS ts_feat
  FROM feat_1m GROUP BY REPLACE(instId,'/','')
),
g1 AS (
  SELECT u.instId, f.ts_feat
  FROM uni u JOIN feat_latest f ON f.instId=u.instId
  WHERE (strftime('%s','now')*1000 - f.ts_feat) <= 120000
),
-- Gate#2: ctx frais
ctx_latest AS (
  SELECT REPLACE(instId,'/','') AS instId, MAX(ts) AS ts_ctx
  FROM A.ctx_a_latest GROUP BY REPLACE(instId,'/','')
),
g2 AS (
  SELECT g1.instId, g1.ts_feat, c.ts_ctx
  FROM g1 LEFT JOIN ctx_latest c ON c.instId=g1.instId
  WHERE (strftime('%s','now')*1000 - c.ts_ctx) <= 300000
),
-- Gate#3: signal produit <= 600s
sig_latest AS (
  SELECT REPLACE(instId,'/','') AS instId, MAX(ts_signal) AS ts_sig
  FROM signals_B GROUP BY REPLACE(instId,'/','')
)
SELECT
  (SELECT COUNT(*) FROM U.universe)                      AS universe,
  (SELECT COUNT(*) FROM g1)                              AS g1_feat_ok,
  (SELECT COUNT(*) FROM g2)                              AS g2_ctx_ok,
  (SELECT COUNT(*) FROM g2 JOIN sig_latest s USING(instId)
     WHERE (strftime('%s','now')*1000 - s.ts_sig) <= 600000) AS g3_sig_ok;

DETACH DATABASE A;
DETACH DATABASE U;
SQL

#!/usr/bin/env bash
set -euo pipefail
DB_B=/opt/scalp/project/data/b.db
DB_A=/opt/scalp/project/data/a.db
DB_OA=/opt/scalp/project/data/oa.db
DB_U=/opt/scalp/project/data/u.db

echo "===== B SOURCES PROBE ($(date '+%F %T')) ====="

sqlite3 -readonly "$DB_B" <<SQL
ATTACH DATABASE '$DB_A'  AS A;
ATTACH DATABASE '$DB_OA' AS OA;
ATTACH DATABASE '$DB_U'  AS U;

.headers on
.mode column
.width 11 8 8 8 19 10 19 10 19 10

WITH uni AS (
  SELECT REPLACE(instId,'/','') AS instId FROM U.universe
),
feat AS (
  SELECT REPLACE(instId,'/','') AS instId,
         MAX(ts) AS ts_feat
  FROM feat_1m
  GROUP BY REPLACE(instId,'/','')
),
ctx AS (
  SELECT REPLACE(instId,'/','') AS instId,
         MAX(ts) AS ts_ctx
  FROM A.ctx_a_latest
  GROUP BY REPLACE(instId,'/','')
),
sig AS (
  SELECT REPLACE(instId,'/','') AS instId,
         MAX(ts_signal) AS ts_sig
  FROM signals_B
  GROUP BY REPLACE(instId,'/','')
),
oa5 AS (
  SELECT REPLACE(instId,'/','') AS instId,
         MAX(ts) AS ts_oa5
  FROM OA.ohlcv_5m
  GROUP BY REPLACE(instId,'/','')
),
mix AS (
  SELECT u.instId,
         f.ts_feat,
         c.ts_ctx,
         o.ts_oa5,
         s.ts_sig
  FROM uni u
  LEFT JOIN feat f ON f.instId=u.instId
  LEFT JOIN ctx  c ON c.instId=u.instId
  LEFT JOIN oa5  o ON o.instId=u.instId
  LEFT JOIN sig  s ON s.instId=u.instId
)
SELECT
  instId AS COIN,
  printf('%9.1f', (strftime('%s','now')*1000 - ts_feat)/1000.0) AS age_feat_s,
  printf('%9.1f', (strftime('%s','now')*1000 - ts_ctx )/1000.0) AS age_ctx_s,
  printf('%9.1f', (strftime('%s','now')*1000 - ts_oa5)/1000.0) AS age_oa5_s,
  datetime(ts_feat/1000,'unixepoch','localtime') AS last_feat,
  CASE WHEN (strftime('%s','now')*1000 - ts_feat) <= 120000 THEN 'OK' ELSE 'OLD' END AS gate_feat,
  datetime(ts_ctx/1000,'unixepoch','localtime') AS last_ctx,

check_table() {
    local name="$1"
    local db="$2"
    local table="$3"
    local tscol="$4"

    if ! sqlite3 "$db" ".tables" | grep -qw "$table"; then
        printf "%-9s → %s ❌ (missing table: %s)\n" "$name" "$name" "$table"
        return
    fi

    local count last
    count=$(sqlite3 "$db" "SELECT COUNT(*) FROM $table;" 2>/dev/null || echo 0)
    last=$(sqlite3 "$db" "
        SELECT 
            printf('%02dh%02dm%02ds ago',
                (strftime('%s','now')-MAX($tscol/1000))/3600,
                ((strftime('%s','now')-MAX($tscol/1000))%3600)/60,
                ((strftime('%s','now')-MAX($tscol/1000))%60)
            )
        FROM $table;
    " 2>/dev/null || echo "00h00m00s ago")

    if [[ "$count" =~ ^[0-9]+$ && "$count" -gt 0 ]]; then
        printf "%-9s → %s ✅  (%s recs, last=%s)\n" "$name" "$name" "$count" "$last"
    else
        printf "%-9s → %s ✅  (0 recs, last=00h00m00s ago)\n" "$name" "$name"
    fi
}

ts_now="$(date '+%Y-%m-%d %H:%M:%S')"

check_db() {
    local name="$1" file="$2" table="$3" ts_col="$4"
    if [[ ! -f "$file" ]]; then
        printf "%-10s ❌ no DB found\n" "$name" | tee -a "$LOG"
        return
    fi
    local count ts_last age_s h m s age_hms status
    count=$(sqlite3 "$file" "SELECT COUNT(*) FROM $table;")
    ts_last=$(sqlite3 "$file" "SELECT MAX($ts_col)/1000 FROM $table;")
    [[ -z "$ts_last" || "$ts_last" == "0" ]] && ts_last=$now_s
    age_s=$(( now_s - ${ts_last%.*:-0} ))
    h=$((age_s/3600)); m=$(( (age_s%3600)/60 )); s=$((age_s%60))
    age_hms=$(printf "%02dh%02dm%02ds" "$h" "$m" "$s")

    if (( age_s < 10 )); then status="✅"
    elif (( age_s < 60 )); then status="⚠️"

    fi

    printf "%-10s %s  %4s recs | last=%s | age=%s\n" \
        "$name" "$status" "$count" "$(date '+%H:%M:%S' -d @$ts_last 2>/dev/null || date -r "$ts_last" '+%H:%M:%S')" "$age_hms" \
        | tee -a "$LOG"
}

#!/usr/bin/env python3
import sqlite3, math, time

DB_A = "/opt/scalp/project/data/a.db"
DB_U = "/opt/scalp/project/data/u.db"
DB_OB = "/opt/scalp/project/data/ob.db"
TF = "5m"

def softmax(x):
    e = [math.exp(i) for i in x]
    s = sum(e)
    return [v/s for v in e]

def get_universe():
    con = sqlite3.connect(DB_U)
    rows = con.execute("SELECT instId FROM universe ORDER BY instId").fetchall()
    con.close()
    return [r[0] for r in rows]

def load_ob(inst):
    con = sqlite3.connect(DB_OB)
    row = con.execute(f"SELECT rsi FROM ob_{TF} WHERE instId=? ORDER BY ts DESC LIMIT 1", (inst,)).fetchone()
    con.close()
    return row[0] if row else None

def compute_ctx(rsi):
    if rsi < 30: return "bullish"
    if rsi > 70: return "bearish"
    if 45 <= rsi <= 55: return "range"
    return "none"

def compute_score_A(ctx):
    if ctx == "bullish": return 0.8
    if ctx == "bearish": return -0.8
    if ctx == "range": return 0.0
    return 0.0

def run():
    insts = get_universe()
    now = int(time.time()*1000)
    out = []

    for inst in insts:
        rsi = load_ob(inst)
        if rsi is None:
            continue
        ctx = compute_ctx(rsi)
        score_A = compute_score_A(ctx)
        p_buy, p_sell = softmax([score_A, -score_A])
        out.append((inst, now, 0.0, score_A, p_buy, p_sell, ctx))

    con = sqlite3.connect(DB_A)
    con.execute("DELETE FROM ctx_a_latest")
    con.executemany("INSERT INTO ctx_a_latest(instId,ts,score_U,score_A,p_buy,p_sell,ctx) VALUES (?,?,?,?,?,?,?)", out)
    con.commit()
    con.close()

if __name__ == "__main__":
    run()

#!/usr/bin/env python3
import sqlite3, time, numpy as np

DB_A  = "/opt/scalp/project/data/a.db"
DB_OA = "/opt/scalp/project/data/oa.db"
DB_U  = "/opt/scalp/project/data/u.db"

# --- paramètres de pondération ---
W = dict(EMA=0.35, MACD=0.25, RSI=0.20, ADX=0.10, OBV=0.10)
TF_WEIGHTS = {"30m":0.5, "15m":0.3, "5m":0.2}
TAU = 0.35

# --- fonctions utilitaires ---
def ema(arr, n):
    if len(arr) < n: return None
    k = 2/(n+1)
    e = arr[0]
    for x in arr[1:]:
        e = k*x + (1-k)*e
    return e

def atr(high, low, close, period=14):
    tr = [max(h-l, abs(h-cp), abs(l-cp)) for h,l,cp in zip(high[1:],low[1:],close[:-1])]
    return np.mean(tr[-period:]) if tr else 0

def rsi(vals, period=14):
    if len(vals) < period+1: return 50
    diffs = np.diff(vals)
    up = np.mean([d for d in diffs[-period:] if d>0]) or 1e-9
    dn = np.mean([-d for d in diffs[-period:] if d<0]) or 1e-9
    rs = up/dn
    return 100 - 100/(1+rs)

def adx(high, low, close, period=14):
    if len(high) < period+2: return 20
    plus_dm = np.maximum(np.diff(high), 0)
    minus_dm = np.maximum(-np.diff(low), 0)
    tr = np.maximum(high[1:]-low[1:], np.abs(high[1:]-close[:-1]))
    tr = np.maximum(tr, np.abs(low[1:]-close[:-1]))
    tr_sm = np.mean(tr[-period:])
    plus = 100*np.mean(plus_dm[-period:])/tr_sm if tr_sm else 0
    minus = 100*np.mean(minus_dm[-period:])/tr_sm if tr_sm else 0
    dx = abs(plus - minus)/(plus + minus + 1e-9)*100
    return dx

def macd(vals):
    e12, e26 = ema(vals,12), ema(vals,26)
    if e12 is None or e26 is None: return 0
    hist = e12 - e26
    return hist

def softmax3(s, tau=TAU):
    exp_pos = np.exp(s/tau)
    exp_neg = np.exp(-s/tau)
    denom = exp_pos + 1 + exp_neg
    return exp_pos/denom, exp_neg/denom, 1 - (exp_pos+exp_neg)/denom

# --- extraction OHLCV ---
def fetch_tf(con, inst, tf):
    q = f"SELECT ts, open, high, low, close, volume FROM ohlcv_{tf} WHERE instId=? ORDER BY ts DESC LIMIT 150"
    rows = con.execute(q, (inst,)).fetchall()
    if not rows: return None
    arr = np.array(rows[::-1])
    return arr[:,1], arr[:,2], arr[:,3], arr[:,4], arr[:,5]

# --- score par timeframe ---
def score_tf(o,h,l,c,v):
    c = c.astype(float)
    atrv = atr(h,l,c)
    s_ema  = np.tanh((ema(c,12) - ema(c,26)) / (max(atrv,1e-8)))
    s_macd = np.tanh(macd(c) / (np.std(c) + 1e-8))
    s_rsi  = (rsi(c)-50)/50
    s_adx  = np.clip((adx(h,l,c)-20)/20, -1, 1)
    s_obv  = np.tanh(np.diff(v)[-1] / (np.median(np.abs(v[-20:])) + 1e-8))
    return np.average([s_ema,s_macd,s_rsi,s_adx,s_obv], weights=list(W.values()))

# --- score multi-TF ---
def score_multi(tf_scores):
    vals = [tf_scores.get(tf,0)*TF_WEIGHTS[tf] for tf in TF_WEIGHTS]
    return sum(vals)

# Calcule les features 1m depuis les ticks/ohlcv → b.db.feat_1m
# ============================================================
import sqlite3, pandas as pd, numpy as np, ta, time, logging, sys
from pathlib import Path

DB_T = Path("/opt/scalp/project/data/ob.db")
DB_B = Path("/opt/scalp/project/data/b.db")
LOG = "/opt/scalp/project/logs/b_feat_builder.log"

def setup_logging():
    logging.basicConfig(
        level=logging.INFO,
        format="%(asctime)s FEAT %(levelname)s %(message)s",
        handlers=[
            logging.FileHandler(LOG),
            logging.StreamHandler(sys.stdout),
        ],
        force=True,
    )

def connect(path: Path) -> sqlite3.Connection:
    c = sqlite3.connect(path, timeout=30, isolation_level=None)
    c.execute("PRAGMA journal_mode=WAL;")
    c.execute("PRAGMA synchronous=NORMAL;")
    return c

def ensure_schema(conn: sqlite3.Connection):
    conn.executescript("""
    CREATE TABLE IF NOT EXISTS feat_1m (
        instId TEXT,
        ts INTEGER,
        open REAL, high REAL, low REAL, close REAL,
        volume REAL, atr REAL,
        ema20 REAL, ema50 REAL,
        rsi7 REAL,
        macd REAL, macd_signal REAL, macd_hist REAL,
        bb_up REAL, bb_mid REAL, bb_low REAL, bb_width REAL,
        adx REAL, vwap REAL, squeeze INTEGER DEFAULT 0,
        PRIMARY KEY(instId, ts)
    );
    """)
    conn.commit()

def compute_features(df: pd.DataFrame) -> pd.DataFrame:
    if len(df) < 20:
        return pd.DataFrame()
    df["atr"] = ta.volatility.AverageTrueRange(df["high"], df["low"], df["close"], window=14).average_true_range()
    df["ema20"] = ta.trend.EMAIndicator(df["close"], window=20).ema_indicator()
    df["ema50"] = ta.trend.EMAIndicator(df["close"], window=50).ema_indicator()
    df["rsi7"] = ta.momentum.RSIIndicator(df["close"], window=7).rsi()
    macd = ta.trend.MACD(df["close"])
    df["macd"] = macd.macd()
    df["macd_signal"] = macd.macd_signal()
    df["macd_hist"] = macd.macd_diff()
    bb = ta.volatility.BollingerBands(df["close"], window=20, window_dev=2)
    df["bb_up"] = bb.bollinger_hband()
    df["bb_mid"] = bb.bollinger_mavg()
    df["bb_low"] = bb.bollinger_lband()
    df["bb_width"] = df["bb_up"] - df["bb_low"]
    df["adx"] = ta.trend.ADXIndicator(df["high"], df["low"], df["close"], window=14).adx()
    # VWAP approximé
    df["vwap"] = (df["close"] * df["volume"]).rolling(window=20, min_periods=1).sum() / df["volume"].rolling(window=20, min_periods=1).sum()
    # squeeze BB < Keltner
    kelt = ta.volatility.KeltnerChannel(df["high"], df["low"], df["close"], window=20)
    df["squeeze"] = np.where(df["bb_width"] < (kelt.keltner_channel_hband() - kelt.keltner_channel_lband()), 1, 0)
    df = df.dropna().reset_index(drop=True)
    return df

def update_feat_1m():
    setup_logging()
    con_t = connect(DB_T)
    con_b = connect(DB_B)
    ensure_schema(con_b)

    instruments = [r[0] for r in con_t.execute("SELECT DISTINCT instId FROM ohlcv_1m;").fetchall()]
    if not instruments:
        logging.warning("Aucun instrument trouvé dans ohlcv_1m.")
        return

    for inst in instruments:
        try:
            df = pd.read_sql_query(f"""
                SELECT ts,o AS open,h AS high, l AS low, c AS close, v AS volume
                FROM ohlcv_1m
                WHERE instId='{inst}'
                ORDER BY ts DESC
                LIMIT 200;
            """, con_t)
            if df.empty:
                continue
            df = df.sort_values("ts").reset_index(drop=True)
            df_feat = compute_features(df)
            if df_feat.empty:
                continue
            last_row = df_feat.iloc[-1]
            con_b.execute("""
                INSERT OR REPLACE INTO feat_1m (
                    instId, ts, open, high, low, close, volume, atr,
                    ema20, ema50, rsi7, macd, macd_signal, macd_hist,
                    bb_up, bb_mid, bb_low, bb_width, adx, vwap, squeeze
                ) VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)
            """, (
                inst, int(last_row["ts"]),
                last_row["open"], last_row["high"], last_row["low"], last_row["close"], last_row["volume"],
                last_row["atr"], last_row["ema20"], last_row["ema50"], last_row["rsi7"],
                last_row["macd"], last_row["macd_signal"], last_row["macd_hist"],
                last_row["bb_up"], last_row["bb_mid"], last_row["bb_low"], last_row["bb_width"],
                last_row["adx"], last_row["vwap"], int(last_row["squeeze"])
            ))
            con_b.commit()
            logging.info(f"FEAT updated → {inst}")
        except Exception as e:
            logging.error(f"{inst} compute error: {e}")
            time.sleep(1)

    con_t.close()
    con_b.close()
    logging.info("Update complete ✓")

if __name__ == "__main__":
    update_feat_1m()

    );
    """)
    conn.commit()

    cur = conn.cursor()
    cur.execute("SELECT DISTINCT instId FROM feat_1m;")
    instruments = [r[0] for r in cur.fetchall()]
    ts_now = int(time.time() * 1000)

    for inst in instruments:
        cur.execute("""
            SELECT close, ema20, ema50, rsi7, adx, atr
            FROM feat_1m WHERE instId=? ORDER BY ts DESC LIMIT 1;
        """, (inst,))

        if not row:
            continue

        # --- Skip si valeurs manquantes ---

            log.warning(f"{inst}: valeurs incomplètes, skip")
            continue

        div = abs(ema20 - ema50) / ema50 if ema50 != 0 else 0
        signals = []

        score_B = min(100, max(0, (adx - 20) * 2 + abs(rsi7 - 50) * 0.5))
        if ema20 > ema50 and rsi7 > 55:
            signals.append(("buy", "MOMENTUM", score_B))
        elif ema20 < ema50 and rsi7 < 45:
            signals.append(("sell", "MOMENTUM", score_B))

        # === BREAKOUT ===
        score_B = min(100, (adx * 1.5) + (div * 5000))
        if ctx in ("range", "neutral") and adx > 20 and div > 0.005:
            side = "buy" if ema20 > ema50 else "sell"
            signals.append((side, "BREAKOUT", score_B))

        # === REVERSAL ===
        if ctx == "bearish" and rsi7 < 35:
            score_B = min(100, (35 - rsi7) * 3)
            signals.append(("buy", "REVERSAL", score_B))
        elif ctx == "bullish" and rsi7 > 65:
            score_B = min(100, (rsi7 - 65) * 3)
            signals.append(("sell", "REVERSAL", score_B))

        # === MEAN REVERSION ===
        if ctx == "range" and abs(ema20 - ema50) / ema50 < 0.002:
            score_B = max(0, 100 - abs(50 - rsi7) * 2)
            side = "buy" if rsi7 < 50 else "sell"

    con = connect_db(DB_H)
    con.execute("""

            trades INTEGER,

    ensure_closed_schema()
    con_c = connect_db(DB_CLOSED)

    con_c.close()

DBU = "/opt/scalp/project/data/u.db"

INIT = 150
MAX_ROWS = 300

def to_bitget_symbol(inst: str) -> str:
    return inst if ":" in inst else f"{inst}:USDT"

def ensure_schema():
    con = sqlite3.connect(DBO, timeout=5)
    cur = con.cursor()
    for tf in TF_LIST:
        cur.execute(f"DROP TABLE IF EXISTS ohlcv_{tf}")
        cur.execute(f"""
        CREATE TABLE IF NOT EXISTS ohlcv_{tf}(
          instId TEXT,
          ts INTEGER,
          open REAL, high REAL, low REAL, close REAL, volume REAL,
          PRIMARY KEY(instId, ts)
        )
        """)

def load_universe():

    rows = [r[0] for r in con.execute("SELECT instId FROM universe")]
    con.close()

def last_ts(con, tf, inst):
    row = con.execute(f"SELECT MAX(ts) FROM ohlcv_{tf} WHERE instId=?", (inst,)).fetchone()
    return row[0] if row and row[0] is not None else None

def upsert_batch(tf, inst, batch):
    if not batch:
        return
    con = sqlite3.connect(DBO, timeout=5)
    cur = con.cursor()
    cur.executemany(
        f"INSERT OR REPLACE INTO ohlcv_{tf}(instId,ts,open,high,low,close,volume) VALUES (?,?,?,?,?,?,?)",
        [(inst, o[0], o[1], o[2], o[3], o[4], o[5]) for o in batch]
    )
    cur.execute(f"""
        DELETE FROM ohlcv_{tf}
        WHERE instId=?
          AND ts NOT IN (
            SELECT ts FROM ohlcv_{tf}
            WHERE instId=?
            ORDER BY ts DESC
            LIMIT {MAX_ROWS}
          )
    """, (inst, inst))
    con.commit()
    con.close()

def sync_inst_tf(inst, tf):
    inst_ccxt = to_bitget_symbol(inst)
    con = sqlite3.connect(DBO, timeout=5)
    ts_last = last_ts(con, tf, inst)
    con.close()

    if ts_last is None:
        data = ex.fetch_ohlcv(inst_ccxt, tf, limit=INIT)
        upsert_batch(tf, inst, data)
        return

    data = ex.fetch_ohlcv(inst_ccxt, tf, since=ts_last + 1, limit=150)
    upsert_batch(tf, inst, data)

#!/usr/bin/env python3
import sqlite3, time

DB_OB = "/opt/scalp/project/data/ob.db"
DB_B  = "/opt/scalp/project/data/b.db"

def ensure_schema():
    con = sqlite3.connect(DB_B, timeout=30, isolation_level=None)
    con.execute("PRAGMA journal_mode=WAL;")
    con.execute("""CREATE TABLE IF NOT EXISTS feat_1m(
        instId TEXT,
        ts INTEGER,
        atr REAL,
        PRIMARY KEY(instId, ts)
    )""")
    con.execute("""CREATE TABLE IF NOT EXISTS feat_3m(
        instId TEXT,
        ts INTEGER,
        atr REAL,
        PRIMARY KEY(instId, ts)
    )""")
    con.close()

def atr14(rows):
    # rows = [(ts,o,h,l,c,v)] triés ASC
    if len(rows) < 15:
        return None
    trs=[]
    prev_close = rows[0][4]
    for (_,o,h,l,c,v) in rows[1:]:
        trs.append(max(h-l, abs(h-prev_close), abs(l-prev_close)))
        prev_close=c
    return sum(trs[-14:]) / 14.0

def process(tf):
    tbl_k = f"kline_{tf}"
    tbl_f = f"feat_{tf}"
    con_ob = sqlite3.connect(DB_OB)
    con_b  = sqlite3.connect(DB_B)

    insts = [r[0] for r in con_ob.execute(f"SELECT DISTINCT instId FROM {tbl_k}")]
    for inst in insts:
        rows = con_ob.execute(
            f"SELECT ts,o,h,l,c,v FROM {tbl_k} WHERE instId=? ORDER BY ts ASC",
            (inst,)
        ).fetchall()
        a = atr14(rows)
        if a is None:
            continue
        ts = rows[-1][0]
        con_b.execute(
            f"INSERT OR REPLACE INTO {tbl_f}(instId,ts,atr) VALUES(?,?,?)",
            (inst,ts,a)
        )
    con_b.commit()
    con_ob.close()
    con_b.close()

def main():
    ensure_schema()
    while True:
        process("1m")
        process("3m")
        time.sleep(20)  # toutes les 20s

if __name__ == "__main__":
    main()

def fetch_and_store(instId: str, conn: sqlite3.Connection, limit: int = 200):
    ex = ccxt.bitget()
    symbol = normalize_symbol(instId)

    cur = conn.cursor()
    cur.execute("SELECT MAX(ts) FROM ohlcv_1m WHERE instId=?;", (instId,))
    last_ts = cur.fetchone()[0]

    if last_ts is None:
        since = int(time.time() * 1000) - limit * 60 * 1000
        log(f"{instId}: initial sync ({symbol}), fetching last {limit} bars")
    else:
        since = int(last_ts + 60_000)
        log(f"{instId}: fetching new data since {datetime.datetime.fromtimestamp(last_ts/1000)}")

    try:
        candles = ex.fetch_ohlcv(symbol, timeframe="1m", since=since, limit=limit)
        if not candles:
            log(f"{instId}: no new candles")
            return

        for ts, o, h, l, c, v in candles:
            cur.execute(
                "INSERT OR REPLACE INTO ohlcv_1m VALUES (?,?,?,?,?,?,?)",
                (instId, ts, o, h, l, c, v)
            )
        conn.commit()

        last_dt = datetime.datetime.fromtimestamp(candles[-1][0] / 1000)
        log(f"{instId}: added {len(candles)} bars → {last_dt:%Y-%m-%d %H:%M:%S}")

def connect_db(path: str) -> sqlite3.Connection:
    con = sqlite3.connect(path, timeout=30, isolation_level=None)
    con.execute("PRAGMA journal_mode=WAL;")
    con.execute("PRAGMA synchronous=NORMAL;")
    con.execute("PRAGMA busy_timeout=5000;")
    return con

SCHEMA_SQL = """
CREATE TABLE IF NOT EXISTS ticks(
  instId TEXT PRIMARY KEY,
  lastPr REAL,
  ts_ms INTEGER
);
CREATE TABLE IF NOT EXISTS ticks_hist(
  instId TEXT,
  lastPr REAL,
  ts_ms INTEGER,
  PRIMARY KEY(instId, ts_ms)
);
CREATE VIEW IF NOT EXISTS v_ticks_latest AS
SELECT
  t.instId,
  t.lastPr,
  t.ts_ms,
  strftime('%H:%M:%S', t.ts_ms/1000, 'unixepoch','localtime') AS hms,
  printf('%02d:%02d:%02d',
     ((strftime('%s','now')*1000 - t.ts_ms)/1000)/3600,
     (((strftime('%s','now')*1000 - t.ts_ms)/1000)%3600)/60,
     ((strftime('%s','now')*1000 - t.ts_ms)/1000)%60) AS age_hms
FROM ticks t
ORDER BY t.instId;
"""

def migrate(conn: sqlite3.Connection) -> None:
    conn.executescript(SCHEMA_SQL)

def upsert_tick(db_t: str, instId: str, price: float, ts_ms: int) -> None:
    con = connect_db(db_t)
    cur = con.cursor()
    cur.execute(
        "INSERT INTO ticks_hist(instId,lastPr,ts_ms) VALUES(?,?,?) "
        "ON CONFLICT(instId,ts_ms) DO NOTHING;",
        (instId, price, ts_ms),
    )
    cur.execute(
        "INSERT INTO ticks(instId,lastPr,ts_ms) VALUES(?,?,?) "
        "ON CONFLICT(instId) DO UPDATE SET lastPr=excluded.lastPr, ts_ms=excluded.ts_ms;",
        (instId, price, ts_ms),
    )
    con.commit()
    con.close()

async def run_ws(db_t: str, db_u: str) -> None:
    import websockets

    while True:
        insts = load_universe(db_u)
        if not insts:
            logging.warning("Aucun instrument actif → attente 5s")
            await asyncio.sleep(5)
            continue

        try:
            async with websockets.connect(WS_URL, ping_interval=None) as ws:
                subs = [
                    {"instType":"USDT-FUTURES","channel":"ticker","instId":inst}
                    for inst in insts
                ]
                msg_sub = {"op":"subscribe","args":subs}
                await ws.send(json.dumps(msg_sub))
                logging.info(f"Abonnement envoyé ({len(subs)} paires)")

                last_ping = time.time()

                while True:
                    try:
                        raw = await asyncio.wait_for(ws.recv(), timeout=25)
                    except asyncio.TimeoutError:
                        # heartbeat toutes 20s
                        await ws.send(json.dumps({"op":"ping"}))
                        last_ping = time.time()
                        continue

                    msg = json.loads(raw)
                    if "action" not in msg:
                        continue
                    if msg["action"] not in ("snapshot","update"):
                        continue
                    data = msg.get("data", [])
                    for d in data:
                        inst = d.get("instId")
                        pr = d.get("lastPr") or d.get("last") or d.get("markPrice")
                        ts_ms = int(d.get("ts") or int(time.time()*1000))
                        if not inst or not pr:
                            continue
                        upsert_tick(db_t, inst, float(pr), ts_ms)
                    await asyncio.sleep(0)  # yield event loop

        except Exception as e:
            logging.error(f"WS error: {e} → reconnect dans 3s")
            await asyncio.sleep(3)

def get_last_tick(inst):
    with sqlite3.connect(DB_T) as con:
        cur = con.cursor()
        cur.execute("SELECT lastPr, ts_ms FROM v_ticks_latest WHERE instId=? LIMIT 1;", (inst,))
        r = cur.fetchone()
    if not r:
        return None, None, None, None
    pr, ts_ms = r
    ts_s = ts_ms / 1000
    hms = datetime.fromtimestamp(ts_s).strftime("%H:%M:%S")
    delta = round(time.time() - ts_s, 1)
    return pr, ts_s, hms, delta

def get_last_signal(inst):
    with sqlite3.connect(DB_B) as con:
        cur = con.cursor()
        cur.execute("SELECT ts FROM signals WHERE instId=? ORDER BY ts DESC LIMIT 1;", (inst,))
        r = cur.fetchone()
    if not r:
        return None, None, None
    ts = float(r[0])
    hms = datetime.fromtimestamp(ts).strftime("%H:%M:%S")
    delta = round(time.time() - ts, 1)
    return ts, hms, delta

    print(f"===== DASH {datetime.now().strftime('%H:%M:%S')} =====")
    for inst, ctx in get_ctx():

        price = ticks[inst]

        if side == "buy" and price <= pos["sl"]:
            reason = "SL_HIT"
        elif side == "sell" and price >= pos["sl"]:
            reason = "SL_HIT"
        elif side == "buy" and price >= pos["tp"]:
            reason = "TP_HIT"
        elif side == "sell" and price <= pos["tp"]:
            reason = "TP_HIT"

        if not reason:
            continue

        pnl = (price - entry) / entry if side == "buy" else (entry - price) / entry
        ts_now = int(time.time() * 1000)

        log.info(f"CLOSE {inst} {side} reason={reason} pnl={pnl:+.4f}")

        cc.execute("""
            INSERT INTO positions_closed(instId, side, entry, exit, pnl, reason_exit, ts_open, ts_close)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        """, (inst, side, entry, price, pnl, reason, pos["ts_follow"], ts_now))
        cc.commit()

        # --- appel du recorder ---
        record_trade(inst, side, entry, price, pnl, reason, pos["ts_follow"], ts_now)

    cf.close(); cc.close(); ct.close()

if __name__ == "__main__":
    try:
        main()
    except Exception as e:
        log.error(f"CLOSER FATAL: {e}")
        raise

import sqlite3, time, logging, sys

def setup_logging():

        level=logging.INFO,
        format="%(asctime)s OPENER %(levelname)s %(message)s",
        handlers=[logging.FileHandler(LOG), logging.StreamHandler(sys.stdout)],
        force=True,
    )

# -----------------------------------------------------
def connect(path: Path) -> sqlite3.Connection:
    c = sqlite3.connect(path, timeout=30, isolation_level=None)
    c.execute("PRAGMA journal_mode=WAL;")
    c.execute("PRAGMA synchronous=NORMAL;")
    c.execute("PRAGMA busy_timeout=5000;")
    return c

# -----------------------------------------------------
def open_from_signals():
    con_b = connect(DB_B)
    con_o = connect(DB_O)

    rows = con_b.execute("""
        SELECT instId, side, reason, price, atr, score_B, ts
        FROM v_signals_for_open
        ORDER BY ts DESC
        LIMIT 20
    """).fetchall()

    if not rows:
        logging.info("No pending signals found in b.db")
        con_b.close(); con_o.close(); return

    inserted = 0
    for instId, side, reason, entry, atr, score_B, ts in rows:
        # skip if already an active or recent trade on same instId
        exists = con_o.execute("""
            SELECT 1 FROM trades_open_init
            WHERE instId=? AND status IN ('pending','open')
              AND ts_open > (strftime('%s','now') - 3600)*1000
        """, (instId,)).fetchone()
        if exists:
            continue

        ts_now = int(time.time() * 1000)
        sl = entry - atr * 2 if side == "buy" else entry + atr * 2
        tp = entry + atr * 4 if side == "buy" else entry - atr * 4

        con_o.execute("""
            INSERT INTO trades_open_init(
                instId, side, entry, qty, sl, tp, atr,
                entry_reason, ctx, score_B, ts_create, ts_open, status
            ) VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?)
        """, (instId, side, entry, 0.0, sl, tp, atr,
              reason, '', score_B, ts_now, ts_now, 'pending'))

        logging.info(f"OPENED {instId} {side} @ {entry:.4f} ATR={atr:.4f} reason={reason}")
        inserted += 1

    con_o.commit(); con_b.commit()
    con_b.close(); con_o.close()
    logging.info(f"✓ opener cycle complete. {inserted} trades opened")

# -----------------------------------------------------
def main(_):
    setup_logging()
    try:
        open_from_signals()
    except Exception as e:
        logging.exception(f"Fatal opener error: {e}")
        raise SystemExit(1)
    return 0

# -----------------------------------------------------
if __name__ == "__main__":
    raise SystemExit(main(sys.argv[1:]))

#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import asyncio
import json
import sqlite3
import threading
import time
from queue import Queue
import websockets

DB_PATH = "/opt/scalp/project/data/t.db"
DB_A_PATH = "/opt/scalp/project/data/a.db"
WS_URL = "wss://ws.bitget.com/mix/v1/stream"
FLUSH_INTERVAL = 0.5  # secondes
QUEUE_MAX = 5000

q = Queue(maxsize=QUEUE_MAX)
lock = threading.Lock()
stop_event = threading.Event()

# --- Connexion SQLite optimisée ---
def init_db():
    conn = sqlite3.connect(DB_PATH, check_same_thread=False, timeout=10)
    cur = conn.cursor()
    cur.executescript("""
        PRAGMA journal_mode=WAL;
        PRAGMA synchronous=OFF;
        CREATE TABLE IF NOT EXISTS ticks(
            instId TEXT PRIMARY KEY,
            price REAL,
            ts_ms INTEGER
        );
        CREATE INDEX IF NOT EXISTS idx_ticks_ts ON ticks(ts_ms);
    """)
    conn.commit()
    return conn

# --- Récupération des symboles depuis le contexte A ---
def get_symbols():
    conn = sqlite3.connect(DB_A_PATH)
    cur = conn.cursor()
    cur.execute("SELECT instId FROM v_ctx_latest;")
    symbols = [r[0].replace("/", "") for r in cur.fetchall()]
    conn.close()
    return symbols

# --- Thread d'écriture SQLite ---
def writer_thread():
    conn = init_db()
    cur = conn.cursor()
    last_flush = 0
    buffer = []
    while not stop_event.is_set():
        try:
            tick = q.get(timeout=FLUSH_INTERVAL)
            buffer.append(tick)
        except:
            pass

        now = time.time()
        if now - last_flush >= FLUSH_INTERVAL and buffer:
            with lock:
                cur.executemany(
                    "INSERT INTO ticks (instId, price, ts_ms) VALUES (?, ?, ?) "
                    "ON CONFLICT(instId) DO UPDATE SET price=excluded.price, ts_ms=excluded.ts_ms;",
                    buffer,
                )
                conn.commit()
                buffer.clear()
                last_flush = now

    conn.commit()
    conn.close()

# --- Fonction WS principale ---
async def ws_consumer():
    symbols = get_symbols()
    args = [
        {"instType": "USDT-FUTURES", "channel": "ticker", "instId": s}
        for s in symbols
    ]
    sub_msg = json.dumps({"op": "subscribe", "args": args})
    print(f"[ticks] Subscribing to {len(symbols)} symbols...")

    async with websockets.connect(WS_URL, ping_interval=15, ping_timeout=10) as ws:
        await ws.send(sub_msg)
        print("[ticks] Connected & subscribed.")
        async for msg in ws:
            data = json.loads(msg)
            if "data" not in data:
                continue
            for d in data["data"]:
                try:
                    instId = d["instId"]
                    price = float(d["lastPr"])
                    ts_ms = int(d["ts"])
                    if not q.full():
                        q.put((instId, price, ts_ms))
                except Exception as e:
                    pass

# --- Main ---
def main():
    wt = threading.Thread(target=writer_thread, daemon=True)
    wt.start()

    try:
        asyncio.run(ws_consumer())
    except KeyboardInterrupt:
        print("Stopping...")
    finally:
        stop_event.set()
        wt.join()

if __name__ == "__main__":
    main()

    SELECT 
        (SELECT COUNT(*) FROM last20 WHERE pnl > 0) * 1.0 / (SELECT COUNT(*) FROM last20) AS winrate,
        (SELECT SUM(pnl) FROM last20 WHERE pnl > 0) AS sum_win,
        (SELECT SUM(ABS(pnl)) FROM last20 WHERE pnl < 0) AS sum_loss
)
SELECT 
    winrate,
    CASE WHEN sum_loss IS NULL OR sum_loss=0 THEN 1.0 ELSE MIN(1.0, (sum_win / sum_loss) / 2.0) END AS pf_norm,
    (0.50*winrate + 0.50*CASE WHEN sum_loss IS NULL OR sum_loss=0 THEN 1.0 ELSE MIN(1.0, (sum_win / sum_loss) / 2.0) END) AS score_H
FROM stats;

-- ======================================================
-- RECORDER H PERFORMANCE ENHANCEMENT
-- ======================================================

-- Nettoyage préalable
DROP VIEW IF EXISTS v_H_coin;
DROP VIEW IF EXISTS v_H_setup;
DROP VIEW IF EXISTS v_H_coin_setup;

-- === SCORE H COIN (EMA par actif) ===
CREATE VIEW v_H_coin AS
WITH ranked AS (
    SELECT
        instId,
        score_H,
        ROW_NUMBER() OVER (PARTITION BY instId ORDER BY ts_record DESC) AS n
    FROM trades_record
)
SELECT
    instId,
    ROUND(SUM(score_H * POWER(0.9, n-1)) / SUM(POWER(0.9, n-1)), 4) AS score_H_coin,
    COUNT(*) AS trades_count,
    ROUND(SUM(CASE WHEN score_H > 0 THEN 1 ELSE 0 END)*100.0/COUNT(*),1) AS winrate
FROM ranked
GROUP BY instId
ORDER BY score_H_coin DESC;

-- === SCORE H SETUP (EMA par setup/entry_reason) ===
CREATE VIEW v_H_setup AS
WITH ranked AS (
    SELECT
        entry_reason,
        score_H,
        ROW_NUMBER() OVER (PARTITION BY entry_reason ORDER BY ts_record DESC) AS n
    FROM trades_record
)
SELECT
    entry_reason,
    ROUND(SUM(score_H * POWER(0.9, n-1)) / SUM(POWER(0.9, n-1)), 4) AS score_H_setup,
    COUNT(*) AS trades_count,
    ROUND(SUM(CASE WHEN score_H > 0 THEN 1 ELSE 0 END)*100.0/COUNT(*),1) AS winrate
FROM ranked
GROUP BY entry_reason
ORDER BY score_H_setup DESC;

-- === SCORE H COIN x SETUP (EMA croisé) ===
CREATE VIEW v_H_coin_setup AS
WITH ranked AS (
    SELECT
        instId,
        entry_reason,
        score_H,
        ROW_NUMBER() OVER (PARTITION BY instId, entry_reason ORDER BY ts_record DESC) AS n
    FROM trades_record
)
SELECT
    instId,
    entry_reason,
    ROUND(SUM(score_H * POWER(0.9, n-1)) / SUM(POWER(0.9, n-1)), 4) AS score_H_pair,
    COUNT(*) AS trades_count,
    ROUND(SUM(CASE WHEN score_H > 0 THEN 1 ELSE 0 END)*100.0/COUNT(*),1) AS winrate
FROM ranked
GROUP BY instId, entry_reason
ORDER BY score_H_pair DESC;
