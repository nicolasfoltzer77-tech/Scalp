-- ===== Univers TOP30 + 5 fixes =====
DROP VIEW IF EXISTS v_universe_35;
CREATE VIEW v_universe_35 AS
WITH latest AS (SELECT MAX(ts) AS ts FROM top_scores WHERE name='TOP30'),
base AS (
  SELECT symbol FROM top_scores
  WHERE name='TOP30' AND ts=(SELECT ts FROM latest)
  ORDER BY score DESC LIMIT 30
),
fix AS (
  SELECT 'BTCUSDT' AS symbol UNION ALL
  SELECT 'ETHUSDT' UNION ALL
  SELECT 'BNBUSDT' UNION ALL
  SELECT 'SOLUSDT' UNION ALL
  SELECT 'XRPUSDT'
)
SELECT DISTINCT symbol FROM (SELECT symbol FROM base UNION ALL SELECT symbol FROM fix)
LIMIT 35;

-- ===== Helpers communs =====
DROP VIEW IF EXISTS v_now_unix;
CREATE VIEW v_now_unix AS SELECT CAST(strftime('%s','now') AS INTEGER) AS now_ts;

-- 5m
DROP VIEW IF EXISTS v_feat_5m;
CREATE VIEW v_feat_5m AS
WITH b AS (
  SELECT o.symbol,o.ts,o.open,o.high,o.low,o.close,o.volume,
         LAG(o.close) OVER (PARTITION BY o.symbol ORDER BY o.ts) AS pc,
         o.close - LAG(o.close) OVER (PARTITION BY o.symbol ORDER BY o.ts) AS dclose
  FROM ohlcv_5m o
  WHERE o.symbol IN (SELECT symbol FROM v_universe_35)
),
c AS (
  SELECT *,
         CASE WHEN dclose>0 THEN dclose ELSE 0 END AS gain,
         CASE WHEN dclose<0 THEN -dclose ELSE 0 END AS loss,
         MAX(high-low, MAX(ABS(high-COALESCE(pc,close)), ABS(low-COALESCE(pc,close)))) AS tr,
         AVG(close) OVER (PARTITION BY symbol ORDER BY ts ROWS BETWEEN 8  PRECEDING AND CURRENT ROW) AS sma9,
         AVG(close) OVER (PARTITION BY symbol ORDER BY ts ROWS BETWEEN 20 PRECEDING AND CURRENT ROW) AS sma21,
         AVG(close) OVER (PARTITION BY symbol ORDER BY ts ROWS BETWEEN 49 PRECEDING AND CURRENT ROW) AS sma50,
         AVG(gain)  OVER (PARTITION BY symbol ORDER BY ts ROWS BETWEEN 6  PRECEDING AND CURRENT ROW) AS avg_gain7,
         AVG(loss)  OVER (PARTITION BY symbol ORDER BY ts ROWS BETWEEN 6  PRECEDING AND CURRENT ROW) AS avg_loss7,
         AVG(tr)    OVER (PARTITION BY symbol ORDER BY ts ROWS BETWEEN 13 PRECEDING AND CURRENT ROW) AS atr14
  FROM b
)
SELECT *,
       CASE WHEN avg_loss7>0 THEN 100 - 100.0/(1.0 + (avg_gain7/avg_loss7)) ELSE 100 END AS rsi7,
       CASE WHEN COALESCE(sma50,0)!=0 THEN (sma9 - sma50)/sma50 ELSE 0 END AS d_9_50,
       CASE WHEN close!=0 THEN ABS(COALESCE(sma21,close)-COALESCE(sma50,close))/close*10000 ELSE 0 END AS adx_proxy
FROM c;

DROP VIEW IF EXISTS v_ctx_tf_5m;
CREATE VIEW v_ctx_tf_5m AS
WITH r AS (
  SELECT f.*,
         ROW_NUMBER() OVER (PARTITION BY f.symbol ORDER BY f.ts DESC) AS rn,
         -- D clampé ~[-1,1]
         ( (CASE WHEN (f.d_9_50/0.0015)>1 THEN 1 WHEN (f.d_9_50/0.0015)<-1 THEN -1 ELSE (f.d_9_50/0.0015) END)
         + (CASE WHEN ((f.rsi7-50)/12)>1 THEN 1 WHEN ((f.rsi7-50)/12)<-1 THEN -1 ELSE ((f.rsi7-50)/12) END)
         )*0.5 AS D
  FROM v_feat_5m f
)
SELECT symbol, ts,
       CASE
         WHEN adx_proxy>=22 AND sma9>sma21 AND sma21>sma50 THEN 'trend_up'
         WHEN adx_proxy>=22 AND sma9<sma21 AND sma21<sma50 THEN 'trend_down'
         WHEN adx_proxy<=18 AND ABS(d_9_50)<=0.0015         THEN 'range'
         ELSE 'squeeze'
       END AS regime,
       D,
       CASE WHEN (CASE
         WHEN adx_proxy>=22 AND sma9>sma21 AND sma21>sma50 THEN 'trend_up'
         WHEN adx_proxy>=22 AND sma9<sma21 AND sma21<sma50 THEN 'trend_down'
         WHEN adx_proxy<=18 AND ABS(d_9_50)<=0.0015         THEN 'range'
         ELSE 'squeeze' END) IN ('trend_up','range')
         THEN MIN(1.0, MAX(0.0, 0.55 + 0.35*D))
         ELSE MAX(0.0, 0.5*D + 0.3) END AS pB5,
       CASE WHEN (CASE
         WHEN adx_proxy>=22 AND sma9>sma21 AND sma21>sma50 THEN 'trend_up'
         WHEN adx_proxy>=22 AND sma9<sma21 AND sma21<sma50 THEN 'trend_down'
         WHEN adx_proxy<=18 AND ABS(d_9_50)<=0.0015         THEN 'range'
         ELSE 'squeeze' END) IN ('trend_down','range')
         THEN MIN(1.0, MAX(0.0, 0.55 - 0.35*D))
         ELSE MAX(0.0, -0.5*D + 0.3) END AS pS5,
       CASE WHEN (CASE
         WHEN adx_proxy>=22 AND sma9>sma21 AND sma21>sma50 THEN 'trend_up'

  pB REAL NOT NULL, pS REAL NOT NULL, pH REAL NOT NULL, pN REAL NOT NULL,
  pB5 REAL, pS5 REAL, pH5 REAL,
  pB15 REAL, pS15 REAL, pH15 REAL,
  pB30 REAL, pS30 REAL, pH30 REAL
);

WITH u AS (SELECT symbol FROM v_universe_35),
a AS (SELECT * FROM v_ctx_tf_5m),
b AS (SELECT * FROM v_ctx_tf_15m),
c AS (SELECT * FROM v_ctx_tf_30m),
m AS (
  SELECT u.symbol,
         COALESCE(a.ts,b.ts,c.ts) AS ts,
         -- régime priorité trend_up/down puis range sinon squeeze/none
         CASE
           WHEN (a.regime='trend_up' OR b.regime='trend_up' OR c.regime='trend_up') THEN 'trend_up'
           WHEN (a.regime='trend_down' OR b.regime='trend_down' OR c.regime='trend_down') THEN 'trend_down'
           WHEN (a.regime='range'     OR b.regime='range'     OR c.regime='range')     THEN 'range'
           ELSE COALESCE(a.regime,b.regime,c.regime,'none')
         END AS regime,
         -- D global simple
         COALESCE(a.D,0)*0.40 + COALESCE(b.D,0)*0.35 + COALESCE(c.D,0)*0.25 AS D_raw,
         -- moyennes pondérées des probas (valeur neutre 0.25 si manquante)
         (0.40*COALESCE(a.pB5,0.25)+0.35*COALESCE(b.pB15,0.25)+0.25*COALESCE(c.pB30,0.25)) AS Bw,
         (0.40*COALESCE(a.pS5,0.25)+0.35*COALESCE(b.pS15,0.25)+0.25*COALESCE(c.pS30,0.25)) AS Sw,
         (0.40*COALESCE(a.pH5,0.25)+0.35*COALESCE(b.pH15,0.25)+0.25*COALESCE(c.pH30,0.25)) AS Hw,
         a.pB5,a.pS5,a.pH5, b.pB15,b.pS15,b.pH15, c.pB30,c.pS30,c.pH30
  FROM u
  LEFT JOIN a ON a.symbol=u.symbol
  LEFT JOIN b ON b.symbol=u.symbol
  LEFT JOIN c ON c.symbol=u.symbol
),
n AS (
  SELECT symbol, ts, regime,
         D_raw AS D,
         -- normalisation
         (Bw/(Bw+Sw+Hw)) AS pB,
         (Sw/(Bw+Sw+Hw)) AS pS,
         (Hw/(Bw+Sw+Hw)) AS pH,
         MAX(0.0, 1.0 - ( (Bw/(Bw+Sw+Hw)) + (Sw/(Bw+Sw+Hw)) + (Hw/(Bw+Sw+Hw)) )) AS pN,
         a_pB5, a_pS5, a_pH5, b_pB15, b_pS15, b_pH15, c_pB30, c_pS30, c_pH30
  FROM (
    SELECT symbol, ts, regime, D_raw, Bw, Sw, Hw,
           a_pB5, a_pS5, a_pH5, b_pB15, b_pS15, b_pH15, c_pB30, c_pS30, c_pH30
    FROM m
  )
)
INSERT INTO ctx_A(symbol,ts,regime,D,pB,pS,pH,pN,
                  pB5,pS5,pH5,pB15,pS15,pH15,pB30,pS30,pH30)
SELECT symbol, (SELECT now_ts FROM v_now_unix), regime, D, pB, pS, pH, pN,
       a_pB5, a_pS5, a_pH5, b_pB15, b_pS15, b_pH15, c_pB30, c_pS30, c_pH30
FROM n;

CREATE VIEW v_ctx_A AS
SELECT symbol, ts, regime, D, pB, pS, pH, pN,
       CASE
         WHEN pB>=0.55 AND pS<=0.35 THEN 'BUY'
         WHEN pS>=0.55 AND pB<=0.35 THEN 'SELL'
         WHEN pH>=pB AND pH>=pS AND pH>=pN THEN 'HOLD'
         ELSE 'NONE'
       END AS decision
FROM ctx_A
ORDER BY symbol;

DROP VIEW IF EXISTS v_ctx_tradeable;
CREATE VIEW v_ctx_tradeable AS
SELECT * FROM v_ctx_A WHERE decision!='NONE';

#!/usr/bin/env python3
import os, time, sqlite3, math
DB="data/scalp.db"
TP_BPS     = float(os.getenv("TP_BPS","15"))       # take profit en bps
SL_BPS     = float(os.getenv("SL_BPS","20"))       # stop loss   en bps
MAX_HOLD   = int(os.getenv("MAX_HOLD_SEC","900"))  # 15 min
FEES_BPS   = float(os.getenv("FEES_BPS","1"))      # aller+retour
POLL       = 0.5

def connect():
    con=sqlite3.connect(DB, timeout=20, isolation_level=None)
    con.execute("PRAGMA journal_mode=WAL")
    con.execute("PRAGMA busy_timeout=20000")
    con.execute("PRAGMA synchronous=NORMAL")
    return con

def last_price(con, symbol, default):
    row=con.execute("SELECT last FROM last_ticks WHERE symbol=?", (symbol,)).fetchone()
    return (row[0] if row else default) or default

def ensure_qty(q): return 1.0 if (q is None or q==0.0) else float(q)

def open_new_positions(con, now):
    rows=con.execute("""
      SELECT id,symbol,side,COALESCE(qty,0),COALESCE(price,0),created_ts
      FROM orders_sim
      WHERE status='SIMULATED' AND id NOT IN (SELECT order_id FROM positions_sim)
      ORDER BY id ASC LIMIT 500
    """).fetchall()
    if not rows: return 0
    con.execute("BEGIN IMMEDIATE")
    for oid,sym,side,qty,px,ts in rows:
        px = px or last_price(con, sym, None) or 0.0
        q  = ensure_qty(qty)
        con.execute("""INSERT OR IGNORE INTO positions_sim
            (order_id,symbol,side,qty,entry_price,entry_ts,status)
            VALUES(?,?,?,?,?,?, 'OPEN')""", (oid,sym,side,q,px,ts or now))
    con.execute("COMMIT")
    return len(rows)

def close_if_needed(con, now):
    rows=con.execute("""
      SELECT id, order_id, symbol, side, qty, entry_price, entry_ts
      FROM positions_sim WHERE status='OPEN'
      ORDER BY id ASC LIMIT 1000
    """).fetchall()
    if not rows: return 0
    n_closed=0
    con.execute("BEGIN IMMEDIATE")
    for pid,oid,sym,side,qty,entry,ets in rows:
        last=last_price(con, sym, entry)
        hold= now - (ets or now)
        # Targets
        tp = entry*(1 + (TP_BPS/10000.0)) if side.upper()=='BUY'  else entry*(1 - (TP_BPS/10000.0))
        sl = entry*(1 - (SL_BPS/10000.0)) if side.upper()=='BUY'  else entry*(1 + (SL_BPS/10000.0))
        reason=None; exit_px=None
        if side.upper()=='BUY':
            if last>=tp: exit_px=last; reason='TP'
            elif last<=sl: exit_px=last; reason='SL'
        else: # SELL
            if last<=tp: exit_px=last; reason='TP'
            elif last>=sl: exit_px=last; reason='SL'
        if reason is None and hold>=MAX_HOLD:
            exit_px=last; reason='TIME'
        if reason is None:
            # Reverse signal ferme la position si ordre plus récent sens opposé
            rev = con.execute("""
              SELECT 1 FROM orders_sim
              WHERE symbol=? AND created_ts>? AND
                    ((side='BUY'  AND ?='SELL') OR (side='SELL' AND ?='BUY'))
              LIMIT 1
            """,(sym, ets or 0, side.upper(), side.upper())).fetchone()
            if rev: exit_px=last; reason='REV'
        if exit_px is not None:
            # PnL: BUY => (exit-entry)*qty ; SELL => (entry-exit)*qty ; moins fees
            gross = (exit_px-entry)*qty if side.upper()=='BUY' else (entry-exit_px)*qty
            fees  = (entry+exit_px)*qty*(FEES_BPS/10000.0)
            pnl   = gross - fees
            pnl_bps = (pnl/ (entry*qty)) * 10000.0 if entry>0 and qty>0 else 0.0
            con.execute("""UPDATE positions_sim
                           SET exit_price=?, exit_ts=?, pnl_abs=?, pnl_bps=?,
                               fees_abs=?, status='CLOSED', close_reason=?
                           WHERE id=?""",
                           (exit_px, now, pnl, pnl_bps, fees, reason, pid))
            n_closed+=1
    con.execute("COMMIT")
    return n_closed

def main():
    con=connect()
    while True:
        now=int(time.time())
        opened=open_new_positions(con, now)
        closed=close_if_needed(con, now)
        if opened==0 and closed==0:
            time.sleep(POLL)

FIX = ["BTCUSDT","ETHUSDT","BNBUSDT","SOLUSDT","XRPUSDT"]

def instid(s):  # "BTC/USDT:USDT" -> "BTCUSDT"
    return s.replace(":USDT","").replace("/","").upper()

def has_90d_history(ex, sym):
    """
    Test léger: demande des bougies 1D autour de J-90.
    Critère: on a au moins une bougie ≤ (now-88j).
    """
    now_ms = int(time.time()*1000)
    since = now_ms - 95*24*3600*1000
    try:
        rows = ex.fetch_ohlcv(sym, timeframe='1d', since=since, limit=10)
    except Exception:
        return False
    if not rows:
        return False
    # une bougie assez ancienne?
    threshold = now_ms - 88*24*3600*1000
    return any(r[0] <= threshold for r in rows)

def quote_volume(ex, sym):
    """
    Récupère le volume quote du ticker si dispo.
    Fallback sur baseVolume sinon.
    """
    try:
        t = ex.fetch_ticker(sym)
    except Exception:
        return 0.0
    for k in ("quoteVolume","quoteVolume24h","info"):
        if k in t and t[k]:
            if k=="info":
                # Bitget renvoie souvent 'baseVolume' ou 'quoteVolume' dans info
                v = t["info"].get("quoteVolume") or t["info"].get("baseVolume")
            else:
                v = t[k]
            try:
                return float(v)
            except Exception:
                pass
    # fallback très faible
    try:
        return float(t.get("baseVolume") or 0.0)
    except Exception:
        return 0.0

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
Paper trader minimal :
- consomme /opt/scalp/project/data/scalp.db
- lit orders_sim (id, signal_id, symbol, side, qty, price, ts, created_ts)
- tient positions_open (netting par symbole) et trades_closed
- met à jour account_sim (balance, equity, used_margin, updated_ts)

import sqlite3, time, math, pathlib, os

DB = "/opt/scalp/project/data/scalp.db"
LOOP_SEC = 2

SCHEMA = """
PRAGMA journal_mode=WAL;
PRAGMA synchronous=NORMAL;

CREATE TABLE IF NOT EXISTS account_sim(
  id INTEGER PRIMARY KEY CHECK(id=1),
  balance REAL NOT NULL,
  equity  REAL NOT NULL,
  used_margin REAL NOT NULL DEFAULT 0,
  updated_ts INTEGER NOT NULL
);

INSERT OR IGNORE INTO account_sim(id,balance,equity,used_margin,updated_ts)
VALUES(1,100.0,100.0,0,CAST(strftime('%s','now') AS INTEGER));

CREATE TABLE IF NOT EXISTS positions_open(
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  symbol TEXT NOT NULL,
  qty    REAL NOT NULL,             -- qty signée (>0 long, <0 short)
  entry_price REAL NOT NULL,
  last   REAL NOT NULL,
  u_pnl  REAL NOT NULL,
  opened_ts INTEGER NOT NULL
);

CREATE INDEX IF NOT EXISTS idx_pos_symbol ON positions_open(symbol);

CREATE TABLE IF NOT EXISTS trades_closed(
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  symbol TEXT NOT NULL,
  qty    REAL NOT NULL,
  entry_price REAL NOT NULL,
  exit_price  REAL NOT NULL,
  pnl   REAL NOT NULL,
  opened_ts INTEGER NOT NULL,
  closed_ts INTEGER NOT NULL
);

CREATE TABLE IF NOT EXISTS paper_state(
  id INTEGER PRIMARY KEY CHECK(id=1),
  last_ord_id INTEGER NOT NULL
);
INSERT OR IGNORE INTO paper_state(id,last_ord_id) VALUES(1,0);
"""

def qrow(cur, q, args=()):
    cur.execute(q, args); r=cur.fetchone(); return r

def qval(cur, q, args=(), default=None):
    r=qrow(cur,q,args); return r[0] if r and r[0] is not None else default

def mark_last_id(cur, last_id):
    cur.execute("UPDATE paper_state SET last_ord_id=? WHERE id=1", (last_id,))

def load_last_id(cur):
    return qval(cur,"SELECT last_ord_id FROM paper_state WHERE id=1",(),0)

def up_equity(cur):
    bal = qval(cur,"SELECT balance FROM account_sim WHERE id=1",(),0.0)
    upnl = qval(cur,"SELECT COALESCE(SUM(u_pnl),0) FROM positions_open",(),0.0)
    now = int(time.time())
    cur.execute("UPDATE account_sim SET equity=?, updated_ts=? WHERE id=1",
                (bal+upnl, now))

def realize(cur, sym, qty_to_close, price, now):
    """Ferme contre la position existante (FIFO simple)."""
    # Une seule ligne par symbole (netting), donc qty_signée suffit
    row = qrow(cur, "SELECT id, qty, entry_price, opened_ts FROM positions_open WHERE symbol=?", (sym,))
    if not row: return 0.0
    pid, pos_qty, entry, opened_ts = row
    # sens inverse pour fermer
    remain = pos_qty + qty_to_close  # qty_to_close est signée (opposée)
    # PnL réalisé sur la portion fermée = (prix_sortie - prix_entree) * qty_signée_initiale_fermée
    closed_qty = -qty_to_close if pos_qty*qty_to_close < 0 else 0.0
    if closed_qty == 0.0:  # rien à fermer
        return 0.0
    pnl = (price - entry) * (closed_qty if pos_qty>0 else -closed_qty)
    # balance++
    bal = qval(cur,"SELECT balance FROM account_sim WHERE id=1",(),0.0)
    cur.execute("UPDATE account_sim SET balance=? WHERE id=1", (bal+pnl,))
    # trade
    cur.execute("""INSERT INTO trades_closed(symbol,qty,entry_price,exit_price,pnl,opened_ts,closed_ts)
                   VALUES(?,?,?,?,?,?,?)""",
                (sym, closed_qty if pos_qty>0 else -closed_qty, entry, price, pnl, opened_ts, now))
    # position restante (si 0 => on efface)
    if abs(remain) < 1e-12:
        cur.execute("DELETE FROM positions_open WHERE id=?", (pid,))
    else:
        # Si on a inversé, on ouvre à ce nouveau prix
        if pos_qty*remain < 0:
            cur.execute("""UPDATE positions_open
                           SET qty=?, entry_price=?, last=?, u_pnl=0, opened_ts=?
                           WHERE id=?""",
                        (remain, price, price, now, pid))
        else:
            cur.execute("""UPDATE positions_open
                           SET qty=?, last=?, u_pnl=(? - entry_price)*CASE WHEN qty>0 THEN qty ELSE -qty END
                           WHERE id=?""",
                        (remain, price, price, pid))
    return pnl

def apply_order(cur, sym, side, qty, price, ts):
    if qty is None: qty = 0.0
    if qty == 0.0:  # rien à faire
        return
    now = int(time.time())
    signed = qty if side=="buy" else -qty
    # existe ?
    row = qrow(cur,"SELECT id, qty, entry_price FROM positions_open WHERE symbol=?", (sym,))
    if not row:
        # ouverture directe si même sens, sinon c'est un short/long selon signed
        cur.execute("""INSERT INTO positions_open(symbol,qty,entry_price,last,u_pnl,opened_ts)
                       VALUES(?,?,?,?,0,?)""",
                    (sym, signed, price, price, now))
        return
    # position existe : fermer partiellement si sens inverse
    pid, pos_qty, entry = row
    if pos_qty * signed < 0:
        # on ferme jusqu'à min(|pos|, |signed|)
        close_part = -math.copysign(min(abs(pos_qty),abs(signed)), signed)  # opposé à la position
        realize(cur, sym, close_part, price, now)
        signed_after = pos_qty + signed
        # si il reste du sens de l'ordre -> se retrouve dans apply suivant
        row2 = qrow(cur,"SELECT id, qty, entry_price FROM positions_open WHERE symbol=?", (sym,))
        if row2:
            pid2, qty2, entry2 = row2
            cur.execute("""UPDATE positions_open
                           SET last=?, u_pnl=(? - entry_price)*CASE WHEN qty>0 THEN qty ELSE -qty END
                           WHERE id=?""",
                        (price, price, pid2))
        else:
            if abs(signed_after) > 1e-12:
                cur.execute("""INSERT INTO positions_open(symbol,qty,entry_price,last,u_pnl,opened_ts)
                               VALUES(?,?,?,?,0,?)""",
                            (sym, signed_after, price, price, now))
    else:
        # même sens : on moyenne le prix d'entrée
        new_qty = pos_qty + signed
        if abs(new_qty) < 1e-12:
            # s'annule exactement -> trade clos
            realize(cur, sym, -signed, price, now)
        else:
            new_entry = (pos_qty*entry + signed*price)/new_qty
            cur.execute("""UPDATE positions_open
                           SET qty=?, entry_price=?, last=?, u_pnl=(? - ?)*CASE WHEN ?>0 THEN ? ELSE -? END
                           WHERE id=?""",
                        (new_qty, new_entry, price, price, new_entry, new_qty, new_qty, new_qty, pid))

def main():
    pathlib.Path("/opt/scalp/project/scripts").mkdir(parents=True, exist_ok=True)
    with sqlite3.connect(DB, timeout=5) as con:

                except Exception:

                           SET u_pnl=(last - entry_price)*CASE WHEN qty>0 THEN qty ELSE -qty END""")

            con.commit()

            # petit résumé console (optionnel)
            acc = qrow(cur,"SELECT balance, equity, updated_ts FROM account_sim WHERE id=1")
            if acc:
                bal, eq, uts = acc
                print(f"updated {time.strftime('%Y-%m-%d %H:%M:%S')}  balance={bal:.2f}  equity={eq:.2f}")

def db_write(sym, ts, last, bid, ask):

async def ws_once(sub):
    async with websockets.connect(URI, ping_interval=20, close_timeout=2) as ws:
        await ws.send(json.dumps(sub))
        t0=time.time()
        while time.time()-t0<25:  # 25s par branche
            msg=json.loads(await asyncio.wait_for(ws.recv(), timeout=8))
            # OLD = {"action":"snapshot"/"update", "data":[...]}
            # NEW = {"event":"subscribe"} puis "data"
            d=None
            if "data" in msg:
                d=msg["data"][0]
            elif msg.get("action") in ("snapshot","update") and msg.get("data"):
                d=msg["data"][0]
            if not d: 
                continue
            last=float(d.get("last") or d.get("lastPr") or d.get("pr") or 0.0)
            bid=float(d.get("bestBid") or d.get("bidPx") or last)
            ask=float(d.get("bestAsk") or d.get("askPx") or last)
            ts=int((d.get("systemTime") or d.get("ts") or time.time()*1000)//1000)
            if last>0:
                db_write(SYM, ts, last, bid, ask)
                print("WROTE", SYM, ts, last, bid, ask)
                return True
        return False

async def rest_fallback():
    url=f"https://api.bitget.com/api/v2/mix/market/tickers?productType=umcbl"  # USDT-perp
    try:
        timeout=aiohttp.ClientTimeout(total=8)
        async with aiohttp.ClientSession(timeout=timeout) as ses:
            async with ses.get(url) as r:
                j=await r.json()
        for it in j.get("data",[]):
            if it.get("symbol")==f"{SYM}_UMCBL":
                last=float(it.get("lastPr") or 0)
                bid=float(it.get("bestBid") or last)
                ask=float(it.get("bestAsk") or last)
                ts=int(time.time())
                if last>0:
                    db_write(SYM, ts, last, bid, ask)
                    print("REST_WROTE", SYM, ts, last, bid, ask)
                    return True
    except Exception as e:
        print("REST_FAIL", e)
    return False

async def main():
    # 3 tentatives avec backoff
    for attempt in range(3):
        # course NEW + OLD, on garde le premier qui écrit
        done, pending = await asyncio.wait(
            [asyncio.create_task(ws_once(SUB_NEW)), asyncio.create_task(ws_once(SUB_OLD))],
            return_when=asyncio.FIRST_COMPLETED)
        ok = any(t.result() for t in done if not t.cancelled())
        for p in pending: p.cancel()
        if ok:
            return
        # fallback REST
        if await rest_fallback():
            return
        await asyncio.sleep(1.5*(attempt+1)+random.random())
    print("TIMEOUT_ALL")
asyncio.run(main())

#!/usr/bin/env python3
import sqlite3, time
DB="/opt/scalp/project/data/scalp.db"

TF_USED=["5m","15m","30m"]
W_TF={"5m":0.20,"15m":0.30,"30m":0.50}
TAU={"5m":0.25,"15m":0.30,"30m":0.35}
ATR_GATE={"5m":0.12,"15m":0.18,"30m":0.25}  # en %

def smx(pos,neg,t):
    import math
    b=math.exp(max(pos,0)/max(t,1e-6))
    s=math.exp(max(neg,0)/max(t,1e-6))
    h=1.0
    z=b+s+h
    return b/z, s/z, h/z

def con():
    c=sqlite3.connect(DB, timeout=10, isolation_level=None)
    c.execute("PRAGMA journal_mode=WAL"); c.execute("PRAGMA synchronous=NORMAL")
    c.execute("""CREATE TABLE IF NOT EXISTS contexts_A(
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      symbol TEXT, ctx TEXT, pb REAL, ps REAL, ph REAL, score REAL, ts INTEGER)""")
    c.execute("""CREATE TABLE IF NOT EXISTS contexts_A_state(
      symbol TEXT PRIMARY KEY, last_ctx TEXT, persist INT, since_ts INT)""")
    c.execute("CREATE INDEX IF NOT EXISTS idx_ctxA_symbol_ts ON contexts_A(symbol,ts)")
    return c

def top30(c):
    return [r[0] for r in c.execute("""
      SELECT symbol FROM v_universe ORDER BY symbol LIMIT 35
    """)]

def last_row(c, tf, sym):
    view={"5m":"indic_5m","15m":"indic_15m","30m":"indic_30m"}[tf]
    q=f"""SELECT close, sma20, sma50, macd, macd_sig, macd_hist, rsi14, atr_pct, obv
          FROM {view}
          WHERE symbol=?
          ORDER BY ts DESC LIMIT 1"""
    r=c.execute(q,(sym,)).fetchone()
    return r

def main():
    c=con(); now=int(time.time())
    syms=top30(c)
    ins=0; with_ctx=0
    for s in syms:
        agg_score=agg_pb=agg_ps=agg_ph=0.0; wsum=0.0; seen=False
        for tf in TF_USED:
            r=last_row(c,tf,s)
            if not r: continue
            seen=True
            close,sma20,sma50,macd,sig,hist,rsi,atr_pct,obv=r
            # scores élémentaires sur signaux SQL
            s_ema = 0.0 if sma20 is None or sma50 is None or close is None else ( (sma20 - sma50) / (close*0.01 + 1e-12) )
            s_macd = 0.0 if macd is None or sig is None else ( (macd - sig) / (abs(macd)+1e-9) )
            s_rsi = 0.0 if rsi is None else ((rsi-50.0)/50.0)
            s_obv = 0.0 if obv is None else max(-1.0, min(1.0, obv / 1e9))
            # simple proxy ADX via amplitude MACD hist
            s_adx = 0.0 if hist is None else max(-1.0, min(1.0, abs(hist)/ (abs(macd)+1e-9) ))
            score_raw = 0.35*s_ema + 0.25*s_macd + 0.20*s_rsi + 0.10*s_adx + 0.10*s_obv
            score_raw = max(-1.0, min(1.0, score_raw))
            gate = 1.0 if (atr_pct or 0.0) >= ATR_GATE[tf] else 0.0
            score = gate * score_raw
            pb,ps,ph = smx(max(score,0), max(-score,0), TAU[tf])
            w=W_TF[tf]; wsum+=w
            agg_score += w*score; agg_pb += w*pb; agg_ps += w*ps; agg_ph += w*ph
        if not seen or wsum==0:
            c.execute("INSERT INTO contexts_A(symbol,ctx,pb,ps,ph,score,ts) VALUES(?,?,?,?,?,?,?)",
                      (s,"none",0.0,0.0,1.0,0.0,now)); ins+=1; continue
        agg_score/=wsum; agg_pb/=wsum; agg_ps/=wsum; agg_ph/=wsum
        # hysteresis minimale
        row=c.execute("SELECT last_ctx,persist FROM contexts_A_state WHERE symbol=?",(s,)).fetchone()
        prev,pers=(row[0],row[1]) if row else ("none",0)
        ctx = "buy" if agg_pb>=0.60 and agg_pb>agg_ps else ("sell" if agg_ps>=0.60 and agg_ps>agg_pb else ("hold" if agg_ph>=max(agg_pb,agg_ps) else "hold"))
        if prev in("buy","sell") and ctx!=prev and max(agg_pb,agg_ps)<0.45: ctx=prev
        pers = (pers+1) if ctx==prev else 1
        if ctx!=prev and pers<2: ctx=prev
        c.execute("INSERT INTO contexts_A(symbol,ctx,pb,ps,ph,score,ts) VALUES(?,?,?,?,?,?,?)",
                  (s,ctx,agg_pb,agg_ps,agg_ph,agg_score,now))
        c.execute("INSERT INTO contexts_A_state(symbol,last_ctx,persist,since_ts) VALUES(?,?,?,?) \
                   ON CONFLICT(symbol) DO UPDATE SET last_ctx=excluded.last_ctx,persist=excluded.persist,since_ts=excluded.since_ts",
                  (s,ctx,pers,now))
        ins+=1; with_ctx+= int(ctx!='none')
    print(f"[AnalyseA] inserted={ins} with_ctx={with_ctx} ts={now}")

if __name__=="__main__":
    main()

#!/usr/bin/env python3
import asyncio, time, ujson as json, websockets

URI = "wss://ws.bitget.com/mix/v1/stream"   # endpoint WS Bitget
# ✅ marché MC (Market Composite)
SUB = {"op":"subscribe","args":[
    {"instType":"MC","channel":"ticker","instId":"BTCUSDT"}
]}

async def main():
    try:
        async with websockets.connect(
            URI, ping_interval=20, ping_timeout=20, close_timeout=5, max_size=2**20
        ) as ws:
            await asyncio.sleep(1)
            await ws.send(json.dumps(SUB))
            print("✅ Subscribed to Bitget MC BTCUSDT ticker")
            start = time.time()
            while time.time() - start < 20:
                msg = await asyncio.wait_for(ws.recv(), timeout=20)
                if isinstance(msg, (bytes, bytearray)):
                    msg = msg.decode(errors="ignore")
                try:
                    data = json.loads(msg)
                except Exception:
                    continue
                if "data" in data:
                    print("📊 WS OK:", str(data)[:200])
                    return
                if data.get("event") in ("subscribe","subscribed") or data.get("code") in (0,"0"):
                    print("📩 SUB ACK:", data)
            print("❌ WS FAIL: no data received in 20s")
    except Exception as e:
        print("🚨 WS FAIL:", type(e).__name__, e)

asyncio.run(main())
