import time, sqlite3, math, json
ADB="data/a.db"
# pondérations TF
W5, W15, W30 = 0.2, 0.3, 0.5
TAU=1.5 # température softmax

def ema(vals, n):
    k=2/(n+1)
    e=None
    for v in vals:
        e = v if e is None else (v*k + e*(1-k))
    return e if e is not None else 0.0

def load_close(conn, table, instId, n=50):
    rows=list(conn.execute(f"SELECT c FROM {table} WHERE instId=? ORDER BY ts DESC LIMIT ?", (instId,n)))
    return [r[0] for r in rows][::-1]

def score_ctx(cs):
    # signal scalaire par TF via pentes EMA
    if len(cs)<20: return 0.0
    s_fast = ema(cs[-10:], 5)
    s_slow = ema(cs[-20:],10)
    d = (s_fast - s_slow)/max(1e-9, s_slow)
    return d

def softmax3(x_buy, x_hold, x_sell, tau=TAU):
    exps=[math.exp(x/tau) for x in (x_buy, x_hold, x_sell)]
    s=sum(exps); 
    return [e/s for e in exps]

def decide(inst, c5, c15, c30):
    s5  = score_ctx(c5)
    s15 = score_ctx(c15)
    s30 = score_ctx(c30)
    x = W5*s5 + W15*s15 + W30*s30
    # map vers 3 classes: buy (x), hold (-|x|), sell (-x)
    p_buy, p_hold, p_sell = softmax3(x, -abs(x)*0.5, -x)
    ctx = "buy" if p_buy>max(p_hold,p_sell) else ("sell" if p_sell>max(p_buy,p_hold) else ("hold" if max(p_buy,p_sell)<0.45 else "none"))
    score_A = (p_buy - p_sell) # [-1..1] centré
    return ctx, p_buy, p_hold, p_sell, score_A

def wait_tf_boundary(tf_sec):
    now = int(time.time())
    wait = tf_sec - (now % tf_sec) + 1
    if wait < tf_sec:
        print(f"[{tf_sec}s] ⏳ attente {wait}s avant synchro...")
        time.sleep(wait)


def last_ts(conn, table, instId):
    r = conn.execute(f"SELECT MAX(ts) FROM {table} WHERE instId=?", (instId,)).fetchone()
    return r[0] or 0


def save_rows(conn, table, instId, candles):
    rows = [
        (instId, int(c[0] // 1000), c[1], c[2], c[3], c[4], c[5])
        for c in candles
    ]
    conn.executemany(
        f"INSERT OR REPLACE INTO {table}(instId,ts,o,h,l,c,v) VALUES(?,?,?,?,?,?,?)",
        rows,
    )


def purge_old(conn, table, instId, limit, tf):
    tf_sec = TF_SECONDS[tf]
    cutoff = int(time.time()) - limit * tf_sec
    conn.execute(f"DELETE FROM {table} WHERE instId=? AND ts < ?", (instId, cutoff))


def fetch_new(pair, tf, since):
    try:
        return EX.fetch_ohlcv(pair, tf, limit=LIMIT, since=since * 1000 if since else None)
    except Exception as e:
        print(f"{pair} {tf} err: {e}")
        return []


def worker(tf, table, syms):
    tf_sec = TF_SECONDS[tf]
    while True:
        wait_tf_boundary(tf_sec)
        with sqlite3.connect(ADB) as ca:
            ca.execute("PRAGMA busy_timeout=8000")
            for sym in syms:
                pair = sym.replace("USDT", "/USDT")
                since = last_ts(ca, table, sym)
                candles = fetch_new(pair, tf, since)
                if not candles:
                    continue
                save_rows(ca, table, sym, candles)
                purge_old(ca, table, sym, LIMIT, tf)
            ca.commit()
        print(f"OA-{tf}: ok ({len(syms)} pairs)")
        sleep_time = tf_sec - (int(time.time()) % tf_sec) + 1
        time.sleep(sleep_time)

#!/usr/bin/env python3
import sqlite3, numpy as np, time, os

ROOT = "/opt/scalp/project"
ADB = f"{ROOT}/data/a.db"
UDB = f"{ROOT}/data/u.db"

# pondérations des TF pour cohérence
WEIGHTS = {"5m": 0.2, "15m": 0.3, "30m": 0.5}
TF_TABLES = {"5m": "ohlcv_5m", "15m": "ohlcv_15m", "30m": "ohlcv_30m"}

def slope(series):
    """pente relative par rapport au dernier close"""
    if len(series) < 2:
        return 0.0
    x = np.arange(len(series))
    y = np.array(series, dtype=float)
    a, _ = np.polyfit(x, y, 1)
    ref = y[-1] or 1
    return float(a / ref)

def context_from_slope(v):
    """détermine un contexte brut (plus sensible)"""
    if v > 0.0005:
        return "buy"
    if v < -0.0005:
        return "sell"
    return "hold"

def analyze_context():
    now = int(time.time())
    with sqlite3.connect(UDB) as cu, sqlite3.connect(ADB) as ca:
        cu.execute("PRAGMA busy_timeout=8000")
        ca.execute("PRAGMA busy_timeout=8000")

        # récupère liste des coins et score_U
        syms = [r[0] for r in cu.execute("SELECT instId FROM u_syms ORDER BY rank")]
        cols = [c[1] for c in cu.execute("PRAGMA table_info(u_syms);")]
        col_scoreU = "score_U" if "score_U" in cols else "score"
        scoresU = dict(cu.execute(f"SELECT instId, {col_scoreU} FROM u_syms"))

        # table contexts_A
        ca.execute("""
        CREATE TABLE IF NOT EXISTS contexts_A (
            instId TEXT PRIMARY KEY,
            ctx_5m TEXT,
            ctx_15m TEXT,
            ctx_30m TEXT,
            ctx_final TEXT,
            score_A REAL,
            score_U REAL,
            updated_ts INTEGER
        );
        """)

        for sym in syms:
            ctxs, slopes = {}, {}
            for tf, tbl in TF_TABLES.items():
                closes = [r[0] for r in ca.execute(
                    f"SELECT c FROM {tbl} WHERE instId=? ORDER BY ts DESC LIMIT 50", (sym,)
                )]
                closes = closes[::-1]
                if not closes:
                    continue
                s = slope(closes)
                slopes[tf] = s
                ctxs[tf] = context_from_slope(s)

            if not ctxs:
                continue

            votes = list(ctxs.values())
            ctx_final = max(set(votes), key=votes.count)
            agreement = sum(WEIGHTS[tf] for tf, c in ctxs.items() if c == ctx_final)
            score_A = round(10 * agreement, 2)

            ca.execute("""
            INSERT INTO contexts_A(instId,ctx_5m,ctx_15m,ctx_30m,ctx_final,score_A,score_U,updated_ts)
            VALUES(?,?,?,?,?,?,?,?)
            ON CONFLICT(instId) DO UPDATE SET
              ctx_5m=excluded.ctx_5m,
              ctx_15m=excluded.ctx_15m,
              ctx_30m=excluded.ctx_30m,
              ctx_final=excluded.ctx_final,
              score_A=excluded.score_A,
              score_U=excluded.score_U,
              updated_ts=excluded.updated_ts
            """, (
                sym,
                ctxs.get("5m"),
                ctxs.get("15m"),
                ctxs.get("30m"),
                ctx_final,
                score_A,
                scoresU.get(sym, 0.0),
                now,
            ))

        ca.commit()
    print(f"A: ok ({len(syms)} coins analysés)")

if __name__ == "__main__":
    analyze_context()

#!/usr/bin/env python3
import asyncio, websockets, ujson as json, sqlite3, time, sys, os
sys.stdout.reconfigure(line_buffering=True)

ROOT = "/opt/scalp/project"
UDB  = f"{ROOT}/data/u.db"
TDB  = f"{ROOT}/data/t.db"
BDB  = f"{ROOT}/data/b.db"
URI  = "wss://ws.bitget.com/mix/v1/stream?compress=false"

TTL_SEC        = 60          # durée de conservation des ticks
FLUSH_INTERVAL = 5           # flush toutes les 5s
GROUP_SIZE     = 10          # nb de paires par WS
BATCH_MAX      = 500         # taille du buffer avant flush

CREATE_SQL = """
CREATE TABLE IF NOT EXISTS ticks (
    instId TEXT PRIMARY KEY,
    lastPr REAL,
    bidPr  REAL,
    askPr  REAL,
    bidSz  REAL,
    askSz  REAL,
    ts_ms  INTEGER
);
"""

def push_to_B():
    """Copie atomique vers b.db"""
    try:
        with sqlite3.connect(TDB, timeout=5) as ct, sqlite3.connect(BDB, timeout=5) as cb:
            cb.execute("""
            CREATE TABLE IF NOT EXISTS ticks_live (
                instId TEXT PRIMARY KEY,
                lastPr REAL,
                bidPr  REAL,
                askPr  REAL,
                bidSz  REAL,
                askSz  REAL,
                ts_ms  INTEGER
            );
            """)
            cb.execute("DELETE FROM ticks_live;")
            for row in ct.execute("SELECT * FROM ticks;"):
                cb.execute("""
                INSERT OR REPLACE INTO ticks_live
                (instId,lastPr,bidPr,askPr,bidSz,askSz,ts_ms)
                VALUES(?,?,?,?,?,?,?);
                """, row)
            cb.commit()
        print(f"T: pushed to b.db @ {time.strftime('%H:%M:%S')}")
    except Exception as e:
        print(f"T: push->B error: {e}")

def save_ticks(batch):
    """Écrit les ticks en lot et purge les anciens"""
    if not batch:
        return
    try:
        with sqlite3.connect(TDB, timeout=5) as c:
            c.execute("PRAGMA busy_timeout=5000")
            c.execute(CREATE_SQL)
            c.executemany("""
            INSERT OR REPLACE INTO ticks(instId,lastPr,bidPr,askPr,bidSz,askSz,ts_ms)
            VALUES(:instId,:lastPr,:bidPr,:askPr,:bidSz,:askSz,:ts_ms);
            """, batch)
            cutoff = int(time.time()*1000) - TTL_SEC*1000
            c.execute("DELETE FROM ticks WHERE ts_ms < ?", (cutoff,))
            c.commit()
        push_to_B()
    except Exception as e:
        print(f"T: save error: {e}")

async def get_symbols():
    with sqlite3.connect(UDB) as c:
        return [r[0] for r in c.execute("SELECT instId FROM u_syms ORDER BY rank;")]

async def handle_socket(subs):
    payload = {"op":"subscribe","args":subs}
    buffer, last_flush = [], time.time()
    async with websockets.connect(URI, ping_interval=20, ping_timeout=20) as ws:
        await ws.send(json.dumps(payload))
        print(f"T: connected ({len(subs)} pairs)")
        async for msg in ws:
            try:
                data = json.loads(msg)
            except:
                continue
            if isinstance(data, dict) and "event" in data:
                if data["event"] == "error":
                    print("T: evt error", data)
                continue
            if "arg" not in data or "data" not in data:
                continue
            try:
                d = data["data"][0]
                tick = dict(
                    instId=data["arg"]["instId"],
                    lastPr=float(d.get("last",0)),
                    bidPr=float(d.get("bestBid",0)),
                    askPr=float(d.get("bestAsk",0)),
                    bidSz=float(d.get("bidSz",0)),
                    askSz=float(d.get("bestAskSz",0)),
                    ts_ms=int(d.get("ts",0)),
                )
                buffer.append(tick)
            except Exception as e:
                print("T: parse error:", e)
            now = time.time()
            if now - last_flush >= FLUSH_INTERVAL or len(buffer) >= BATCH_MAX:
                print(f"T: flush {len(buffer)} ticks @ {time.strftime('%H:%M:%S')}")
                save_ticks(buffer)
                buffer.clear()
                last_flush = now

async def main():
    with sqlite3.connect(TDB) as c:
        c.execute(CREATE_SQL)
    symbols = await get_symbols()
    if not symbols:
        print("T: no symbols found")
        return
    groups = [symbols[i:i+GROUP_SIZE] for i in range(0, len(symbols), GROUP_SIZE)]
    while True:
        try:
            await asyncio.gather(*[
                handle_socket([{"instType":"mc","channel":"ticker","instId":s} for s in g])
                for g in groups
            ])
        except Exception as e:
            print(f"T: reconnect all sockets ({e})")
            await asyncio.sleep(5)

if __name__ == "__main__":
    asyncio.run(main())

def last_ts(conn, table, instId):
    r = conn.execute(f"SELECT MAX(ts) FROM {table} WHERE instId=?", (instId,)).fetchone()
    return r[0] or 0


def save_rows(conn, table, instId, candles):
    rows = [
        (instId, int(c[0] // 1000), c[1], c[2], c[3], c[4], c[5]) for c in candles
    ]
    conn.executemany(
        f"INSERT OR REPLACE INTO {table}(instId,ts,o,h,l,c,v) VALUES(?,?,?,?,?,?,?)",
        rows,
    )


def purge_old(conn, table, instId, limit, tf):
    tf_sec = TF_SECONDS[tf]
    cutoff = int(time.time()) - limit * tf_sec
    conn.execute(f"DELETE FROM {table} WHERE instId=? AND ts < ?", (instId, cutoff))


def fetch_new(pair, tf, since):
    try:
        return EX.fetch_ohlcv(pair, tf, limit=LIMIT, since=since * 1000 if since else None)
    except Exception as e:
        print(f"{pair} {tf} err: {e}")
        return []


def aggregate_3m(conn):
    """génère les bougies 3m à partir des 1m"""
    conn.execute("""
    CREATE TABLE IF NOT EXISTS ohlcv_3m (
        instId TEXT,
        ts INTEGER,
        o REAL, h REAL, l REAL, c REAL, v REAL,
        PRIMARY KEY(instId, ts)
    );
    """)
    conn.execute("DELETE FROM ohlcv_3m;")
    conn.execute("""
    INSERT OR REPLACE INTO ohlcv_3m(instId, ts, o, h, l, c, v)
    SELECT
      instId,
      (ts/180)*180 AS ts_group,
      FIRST_VALUE(o) OVER w,
      MAX(h)        OVER w,
      MIN(l)        OVER w,
      LAST_VALUE(c) OVER w,
      SUM(v)        OVER w
    FROM ohlcv_1m
    WINDOW w AS (PARTITION BY instId,(ts/180)*180 ORDER BY ts
                 ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING);
    """)
    conn.commit()

                save_rows(co, f"ohlcv_{tf}", sym, candles)

        aggregate_3m(co)
        co.commit()
    print("OB: ok (1m/3m/5m maj incrémentale + purge + agrégation 3m)")
