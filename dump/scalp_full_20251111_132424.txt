OUT="${DUMP_DIR}/scalp_full_${TS}.txt"
MAX_FILE_SIZE_KB="${MAX_FILE_SIZE_KB:-512}"

# ---------- GIT ENV ----------
[ -f /etc/scalp.env ] && . /etc/scalp.env
: "${GIT_USERNAME:?}"
: "${GIT_TOKEN:?}"
GIT_BRANCH="${GIT_BRANCH:-main}"
GIT_EMAIL_USE="${GIT_EMAIL:-${GIT_USERNAME}@users.noreply.github.com}"
REMOTE_URL="https://${GIT_HOST:-github.com}/${GIT_OWNER:-$GIT_USERNAME}/${GIT_REPO:-Scalp}.git"

# ---------- HEADER ----------
{
  echo "# ===== SCALP PROJECT DUMP ====="
  echo "# $(date -u +'%Y-%m-%d %H:%M:%S UTC')"
  echo "# Root: $ROOT"
  echo
  echo "========== TREE =========="
} > "$OUT"

find "$ROOT" -type f \
  -not -path '*/.git/*' -not -path '*/venv/*' \
  -not -path '*/dump/*' -not -path '*/logs/*' \
  -size -"${MAX_FILE_SIZE_KB}"k \
  \( -name '*.py' -o -name '*.sh' -o -name '*.conf' -o -name '*.sql' -o -name '*.txt' \) |
  sort | while IFS= read -r f; do
    sz=$(stat -c%s "$f" 2>/dev/null || echo 0)
    mt=$(stat -c%y "$f" 2>/dev/null || echo unknown)
    printf "%-80s %12s %s\n" "$f" "$sz" "$mt" >> "$OUT"
done

{
  echo
  echo "========== FILE CONTENT =========="
} >> "$OUT"

  -not -path '*/.git/*' -not -path '*/venv/*' \

#!/usr/bin/env bash
set -euo pipefail
DB_B=/opt/scalp/project/data/b.db
DB_A=/opt/scalp/project/data/a.db
DB_U=/opt/scalp/project/data/u.db

echo "===== B GATE COUNTER ($(date '+%F %T')) ====="
sqlite3 -readonly "$DB_B" <<SQL
ATTACH DATABASE '$DB_A' AS A;
ATTACH DATABASE '$DB_U' AS U;

.headers on
.mode column

-- Gate#1: candidats (universe ∩ feat_1m frais)
WITH uni AS (
  SELECT REPLACE(instId,'/','') AS instId FROM U.universe
),
feat_latest AS (
  SELECT REPLACE(instId,'/','') AS instId, MAX(ts) AS ts_feat
  FROM feat_1m GROUP BY REPLACE(instId,'/','')
),
g1 AS (
  SELECT u.instId, f.ts_feat
  FROM uni u JOIN feat_latest f ON f.instId=u.instId
  WHERE (strftime('%s','now')*1000 - f.ts_feat) <= 120000
),
-- Gate#2: ctx frais
ctx_latest AS (
  SELECT REPLACE(instId,'/','') AS instId, MAX(ts) AS ts_ctx
  FROM A.ctx_a_latest GROUP BY REPLACE(instId,'/','')
),
g2 AS (
  SELECT g1.instId, g1.ts_feat, c.ts_ctx
  FROM g1 LEFT JOIN ctx_latest c ON c.instId=g1.instId
  WHERE (strftime('%s','now')*1000 - c.ts_ctx) <= 300000
),
-- Gate#3: signal produit <= 600s
sig_latest AS (
  SELECT REPLACE(instId,'/','') AS instId, MAX(ts_signal) AS ts_sig
  FROM signals_B GROUP BY REPLACE(instId,'/','')
)
SELECT
  (SELECT COUNT(*) FROM U.universe)                      AS universe,
  (SELECT COUNT(*) FROM g1)                              AS g1_feat_ok,
  (SELECT COUNT(*) FROM g2)                              AS g2_ctx_ok,
  (SELECT COUNT(*) FROM g2 JOIN sig_latest s USING(instId)
     WHERE (strftime('%s','now')*1000 - s.ts_sig) <= 600000) AS g3_sig_ok;

DETACH DATABASE A;
DETACH DATABASE U;
SQL

ATTACH DATABASE '$DB_A'  AS A;
ATTACH DATABASE '$DB_OA' AS OA;
ATTACH DATABASE '$DB_U'  AS U;

.headers on
.mode column
.width 11 8 8 8 19 10 19 10 19 10

WITH uni AS (
  SELECT REPLACE(instId,'/','') AS instId FROM U.universe
),
feat AS (
  SELECT REPLACE(instId,'/','') AS instId,
         MAX(ts) AS ts_feat
  FROM feat_1m
  GROUP BY REPLACE(instId,'/','')
),
ctx AS (
  SELECT REPLACE(instId,'/','') AS instId,
         MAX(ts) AS ts_ctx
  FROM A.ctx_a_latest
  GROUP BY REPLACE(instId,'/','')
),
sig AS (
  SELECT REPLACE(instId,'/','') AS instId,
         MAX(ts_signal) AS ts_sig
  FROM signals_B
  GROUP BY REPLACE(instId,'/','')
),
oa5 AS (
  SELECT REPLACE(instId,'/','') AS instId,
         MAX(ts) AS ts_oa5
  FROM OA.ohlcv_5m
  GROUP BY REPLACE(instId,'/','')
),
mix AS (
  SELECT u.instId,
         f.ts_feat,
         c.ts_ctx,
         o.ts_oa5,
         s.ts_sig
  FROM uni u
  LEFT JOIN feat f ON f.instId=u.instId
  LEFT JOIN ctx  c ON c.instId=u.instId
  LEFT JOIN oa5  o ON o.instId=u.instId
  LEFT JOIN sig  s ON s.instId=u.instId
)
SELECT
  instId AS COIN,
  printf('%9.1f', (strftime('%s','now')*1000 - ts_feat)/1000.0) AS age_feat_s,
  printf('%9.1f', (strftime('%s','now')*1000 - ts_ctx )/1000.0) AS age_ctx_s,
  printf('%9.1f', (strftime('%s','now')*1000 - ts_oa5)/1000.0) AS age_oa5_s,
  datetime(ts_feat/1000,'unixepoch','localtime') AS last_feat,
  CASE WHEN (strftime('%s','now')*1000 - ts_feat) <= 120000 THEN 'OK' ELSE 'OLD' END AS gate_feat,
  datetime(ts_ctx/1000,'unixepoch','localtime') AS last_ctx,
  CASE WHEN (strftime('%s','now')*1000 - ts_ctx ) <= 300000 THEN 'OK' ELSE 'OLD' END AS gate_ctx,
  datetime(ts_sig/1000,'unixepoch','localtime') AS last_sig,
  CASE WHEN (strftime('%s','now')*1000 - ts_sig ) <= 600000 THEN 'OK' ELSE 'OLD' END AS gate_sig
FROM mix
ORDER BY age_sig_s IS NULL, gate_feat DESC, gate_ctx DESC, age_feat_s DESC
LIMIT 25;

.print ''
.print '--- GATES SUMMARY (Universe only) ---'
.headers off
SELECT 'Universe coins: ' || COUNT(*) FROM U.universe;
SELECT 'feat_1m fresh <=120s: ' || COUNT(*) FROM (
  SELECT 1 FROM feat_1m f
  JOIN U.universe u ON REPLACE(f.instId,'/','')=REPLACE(u.instId,'/','')
  GROUP BY REPLACE(f.instId,'/','')
  HAVING (strftime('%s','now')*1000 - MAX(f.ts)) <= 120000
);
SELECT 'ctx_a_latest fresh <=300s: ' || COUNT(*) FROM (
  SELECT 1 FROM A.ctx_a_latest a
  JOIN U.universe u ON REPLACE(a.instId,'/','')=REPLACE(u.instId,'/','')
  GROUP BY REPLACE(a.instId,'/','')
  HAVING (strftime('%s','now')*1000 - MAX(a.ts)) <= 300000
);
SELECT 'signals_B updated <=600s: ' || COUNT(*) FROM (
  SELECT 1 FROM signals_B s
  JOIN U.universe u ON REPLACE(s.instId,'/','')=REPLACE(u.instId,'/','')

);

DETACH DATABASE A;
DETACH DATABASE OA;
DETACH DATABASE U;

#!/usr/bin/env bash
set -euo pipefail

echo "===== CHECK O-F-C-R PIPELINE ====="

check_table() {
    local name="$1"
    local db="$2"
    local table="$3"
    local tscol="$4"

    if ! sqlite3 "$db" ".tables" | grep -qw "$table"; then
        printf "%-9s → %s ❌ (missing table: %s)\n" "$name" "$name" "$table"
        return
    fi

    local count last
    count=$(sqlite3 "$db" "SELECT COUNT(*) FROM $table;" 2>/dev/null || echo 0)
    last=$(sqlite3 "$db" "
        SELECT 
            printf('%02dh%02dm%02ds ago',
                (strftime('%s','now')-MAX($tscol/1000))/3600,
                ((strftime('%s','now')-MAX($tscol/1000))%3600)/60,
                ((strftime('%s','now')-MAX($tscol/1000))%60)
            )
        FROM $table;
    " 2>/dev/null || echo "00h00m00s ago")

    if [[ "$count" =~ ^[0-9]+$ && "$count" -gt 0 ]]; then
        printf "%-9s → %s ✅  (%s recs, last=%s)\n" "$name" "$name" "$count" "$last"
    else
        printf "%-9s → %s ✅  (0 recs, last=00h00m00s ago)\n" "$name" "$name"
    fi
}

# --- CHECK PIPELINE (tolérant) ---
check_table opener   "$DB_OPEN" "trades_open_init" "ts_open"
check_table follower "$DB_FOL"  "trades_follow"    "ts_update"

ORDER BY ts_update DESC
LIMIT 5;
" 2>/dev/null || echo "(no table trades_follow)"

#!/usr/bin/env bash
set -euo pipefail
DATA="/opt/scalp/project/data"
echo "===== CHECK O-F-R STATUS $(date '+%H:%M:%S') ====="

check_db() {
  local label=$1 db=$2 table=$3 ts_col=$4
  [[ -f "$db" ]] || { echo "$label ❌ no DB"; return; }

  local count ts_last age_s age_hms
  count=$(sqlite3 "$db" "SELECT COUNT(*) FROM $table;" 2>/dev/null || echo 0)
  ts_last=$(sqlite3 "$db" "SELECT MAX($ts_col) FROM $table;" 2>/dev/null || echo 0)
  [[ -z "$ts_last" || "$ts_last" == "0" ]] && ts_last=0

  local now=$(($(date +%s%3N)))
  age_s=$(( (now - ts_last) / 1000 ))
  if (( age_s < 10 )); then s="✅"
  elif (( age_s < 60 )); then s="⚠️"
  else s="❌"
  fi
  age_hms=$(printf "%02dh%02dm%02ds" $((age_s/3600)) $(((age_s%3600)/60)) $((age_s%60)))
  echo "$(basename "$db") → $label $s ($count rows, last=$age_hms ago)"
}

check_db "FOLLOWER" "$DATA/follower.db" "trades_follow" "ts_update"
check_db "CLOSER"   "$DATA/closer.db"   "trades_close"  "ts_close"
check_db "RECORDER" "$DATA/recorder.db" "trades_record" "ts_record"

echo "====================================="

----- FILE: /opt/scalp/project/bin/dash.sh -----
#!/usr/bin/env bash
set -euo pipefail

DATA="/opt/scalp/project/data"
LOG="/opt/scalp/project/logs/dash.log"
now_s=$(date +%s)
ts_now="$(date '+%Y-%m-%d %H:%M:%S')"
printf "\n===== SCALP PIPE HEALTH %s =====\n" "$(date '+%H:%M:%S')" | tee -a "$LOG"

check_db() {
    local name="$1" file="$2" table="$3" ts_col="$4"
    if [[ ! -f "$file" ]]; then
        printf "%-10s ❌ no DB found\n" "$name" | tee -a "$LOG"
        return
    fi
    local count ts_last age_s h m s age_hms status
    count=$(sqlite3 "$file" "SELECT COUNT(*) FROM $table;")
    ts_last=$(sqlite3 "$file" "SELECT MAX($ts_col)/1000 FROM $table;")
    [[ -z "$ts_last" || "$ts_last" == "0" ]] && ts_last=$now_s
    age_s=$(( now_s - ${ts_last%.*:-0} ))
    h=$((age_s/3600)); m=$(( (age_s%3600)/60 )); s=$((age_s%60))
    age_hms=$(printf "%02dh%02dm%02ds" "$h" "$m" "$s")

    if (( age_s < 10 )); then status="✅"
    elif (( age_s < 60 )); then status="⚠️"
    else status="❌"
    fi

    printf "%-10s %s  %4s recs | last=%s | age=%s\n" \
        "$name" "$status" "$count" "$(date '+%H:%M:%S' -d @$ts_last 2>/dev/null || date -r "$ts_last" '+%H:%M:%S')" "$age_hms" \
        | tee -a "$LOG"
}

check_db "opener"   "$DATA/opener.db"   "trades_open_init" "ts_create"
check_db "follower" "$DATA/follower.db" "trades_follow"    "ts_update"
check_db "closer"   "$DATA/closer.db"   "trades_close"     "ts_close"
check_db "recorder" "$DATA/recorder.db" "trades_record"    "ts_record"

printf "==============================================\n" | tee -a "$LOG"
echo "" >> "$LOG"

add_col_safe() {
  local db="$1" table="$2" col="$3" type="$4"
  if ! sqlite3 "$db" "PRAGMA table_info($table);" | awk -F'|' '{print $2}' | grep -qx "$col"; then
    sqlite3 "$db" "ALTER TABLE $table ADD COLUMN $col $type;"
    echo "[$(basename "$db")] +$col"
  fi
}

    f.instId,

FROM trades_follow f

import sqlite3, math, time

DB_A = "/opt/scalp/project/data/a.db"

DB_OB = "/opt/scalp/project/data/ob.db"
TF = "5m"

def softmax(x):
    e = [math.exp(i) for i in x]
    s = sum(e)
    return [v/s for v in e]

def get_universe():
    con = sqlite3.connect(DB_U)
    rows = con.execute("SELECT instId FROM universe ORDER BY instId").fetchall()
    con.close()
    return [r[0] for r in rows]

def load_ob(inst):
    con = sqlite3.connect(DB_OB)
    row = con.execute(f"SELECT rsi FROM ob_{TF} WHERE instId=? ORDER BY ts DESC LIMIT 1", (inst,)).fetchone()
    con.close()
    return row[0] if row else None

def compute_ctx(rsi):
    if rsi < 30: return "bullish"
    if rsi > 70: return "bearish"
    if 45 <= rsi <= 55: return "range"
    return "none"

def compute_score_A(ctx):
    if ctx == "bullish": return 0.8
    if ctx == "bearish": return -0.8
    if ctx == "range": return 0.0
    return 0.0

def run():
    insts = get_universe()
    now = int(time.time()*1000)
    out = []

    for inst in insts:
        rsi = load_ob(inst)
        if rsi is None:
            continue
        ctx = compute_ctx(rsi)
        score_A = compute_score_A(ctx)
        p_buy, p_sell = softmax([score_A, -score_A])
        out.append((inst, now, 0.0, score_A, p_buy, p_sell, ctx))

    con = sqlite3.connect(DB_A)
    con.execute("DELETE FROM ctx_a_latest")
    con.executemany("INSERT INTO ctx_a_latest(instId,ts,score_U,score_A,p_buy,p_sell,ctx) VALUES (?,?,?,?,?,?,?)", out)

    con.close()

if __name__ == "__main__":

#!/usr/bin/env python3
import sqlite3, time, numpy as np

DB_A  = "/opt/scalp/project/data/a.db"
DB_OA = "/opt/scalp/project/data/oa.db"
DB_U  = "/opt/scalp/project/data/u.db"

# --- paramètres de pondération ---
W = dict(EMA=0.35, MACD=0.25, RSI=0.20, ADX=0.10, OBV=0.10)
TF_WEIGHTS = {"30m":0.5, "15m":0.3, "5m":0.2}
TAU = 0.35

# --- fonctions utilitaires ---
def ema(arr, n):
    if len(arr) < n: return None
    k = 2/(n+1)
    e = arr[0]
    for x in arr[1:]:
        e = k*x + (1-k)*e
    return e

def atr(high, low, close, period=14):
    tr = [max(h-l, abs(h-cp), abs(l-cp)) for h,l,cp in zip(high[1:],low[1:],close[:-1])]
    return np.mean(tr[-period:]) if tr else 0

def rsi(vals, period=14):
    if len(vals) < period+1: return 50
    diffs = np.diff(vals)
    up = np.mean([d for d in diffs[-period:] if d>0]) or 1e-9
    dn = np.mean([-d for d in diffs[-period:] if d<0]) or 1e-9
    rs = up/dn
    return 100 - 100/(1+rs)

def adx(high, low, close, period=14):
    if len(high) < period+2: return 20
    plus_dm = np.maximum(np.diff(high), 0)
    minus_dm = np.maximum(-np.diff(low), 0)
    tr = np.maximum(high[1:]-low[1:], np.abs(high[1:]-close[:-1]))
    tr = np.maximum(tr, np.abs(low[1:]-close[:-1]))
    tr_sm = np.mean(tr[-period:])
    plus = 100*np.mean(plus_dm[-period:])/tr_sm if tr_sm else 0
    minus = 100*np.mean(minus_dm[-period:])/tr_sm if tr_sm else 0
    dx = abs(plus - minus)/(plus + minus + 1e-9)*100
    return dx

def macd(vals):
    e12, e26 = ema(vals,12), ema(vals,26)
    if e12 is None or e26 is None: return 0
    hist = e12 - e26
    return hist

def softmax3(s, tau=TAU):
    exp_pos = np.exp(s/tau)
    exp_neg = np.exp(-s/tau)
    denom = exp_pos + 1 + exp_neg
    return exp_pos/denom, exp_neg/denom, 1 - (exp_pos+exp_neg)/denom

# --- extraction OHLCV ---
def fetch_tf(con, inst, tf):
    q = f"SELECT ts, open, high, low, close, volume FROM ohlcv_{tf} WHERE instId=? ORDER BY ts DESC LIMIT 150"
    rows = con.execute(q, (inst,)).fetchall()
    if not rows: return None
    arr = np.array(rows[::-1])
    return arr[:,1], arr[:,2], arr[:,3], arr[:,4], arr[:,5]

# --- score par timeframe ---
def score_tf(o,h,l,c,v):
    c = c.astype(float)
    atrv = atr(h,l,c)
    s_ema  = np.tanh((ema(c,12) - ema(c,26)) / (max(atrv,1e-8)))
    s_macd = np.tanh(macd(c) / (np.std(c) + 1e-8))
    s_rsi  = (rsi(c)-50)/50
    s_adx  = np.clip((adx(h,l,c)-20)/20, -1, 1)
    s_obv  = np.tanh(np.diff(v)[-1] / (np.median(np.abs(v[-20:])) + 1e-8))
    return np.average([s_ema,s_macd,s_rsi,s_adx,s_obv], weights=list(W.values()))

# --- score multi-TF ---
def score_multi(tf_scores):
    vals = [tf_scores.get(tf,0)*TF_WEIGHTS[tf] for tf in TF_WEIGHTS]
    return sum(vals)

def main():
    ts_now = int(time.time()*1000)
    con_a = sqlite3.connect(DB_A)
    con_oa = sqlite3.connect(DB_OA)
    con_u = sqlite3.connect(DB_U)

    insts = [r[0] for r in con_u.execute("SELECT instId FROM universe")]

    rows_out=[]
    for inst in insts:
        tf_scores={}
        for tf in TF_WEIGHTS.keys():
            data = fetch_tf(con_oa, inst, tf)
            if data is None: continue
            o,h,l,c,v = data
            tf_scores[tf] = score_tf(o,h,l,c,v)
        if not tf_scores: continue
        s = score_multi(tf_scores)

        p_buy,p_sell,p_hold = softmax3(s)
        if p_buy>=0.6: ctx="bullish"
        elif p_sell>=0.6: ctx="bearish"
        else: ctx="range"

        score_U = 0.0
        score_A = abs(s)
        rows_out.append((inst, ts_now, score_U, score_A, p_buy, p_sell, ctx))

    cur = con_a.cursor()
    cur.execute("DELETE FROM ctx_a_latest")
    cur.executemany("""
        INSERT INTO ctx_a_latest(instId,ts,score_U,score_A,p_buy,p_sell,ctx)
        VALUES (?,?,?,?,?,?,?)
    """, rows_out)
    con_a.commit()
    con_a.close(); con_oa.close(); con_u.close()
    print(f"[A] ✓ ctx_A updated (multi-indicator, {len(rows_out)} inst.)")

if __name__ == "__main__":
    main()

#!/usr/bin/env python3
# ==============================================================
#  SCALP - B_FEAT_BUILDER (adapté au ticks minimal instId/lastPr/ts_ms)
# ==============================================================

import sqlite3
import pandas as pd
import numpy as np
import logging
import time

DB_B = "/opt/scalp/project/data/b.db"
DB_T = "/opt/scalp/project/data/t.db"

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s B_FEAT_BUILDER %(levelname)s %(message)s",
    datefmt="%Y-%m-%d %H:%M:%S"
)

# --------------------------------------------------------------
# Fonctions utilitaires
# --------------------------------------------------------------
def ema(series, span):
    return series.ewm(span=span, adjust=False).mean()

def rsi(series, period=14):
    delta = series.diff()
    gain = delta.clip(lower=0).rolling(period).mean()
    loss = (-delta.clip(upper=0)).rolling(period).mean()
    rs = gain / (loss + 1e-9)
    return 100 - (100 / (1 + rs))

def macd(series, fast=12, slow=26, signal=9):
    ema_fast = ema(series, fast)
    ema_slow = ema(series, slow)
    macd_line = ema_fast - ema_slow
    signal_line = ema(macd_line, signal)
    hist = macd_line - signal_line
    return macd_line, signal_line, hist

def bollinger(series, period=20, mult=2):
    mid = series.rolling(period).mean()
    std = series.rolling(period).std()
    up = mid + mult * std
    low = mid - mult * std
    width = (up - low) / mid
    return up, mid, low, width

# --------------------------------------------------------------
# Construction des features à partir des ticks
# --------------------------------------------------------------
def build_features(instId):
    con = sqlite3.connect(DB_T)
    df = pd.read_sql_query(f"""
        SELECT instId, lastPr AS close, ts_ms AS ts
        FROM ticks
        WHERE instId = '{instId}'
        ORDER BY ts ASC
        LIMIT 1000;
    """, con)
    con.close()

    if df.empty or len(df) < 20:
        return 0

    df["open"] = df["close"].shift(1)
    df["high"] = df["close"].rolling(3).max()
    df["low"] = df["close"].rolling(3).min()
    df["volume"] = 0.0
    df["price"] = df["close"]

    # ---- indicateurs ----
    df["ema12"] = ema(df["close"], 12)
    df["ema26"] = ema(df["close"], 26)
    df["ema20"] = ema(df["close"], 20)
    df["ema50"] = ema(df["close"], 50)
    df["ema100"] = ema(df["close"], 100)
    df["ema200"] = ema(df["close"], 200)

    df["rsi7"] = rsi(df["close"], 7)
    df["rsi14"] = rsi(df["close"], 14)
    df["macd"], df["macd_signal"], df["macd_hist"] = macd(df["close"])
    df["bb_up"], df["bb_mid"], df["bb_low"], df["bb_width"] = bollinger(df["close"])

    # ATR simplifié à partir de range local
    df["atr"] = (df["high"] - df["low"]).rolling(14).mean()

    # champs placeholder
    df["adx"] = 0
    df["plus_di"] = 0
    df["minus_di"] = 0
    df["trigger"] = ""
    df["ctx"] = "range"

    df = df.dropna().tail(200)

    # ---- insert dans feat_1m ----
    con_b = sqlite3.connect(DB_B)
    cur = con_b.cursor()

    cur.execute("""
        CREATE TABLE IF NOT EXISTS feat_1m (
            instId TEXT,
            ts INTEGER,
            open REAL, high REAL, low REAL, close REAL,
            volume REAL, price REAL,
            ema12 REAL, ema26 REAL, ema20 REAL, ema50 REAL, ema100 REAL, ema200 REAL,
            rsi7 REAL, rsi14 REAL, macd REAL, macd_signal REAL, macd_hist REAL,
            atr REAL, bb_up REAL, bb_mid REAL, bb_low REAL, bb_width REAL,
            adx REAL, plus_di REAL, minus_di REAL,
            trigger TEXT, ctx TEXT, ts_update INTEGER,
            PRIMARY KEY (instId, ts)
        );
    """)

    for _, r in df.iterrows():
        cur.execute("""
            INSERT OR REPLACE INTO feat_1m VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)
        """, tuple(r[k] for k in [
            "instId","ts","open","high","low","close","volume","price",
            "ema12","ema26","ema20","ema50","ema100","ema200",
            "rsi7","rsi14","macd","macd_signal","macd_hist",
            "atr","bb_up","bb_mid","bb_low","bb_width",
            "adx","plus_di","minus_di","trigger","ctx","ts"
        ]))
    con_b.commit()
    con_b.close()
    return len(df)

# Main
# --------------------------------------------------------------
def main():

    insts = [r[0] for r in con.execute("SELECT DISTINCT instId FROM ticks;")]

def main():
    conn_feat = sqlite3.connect(DB_FEAT)
    conn_trig = sqlite3.connect(DB_TRIG)
    conn_trig.execute("""
    CREATE TABLE IF NOT EXISTS triggers_B (
        instId TEXT,
        side TEXT,
        reason TEXT,
        score_B REAL,
        price REAL,
        atr REAL,
        ts_trigger INTEGER,
        PRIMARY KEY(instId, reason)
    );
    """)

    cur = conn_feat.cursor()
    cur.execute("SELECT DISTINCT instId FROM feat_1m;")
    instruments = [r[0] for r in cur.fetchall()]
    ts_now = int(time.time() * 1000)

    for inst in instruments:
        cur.execute("""
            SELECT close, ema20, ema50, rsi7, adx, atr
            FROM feat_1m WHERE instId=? ORDER BY ts DESC LIMIT 1;
        """, (inst,))
        row = cur.fetchone()
        if not row:
            continue
        close, ema20, ema50, rsi7, adx, atr = row
        if None in (close, ema20, ema50, rsi7, adx, atr):
            log.warning(f"{inst}: valeurs incomplètes, skip")

        div = abs(ema20 - ema50) / ema50 if ema50 != 0 else 0
        signals = []

        # === MOMENTUM ===
        score_B = min(100, max(0, (adx - 20) * 2 + abs(rsi7 - 50) * 0.5))
        if ema20 > ema50 and rsi7 > 55:
            signals.append(("buy", "MOMENTUM", score_B))
        elif ema20 < ema50 and rsi7 < 45:
            signals.append(("sell", "MOMENTUM", score_B))

        # === BREAKOUT ===
        score_B = min(100, (adx * 1.5) + (div * 5000))
        if ctx in ("range", "neutral") and adx > 20 and div > 0.005:
            side = "buy" if ema20 > ema50 else "sell"
            signals.append((side, "BREAKOUT", score_B))

        # === REVERSAL ===
        if ctx == "bearish" and rsi7 < 35:
            score_B = min(100, (35 - rsi7) * 3)
            signals.append(("buy", "REVERSAL", score_B))
        elif ctx == "bullish" and rsi7 > 65:
            score_B = min(100, (rsi7 - 65) * 3)
            signals.append(("sell", "REVERSAL", score_B))

        # === MEAN REVERSION ===
        if ctx == "range" and abs(ema20 - ema50) / ema50 < 0.002:
            score_B = max(0, 100 - abs(50 - rsi7) * 2)
            side = "buy" if rsi7 < 50 else "sell"

MAX_ROWS = 300

def to_bitget_symbol(inst: str) -> str:
    return inst if ":" in inst else f"{inst}:USDT"

def ensure_schema():
    con = sqlite3.connect(DBO, timeout=5)
    cur = con.cursor()
    for tf in TF_LIST:
        cur.execute(f"DROP TABLE IF EXISTS ohlcv_{tf}")
        cur.execute(f"""
        CREATE TABLE IF NOT EXISTS ohlcv_{tf}(
          instId TEXT,
          ts INTEGER,
          open REAL, high REAL, low REAL, close REAL, volume REAL,
          PRIMARY KEY(instId, ts)
        )
        """)
    con.commit()
    con.close()

def load_universe():
    con = sqlite3.connect(DBU, timeout=5)
    rows = [r[0] for r in con.execute("SELECT instId FROM universe")]
    con.close()
    return rows

def last_ts(con, tf, inst):
    row = con.execute(f"SELECT MAX(ts) FROM ohlcv_{tf} WHERE instId=?", (inst,)).fetchone()
    return row[0] if row and row[0] is not None else None

def upsert_batch(tf, inst, batch):
    if not batch:
        return
    con = sqlite3.connect(DBO, timeout=5)
    cur = con.cursor()
    cur.executemany(
        f"INSERT OR REPLACE INTO ohlcv_{tf}(instId,ts,open,high,low,close,volume) VALUES (?,?,?,?,?,?,?)",
        [(inst, o[0], o[1], o[2], o[3], o[4], o[5]) for o in batch]
    )
    cur.execute(f"""
        DELETE FROM ohlcv_{tf}
        WHERE instId=?
          AND ts NOT IN (
            SELECT ts FROM ohlcv_{tf}
            WHERE instId=?
            ORDER BY ts DESC
            LIMIT {MAX_ROWS}
          )
    """, (inst, inst))
    con.commit()
    con.close()

def sync_inst_tf(inst, tf):
    inst_ccxt = to_bitget_symbol(inst)
    con = sqlite3.connect(DBO, timeout=5)
    ts_last = last_ts(con, tf, inst)
    con.close()

    if ts_last is None:
        data = ex.fetch_ohlcv(inst_ccxt, tf, limit=INIT)
        upsert_batch(tf, inst, data)
        return

    data = ex.fetch_ohlcv(inst_ccxt, tf, since=ts_last + 1, limit=150)
    upsert_batch(tf, inst, data)

def main():

        atr REAL,
        PRIMARY KEY(instId, ts)
    )""")

def atr14(rows):
    # rows = [(ts,o,h,l,c,v)] triés ASC
    if len(rows) < 15:
        return None
    trs=[]
    prev_close = rows[0][4]
    for (_,o,h,l,c,v) in rows[1:]:
        trs.append(max(h-l, abs(h-prev_close), abs(l-prev_close)))
        prev_close=c
    return sum(trs[-14:]) / 14.0

def process(tf):
    tbl_k = f"kline_{tf}"
    tbl_f = f"feat_{tf}"
    con_ob = sqlite3.connect(DB_OB)
    con_b  = sqlite3.connect(DB_B)

    for inst in insts:
        rows = con_ob.execute(
            f"SELECT ts,o,h,l,c,v FROM {tbl_k} WHERE instId=? ORDER BY ts ASC",
            (inst,)

        a = atr14(rows)

            continue
        ts = rows[-1][0]
        con_b.execute(
            f"INSERT OR REPLACE INTO {tbl_f}(instId,ts,atr) VALUES(?,?,?)",
            (inst,ts,a)
        )
    con_b.commit()

import ccxt
import sqlite3
import time
import datetime
import os
import subprocess
from pathlib import Path

DB_OB = Path("/opt/scalp/project/data/ob.db")
DB_U  = Path("/opt/scalp/project/data/u.db")
DB_X  = Path("/opt/scalp/project/data/opener.db")
DB_B  = Path("/opt/scalp/project/data/b.db")
LOG_PREFIX = "OB_1M"

# ============================================================
def log(msg: str):
    print(f"{datetime.datetime.now():%Y-%m-%d %H:%M:%S} {LOG_PREFIX} {msg}", flush=True)

# ============================================================
def get_watchlist():
    watch = set()
    try:
        with sqlite3.connect(DB_U) as conn_u:
            for (instId,) in conn_u.execute("SELECT DISTINCT instId FROM universe WHERE active=1;"):
                watch.add(instId)
    except Exception as e:
        log(f"ERR universe: {e}")

    try:
        with sqlite3.connect(DB_X) as conn_x:
            for (instId,) in conn_x.execute("SELECT DISTINCT instId FROM positions_open;"):
                watch.add(instId)
    except Exception as e:
        log(f"ERR x_open: {e}")

    if not watch:
        log("⚠️  Aucun instrument actif trouvé")
    else:
        log(f"Watchlist: {', '.join(sorted(watch))}")

    return sorted(list(watch))

# ============================================================
def init_db():
    conn = sqlite3.connect(DB_OB)
    conn.execute("""
        CREATE TABLE IF NOT EXISTS ohlcv_1m (
            instId TEXT,
            ts INTEGER,
            o REAL, h REAL, l REAL, c REAL, v REAL,
            PRIMARY KEY (instId, ts)
        );
    """)
    conn.commit()
    return conn

# ============================================================
def normalize_symbol(instId: str) -> str:
    if "/" in instId:
        return instId.upper()
    if instId.endswith("USDT"):
        return instId.replace("USDT", "/USDT")
    return instId

# ============================================================
def fetch_and_store(instId: str, conn: sqlite3.Connection, limit: int = 200):
    ex = ccxt.bitget()
    symbol = normalize_symbol(instId)

    cur = conn.cursor()
    cur.execute("SELECT MAX(ts) FROM ohlcv_1m WHERE instId=?;", (instId,))
    last_ts = cur.fetchone()[0]

    if last_ts is None:
        since = int(time.time() * 1000) - limit * 60 * 1000
        log(f"{instId}: initial sync ({symbol}), fetching last {limit} bars")
    else:
        since = int(last_ts + 60_000)
        log(f"{instId}: fetching new data since {datetime.datetime.fromtimestamp(last_ts/1000)}")

    try:
        candles = ex.fetch_ohlcv(symbol, timeframe="1m", since=since, limit=limit)
        if not candles:
            log(f"{instId}: no new candles")
            return

        for ts, o, h, l, c, v in candles:
            cur.execute(
                "INSERT OR REPLACE INTO ohlcv_1m VALUES (?,?,?,?,?,?,?)",
                (instId, ts, o, h, l, c, v)
            )
        conn.commit()

        last_dt = datetime.datetime.fromtimestamp(candles[-1][0] / 1000)
        log(f"{instId}: added {len(candles)} bars → {last_dt:%Y-%m-%d %H:%M:%S}")
    except Exception as e:
        log(f"ERR fetch {instId}: {e}")

# ============================================================
def trigger_feat():
    """Exécute le script de features après collecte."""
    try:
        log("▶️  Lancement des calculs feat_1m ...")
        subprocess.run(["/opt/scalp/project/venv/bin/python3", SCRIPT_FEAT], timeout=180)
        log("✅  Calculs feat_1m terminés")
    except Exception as e:
        log(f"ERR feat trigger: {e}")

# ============================================================
def main():
    log("=== OB_COLLECT sync minute start ===")

    # Alignement initial sur la minute pile
    wait = 60 - (time.time() % 60)
    log(f"Attente initiale {wait:.1f}s jusqu'à minute pile")
    time.sleep(wait)

    while True:
        conn = init_db()
        watchlist = get_watchlist()
        if not watchlist:
            log("No instruments to collect → sleep")
        else:
            for instId in watchlist:
                fetch_and_store(instId, conn)
        conn.close()
        trigger_feat()

        # attente prochaine minute pile
        wait = 60 - (time.time() % 60)
        time.sleep(wait)

# ============================================================
if __name__ == "__main__":
    main()

#!/usr/bin/env python3
# --- SCALP - TICKS COLLECTOR (Bitget Futures v2 WS) ---
# Inputs:  u.db (universe active=1)
# Outputs: t.db (ticks, ticks_hist, v_ticks_latest)
#
# WebSocket public Futures v2: wss://ws.bitget.com/v2/ws/public
# Maintient ticks temps réel, reconnecte automatiquement,
# logue tout, idempotent, conforme guide IA.

from __future__ import annotations
import argparse, asyncio, json, logging, sqlite3, sys, time
from pathlib import Path
from typing import Any, List

DB_T = "/opt/scalp/project/data/t.db"
DB_U = "/opt/scalp/project/data/u.db"
LOGF = "/opt/scalp/project/logs/t_ticks.log"
WS_URL = "wss://ws.bitget.com/v2/ws/public"

# --- LOGGING ---------------------------------------------------
def setup_logging() -> None:
    logging.basicConfig(
        level=logging.INFO,
        format="%(asctime)s T_TICKS %(levelname)s %(message)s",
        handlers=[
            logging.FileHandler(LOGF),
            logging.StreamHandler(sys.stdout),
        ],
        force=True,
    )

# --- SQLITE ----------------------------------------------------
def connect_db(path: str) -> sqlite3.Connection:
    con = sqlite3.connect(path, timeout=30, isolation_level=None)
    con.execute("PRAGMA journal_mode=WAL;")
    con.execute("PRAGMA synchronous=NORMAL;")
    con.execute("PRAGMA busy_timeout=5000;")
    return con

SCHEMA_SQL = """
CREATE TABLE IF NOT EXISTS ticks(
  instId TEXT PRIMARY KEY,
  lastPr REAL,
  ts_ms INTEGER
);
CREATE TABLE IF NOT EXISTS ticks_hist(
  instId TEXT,
  lastPr REAL,
  ts_ms INTEGER,
  PRIMARY KEY(instId, ts_ms)
);
CREATE VIEW IF NOT EXISTS v_ticks_latest AS
SELECT
  t.instId,
  t.lastPr,
  t.ts_ms,
  strftime('%H:%M:%S', t.ts_ms/1000, 'unixepoch','localtime') AS hms,
  printf('%02d:%02d:%02d',
     ((strftime('%s','now')*1000 - t.ts_ms)/1000)/3600,
     (((strftime('%s','now')*1000 - t.ts_ms)/1000)%3600)/60,
     ((strftime('%s','now')*1000 - t.ts_ms)/1000)%60) AS age_hms
FROM ticks t
ORDER BY t.instId;
"""

def migrate(conn: sqlite3.Connection) -> None:
    conn.executescript(SCHEMA_SQL)

def upsert_tick(db_t: str, instId: str, price: float, ts_ms: int) -> None:
    con = connect_db(db_t)
    cur = con.cursor()
    cur.execute(
        "INSERT INTO ticks_hist(instId,lastPr,ts_ms) VALUES(?,?,?) "
        "ON CONFLICT(instId,ts_ms) DO NOTHING;",
        (instId, price, ts_ms),
    )
    cur.execute(
        "INSERT INTO ticks(instId,lastPr,ts_ms) VALUES(?,?,?) "
        "ON CONFLICT(instId) DO UPDATE SET lastPr=excluded.lastPr, ts_ms=excluded.ts_ms;",
        (instId, price, ts_ms),
    )
    con.commit()
    con.close()

# --- UNIVERSE --------------------------------------------------
def load_universe(db_u: str) -> List[str]:
    con = connect_db(db_u)
    cur = con.cursor()
    try:
        insts = [r[0] for r in cur.execute(
            "SELECT instId FROM universe WHERE active=1 ORDER BY instId;"
        ).fetchall()]
        if not insts:
            raise Exception
    except Exception:
        insts = [r[0] for r in cur.execute(
            "SELECT DISTINCT instId FROM universe ORDER BY instId;"
        ).fetchall()]
    con.close()
    return [s.replace("/", "").upper() for s in insts]

# --- SELF TEST -------------------------------------------------
def smoke_test() -> None:
    tmp = Path("/tmp/t_ticks_ws_test.db")
    if tmp.exists():
        tmp.unlink()
    con = connect_db(str(tmp))
    migrate(con)
    now_ms = int(time.time() * 1000)
    con.execute("INSERT INTO ticks VALUES(?,?,?)", ("TESTUSDT", 1.2345, now_ms))
    row = con.execute("SELECT instId FROM v_ticks_latest WHERE instId='TESTUSDT'").fetchone()
    assert row and row[0] == "TESTUSDT", "self-test failed"
    con.close()
    tmp.unlink(missing_ok=True)
    logging.info("SELF-TEST OK")

# --- WEBSOCKET HANDLER ----------------------------------------
async def run_ws(db_t: str, db_u: str) -> None:
    import websockets

    while True:
        insts = load_universe(db_u)
        if not insts:
            logging.warning("Aucun instrument actif → attente 5s")
            await asyncio.sleep(5)
            continue

        try:
            async with websockets.connect(WS_URL, ping_interval=None) as ws:
                subs = [
                    {"instType":"USDT-FUTURES","channel":"ticker","instId":inst}
                    for inst in insts
                ]
                msg_sub = {"op":"subscribe","args":subs}
                await ws.send(json.dumps(msg_sub))
                logging.info(f"Abonnement envoyé ({len(subs)} paires)")

                last_ping = time.time()

                while True:
                    try:
                        raw = await asyncio.wait_for(ws.recv(), timeout=25)
                    except asyncio.TimeoutError:
                        # heartbeat toutes 20s
                        await ws.send(json.dumps({"op":"ping"}))
                        last_ping = time.time()
                        continue

                    msg = json.loads(raw)
                    if "action" not in msg:
                        continue
                    if msg["action"] not in ("snapshot","update"):
                        continue
                    data = msg.get("data", [])
                    for d in data:
                        inst = d.get("instId")
                        pr = d.get("lastPr") or d.get("last") or d.get("markPrice")
                        ts_ms = int(d.get("ts") or int(time.time()*1000))
                        if not inst or not pr:
                            continue
                        upsert_tick(db_t, inst, float(pr), ts_ms)
                    await asyncio.sleep(0)  # yield event loop

        except Exception as e:
            logging.error(f"WS error: {e} → reconnect dans 3s")
            await asyncio.sleep(3)

# --- MAIN ------------------------------------------------------
def parse(argv: List[str]) -> argparse.Namespace:
    p = argparse.ArgumentParser(description="SCALP - Ticks collector (Bitget Futures WS)")
    p.add_argument("--db-t", default=DB_T)
    p.add_argument("--db-u", default=DB_U)
    p.add_argument("--self-test", action="store_true")
    return p.parse_args(argv)

def main(argv: List[str]) -> int:
    setup_logging()
    args = parse(argv)
    if args.self_test:
        smoke_test()
        return 0

    con = connect_db(args.db_t)
    migrate(con)
    con.close()

    asyncio.run(run_ws(args.db_t, args.db_u))
    return 0

if __name__ == "__main__":
    raise SystemExit(main(sys.argv[1:]))

#!/usr/bin/env python3
import sqlite3, time
from datetime import datetime

DB_A = "/opt/scalp/project/data/a.db"
DB_B = "/opt/scalp/project/data/b.db"
DB_T = "/opt/scalp/project/data/t.db"

def get_ctx():
    with sqlite3.connect(DB_A) as con:
        cur = con.cursor()
        cur.execute("SELECT instId, ctx FROM v_ctx_latest WHERE ctx IN ('bullish','bearish','range');")
        return cur.fetchall()

def get_last_tick(inst):
    with sqlite3.connect(DB_T) as con:
        cur = con.cursor()
        cur.execute("SELECT lastPr, ts_ms FROM v_ticks_latest WHERE instId=? LIMIT 1;", (inst,))
        r = cur.fetchone()
    if not r:
        return None, None, None, None
    pr, ts_ms = r
    ts_s = ts_ms / 1000
    hms = datetime.fromtimestamp(ts_s).strftime("%H:%M:%S")
    delta = round(time.time() - ts_s, 1)
    return pr, ts_s, hms, delta

def get_last_signal(inst):
    with sqlite3.connect(DB_B) as con:
        cur = con.cursor()
        cur.execute("SELECT ts FROM signals WHERE instId=? ORDER BY ts DESC LIMIT 1;", (inst,))
        r = cur.fetchone()
    if not r:
        return None, None, None
    ts = float(r[0])
    hms = datetime.fromtimestamp(ts).strftime("%H:%M:%S")
    delta = round(time.time() - ts, 1)
    return ts, hms, delta

def dash():
    print(f"===== DASH {datetime.now().strftime('%H:%M:%S')} =====")
    for inst, ctx in get_ctx():
        pr, ts_tick, hms_tick, delta_tick = get_last_tick(inst)
        ts_sig, hms_sig, delta_sig = get_last_signal(inst)

        delta = "-"
        if ts_tick and ts_sig:
            delta = round(ts_tick - ts_sig, 1)

        flag = "⚠️" if delta_tick and delta_tick > 5 else "✅"
        print(f"{inst:<10} | ctx={ctx:<8} | price={pr or '-':<10} | "
              f"tick={hms_tick or '-':>8} | sig={hms_sig or '-':>8} | "
              f"Δ={delta:<6} | {flag}")
    print("=" * 90)

if __name__ == "__main__":
    while True:
        try:
            dash()
            time.sleep(3)
        except KeyboardInterrupt:
            break
        except Exception as e:
            print("B_DASH ERROR:", e)
            time.sleep(5)

#!/usr/bin/env python3
from __future__ import annotations
import sqlite3, time, logging, sys
from pathlib import Path

DB_G = Path("/opt/scalp/project/data/gest.db")
LOG_PATH = Path("/opt/scalp/project/logs/budget.log")

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s BUDGET %(levelname)s %(message)s",
    handlers=[logging.FileHandler(LOG_PATH), logging.StreamHandler(sys.stdout)]
)

def connect_db(path: Path):
    con = sqlite3.connect(path, timeout=15, isolation_level=None)
    con.execute("PRAGMA journal_mode=WAL;")
    con.execute("PRAGMA synchronous=NORMAL;")
    con.execute("PRAGMA busy_timeout=4000;")
    return con

def ensure_state(con):
    r = con.execute("SELECT COUNT(*) FROM budget_state;").fetchone()[0]
    if r == 0:
        con.execute("INSERT INTO budget_state (capital_total, capital_used, capital_free) VALUES (1000,0,1000);")
        con.commit()

def handle_open_updates(con):
    rows = con.execute("SELECT * FROM v_budget_for_open;").fetchall()
    if not rows:
        return 0
    for trade_uid, instId, side, margin, ts_open in rows:
        state = con.execute("SELECT capital_total, capital_used, capital_free FROM budget_state;").fetchone()
        total, used, free = state
        if free < margin:
            logging.warning(f"OPEN REFUSED {instId}: marge {margin} > capital libre {free}")
            continue
        used += margin
        free -= margin
        con.execute("""
            UPDATE budget_state SET capital_used=?, capital_free=?, updated_at=strftime('%s','now')*1000;
        """, (used, free))
        con.execute("""
            INSERT INTO budget_ack_open (trade_uid, capital_used, capital_free, ts_ack)
            VALUES (?, ?, ?, strftime('%s','now')*1000);
        """, (trade_uid, used, free))
        logging.info(f"OPEN BUDGET {instId} → marge {margin} engagée (used={used}, free={free})")
    con.commit()
    return len(rows)

def handle_close_updates(con):
    rows = con.execute("SELECT * FROM v_budget_for_close;").fetchall()
    if not rows:
        return 0
    for trade_uid, pnl_real, ts_close in rows:
        state = con.execute("SELECT capital_total, capital_used, capital_free FROM budget_state;").fetchone()
        total, used, free = state
        margin = con.execute("SELECT margin FROM budget_open WHERE trade_uid=?;", (trade_uid,)).fetchone()
        if margin:
            margin = margin[0]
        else:
            margin = 0
        used -= margin
        profit_loss = total * (pnl_real / 100)
        free += margin + profit_loss
        total = free + used
        con.execute("""
            UPDATE budget_state SET capital_total=?, capital_used=?, capital_free=?, updated_at=strftime('%s','now')*1000;
        """, (total, used, free))
        con.execute("""
            INSERT INTO budget_ack_close (trade_uid, capital_used, capital_free, ts_ack)
            VALUES (?, ?, ?, strftime('%s','now')*1000);
        """, (trade_uid, used, free))
        logging.info(f"CLOSE BUDGET {trade_uid} → libéré {margin}, PnL={pnl_real:.3f}%, free={free:.2f}")
    con.commit()
    return len(rows)

def loop():
    con = connect_db(DB_G)
    ensure_state(con)
    logging.info("BUDGET loop started (0.2Hz)")
    while True:
        try:
            n1 = handle_open_updates(con)
            n2 = handle_close_updates(con)
            if n1 + n2 > 0:
                state = con.execute("SELECT capital_total, capital_used, capital_free FROM budget_state;").fetchone()
                logging.info(f"STATE total={state[0]:.2f} used={state[1]:.2f} free={state[2]:.2f}")
            time.sleep(5)
        except KeyboardInterrupt:
            break
        except Exception as e:
            logging.warning(f"loop: {e}")
            time.sleep(3)
    con.close()
    logging.info("BUDGET stopped.")

if __name__ == "__main__":
    loop()

def get_margin_engaged(con_o):
    """Somme de la marge estimée sur les positions ouvertes/pending."""
    try:
        res = con_o.execute("""
            SELECT SUM(entry * qty / leverage)
            FROM trades_open_init
            WHERE status IN ('pending','open');
        """).fetchone()[0]
        return round(res or 0.0, 4)
    except Exception:
        return 0.0

def get_pnl_real(con_r):
    """Somme du PnL réalisé total depuis l’historique."""
    try:
        res = con_r.execute("SELECT SUM(pnl_net) FROM trades_recorded;").fetchone()[0]
        if res is None:
            res = con_r.execute("SELECT SUM(pnl) FROM trades_recorded;").fetchone()[0]
        return round(res or 0.0, 4)
    except Exception:
        return 0.0

def get_capital_initial(con_b):
    """Capital initial du budget (si non encore présent, crée 1000 USDT)."""
    con_b.execute("""
        CREATE TABLE IF NOT EXISTS budget_state(
            id INTEGER PRIMARY KEY CHECK (id=1),
            balance REAL DEFAULT 1000.0,
            margin REAL DEFAULT 0.0,
            pnl_real REAL DEFAULT 0.0,
            ts_update INTEGER
        );
    """)
    con_b.commit()

import sqlite3, time, logging

DB_FOLLOW = "/opt/scalp/project/data/follower.db"
DB_CLOSER = "/opt/scalp/project/data/closer.db"
DB_REC = "/opt/scalp/project/data/recorder.db"
DB_BUDGET = "/opt/scalp/project/data/budget.db"
DB_TICKS = "/opt/scalp/project/data/t.db"

logging.basicConfig(level=logging.INFO, format="%(asctime)s CLOSER FIX %(levelname)s %(message)s")

def get_last_price(instId):
    """Récupère le dernier prix connu depuis t.db"""
    try:
        with sqlite3.connect(DB_TICKS) as ct:
            cur = ct.cursor()
            cur.execute("SELECT lastPr FROM v_ticks_latest WHERE instId=? LIMIT 1;", (instId,))
            row = cur.fetchone()
            return float(row[0]) if row else None
    except Exception as e:
        logging.warning(f"{instId}: erreur lecture ticks → {e}")
        return None

def finalize_closures():
    con_f = sqlite3.connect(DB_FOLLOW)
    con_c = sqlite3.connect(DB_CLOSER)
    con_r = sqlite3.connect(DB_REC)
    con_b = sqlite3.connect(DB_BUDGET)

    cur_f = con_f.cursor()
    cur_c = con_c.cursor()
    cur_r = con_r.cursor()
    cur_b = con_b.cursor()

    rows = cur_f.execute("""
        SELECT instId, side, entry, last, qty, reason_close,
               score_U, score_A, score_B, ts_open, ts_update
        FROM trades_follow
        WHERE status='to_close'
    """).fetchall()

    if not rows:
        logging.info("Aucune position à finaliser.")
        return

    for instId, side, entry, last, qty, reason, sU, sA, sB, ts_open, ts_close in rows:
        if last is None:
            last = get_last_price(instId)
            if last is None:
                last = entry
                logging.warning(f"{instId}: aucun prix disponible → fallback entry={entry}")
            else:
                logging.info(f"{instId}: récupéré last depuis ticks = {last}")

        pnl = round((last - entry) * qty * (1 if side == 'buy' else -1), 6)
        logging.info(f"→ CLOSE {instId} {side} pnl={pnl} reason={reason}")

        cur_c.execute("""
            INSERT INTO positions_closed(instId,side,entry,exit,qty,pnl,sl,tp,ts_open,ts_close,reason_exit,score_U,score_A,score_B)
            VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?)
        """, (instId,side,entry,last,qty,pnl,0,0,ts_open,ts_close,reason,sU,sA,sB))

        cur_r.execute("""
            INSERT INTO trades_recorded(instId,side,entry,exit,pnl,reason_exit,ts_open,ts_close,score_U,score_A,score_B)
            VALUES (?,?,?,?,?,?,?,?,?,?,?)
        """, (instId,side,entry,last,pnl,reason,ts_open,ts_close,sU,sA,sB))

        cur_b.execute("""
            INSERT INTO ledger(src,instId,type,amount,ts)
            VALUES ('closer',?, 'pnl_real', ?, strftime('%s','now')*1000)
        """, (instId, pnl))

        cur_f.execute("UPDATE trades_follow SET status='closed' WHERE instId=?;", (instId,))

    con_c.commit()
    con_r.commit()
    con_b.commit()
    con_f.commit()
    logging.info(f"✓ {len(rows)} clôture(s) finalisée(s).")

if __name__ == "__main__":
    finalize_closures()

    return f"T{ts_signal}_{instId.replace('/','')}_{side}"

def copy_signals_to_trades(limit=500):
    with sqlite3.connect(DB_B) as src, sqlite3.connect(DB_G) as dst:
        src.row_factory = sqlite3.Row
        dst.row_factory = sqlite3.Row

        # Signaux nouveaux uniquement
        rows = src.execute("""
            SELECT b.instId, b.side, b.reason, b.price, b.atr, b.score_B, b.ts_signal,
                   COALESCE(c.ctx,'range') AS ctx
            FROM v_opener_for_gest b
            LEFT JOIN gest.trades_all g
                ON g.instId=b.instId AND g.side=b.side AND g.ts_signal=b.ts_signal
            LEFT JOIN ctx_cache c ON c.instId=b.instId
            WHERE b.price>0 AND g.trade_uid IS NULL
            ORDER BY b.ts_signal DESC
            LIMIT ?
        """, (limit,)).fetchall()

        n_new = 0
        for r in rows:
            t_uid = make_uid(r["instId"], r["side"], r["ts_signal"])
            try:
                dst.execute("""
                    INSERT INTO trades_all(
                        trade_uid, instId, side, reason, ctx,
                        score_B, atr, price, ts_signal,
                        state, ts_create
                    ) VALUES (?,?,?,?,?,?,?,?,?,'NEW',strftime('%s','now')*1000)
                """, (
                    t_uid, r["instId"], r["side"], r["reason"], r["ctx"],
                    r["score_B"], r["atr"], r["price"], r["ts_signal"]
                ))
                n_new += 1
            except sqlite3.IntegrityError:
                continue

        dst.commit()
        if n_new > 0:
            logging.info(f"SYNC {n_new} nouveaux signaux ajoutés")
        return n_new

def section(title: str):
    print(f"\n===== {title.upper()} =====")

def show_table(rows, headers):
    if not rows:
        print("(aucun)")
        return
    widths = [max(len(str(row[h])) for row in rows) for h in headers]
    widths = [max(w, len(h)) for w, h in zip(widths, headers)]
    header_line = "  ".join(h.ljust(w) for h, w in zip(headers, widths))
    print(header_line)
    print("-" * len(header_line))
    for row in rows:
        print("  ".join(str(row[h]).ljust(w) for h, w in zip(headers, widths)))

def dash():
    os.system("clear")
    con = connect()

    # --- 3 derniers signaux reçus
    section("DERNIERS SIGNAUX")
    rows = con.execute("""
        SELECT trade_uid, instId, side, reason_close AS reason,
               datetime(ts_signal_close/1000,'unixepoch','localtime') AS ts
        FROM signals_for_close
        ORDER BY ts_signal_close DESC LIMIT 3;
    """).fetchall()
    show_table(rows, ["trade_uid", "instId", "side", "reason", "ts"])

    # --- 3 dernières ouvertures (jointure pour récupérer instId, side)
    section("DERNIERES OUVERTURES")
    rows = con.execute("""
        SELECT o.trade_uid, t.instId, t.side,
               ROUND(o.entry,4) AS entry,
               datetime(o.ts_open/1000,'unixepoch','localtime') AS ts
        FROM open_snapshot o
        JOIN trades t ON t.trade_uid = o.trade_uid
        ORDER BY o.ts_open DESC LIMIT 3;
    """).fetchall()
    show_table(rows, ["trade_uid", "instId", "side", "entry", "ts"])

    # --- 2 meilleurs / 2 pires trades en cours
    section("BEST / WORST EN COURS")
    rows = con.execute("""
        SELECT f.trade_uid, t.instId, t.side, ROUND(f.pnl_float,3) AS pnl, f.reason_close
        FROM follow_snapshot f
        JOIN trades t ON t.trade_uid = f.trade_uid
        ORDER BY f.pnl_float DESC LIMIT 2;
    """).fetchall()
    show_table(rows, ["trade_uid", "instId", "side", "pnl", "reason_close"])
    rows = con.execute("""
        SELECT f.trade_uid, t.instId, t.side, ROUND(f.pnl_float,3) AS pnl, f.reason_close
        FROM follow_snapshot f
        JOIN trades t ON t.trade_uid = f.trade_uid
        ORDER BY f.pnl_float ASC LIMIT 2;
    """).fetchall()
    show_table(rows, ["trade_uid", "instId", "side", "pnl", "reason_close"])

    # --- 3 dernières clôtures
    section("DERNIERES CLOTURES")
    rows = con.execute("""
        SELECT c.trade_uid, t.instId, t.side,
               ROUND(c.pnl_real,3) AS pnl, c.reason_exit,
               datetime(c.ts_close/1000,'unixepoch','localtime') AS ts
        FROM close_snapshot c
        JOIN trades t ON t.trade_uid = c.trade_uid
        ORDER BY c.ts_close DESC LIMIT 3;
    """).fetchall()
    show_table(rows, ["trade_uid", "instId", "side", "pnl", "reason_exit", "ts"])

    # --- 3 derniers enregistrements recorder
    section("DERNIERS RECORDS")

               datetime(recorded_at/1000,'unixepoch','localtime') AS ts

    """).fetchall()

    # --- Budget global
    section("BUDGET GLOBAL")
    state = con.execute("SELECT capital_total, capital_used, capital_free FROM budget_state;").fetchone()

DB_PATH = "/opt/scalp/project/data/t.db"
DB_A_PATH = "/opt/scalp/project/data/a.db"
WS_URL = "wss://ws.bitget.com/mix/v1/stream"
FLUSH_INTERVAL = 0.5  # secondes
QUEUE_MAX = 5000

q = Queue(maxsize=QUEUE_MAX)
lock = threading.Lock()
stop_event = threading.Event()

# --- Connexion SQLite optimisée ---
def init_db():
    conn = sqlite3.connect(DB_PATH, check_same_thread=False, timeout=10)
    cur = conn.cursor()
    cur.executescript("""
        PRAGMA journal_mode=WAL;
        PRAGMA synchronous=OFF;
        CREATE TABLE IF NOT EXISTS ticks(
            instId TEXT PRIMARY KEY,
            price REAL,
            ts_ms INTEGER
        );
        CREATE INDEX IF NOT EXISTS idx_ticks_ts ON ticks(ts_ms);
    """)
    conn.commit()
    return conn

# --- Récupération des symboles depuis le contexte A ---
def get_symbols():
    conn = sqlite3.connect(DB_A_PATH)
    cur = conn.cursor()
    cur.execute("SELECT instId FROM v_ctx_latest;")
    symbols = [r[0].replace("/", "") for r in cur.fetchall()]
    conn.close()
    return symbols

# --- Thread d'écriture SQLite ---
def writer_thread():
    conn = init_db()
    cur = conn.cursor()
    last_flush = 0
    buffer = []
    while not stop_event.is_set():
        try:
            tick = q.get(timeout=FLUSH_INTERVAL)
            buffer.append(tick)
        except:
            pass

        now = time.time()
        if now - last_flush >= FLUSH_INTERVAL and buffer:
            with lock:
                cur.executemany(
                    "INSERT INTO ticks (instId, price, ts_ms) VALUES (?, ?, ?) "
                    "ON CONFLICT(instId) DO UPDATE SET price=excluded.price, ts_ms=excluded.ts_ms;",
                    buffer,
                )
                conn.commit()
                buffer.clear()
                last_flush = now

    conn.commit()
    conn.close()

# --- Fonction WS principale ---
async def ws_consumer():
    symbols = get_symbols()
    args = [
        {"instType": "USDT-FUTURES", "channel": "ticker", "instId": s}
        for s in symbols
    ]
    sub_msg = json.dumps({"op": "subscribe", "args": args})
    print(f"[ticks] Subscribing to {len(symbols)} symbols...")

    async with websockets.connect(WS_URL, ping_interval=15, ping_timeout=10) as ws:
        await ws.send(sub_msg)
        print("[ticks] Connected & subscribed.")
        async for msg in ws:
            data = json.loads(msg)
            if "data" not in data:
                continue
            for d in data["data"]:
                try:
                    instId = d["instId"]
                    price = float(d["lastPr"])
                    ts_ms = int(d["ts"])
                    if not q.full():
                        q.put((instId, price, ts_ms))
                except Exception as e:
                    pass

# --- Main ---
def main():
    wt = threading.Thread(target=writer_thread, daemon=True)
    wt.start()

    try:
        asyncio.run(ws_consumer())
    except KeyboardInterrupt:
        print("Stopping...")
    finally:
        stop_event.set()
        wt.join()

if __name__ == "__main__":
    main()

    f30.instId,
    f30.ts,
    COALESCE((ABS(f5.macd) + ABS(f15.macd) + ABS(f30.macd)) / 3.0, 0) AS score_A,

DROP VIEW IF EXISTS v_H_latest;
CREATE VIEW v_H_latest AS
WITH last20 AS (
    SELECT pnl, pnl_pct
    FROM history
    ORDER BY ts_close DESC
    LIMIT 20
),
stats AS (
    SELECT 
        (SELECT COUNT(*) FROM last20 WHERE pnl > 0) * 1.0 / (SELECT COUNT(*) FROM last20) AS winrate,
        (SELECT SUM(pnl) FROM last20 WHERE pnl > 0) AS sum_win,
        (SELECT SUM(ABS(pnl)) FROM last20 WHERE pnl < 0) AS sum_loss
)
SELECT 
    winrate,
    CASE WHEN sum_loss IS NULL OR sum_loss=0 THEN 1.0 ELSE MIN(1.0, (sum_win / sum_loss) / 2.0) END AS pf_norm,
    (0.50*winrate + 0.50*CASE WHEN sum_loss IS NULL OR sum_loss=0 THEN 1.0 ELSE MIN(1.0, (sum_win / sum_loss) / 2.0) END) AS score_H
FROM stats;

-- ======================================================
-- RECORDER H PERFORMANCE ENHANCEMENT
-- ======================================================

-- Nettoyage préalable
DROP VIEW IF EXISTS v_H_coin;
DROP VIEW IF EXISTS v_H_setup;
DROP VIEW IF EXISTS v_H_coin_setup;

-- === SCORE H COIN (EMA par actif) ===
CREATE VIEW v_H_coin AS
WITH ranked AS (
    SELECT
        instId,
        score_H,
        ROW_NUMBER() OVER (PARTITION BY instId ORDER BY ts_record DESC) AS n
    FROM trades_record
)
SELECT
    instId,
    ROUND(SUM(score_H * POWER(0.9, n-1)) / SUM(POWER(0.9, n-1)), 4) AS score_H_coin,
    COUNT(*) AS trades_count,
    ROUND(SUM(CASE WHEN score_H > 0 THEN 1 ELSE 0 END)*100.0/COUNT(*),1) AS winrate
FROM ranked
GROUP BY instId
ORDER BY score_H_coin DESC;

-- === SCORE H SETUP (EMA par setup/entry_reason) ===
CREATE VIEW v_H_setup AS
WITH ranked AS (
    SELECT
        entry_reason,
        score_H,
        ROW_NUMBER() OVER (PARTITION BY entry_reason ORDER BY ts_record DESC) AS n
    FROM trades_record
)
SELECT
    entry_reason,
    ROUND(SUM(score_H * POWER(0.9, n-1)) / SUM(POWER(0.9, n-1)), 4) AS score_H_setup,
    COUNT(*) AS trades_count,
    ROUND(SUM(CASE WHEN score_H > 0 THEN 1 ELSE 0 END)*100.0/COUNT(*),1) AS winrate
FROM ranked
GROUP BY entry_reason
ORDER BY score_H_setup DESC;

-- === SCORE H COIN x SETUP (EMA croisé) ===
CREATE VIEW v_H_coin_setup AS
WITH ranked AS (
    SELECT
        instId,
        entry_reason,
        score_H,
        ROW_NUMBER() OVER (PARTITION BY instId, entry_reason ORDER BY ts_record DESC) AS n
    FROM trades_record
)
SELECT
    instId,
    entry_reason,
    ROUND(SUM(score_H * POWER(0.9, n-1)) / SUM(POWER(0.9, n-1)), 4) AS score_H_pair,
    COUNT(*) AS trades_count,
    ROUND(SUM(CASE WHEN score_H > 0 THEN 1 ELSE 0 END)*100.0/COUNT(*),1) AS winrate
FROM ranked
GROUP BY instId, entry_reason
ORDER BY score_H_pair DESC;
