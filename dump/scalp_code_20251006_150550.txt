#!/usr/bin/env python3
import os, time, sqlite3, math
DB="data/scalp.db"
TP_BPS     = float(os.getenv("TP_BPS","15"))       # take profit en bps
SL_BPS     = float(os.getenv("SL_BPS","20"))       # stop loss   en bps
MAX_HOLD   = int(os.getenv("MAX_HOLD_SEC","900"))  # 15 min
FEES_BPS   = float(os.getenv("FEES_BPS","1"))      # aller+retour
POLL       = 0.5

def connect():
    con=sqlite3.connect(DB, timeout=20, isolation_level=None)
    con.execute("PRAGMA journal_mode=WAL")
    con.execute("PRAGMA busy_timeout=20000")
    con.execute("PRAGMA synchronous=NORMAL")
    return con

def last_price(con, symbol, default):
    row=con.execute("SELECT last FROM last_ticks WHERE symbol=?", (symbol,)).fetchone()
    return (row[0] if row else default) or default

def ensure_qty(q): return 1.0 if (q is None or q==0.0) else float(q)

def open_new_positions(con, now):
    rows=con.execute("""
      SELECT id,symbol,side,COALESCE(qty,0),COALESCE(price,0),created_ts
      FROM orders_sim
      WHERE status='SIMULATED' AND id NOT IN (SELECT order_id FROM positions_sim)
      ORDER BY id ASC LIMIT 500
    """).fetchall()
    if not rows: return 0
    con.execute("BEGIN IMMEDIATE")
    for oid,sym,side,qty,px,ts in rows:
        px = px or last_price(con, sym, None) or 0.0
        q  = ensure_qty(qty)
        con.execute("""INSERT OR IGNORE INTO positions_sim
            (order_id,symbol,side,qty,entry_price,entry_ts,status)
            VALUES(?,?,?,?,?,?, 'OPEN')""", (oid,sym,side,q,px,ts or now))
    con.execute("COMMIT")
    return len(rows)

def close_if_needed(con, now):
    rows=con.execute("""
      SELECT id, order_id, symbol, side, qty, entry_price, entry_ts
      FROM positions_sim WHERE status='OPEN'
      ORDER BY id ASC LIMIT 1000
    """).fetchall()
    if not rows: return 0
    n_closed=0
    con.execute("BEGIN IMMEDIATE")
    for pid,oid,sym,side,qty,entry,ets in rows:
        last=last_price(con, sym, entry)
        hold= now - (ets or now)
        # Targets
        tp = entry*(1 + (TP_BPS/10000.0)) if side.upper()=='BUY'  else entry*(1 - (TP_BPS/10000.0))
        sl = entry*(1 - (SL_BPS/10000.0)) if side.upper()=='BUY'  else entry*(1 + (SL_BPS/10000.0))
        reason=None; exit_px=None
        if side.upper()=='BUY':
            if last>=tp: exit_px=last; reason='TP'
            elif last<=sl: exit_px=last; reason='SL'
        else: # SELL
            if last<=tp: exit_px=last; reason='TP'
            elif last>=sl: exit_px=last; reason='SL'
        if reason is None and hold>=MAX_HOLD:
            exit_px=last; reason='TIME'
        if reason is None:
            # Reverse signal ferme la position si ordre plus récent sens opposé
            rev = con.execute("""
              SELECT 1 FROM orders_sim
              WHERE symbol=? AND created_ts>? AND
                    ((side='BUY'  AND ?='SELL') OR (side='SELL' AND ?='BUY'))
              LIMIT 1
            """,(sym, ets or 0, side.upper(), side.upper())).fetchone()
            if rev: exit_px=last; reason='REV'
        if exit_px is not None:
            # PnL: BUY => (exit-entry)*qty ; SELL => (entry-exit)*qty ; moins fees
            gross = (exit_px-entry)*qty if side.upper()=='BUY' else (entry-exit_px)*qty
            fees  = (entry+exit_px)*qty*(FEES_BPS/10000.0)
            pnl   = gross - fees
            pnl_bps = (pnl/ (entry*qty)) * 10000.0 if entry>0 and qty>0 else 0.0
            con.execute("""UPDATE positions_sim
                           SET exit_price=?, exit_ts=?, pnl_abs=?, pnl_bps=?,
                               fees_abs=?, status='CLOSED', close_reason=?
                           WHERE id=?""",
                           (exit_px, now, pnl, pnl_bps, fees, reason, pid))
            n_closed+=1
    con.execute("COMMIT")
    return n_closed

def main():
    con=connect()
    while True:
        now=int(time.time())
        opened=open_new_positions(con, now)
        closed=close_if_needed(con, now)
        if opened==0 and closed==0:
            time.sleep(POLL)

#!/usr/bin/env python3
import os, time, json, sqlite3, asyncio, websockets

DB = "/opt/scalp/project/data/scalp.db"
URI = os.getenv("WS_URL", "wss://ws.bitget.com/mix/v1/stream")   # pas de ?compress=
HEARTBEAT = int(os.getenv("WS_HEARTBEAT_SEC", "20"))
RESUB_AFTER = int(os.getenv("WS_RESUB_AFTER_SEC", "10"))
MAX_BACKOFF = int(os.getenv("WS_MAX_BACKOFF_SEC", "60"))

def connect_db():
    con = sqlite3.connect(DB, timeout=20, isolation_level=None)
    con.execute("PRAGMA journal_mode=WAL")
    con.execute("PRAGMA busy_timeout=20000")
    con.execute("""CREATE TABLE IF NOT EXISTS last_ticks(
        symbol TEXT PRIMARY KEY, last REAL NOT NULL, ts INTEGER NOT NULL)""")
    con.execute("""CREATE TABLE IF NOT EXISTS signals_B(
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        symbol TEXT, side TEXT, ctx TEXT, size REAL, leverage INTEGER,
        sl REAL, tp REAL, reason TEXT, created_ts INTEGER)""")
    return con

def load_symbols():
    # 1) top90_active.json si présent, sinon BTCUSDT
    p = "/opt/scalp/project/data/top90_active.json"
    try:
        with open(p,"r") as f:
            data = json.load(f)
        if isinstance(data, dict):
            syms = list(data.keys())
        elif isinstance(data, list):
            syms = data
        else:
            syms = []
    except Exception:
        syms = []
    if not syms:
        syms = ["BTCUSDT"]
    # normalise
    syms = [s.upper().replace("-","").strip() for s in syms if isinstance(s,str)]
    return sorted(set(syms))

def build_sub(syms):
    # Bitget: {"op":"subscribe","args":[{"instType":"MC","channel":"ticker","instId":"BTCUSDT"}, ...]}
    return {"op":"subscribe","args":[{"instType":"MC","channel":"ticker","instId":s} for s in syms]}

async def run():
    con = connect_db()
    syms = load_symbols()
    sub = build_sub(syms)
    print("Connecting to:", URI, "symbols:", len(syms))
    backoff = 1
    px_prev = {}  # mémoire pour delta
    while True:
        try:
            async with websockets.connect(
                URI, ping_interval=HEARTBEAT, ping_timeout=HEARTBEAT, close_timeout=5, max_size=2**20
            ) as ws:
                await asyncio.sleep(1)
                await ws.send(json.dumps(sub))
                print("SUB sent")
                start = time.time()
                while True:
                    msg = await asyncio.wait_for(ws.recv(), timeout=max(HEARTBEAT*2, 30))
                    if isinstance(msg,(bytes,bytearray)):
                        msg = msg.decode(errors="ignore")
                    try:
                        d = json.loads(msg)
                    except Exception:
                        continue
                    # ACK subscribe
                    if d.get("event") in ("subscribe","subscribed") or d.get("code") in (0,"0"):
                        continue
                    # données
                    data = d.get("data")
                    if not data:
                        continue
                    now = int(time.time())
                    for it in data:
                        inst = (it.get("instId") or it.get("symbol") or "").upper().replace("-","")
                        last_str = it.get("last") or it.get("price") or it.get("close")
                        try:
                            last = float(last_str)
                        except Exception:
                            continue
                        # UPSERT last_ticks
                        con.execute("""INSERT INTO last_ticks(symbol,last,ts)
                                       VALUES(?,?,?)
                                       ON CONFLICT(symbol) DO UPDATE SET last=excluded.last, ts=excluded.ts""",
                                    (inst, last, now))
                        # émission simple côté B sur variation
                        prev = px_prev.get(inst)
                        if prev and prev > 0:
                            delta_bps = abs((last-prev)/prev)*10000.0
                            if delta_bps > 0:  # seuil minimal si besoin
                                side = "buy" if last>prev else "sell"
                                con.execute("""INSERT INTO signals_B(symbol,side,ctx,size,leverage,sl,tp,reason,created_ts)
                                               VALUES(?,?,?,?,?,?,?,?,?)""",
                                            (inst, side, "TICK", 0.0, 3, None, None, f"tick={delta_bps:.5f}", now))
                        px_prev[inst] = last
                    # resub périodique pour sécurité
                    if time.time() - start > RESUB_AFTER*60:
                        await ws.send(json.dumps(sub))
                        start = time.time()
                # fin boucle messages
        except Exception as e:
            print("WS error:", type(e).__name__, str(e)[:200])
            await asyncio.sleep(backoff)
            backoff = min(backoff*2, MAX_BACKOFF)

if __name__ == "__main__":
    asyncio.run(run())

#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import sqlite3, time, random, os

DB = "/opt/scalp/project/data/scalp.db"
BUSY_MS = 20000
INIT_BALANCE = 100.0

def conn():
    cx = sqlite3.connect(DB, check_same_thread=False)
    cx.row_factory = sqlite3.Row
    cx.execute("PRAGMA journal_mode=WAL;")
    cx.execute(f"PRAGMA busy_timeout={BUSY_MS};")
    return cx

def init_db(cx):
    cx.executescript("""
    CREATE TABLE IF NOT EXISTS account_sim (
        id INTEGER PRIMARY KEY CHECK (id=1),
        balance REAL DEFAULT 0,
        equity REAL DEFAULT 0,
        used_margin REAL DEFAULT 0,
        updated_ts INTEGER
    );
    CREATE TABLE IF NOT EXISTS positions_open (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        symbol TEXT,
        side TEXT,
        qty REAL,
        entry_price REAL,
        last REAL,
        u_pnl REAL,
        opened_ts INTEGER
    );
    CREATE TABLE IF NOT EXISTS trades_closed (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        symbol TEXT,
        side TEXT,
        qty REAL,
        entry_price REAL,
        exit_price REAL,
        pnl REAL,
        reason TEXT,
        opened_ts INTEGER,
        closed_ts INTEGER
    );
    """)

    cur = cx.execute("SELECT COUNT(*) AS n FROM account_sim")
    if cur.fetchone()["n"] == 0:
        ts = int(time.time())
        cx.execute("INSERT INTO account_sim(id,balance,equity,used_margin,updated_ts) VALUES(1,?,?,?,?)",
                   (INIT_BALANCE, INIT_BALANCE, 0, ts))
        cx.commit()

def fetch_signal(cx):
    row = cx.execute("""
        SELECT id, symbol, side, reason, created_ts
        FROM signals_B
        WHERE id NOT IN (SELECT signal_id FROM orders_sim)
        ORDER BY id DESC LIMIT 1
    """).fetchone()
    return row

def update_equity(cx):
    ts = int(time.time())
    bal, eq = cx.execute("SELECT balance, equity FROM account_sim WHERE id=1").fetchone()
    cx.execute("UPDATE account_sim SET equity=?, updated_ts=? WHERE id=1", (eq, ts))
    cx.commit()

def simulate_order(cx, sig):
    symbol = sig["symbol"]
    side = sig["side"]
    price = random.uniform(0.95, 1.05) * random.uniform(1, 100)
    qty = 1.0
    ts = int(time.time())
    cx.execute("""
        INSERT INTO positions_open(symbol,side,qty,entry_price,last,u_pnl,opened_ts)
        VALUES(?,?,?,?,?,?,?)
    """, (symbol, side, qty, price, price, 0, ts))
    cx.commit()

def close_random_positions(cx):
    rows = cx.execute("SELECT * FROM positions_open ORDER BY opened_ts ASC LIMIT 5").fetchall()
    for r in rows:
        pnl = random.uniform(-2, 5)
        ts = int(time.time())
        cx.execute("""
            INSERT INTO trades_closed(symbol,side,qty,entry_price,exit_price,pnl,reason,opened_ts,closed_ts)
            VALUES(?,?,?,?,?,?,?,?,?)
        """, (r["symbol"], r["side"], r["qty"], r["entry_price"],
              r["entry_price"]*(1+pnl/100), pnl, "SIM_CLOSE", r["opened_ts"], ts))
        cx.execute("DELETE FROM positions_open WHERE id=?", (r["id"],))
    cx.commit()

def loop():
    cx = conn()
    init_db(cx)
    print("=== PAPER EXEC SIM ===")
    for _ in range(5):
        sig = fetch_signal(cx)
        if sig:
            simulate_order(cx, sig)
        close_random_positions(cx)
        update_equity(cx)
        time.sleep(1)
    print("Simulation terminée.")

if __name__ == "__main__":
    loop()
