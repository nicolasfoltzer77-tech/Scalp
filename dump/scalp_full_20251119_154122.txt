#!/usr/bin/env bash
set -euo pipefail

DB_OB="/opt/scalp/project/data/ob.db"
DB_U="/opt/scalp/project/data/u.db"
LOG="/opt/scalp/project/logs/ob_collect.log"

echo "[$(date)] OB_COLLECT START" >> "$LOG"

# load universe
mapfile -t UNIVERSE < <(sqlite3 -readonly "$DB_U" "SELECT instId FROM universe;")

# transform AAVE/USDT → AAVEUSDT_UMCBL
convert_inst() {
    echo "${1//\//}"_UMCBL
}

while true; do
    for inst in "${UNIVERSE[@]}"; do
        sym=$(convert_inst "$inst")

        URL="https://api.bitget.com/api/mix/v1/market/candles?symbol=${sym}&granularity=60"
        JSON=$(curl -s "$URL")

        # Check valid response (must be array)
        if echo "$JSON" | jq -e '.[0]' >/dev/null 2>&1; then
            ts=$(echo "$JSON" | jq -r '.[0][0]')
            o=$(echo "$JSON"  | jq -r '.[0][1]')
            h=$(echo "$JSON"  | jq -r '.[0][2]')
            l=$(echo "$JSON"  | jq -r '.[0][3]')
            c=$(echo "$JSON"  | jq -r '.[0][4]')
            v=$(echo "$JSON"  | jq -r '.[0][5]')

            sqlite3 "$DB_OB" <<SQL
INSERT INTO ohlcv_1m(instId, ts, o,h,l,c,v)
VALUES("$sym", $ts, $o, $h, $l, $c, $v)
ON CONFLICT(instId, ts) DO UPDATE SET
    o=excluded.o,
    h=excluded.h,
    l=excluded.l,
    c=excluded.c,
    v=excluded.v;
SQL

        else
            echo "[$(date)] ERR $sym Invalid JSON: $JSON" >> "$LOG"
        fi
    done

    sleep 1
done

#!/usr/bin/env bash
set -euo pipefail
DB="/opt/scalp/project/data/ob.db"

# Sécurité: vérifier que la base n'est pas verrouillée
if lsof "$DB" | grep -q ob.db; then
  echo "❌ ob.db en cours d'utilisation. Ferme les processus avant d'agréger."
  lsof "$DB"
  exit 1
fi

sqlite3 "$DB" <<'SQL'
PRAGMA busy_timeout=5000;

-- (Optionnel mais recommandé si non créés) Index pour accélérer l'agrégation
-- CREATE INDEX IF NOT EXISTS idx_ohlcv_1m_inst_ts ON ohlcv_1m(instId, ts);

-- Nettoyage 3m
DELETE FROM ohlcv_3m;

-- Agrégation 1m -> 3m sans fonctions analytiques (compat SQLite 3.37)
WITH g AS (
  SELECT
    instId,
    (ts/180000)*180000 AS ts3,
    MIN(ts) AS ts_min,
    MAX(ts) AS ts_max,
    MAX(h) AS h,
    MIN(l) AS l,
    SUM(v) AS v
  FROM ohlcv_1m
  GROUP BY instId, ts3
)
INSERT INTO ohlcv_3m(instId, ts, o, h, l, c, v)
SELECT
  g.instId,
  g.ts3 AS ts,
  (SELECT o FROM ohlcv_1m WHERE instId=g.instId AND ts=g.ts_min) AS o,
  g.h,
  g.l,
  (SELECT c FROM ohlcv_1m WHERE instId=g.instId AND ts=g.ts_max) AS c,
  g.v
FROM g;

-- Vérification
SELECT '3m' AS tf, COUNT(*) AS rows, datetime(MAX(ts)/1000,'unixepoch','localtime') AS last_ts
FROM ohlcv_3m;
SQL

is_kept() {

#!/usr/bin/env bash
set -euo pipefail

DATA="/opt/scalp/project/data"
LOG="/opt/scalp/project/logs/dash.log"
now_s=$(date +%s)
ts_now="$(date '+%Y-%m-%d %H:%M:%S')"
printf "\n===== SCALP PIPE HEALTH %s =====\n" "$(date '+%H:%M:%S')" | tee -a "$LOG"

check_db() {
    local name="$1" file="$2" table="$3" ts_col="$4"
    if [[ ! -f "$file" ]]; then
        printf "%-10s ❌ no DB found\n" "$name" | tee -a "$LOG"
        return
    fi
    local count ts_last age_s h m s age_hms status
    count=$(sqlite3 "$file" "SELECT COUNT(*) FROM $table;")
    ts_last=$(sqlite3 "$file" "SELECT MAX($ts_col)/1000 FROM $table;")
    [[ -z "$ts_last" || "$ts_last" == "0" ]] && ts_last=$now_s
    age_s=$(( now_s - ${ts_last%.*:-0} ))
    h=$((age_s/3600)); m=$(( (age_s%3600)/60 )); s=$((age_s%60))
    age_hms=$(printf "%02dh%02dm%02ds" "$h" "$m" "$s")

    if (( age_s < 10 )); then status="✅"
    elif (( age_s < 60 )); then status="⚠️"
    else status="❌"
    fi

    printf "%-10s %s  %4s recs | last=%s | age=%s\n" \
        "$name" "$status" "$count" "$(date '+%H:%M:%S' -d @$ts_last 2>/dev/null || date -r "$ts_last" '+%H:%M:%S')" "$age_hms" \
        | tee -a "$LOG"
}

check_db "opener"   "$DATA/opener.db"   "trades_open_init" "ts_create"
check_db "follower" "$DATA/follower.db" "trades_follow"    "ts_update"
check_db "closer"   "$DATA/closer.db"   "trades_close"     "ts_close"
check_db "recorder" "$DATA/recorder.db" "trades_record"    "ts_record"

printf "==============================================\n" | tee -a "$LOG"
echo "" >> "$LOG"

  echo \"\$(date '+%F %T') [AGG3M] Start aggregation\" >> \$LOG
  sqlite3 \$DB '
    DELETE FROM ohlcv_3m;
    INSERT INTO ohlcv_3m (instId, ts, o, h, l, c, v)
    SELECT instId,
           (ts/180000)*180000 AS ts_group,
           first_value(o) OVER w AS o,
           MAX(h) OVER w AS h,
           MIN(l) OVER w AS l,
           last_value(c) OVER w AS c,
           SUM(v) OVER w AS v
    FROM ohlcv_1m
    WINDOW w AS (PARTITION BY instId, (ts/180000) ORDER BY ts ROWS BETWEEN 2 PRECEDING AND CURRENT ROW)
    GROUP BY instId, ts_group;
  '
  echo \"\$(date '+%F %T') [AGG3M] Done\" >> \$LOG
"

    ["1m"]="ohlcv_1m:60"
    ["3m"]="ohlcv_3m:180"
    ["5m"]="ohlcv_5m:300"
)

now_ms=$(($(date +%s) * 1000))

echo "=== LAST OB CANDLES USED ==="
printf "%-3s %-19s %-7s %s\n" "tf" "last_ts" "age_s" "status"
echo "--  -------------------  -------  -------"

for tf in 1m 3m 5m; do
    entry="${TF_TABLES[$tf]}"
    table="${entry%%:*}"
    limit="${entry##*:}"

    last_ts=$(sqlite3 "$DB_OB" "SELECT MAX(ts) FROM $table;")

    if [[ -z "$last_ts" || "$last_ts" == "NULL" ]]; then
        printf "%-3s %-19s %-7s %s\n" "$tf" "---" "---" "❌"
        continue
    fi

    age=$(( (now_ms - last_ts) / 1000 ))
    last_human=$(date -d "@$((last_ts / 1000))" +"%Y-%m-%d %H:%M:%S")

    status="⚠️"
    if (( age < limit )); then
        status="✅"
    fi

    printf "%-3s %-19s %-7d %s\n" "$tf" "$last_human" "$age" "$status"
done

ATTACH DATABASE '"$DB_F"' AS F;

.mode column

WITH sig AS (
  SELECT instId, MAX(ts) AS ts_signal_s
  FROM B.signals_B
  GROUP BY instId
),
open AS (
  SELECT instId, MAX(ts_open) AS ts_open
  FROM O.trades_open_init
  GROUP BY instId
),
closed AS (
  SELECT instId, side, entry, exit, pnl, reason_exit, atr, ts_open, ts_close
  FROM main.positions_closed
)
SELECT
  c.instId,
  c.side,
  c.reason_exit,
  printf('%.4f', c.pnl) AS pnl,
  (c.ts_close - c.ts_open) AS trade_ms,
  CASE WHEN s.ts_signal_s IS NULL THEN NULL
       ELSE (c.ts_open - s.ts_signal_s*1000)
  END AS open_latency_ms
FROM closed c
LEFT JOIN sig s   ON s.instId = c.instId
LEFT JOIN open o  ON o.instId = c.instId AND o.ts_open = c.ts_open
ORDER BY c.ts_close DESC
LIMIT 50;

DETACH DATABASE O;

#!/usr/bin/env python3
import sqlite3, math, time

DB_A = "/opt/scalp/project/data/a.db"
DB_U = "/opt/scalp/project/data/u.db"
DB_OB = "/opt/scalp/project/data/ob.db"
TF = "5m"

def softmax(x):
    e = [math.exp(i) for i in x]
    s = sum(e)
    return [v/s for v in e]

def get_universe():
    con = sqlite3.connect(DB_U)
    rows = con.execute("SELECT instId FROM universe ORDER BY instId").fetchall()
    con.close()
    return [r[0] for r in rows]

def load_ob(inst):
    con = sqlite3.connect(DB_OB)
    row = con.execute(f"SELECT rsi FROM ob_{TF} WHERE instId=? ORDER BY ts DESC LIMIT 1", (inst,)).fetchone()
    con.close()
    return row[0] if row else None

def compute_ctx(rsi):
    if rsi < 30: return "bullish"
    if rsi > 70: return "bearish"
    if 45 <= rsi <= 55: return "range"
    return "none"

def compute_score_A(ctx):
    if ctx == "bullish": return 0.8
    if ctx == "bearish": return -0.8
    if ctx == "range": return 0.0
    return 0.0

def run():
    insts = get_universe()
    now = int(time.time()*1000)
    out = []

    for inst in insts:
        rsi = load_ob(inst)
        if rsi is None:
            continue
        ctx = compute_ctx(rsi)
        score_A = compute_score_A(ctx)
        p_buy, p_sell = softmax([score_A, -score_A])
        out.append((inst, now, 0.0, score_A, p_buy, p_sell, ctx))

    con = sqlite3.connect(DB_A)
    con.execute("DELETE FROM ctx_a_latest")
    con.executemany("INSERT INTO ctx_a_latest(instId,ts,score_U,score_A,p_buy,p_sell,ctx) VALUES (?,?,?,?,?,?,?)", out)
    con.commit()
    con.close()

if __name__ == "__main__":
    run()

#!/usr/bin/env python3
# SCALP - A CTX (contexte directionnel + intensité)
# score_U : ratio volume/volatilité (0→1)
# score_A : force du contexte directionnel (0→1)

import sqlite3
import numpy as np
import pandas as pd
import logging
import sys
from datetime import datetime

# ---------------------------------------
# LOGGING
# ---------------------------------------
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s A_CTX %(levelname)s %(message)s",
    handlers=[
        logging.StreamHandler(sys.stdout),
        logging.FileHandler("/opt/scalp/project/logs/a_ctx.log", mode='a')
    ]
)

# ---------------------------------------
# DB CONNECT
# ---------------------------------------
def connect_db(path):
    conn = sqlite3.connect(path, timeout=30, isolation_level=None)
    conn.execute("PRAGMA journal_mode=WAL;")
    conn.execute("PRAGMA synchronous=NORMAL;")
    return conn

# ---------------------------------------
# LOAD FEAT TABLES
# ---------------------------------------
def load_feat(conn, table):
    q = f"""
        SELECT instId, ts, ema12, ema26, macd, rsi14
        FROM {table}
        ORDER BY instId, ts;
    """
    return pd.read_sql_query(q, conn)

def load_ohlcv_with_volume(conn, table):
    q = f"""
        SELECT instId, ts, close, volume
        FROM {table}
        ORDER BY instId, ts;
    """
    return pd.read_sql_query(q, conn)

# ---------------------------------------
# COMPUTE SCORE_U : volume / volatilité
# ---------------------------------------
def compute_score_U(df_feat, df_ohlcv):
    df = df_feat.merge(df_ohlcv, on=["instId","ts"], how="inner")

    # volatilité : |close - ema26|
    df["volatility"] = (df["close"] - df["ema26"]).abs()

    df["raw_U"] = df["volume"] / (df["volatility"] + 1e-9)

    # NORMALISATION DOUCE
    df["score_U"] = np.tanh(df["raw_U"])

    return df[["instId","ts","score_U"]]

# ---------------------------------------
# SCORE TF DIRECTIONNEL
# ---------------------------------------
def compute_score_tf(df):
    # signaux directionnels simples
    s_ema  = np.tanh((df["ema12"] - df["ema26"]) / (1e-8 + abs(df["ema26"])))
    s_macd = np.tanh(df["macd"])
    s_rsi  = np.tanh((df["rsi14"] - 50) / 50)

    S = 0.5 * s_ema + 0.3 * s_macd + 0.2 * s_rsi
    return float(S)

# ---------------------------------------
# SOFTMAX DIRECTIONNEL
# ---------------------------------------
def softmax_ctx(S):
    e_buy  = np.exp(S)
    e_sell = np.exp(-S)
    e_rng  = 1.0

    Z = e_buy + e_sell + e_rng
    p_buy  = float(e_buy  / Z)
    p_sell = float(e_sell / Z)

    if p_buy >= max(p_sell, 1 - p_buy - p_sell):
        ctx = "bullish"
    elif p_sell >= max(p_buy, 1 - p_buy - p_sell):
        ctx = "bearish"
    else:
        ctx = "range"

    return p_buy, p_sell, ctx

# ---------------------------------------
# MAIN CTX PIPELINE
# ---------------------------------------
def main():
    logging.info("Start A_CTX (light)")

    conn_a  = connect_db("/opt/scalp/project/data/a.db")
    conn_oa = connect_db("/opt/scalp/project/data/oa.db")

    # Load features
    f5  = load_feat(conn_a, "feat_5m")
    f15 = load_feat(conn_a, "feat_15m")
    f30 = load_feat(conn_a, "feat_30m")

    # Load ohlcv for score_U
    ohlcv5 = load_ohlcv_with_volume(conn_oa, "ohlcv_5m")

    # Compute score_U
    dfU = compute_score_U(f5, ohlcv5)  # basé sur TF rapide (5m)
    dfU = dfU.groupby("instId").tail(1).set_index("instId")

    # Latest rows per TF
    f5_latest  = f5.groupby("instId").tail(1).set_index("instId")
    f15_latest = f15.groupby("instId").tail(1).set_index("instId")
    f30_latest = f30.groupby("instId").tail(1).set_index("instId")

    # Clear ctx table
    conn_a.execute("DELETE FROM ctx_a_latest;")

    ts_now = int(datetime.now().timestamp() * 1000)

    insts = sorted(f5_latest.index.unique())
    for inst in insts:
        try:
            r5  = f5_latest.loc[inst]
            r15 = f15_latest.loc[inst]
            r30 = f30_latest.loc[inst]
            u   = dfU.loc[inst]["score_U"]

            # Direction TF
            S5  = compute_score_tf(r5)
            S15 = compute_score_tf(r15)
            S30 = compute_score_tf(r30)

            # Fusion multi-TF
            S = 0.5*S5 + 0.3*S15 + 0.2*S30

            # score_A = force directionnelle (0→1)
            score_A = abs(S)

            # softmax
            p_buy, p_sell, ctx = softmax_ctx(S)

            logging.info(f"{inst} S={S:.4f} scoreU={u:.3f} pbuy={p_buy:.3f} psell={p_sell:.3f} ctx={ctx}")

            conn_a.execute("""
                INSERT INTO ctx_a_latest (instId, ts, score_U, score_A, p_buy, p_sell, ctx)
                VALUES (?, ?, ?, ?, ?, ?, ?)
            """, (inst, ts_now, float(u), float(score_A), p_buy, p_sell, ctx))

        except Exception as e:
            logging.error(f"{inst} error: {e}")

    logging.info("Done.")

if __name__ == "__main__":
    main()

    conn.execute("PRAGMA journal_mode=WAL;")
    conn.execute("PRAGMA synchronous=NORMAL;")
    return conn

# ---------------------------
# Indicators
# ---------------------------
def compute_indicators(df):
    df = df.sort_values("ts")

    df["ema12"] = df.groupby("instId")["close"].transform(lambda x: x.ewm(span=12).mean())
    df["ema26"] = df.groupby("instId")["close"].transform(lambda x: x.ewm(span=26).mean())
    df["macd"]  = df["ema12"] - df["ema26"]

    def rsi(close):
        delta = close.diff()
        gain = delta.clip(lower=0)
        loss = -delta.clip(upper=0)
        avg_gain = gain.rolling(14).mean()
        avg_loss = loss.rolling(14).mean()
        rs = avg_gain / (avg_loss + 1e-9)
        return 100 - (100 / (1 + rs))

    df["rsi14"] = df.groupby("instId")["close"].transform(rsi)
    return df

# ---------------------------
# Load OHLCV
# ---------------------------
def load_tf(conn, table):
    q = f"""
        SELECT instId, ts, open, high, low, close, volume
        FROM {table}
        ORDER BY instId, ts;
    """
    return pd.read_sql_query(q, conn)

# ---------------------------
# Write FEAT
# ---------------------------
def write_features(conn, df, table):
    conn.execute(f"DELETE FROM {table};")
    out = df[["instId", "ts", "ema12", "ema26", "macd", "rsi14"]]
    out.to_sql(table, conn, if_exists="append", index=False)

# ---------------------------
# MAIN
# ---------------------------
def main():
    logging.info("Start A_FEAT...")

    conn_oa = connect_db("/opt/scalp/project/data/oa.db")
    conn_a  = connect_db("/opt/scalp/project/data/a.db")

    TF = {
        "5m":  ("ohlcv_5m",  "feat_5m"),
        "15m": ("ohlcv_15m", "feat_15m"),
        "30m": ("ohlcv_30m", "feat_30m")
    }

        df = load_tf(conn_oa, ohlcv_table)

        if df.empty:
            logging.warning(f"{tf} empty, skip")
            continue

        logging.info(f"Compute indicators {tf}…")
        df = compute_indicators(df)

        logging.info(f"Write {feat_table}…")
        write_features(conn_a, df, feat_table)

    logging.info("A_FEAT completed OK")

    delta = series.diff()

    loss = np.where(delta < 0, -delta, 0)
    roll_up = pd.Series(gain).rolling(period).mean()

    high = df["h"]
    low = df["l"]
    close = df["c"]
    prev_close = close.shift(1)
    tr = pd.concat([
        (high - low),

def ensure_closed_schema():
    con = connect_db(DB_CLOSED)
    con.execute("""
        CREATE TABLE IF NOT EXISTS positions_closed(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            instId TEXT, side TEXT, entry REAL, exit REAL,
            sl REAL, tp REAL, pnl REAL, reason TEXT,
            ts_open INTEGER, ts_close INTEGER
        )
    """)
    con.close()

def migrate_h():
    con = connect_db(DB_H)
    con.execute("""
        CREATE TABLE IF NOT EXISTS score_H(
            instId TEXT PRIMARY KEY,
            avg_pnl REAL,
            winrate REAL,
            trades INTEGER,
            last_update INTEGER
        )
    """)
    con.close()

def compute_score():
    ensure_closed_schema()
    con_c = connect_db(DB_CLOSED)
    rows = con_c.execute("SELECT instId, pnl FROM positions_closed").fetchall()
    con_c.close()
    if not rows:
        logging.info("Aucun trade clos, pas de calcul.")
        return
    data = {}
    for inst,pnl in rows:
        data.setdefault(inst,[]).append(pnl)
    con_h = connect_db(DB_H)
    for inst,vals in data.items():
        avg = statistics.mean(vals)
        winrate = sum(1 for v in vals if v>0)/len(vals)
        con_h.execute("""

            VALUES(?,?,?,?,?)
        """,(inst,avg,winrate,len(vals),int(time.time()*1000)))

#!/usr/bin/env python3
import time

from datetime import datetime

DB_WS = "/opt/scalp/project/data/ws.db"
DB_OA = "/opt/scalp/project/data/oa.db"
DB_U  = "/opt/scalp/project/data/u.db"

TF_MS = {
    "5m":  5  * 60 * 1000,
    "15m": 15 * 60 * 1000,
    "30m": 30 * 60 * 1000
}

WS_TABLE = {
    "5m":  "ws_ohlcv_5m",
    "15m": "ws_ohlcv_15m",
    "30m": "ws_ohlcv_30m"
}

OA_TABLE = {
    "5m":  "ohlcv_5m",
    "15m": "ohlcv_15m",
    "30m": "ohlcv_30m"
}

###############################################################################
# DB helpers
###############################################################################
def conn_ws():
    c = sqlite3.connect(DB_WS, timeout=10)
    c.execute("PRAGMA journal_mode=WAL;")
    return c

def conn_oa():
    c = sqlite3.connect(DB_OA, timeout=10)
    c.execute("PRAGMA journal_mode=WAL;")
    return c

def load_universe():
    conn = sqlite3.connect(DB_U)
    rows = conn.execute("SELECT instId FROM universe ORDER BY instId;").fetchall()
    conn.close()
    return [r[0] for r in rows]

###############################################################################
# WS query
###############################################################################
def get_last_ws_closed(ws_con, instId, tf):
    table = WS_TABLE[tf]
    row = ws_con.execute(
        f"SELECT ts, open, high, low, close, volume FROM {table} WHERE instId=? ORDER BY ts DESC LIMIT 1",
        (instId,)
    ).fetchone()
    return row if row else None

###############################################################################
# CCXT fallback
###############################################################################
def fetch_ccxt(instId, tf, since_ms):
    ex = ccxt.bitget({
        "options":{"defaultType":"swap"},
        "rateLimit":10
    })
    market = instId.replace("/", "")
    tf_ccxt = tf
    return ex.fetch_ohlcv(market, timeframe=tf_ccxt, limit=5, since=since_ms)

###############################################################################
# Insert into OA
###############################################################################
def insert_oa(oa_con, tf, instId, ts, o, h, l, c, v):
    table = OA_TABLE[tf]
    oa_con.execute(
        f"""
        INSERT OR REPLACE INTO {table}(instId, ts, open, high, low, close, volume)
        VALUES (?,?,?,?,?,?,?)
        """,
        (instId, ts, o, h, l, c, v)
    )

###############################################################################
# HYBRID LOGIC
###############################################################################
def process_instrument(instId, ws_con, oa_con):
    now_ms = int(time.time() * 1000)

    for tf in ("5m","15m","30m"):
        tf_size = TF_MS[tf]

        # 1. Get latest WS closed candle
        ws = get_last_ws_closed(ws_con, instId, tf)
        ws_ts = ws[0] if ws else None

        # 2. Get OA last candle
        row = oa_con.execute(
            f"SELECT ts FROM {OA_TABLE[tf]} WHERE instId=? ORDER BY ts DESC LIMIT 1",
            (instId,)
        ).fetchone()
        oa_ts = row[0] if row else 0

        # CASE A: WS is fresh → use WS
        if ws_ts and (now_ms - ws_ts) < (tf_size * 1.5):
            if ws_ts > oa_ts:
                insert_oa(oa_con, tf, instId, ws_ts, *ws[1:])
                print(f"[OA] WS {tf} {instId} ts={ws_ts} applied")
            continue

        # CASE B: WS missing or stale → fallback REST
        since_ms = oa_ts if oa_ts else now_ms - (tf_size * 5)
        ohlcv = fetch_ccxt(instId, tf, since_ms)
        if not ohlcv:
            continue

        for ts, o, h, l, c, v in ohlcv:
            if ts > oa_ts:
                insert_oa(oa_con, tf, instId, ts, o, h, l, c, v)
                print(f"[OA] CCXT {tf} {instId} ts={ts} applied")

###############################################################################
# MAIN
###############################################################################
def main():
    t0 = time.time()
    ws_con = conn_ws()
    oa_con = conn_oa()

    universe = load_universe()
    for instId in universe:
        try:
            process_instrument(instId, ws_con, oa_con)
        except Exception as e:
            print(f"[OA] ERROR {instId}: {e}")

    oa_con.commit()
    ws_con.close()
    oa_con.close()

    print(f"[OA] HYBRID sync completed in {time.time()-t0:.3f}s")

if __name__ == "__main__":
    main()

###############################################################################
# Insert rows
###############################################################################
def insert_row(con, tf, instId, ts, o, h, l, c, v):
    con.execute(
        f"INSERT OR REPLACE INTO {TABLE[tf]}(instId, ts, open, high, low, close, volume)"
        f" VALUES (?,?,?,?,?,?,?)",
        (instId, ts, o, h, l, c, v)
    )

###############################################################################
# Main OA logic
###############################################################################
def sync_inst(con, instId):
    now = int(time.time() * 1000)

    for tf, tf_ms in TF_MS.items():

        row = con.execute(
            f"SELECT ts FROM {TABLE[tf]} WHERE instId=? ORDER BY ts DESC LIMIT 1",
            (instId,)
        ).fetchone()

        last_ts = row[0] if row else (now - tf_ms * 10)

        # Fetch only the missing window
        ohlcv = fetch_missing(instId, tf, last_ts)
        if not ohlcv:
            continue

        for ts, o, h, l, c, v in ohlcv:

            # Normaliser timestamp (sécurité)
            if ts < 10**11:   # seconde → millisecondes
                ts = ts * 1000

            # Validation strict millisecondes
            if ts > now + 3600 * 1000:
                print(f"[OA] WARNING {instId} {tf} future-ts={ts}")
                continue

            if ts > last_ts:
                insert_row(con, tf, instId, ts, o, h, l, c, v)
                print(f"[OA] {instId} {tf} synced ts={ts}")

###############################################################################
# Entrypoint
###############################################################################
def main():
    t0 = time.time()

    universe = load_universe()
    con = conn_oa()

    for instId in universe:
        try:
            sync_inst(con, instId)
        except Exception as e:
            print(f"[OA] ERROR {instId}: {e}")

    con.commit()
    con.close()

    print(f"[OA] Sync OK in {time.time() - t0:.3f}s")

if __name__ == "__main__":
    main()

def aggregate_3m(conn):

    INSERT OR REPLACE INTO ohlcv_3m(instId, ts, o, h, l, c, v)
    SELECT instId,
           (ts/180000)*180000 AS ts,

           LAST_VALUE(c)  OVER w,
           SUM(v)         OVER w
    FROM ohlcv_1m

        ORDER BY ts

# ==========================================================
# FEAT FUNCTIONS
# ==========================================================
def ema(s, span): return s.ewm(span=span, adjust=False).mean()

def rsi(s, period=14):
    d = s.diff()
    up = d.clip(lower=0).ewm(alpha=1/period).mean()
    dn = -d.clip(upper=0).ewm(alpha=1/period).mean()
    rs = up / dn.replace({0: np.nan})
    return 100 - (100/(1+rs))

def atr(df, period=14):
    hl = (df.h - df.l).abs()
    hc = (df.h - df.c.shift()).abs()
    lc = (df.l - df.c.shift()).abs()
    tr = pd.concat([hl, hc, lc], axis=1).max(axis=1)
    return tr.ewm(alpha=1/period, adjust=False).mean()

def macd_full(c, f=12, sl=26, sig=9):
    e1 = ema(c,f)
    e2 = ema(c,sl)
    macd = e1 - e2
    signal = macd.ewm(span=sig, adjust=False).mean()
    hist = macd - signal
    return macd, signal, hist

def bollinger(s, period=20, k=2):
    mid = s.rolling(period).mean()
    std = s.rolling(period).std()
    up = mid + k*std
    low = mid - k*std
    width = up - low
    return mid, std, up, low, width

def _f(x):
    try:
        if pd.isna(x):
            return None
    except:
        pass
    return float(x) if x is not None else None

def build_feat(inst, tf, co, cb):
    t_ob   = f"ohlcv_{tf}"
    t_feat = f"feat_{tf}"

    df = pd.read_sql_query(
        f"SELECT ts,o,h,l,c,v FROM {t_ob} WHERE instId=? ORDER BY ts",
        co, params=(inst,)
    )
    if len(df) < 20:
        return 0

    c = df.c

    df["ema9"]  = ema(c,9)
    df["ema12"] = ema(c,12)
    df["ema21"] = ema(c,21)
    df["ema26"] = ema(c,26)
    df["ema50"] = ema(c,50)

    df["macd"], df["macdsignal"], df["macdhist"] = macd_full(c)
    df["rsi"] = rsi(c)
    df["atr"] = atr(df)

    mid,std,up,low,width = bollinger(c)
    df["bb_mid"]   = mid
    df["bb_std"]   = std
    df["bb_up"]    = up
    df["bb_low"]   = low
    df["bb_width"] = width

    df["mom"]   = c.diff(10)
    df["roc"]   = (c / c.shift(10) - 1) * 100
    df["slope"] = (df["ema12"] - df["ema12"].shift(10)) / 10
    df["ctx"]   = None

    rows = []
    for r in df.itertuples(index=False):
        rows.append((
            inst, int(r.ts),
            _f(r.o), _f(r.h), _f(r.l), _f(r.c), _f(r.v),
            _f(r.ema9), _f(r.ema12), _f(r.ema21), _f(r.ema26), _f(r.ema50),
            _f(r.macd), _f(r.macdsignal), _f(r.macdhist),
            _f(r.rsi), _f(r.atr),
            _f(r.bb_mid), _f(r.bb_std), _f(r.bb_up), _f(r.bb_low), _f(r.bb_width),
            _f(r.mom), _f(r.roc), _f(r.slope),
            None
        ))

    cb.executemany(f"""
    INSERT OR REPLACE INTO {t_feat} (
        instId,ts,
        o,h,l,c,v,
        ema9,ema12,ema21,ema26,ema50,

        rsi,atr,
        bb_mid,bb_std,bb_up,bb_low,bb_width,
        mom,roc,slope,
        ctx
    ) VALUES ({','.join(['?']*26)})
    """, rows)

    return len(rows)

#!/usr/bin/env python3
import asyncio
import json
import websockets
import sqlite3
import time

WS_URL = "wss://ws.bitget.com/v2/ws/public"
DB_U = "/opt/scalp/project/data/u.db"

# Timeframes in ms for identifying CLOSED candles
TF = {
    "candle5m":  5 * 60 * 1000,
    "candle15m": 15 * 60 * 1000,
    "candle30m": 30 * 60 * 1000
}

################################################################
# Load universe from u.db
################################################################
def load_universe():
    conn = sqlite3.connect(DB_U)
    rows = conn.execute("SELECT instId FROM universe ORDER BY instId;").fetchall()
    conn.close()
    return [r[0] for r in rows]


################################################################
# Build subscription message
################################################################
def build_sub(universe):
    args = []
    for inst in universe:
        args.append({"instType":"USDT-FUTURES","channel":"candle5m","instId":inst})
        args.append({"instType":"USDT-FUTURES","channel":"candle15m","instId":inst})
        args.append({"instType":"USDT-FUTURES","channel":"candle30m","instId":inst})
    return {"op":"subscribe","args":args}


################################################################
# Core WS loop
################################################################
async def ws_loop():
    universe = load_universe()
    sub = build_sub(universe)

    while True:
        try:
            print("[WS] Connecting...")
            async with websockets.connect(WS_URL, ping_interval=20) as ws:
                print("[WS] Connected, subscribing…")
                await ws.send(json.dumps(sub))

                while True:
                    raw = await ws.recv()
                    data = json.loads(raw)

                    # Ignore heartbeats
                    if isinstance(data, dict) and data.get("event") in ("ping","pong"):
                        continue

                    # Only candle messages
                    if "arg" not in data or "data" not in data:
                        continue

                    channel = data["arg"].get("channel")
                    inst    = data["arg"].get("instId")
                    rows    = data["data"]

                    if channel not in TF:
                        continue

                    tf_ms = TF[channel]

                    # Process each kline
                    for k in rows:
                        ts     = int(k[0])
                        open_  = float(k[1])
                        high   = float(k[2])
                        low    = float(k[3])
                        close  = float(k[4])
                        volume = float(k[5])

                        # === FILTER: closed candles only ===
                        if ts % tf_ms != 0:
                            continue

                        print(f"[{channel}] CLOSED {inst}  ts={ts}  "
                              f"open={open_} high={high} low={low} "
                              f"close={close} vol={volume}")

        except Exception as e:
            print(f"[WS] Error: {e}, retry in 5s…")
            await asyncio.sleep(5)


################################################################
# Entry point
################################################################
if __name__ == "__main__":
    asyncio.run(ws_loop())

def get_ctx():
    with sqlite3.connect(DB_A) as con:
        cur = con.cursor()
        cur.execute("SELECT instId, ctx FROM v_ctx_latest WHERE ctx IN ('bullish','bearish','range');")
        return cur.fetchall()

def get_last_tick(inst):
    with sqlite3.connect(DB_T) as con:
        cur = con.cursor()
        cur.execute("SELECT lastPr, ts_ms FROM v_ticks_latest WHERE instId=? LIMIT 1;", (inst,))
        r = cur.fetchone()
    if not r:
        return None, None, None, None
    pr, ts_ms = r
    ts_s = ts_ms / 1000
    hms = datetime.fromtimestamp(ts_s).strftime("%H:%M:%S")
    delta = round(time.time() - ts_s, 1)
    return pr, ts_s, hms, delta

def get_last_signal(inst):
    with sqlite3.connect(DB_B) as con:
        cur = con.cursor()
        cur.execute("SELECT ts FROM signals WHERE instId=? ORDER BY ts DESC LIMIT 1;", (inst,))
        r = cur.fetchone()
    if not r:
        return None, None, None
    ts = float(r[0])
    hms = datetime.fromtimestamp(ts).strftime("%H:%M:%S")
    delta = round(time.time() - ts, 1)
    return ts, hms, delta

def dash():
    print(f"===== DASH {datetime.now().strftime('%H:%M:%S')} =====")
    for inst, ctx in get_ctx():
        pr, ts_tick, hms_tick, delta_tick = get_last_tick(inst)
        ts_sig, hms_sig, delta_sig = get_last_signal(inst)

        delta = "-"
        if ts_tick and ts_sig:
            delta = round(ts_tick - ts_sig, 1)

        flag = "⚠️" if delta_tick and delta_tick > 5 else "✅"
        print(f"{inst:<10} | ctx={ctx:<8} | price={pr or '-':<10} | "
              f"tick={hms_tick or '-':>8} | sig={hms_sig or '-':>8} | "
              f"Δ={delta:<6} | {flag}")
    print("=" * 90)

if __name__ == "__main__":

        try:
            dash()

def connect(path):
    c = sqlite3.connect(path, timeout=30, isolation_level=None)

def get_margin_engaged(con_o):
    """Somme de la marge estimée sur les positions ouvertes/pending."""
    try:
        res = con_o.execute("""
            SELECT SUM(entry * qty / leverage)
            FROM trades_open_init
            WHERE status IN ('pending','open');
        """).fetchone()[0]
        return round(res or 0.0, 4)
    except Exception:
        return 0.0

def get_pnl_real(con_r):
    """Somme du PnL réalisé total depuis l’historique."""
    try:
        res = con_r.execute("SELECT SUM(pnl_net) FROM trades_recorded;").fetchone()[0]
        if res is None:
            res = con_r.execute("SELECT SUM(pnl) FROM trades_recorded;").fetchone()[0]
        return round(res or 0.0, 4)
    except Exception:
        return 0.0

def get_capital_initial(con_b):
    """Capital initial du budget (si non encore présent, crée 1000 USDT)."""
    con_b.execute("""
        CREATE TABLE IF NOT EXISTS budget_state(
            id INTEGER PRIMARY KEY CHECK (id=1),
            balance REAL DEFAULT 1000.0,
            margin REAL DEFAULT 0.0,
            pnl_real REAL DEFAULT 0.0,
            ts_update INTEGER
        );
    """)
    con_b.commit()
    row = con_b.execute("SELECT balance FROM budget_state WHERE id=1;").fetchone()
    if not row:
        con_b.execute("INSERT OR REPLACE INTO budget_state(id,balance,margin,pnl_real,ts_update) VALUES (1,1000.0,0.0,0.0,strftime('%s','now'));")
        con_b.commit()
        return 1000.0
    return float(row[0])

def update_budget():
    con_b = connect(DB_BUDGET)
    con_o = connect(DB_OPENER)
    con_r = connect(DB_REC)

    capital = get_capital_initial(con_b)
    margin  = get_margin_engaged(con_o)
    pnl_real = get_pnl_real(con_r)

    ts_now = int(time.time())

    con_b.execute("""

def finalize_closures():
    con_f = sqlite3.connect(DB_FOLLOW)
    con_c = sqlite3.connect(DB_CLOSER)
    con_r = sqlite3.connect(DB_REC)
    con_b = sqlite3.connect(DB_BUDGET)

    cur_f = con_f.cursor()
    cur_c = con_c.cursor()
    cur_r = con_r.cursor()
    cur_b = con_b.cursor()

               score_U, score_A, score_B, ts_open, ts_update
        FROM trades_follow
        WHERE status='to_close'
    """).fetchall()

        logging.info("Aucune position à finaliser.")
        return

    for instId, side, entry, last, qty, reason, sU, sA, sB, ts_open, ts_close in rows:

            last = get_last_price(instId)
            if last is None:

            else:
                logging.info(f"{instId}: récupéré last depuis ticks = {last}")

        """, (instId,side,entry,last,pnl,reason,ts_open,ts_close,sU,sA,sB))

    con_b.commit()

def conn(path):

    c.execute("PRAGMA busy_timeout=3000;")
    return c


def get_last_price(inst):
    ct = conn(DB_T)
    r = ct.execute(
        "SELECT price FROM v_ticks_latest WHERE instId=? LIMIT 1;",
        (inst.replace("/", ""),)
    ).fetchone()
    ct.close()
    return r[0] if r else None


def compute_risk(price, atr, side):
    if price <= 0 or atr <= 0:
        return None
    sl = price - 1.5 * atr if side == "buy" else price + 1.5 * atr
    tp = price + 3.0 * atr if side == "buy" else price - 3.0 * atr
    qty = 1.0
    lev = 5.0
    return sl, tp, qty, lev


def load_new():
    cg = conn(DB_G)
    rows = cg.execute("""
        SELECT trade_uid, instId, side, price, atr, ts_signal
        FROM v_for_opener
    """).fetchall()
    cg.close()
    return rows


def store(uid, inst, side, entry, sl, tp, qty, lev, ts_signal):
    co = conn(DB_O)
    ts = int(time.time()*1000)

    co.execute("""
        INSERT OR REPLACE INTO trades_open
        (uid, instId, side, entry, sl, tp, qty, leverage, ts_signal, ts_open)
        VALUES (?,?,?,?,?,?,?,?,?,?)
    """, (uid, inst, side, entry, sl, tp, qty, lev, ts_signal, ts))

    co.close()
    return ts


def main():
    log.info("OPENER started")

    while True:
        try:
            rows = load_new()

def build_signal(trg):
    inst, side, reason, score_B, ts, atr_trig = trg

    sig = (
        uid, inst, side, reason, ctx,
        float(price), float(atr),
        0, 0, float(score_B),
        int(time.time()*1000)

#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import asyncio
import json
import sqlite3
import threading
import time
from queue import Queue
import websockets

DB_PATH = "/opt/scalp/project/data/t.db"
DB_A_PATH = "/opt/scalp/project/data/a.db"
WS_URL = "wss://ws.bitget.com/mix/v1/stream"
FLUSH_INTERVAL = 0.5  # secondes
QUEUE_MAX = 5000

q = Queue(maxsize=QUEUE_MAX)
lock = threading.Lock()
stop_event = threading.Event()

# --- Connexion SQLite optimisée ---
def init_db():
    conn = sqlite3.connect(DB_PATH, check_same_thread=False, timeout=10)
    cur = conn.cursor()
    cur.executescript("""
        PRAGMA journal_mode=WAL;
        PRAGMA synchronous=OFF;
        CREATE TABLE IF NOT EXISTS ticks(
            instId TEXT PRIMARY KEY,
            price REAL,
            ts_ms INTEGER
        );
        CREATE INDEX IF NOT EXISTS idx_ticks_ts ON ticks(ts_ms);
    """)
    conn.commit()
    return conn

# --- Récupération des symboles depuis le contexte A ---
def get_symbols():
    conn = sqlite3.connect(DB_A_PATH)
    cur = conn.cursor()
    cur.execute("SELECT instId FROM v_ctx_latest;")
    symbols = [r[0].replace("/", "") for r in cur.fetchall()]
    conn.close()
    return symbols

# --- Thread d'écriture SQLite ---
def writer_thread():
    conn = init_db()
    cur = conn.cursor()
    last_flush = 0
    buffer = []
    while not stop_event.is_set():
        try:
            tick = q.get(timeout=FLUSH_INTERVAL)
            buffer.append(tick)
        except:
            pass

        now = time.time()
        if now - last_flush >= FLUSH_INTERVAL and buffer:
            with lock:
                cur.executemany(
                    "INSERT INTO ticks (instId, price, ts_ms) VALUES (?, ?, ?) "
                    "ON CONFLICT(instId) DO UPDATE SET price=excluded.price, ts_ms=excluded.ts_ms;",
                    buffer,
                )
                conn.commit()
                buffer.clear()
                last_flush = now

    conn.commit()
    conn.close()

# --- Fonction WS principale ---
async def ws_consumer():
    symbols = get_symbols()
    args = [
        {"instType": "USDT-FUTURES", "channel": "ticker", "instId": s}
        for s in symbols
    ]
    sub_msg = json.dumps({"op": "subscribe", "args": args})
    print(f"[ticks] Subscribing to {len(symbols)} symbols...")

    async with websockets.connect(WS_URL, ping_interval=15, ping_timeout=10) as ws:
        await ws.send(sub_msg)
        print("[ticks] Connected & subscribed.")
        async for msg in ws:
            data = json.loads(msg)
            if "data" not in data:
                continue
            for d in data["data"]:
                try:
                    instId = d["instId"]
                    price = float(d["lastPr"])
                    ts_ms = int(d["ts"])
                    if not q.full():
                        q.put((instId, price, ts_ms))
                except Exception as e:
                    pass

# --- Main ---
def main():
    wt = threading.Thread(target=writer_thread, daemon=True)
    wt.start()

    try:
        asyncio.run(ws_consumer())
    except KeyboardInterrupt:
        print("Stopping...")
    finally:
        stop_event.set()
        wt.join()

if __name__ == "__main__":
    main()

#!/usr/bin/env python3
# SCALP - TRIGGERS MODULE (V5 HYBRIDE PRO)
# Features 1m + Validation Tick + Filtre Expert
# Compatible SCALP guidelines (U/OA/A/OB/T/B/X + gest bus)

import sqlite3
import time
import logging

DB_B   = "/opt/scalp/project/data/b.db"
DB_T   = "/opt/scalp/project/data/t.db"
DB_TRG = "/opt/scalp/project/data/triggers.db"

LOOP_SLEEP   = 1.0
COOLDOWN_MS  = 90_000
BOOST_FACTOR = 1.20   # +20% pour re-déclencher

logging.basicConfig(
    filename="/opt/scalp/project/logs/triggers.log",
    level=logging.INFO,
    format="%(asctime)s TRIG %(levelname)s %(message)s"
)

def connect(db):
    conn = sqlite3.connect(db, timeout=5, isolation_level=None)
    conn.row_factory = sqlite3.Row
    return conn

def normalize(x, low, high):
    if x is None:
        return 0.0
    if x <= low:
        return 0.0
    if x >= high:
        return 1.0
    return (x - low) / (high - low)

# -------------------------------------------------------
# LOAD FEATURES
# -------------------------------------------------------
def load_feat(instId):
    conn = connect(DB_B)
    q = """
        SELECT 
            ema12, ema26,
            macd, macdsignal,
            atr, bb_width, 
            mom, roc, slope,
            c AS price
        FROM feat_1m
        WHERE instId=?
        ORDER BY ts DESC
        LIMIT 1;
    """
    row = conn.execute(q, (instId,)).fetchone()
    conn.close()
    return row

# -------------------------------------------------------
# LOAD LAST TICK  (AAVE/USDT → AAVEUSDT)
# -------------------------------------------------------
def load_tick(instId):
    tick_id = instId.replace("/", "")
    conn = connect(DB_T)
    row = conn.execute(
        "SELECT lastPr FROM v_ticks_latest WHERE instId=? LIMIT 1;",
        (tick_id,)
    ).fetchone()
    conn.close()
    return row["lastPr"] if row else None

# -------------------------------------------------------
# DETECT (features only)
# -------------------------------------------------------
def detect(inst, f):
    if f is None:
        return None

    price = f["price"]
    if not price or price <= 0:
        return None

    bb     = f["bb_width"] or 0.0
    atr    = f["atr"] or 0.0
    bbw    = bb / price
    slope  = f["slope"] or 0.0
    roc    = f["roc"] or 0.0
    mom    = f["mom"] or 0.0
    ema12  = f["ema12"]
    ema26  = f["ema26"]
    macd   = f["macd"]
    macds  = f["macdsignal"]

    # EXPANSION
    if bbw > 0.007:
        score = normalize(bbw, 0.005, 0.02)
        side  = "buy" if mom > 0 else "sell"
        return ("VOLATILITY_EXPANSION", side, score, price, atr)

    # SQUEEZE
    if bbw < 0.002:
        score = normalize(0.003 - bbw, 0.0, 0.003)
        side  = "buy" if macd > macds else "sell"
        return ("VOLATILITY_SQUEEZE", side, score, price, atr)

    # MOMENTUM
    if abs(roc) > 0.30:
        score = normalize(abs(roc), 0.30, 1.0)
        side = "buy" if roc > 0 else "sell"
        return ("MOMENTUM", side, score, price, atr)

    # BREAKOUT
    if abs(slope) > 0.0015:
        score = normalize(abs(slope), 0.0015, 0.008)
        side  = "buy" if slope > 0 else "sell"
        return ("BREAKOUT", side, score, price, atr)

    # EMA SHIFT
    if ema12 and ema26:
        diff = (ema12 - ema26) / price
        if abs(diff) > 0.002:
            score = normalize(abs(diff), 0.002, 0.01)
            side  = "buy" if diff > 0 else "sell"
            return ("EMA_TREND_SHIFT", side, score, price, atr)

    return None

# -------------------------------------------------------
# TICK VALIDATION
# -------------------------------------------------------
def validate_tick(reason, side, score, price, atr, tick_price):
    if tick_price is None:
        return False

    # BREAKOUT
    if reason == "BREAKOUT":
        if side == "buy"  and tick_price > price * 1.0004:
            return True
        if side == "sell" and tick_price < price * 0.9996:
            return True
        return False

    # MOMENTUM
    if reason == "MOMENTUM":
        return abs(tick_price - price) > atr * 0.50

    # SQUEEZE
    if reason == "VOLATILITY_SQUEEZE":
        return abs(tick_price - price) < atr * 0.20

    # EXPANSION
    if reason == "VOLATILITY_EXPANSION":
        return abs(tick_price - price) > atr * 0.80

    # EMA SHIFT
    if reason == "EMA_TREND_SHIFT":
        if side == "buy"  and tick_price > price * 1.0002:
            return True
        if side == "sell" and tick_price < price * 0.9998:
            return True
        return False

    return False

# -------------------------------------------------------
# LOAD / SAVE STATE
# -------------------------------------------------------
def load_state(inst):
    conn = connect(DB_TRG)
    row = conn.execute(
        "SELECT reason, side, score, ts FROM trig_state WHERE instId=?",
        (inst,)
    ).fetchone()
    conn.close()
    return row

def save_state(inst, reason, side, score, ts):
    conn = connect(DB_TRG)
    conn.execute("""
        INSERT INTO trig_state(instId, reason, side, score, ts)
        VALUES(?,?,?,?,?)
        ON CONFLICT(instId)
        DO UPDATE SET reason=excluded.reason,
                      side=excluded.side,
                      score=excluded.score,
                      ts=excluded.ts;
    """, (inst, reason, side, score, ts))
    conn.close()

# -------------------------------------------------------
# SHOULD EMIT?
# -------------------------------------------------------
def should_emit(inst, reason, side, score):
    prev = load_state(inst)
    now  = int(time.time() * 1000)

    if prev is None:
        return True
    if reason != prev["reason"]:
        return True
    if side != prev["side"]:
        return True
    if score > prev["score"] * BOOST_FACTOR:
        return True
    if now - prev["ts"] >= COOLDOWN_MS:
        return True
    return False

# -------------------------------------------------------
# INSERT TRIGGER
# -------------------------------------------------------
def push_trigger(inst, reason, side, score, price, atr):
    ts = int(time.time() * 1000)
    conn = connect(DB_TRG)
    conn.execute("""
        INSERT INTO triggers(
            instId, tf, side, reason, score,
            ctx, confidence, price, atr, ts, consumed
        )
        VALUES(?, '1m', ?, ?, ?, NULL, ?, ?, ?, ?, 0)
    """, (inst, side, reason, score, score, price, atr, ts))
    conn.close()

    logging.info(f"TRIGGER {inst} {side} {reason} score={score:.3f}")
    save_state(inst, reason, side, score, ts)

def main():

    SELECT instId, MAX(ts) AS ts_last
    FROM triggers
    GROUP BY instId
)

    t.instId,
    t.side,
    t.reason,
    t.score   AS score_B,
    t.ctx,

JOIN last_ts lt
  ON lt.instId = t.instId AND lt.ts_last = t.ts
ORDER BY t.ts DESC
