import time, sqlite3, math, json
ADB="data/a.db"
# pondérations TF
W5, W15, W30 = 0.2, 0.3, 0.5
TAU=1.5 # température softmax

def ema(vals, n):
    k=2/(n+1)
    e=None
    for v in vals:
        e = v if e is None else (v*k + e*(1-k))
    return e if e is not None else 0.0

def load_close(conn, table, instId, n=50):
    rows=list(conn.execute(f"SELECT c FROM {table} WHERE instId=? ORDER BY ts DESC LIMIT ?", (instId,n)))
    return [r[0] for r in rows][::-1]

def score_ctx(cs):
    # signal scalaire par TF via pentes EMA
    if len(cs)<20: return 0.0
    s_fast = ema(cs[-10:], 5)
    s_slow = ema(cs[-20:],10)
    d = (s_fast - s_slow)/max(1e-9, s_slow)
    return d

def softmax3(x_buy, x_hold, x_sell, tau=TAU):
    exps=[math.exp(x/tau) for x in (x_buy, x_hold, x_sell)]
    s=sum(exps); 
    return [e/s for e in exps]

def decide(inst, c5, c15, c30):
    s5  = score_ctx(c5)
    s15 = score_ctx(c15)
    s30 = score_ctx(c30)
    x = W5*s5 + W15*s15 + W30*s30
    # map vers 3 classes: buy (x), hold (-|x|), sell (-x)
    p_buy, p_hold, p_sell = softmax3(x, -abs(x)*0.5, -x)
    ctx = "buy" if p_buy>max(p_hold,p_sell) else ("sell" if p_sell>max(p_buy,p_hold) else ("hold" if max(p_buy,p_sell)<0.45 else "none"))
    score_A = (p_buy - p_sell) # [-1..1] centré
    return ctx, p_buy, p_hold, p_sell, score_A

def run():
    with sqlite3.connect(ADB) as c:
        c.execute("PRAGMA busy_timeout=8000")
        syms=[r[0] for r in c.execute("SELECT DISTINCT instId FROM ohlcv_5m")]
        now=int(time.time())

            c30 = load_close(c,"ohlcv_30m",s, 50)

            ctx,pb,ph,ps,scoreA = decide(s,c5,c15,c30)

                         VALUES(?,?,?,?,?,?,?,?,?)""",(s,ts,ctx,pb,ph,ps,scoreA,det,now))
        c.commit()

def ensure_schema(c):
    c.execute("""CREATE TABLE IF NOT EXISTS ctx_A(
        instId TEXT PRIMARY KEY,
        ctx TEXT,
        updated_ts INTEGER
    );""")

def get_closes(conn, sym, n=3):
    rows = [r[0] for r in conn.execute(
        "SELECT close FROM ohlcv_5m WHERE instId=? ORDER BY ts DESC LIMIT ?",
        (sym, n)
    )]
    return rows if len(rows) == n else None

def context_from_closes(closes):
    if not closes: return "none"
    a, b = closes[0], closes[-1]
    if b > a * 1.002: return "buy"
    if b < a * 0.998: return "sell"
    return "hold"

def compute_score_A(closes):
    if not closes or closes[0] == 0: return 0.0
    rel = (closes[-1] - closes[0]) / closes[0]
    score = min(1.0, max(0.0, abs(rel) * 50))  # ~2% => 1.0
    return round(score, 3)

now = int(time.time())

        score_A = compute_score_A(closes)

#!/usr/bin/env python3
import asyncio, websockets, ujson as json, sqlite3, time, sys, os
sys.stdout.reconfigure(line_buffering=True)

ROOT = "/opt/scalp/project"
UDB  = f"{ROOT}/data/u.db"
TDB  = f"{ROOT}/data/t.db"
BDB  = f"{ROOT}/data/b.db"
URI  = "wss://ws.bitget.com/mix/v1/stream?compress=false"

TTL_SEC        = 60          # durée de conservation des ticks
FLUSH_INTERVAL = 5           # flush toutes les 5s
GROUP_SIZE     = 10          # nb de paires par WS
BATCH_MAX      = 500         # taille du buffer avant flush

CREATE_SQL = """
CREATE TABLE IF NOT EXISTS ticks (
    instId TEXT PRIMARY KEY,
    lastPr REAL,
    bidPr  REAL,
    askPr  REAL,
    bidSz  REAL,
    askSz  REAL,
    ts_ms  INTEGER
);
"""

def push_to_B():
    """Copie atomique vers b.db"""
    try:
        with sqlite3.connect(TDB, timeout=5) as ct, sqlite3.connect(BDB, timeout=5) as cb:
            cb.execute("""
            CREATE TABLE IF NOT EXISTS ticks_live (
                instId TEXT PRIMARY KEY,
                lastPr REAL,
                bidPr  REAL,
                askPr  REAL,
                bidSz  REAL,
                askSz  REAL,
                ts_ms  INTEGER
            );
            """)
            cb.execute("DELETE FROM ticks_live;")
            for row in ct.execute("SELECT * FROM ticks;"):
                cb.execute("""
                INSERT OR REPLACE INTO ticks_live
                (instId,lastPr,bidPr,askPr,bidSz,askSz,ts_ms)
                VALUES(?,?,?,?,?,?,?);
                """, row)
            cb.commit()
        print(f"T: pushed to b.db @ {time.strftime('%H:%M:%S')}")
    except Exception as e:
        print(f"T: push->B error: {e}")

def save_ticks(batch):
    """Écrit les ticks en lot et purge les anciens"""
    if not batch:
        return
    try:
        with sqlite3.connect(TDB, timeout=5) as c:
            c.execute("PRAGMA busy_timeout=5000")
            c.execute(CREATE_SQL)
            c.executemany("""
            INSERT OR REPLACE INTO ticks(instId,lastPr,bidPr,askPr,bidSz,askSz,ts_ms)
            VALUES(:instId,:lastPr,:bidPr,:askPr,:bidSz,:askSz,:ts_ms);
            """, batch)
            cutoff = int(time.time()*1000) - TTL_SEC*1000
            c.execute("DELETE FROM ticks WHERE ts_ms < ?", (cutoff,))
            c.commit()
        push_to_B()
    except Exception as e:
        print(f"T: save error: {e}")

async def get_symbols():
    with sqlite3.connect(UDB) as c:
        return [r[0] for r in c.execute("SELECT instId FROM u_syms ORDER BY rank;")]

async def handle_socket(subs):
    payload = {"op":"subscribe","args":subs}
    buffer, last_flush = [], time.time()
    async with websockets.connect(URI, ping_interval=20, ping_timeout=20) as ws:
        await ws.send(json.dumps(payload))
        print(f"T: connected ({len(subs)} pairs)")
        async for msg in ws:
            try:
                data = json.loads(msg)
            except:
                continue
            if isinstance(data, dict) and "event" in data:
                if data["event"] == "error":
                    print("T: evt error", data)
                continue
            if "arg" not in data or "data" not in data:
                continue
            try:
                d = data["data"][0]
                tick = dict(
                    instId=data["arg"]["instId"],
                    lastPr=float(d.get("last",0)),
                    bidPr=float(d.get("bestBid",0)),
                    askPr=float(d.get("bestAsk",0)),
                    bidSz=float(d.get("bidSz",0)),
                    askSz=float(d.get("bestAskSz",0)),
                    ts_ms=int(d.get("ts",0)),
                )
                buffer.append(tick)
            except Exception as e:
                print("T: parse error:", e)
            now = time.time()
            if now - last_flush >= FLUSH_INTERVAL or len(buffer) >= BATCH_MAX:
                print(f"T: flush {len(buffer)} ticks @ {time.strftime('%H:%M:%S')}")
                save_ticks(buffer)
                buffer.clear()
                last_flush = now

async def main():
    with sqlite3.connect(TDB) as c:
        c.execute(CREATE_SQL)
    symbols = await get_symbols()
    if not symbols:
        print("T: no symbols found")
        return
    groups = [symbols[i:i+GROUP_SIZE] for i in range(0, len(symbols), GROUP_SIZE)]
    while True:
        try:
            await asyncio.gather(*[
                handle_socket([{"instType":"mc","channel":"ticker","instId":s} for s in g])
                for g in groups
            ])
        except Exception as e:
            print(f"T: reconnect all sockets ({e})")
            await asyncio.sleep(5)

if __name__ == "__main__":
    asyncio.run(main())

import time, sqlite3, json, math

ADB="data/a.db"; ODB="data/ob.db"; TDB="data/t.db"; BDB="data/b.db"

def ema(vals, n):
    k=2/(n+1); e=None
    for v in vals:
        e=v if e is None else (v*k + e*(1-k))
    return e if e is not None else 0.0

def slope(vals, n=10):
    if len(vals)<n+1: return 0.0
    a=ema(vals[-n-1:-1], n//2 or 1)
    b=ema(vals[-n:],   n//2 or 1)
    return (b-a)/max(1e-9,a)

def last_closes(conn, table, inst, n):

async def ws_check_one(sym: str, timeout: float = 1.5) -> bool:
    payload = {"op": "subscribe", "args": [{"instType": "mc", "channel": "ticker", "instId": sym}]}
    try:
        async with websockets.connect(WS_URI, ping_interval=20, ping_timeout=20, close_timeout=1) as ws:
            await ws.send(json.dumps(payload))
            msg = await asyncio.wait_for(ws.recv(), timeout=timeout)
            if '"event":"error"' in msg or ('"code":' in msg and '"code":0' not in msg):
                return False
            return True

async def ws_check_batch(symbols, max_conc: int = 8) -> dict:
    sem = asyncio.Semaphore(max_conc)

    tasks = [wrapped(s) for s in symbols]
    out = {}
    for coro in asyncio.as_completed(tasks):

        out[sym] = ok
    return out

BDB="data/b.db"; XDB="data/x.db"; TDB="data/t.db"

def acct_get(k,default):
    with sqlite3.connect(XDB) as c:
        r=c.execute("SELECT v FROM acct WHERE k=?",(k,)).fetchone()
        return float(r[0]) if r else default

def acct_set(k,v):
    with sqlite3.connect(XDB) as c:
        c.execute("INSERT OR REPLACE INTO acct(k,v) VALUES(?,?)",(k,str(v)))
        c.commit()

def last_signal():
    with sqlite3.connect(BDB) as c:
        r=c.execute("SELECT instId,ts_ms,side,strength,ctx_score FROM signals_B ORDER BY ts_ms DESC LIMIT 20").fetchall()
        return r or []

def last_price(inst):
    with sqlite3.connect(TDB) as c:
        r=c.execute("SELECT lastPr FROM ticks_latest WHERE instId=?",(inst,)).fetchone()
        return float(r[0]) if r else None

def open_position(inst, side, px, budget_usdt):
    qty = budget_usdt/max(1e-9, px)
    ts=int(time.time()*1000)
    with sqlite3.connect(XDB) as c:
        c.execute("""INSERT INTO orders_sim(instId,side,px_open,qty,ts_open,sl,tp,plan_ref)
                     VALUES(?,?,?,?,?,?,?,?)""",(inst,side,px,qty,ts,None,None,None))
        oid=c.execute("SELECT last_insert_rowid()").fetchone()[0]
        c.execute("""INSERT OR REPLACE INTO positions_open(order_id,instId,side,px_open,qty,tp,sl,ts_open)
                     VALUES(?,?,?,?,?,?,?,?)""",(oid,inst,side,px,qty,None,None,ts))
        used=acct_get("used_budget_usdt",0)
        acct_set("used_budget_usdt", used+budget_usdt)
        c.commit()

def maybe_close():
    with sqlite3.connect(XDB) as c:
        pos=c.execute("SELECT order_id,instId,side,px_open,qty,ts_open FROM positions_open").fetchall()
    for oid,inst,side,px_open,qty,ts_open in pos:
        px = last_price(inst)
        if px is None: continue
        gain = (px - px_open) * (qty if side=="buy" else -qty)
        pnl = gain
        age = (int(time.time()*1000)-ts_open)/60000.0
        if abs((px/px_open)-1)>=0.004 or age>=10:
            with sqlite3.connect(XDB) as c:
                c.execute("DELETE FROM positions_open WHERE order_id=?", (oid,))
                c.execute("""INSERT INTO positions_closed(order_id,instId,side,px_open,px_close,qty,pnl_usdt,ts_open,ts_close,reason)
                             VALUES(?,?,?,?,?,?,?,?,?,?)""",(oid,inst,side,px_open,px,qty,pnl,ts_open,int(time.time()*1000),"rule"))
                used=acct_get("used_budget_usdt",0)
                acct_set("used_budget_usdt", max(0.0, used-px_open*qty))
                c.commit()

def loop():
    while True:
        base=acct_get("base_budget_usdt",1000)
        used=acct_get("used_budget_usdt",0)
        free=max(0.0, base-used)
        per_trade = max(5.0, base*0.02)
        if free>=per_trade:
            for inst,ts,side,strength,ctx_score in last_signal():
                px=last_price(inst)
                if px is None: continue
                open_position(inst, side, px, per_trade)
                break
        maybe_close()
        print("X: ok")
        time.sleep(10)

if __name__=="__main__":
    loop()

import sys, sqlite3, time
OADB = sys.argv[1]
TFs = {"5m":300, "15m":900, "30m":1800}

with sqlite3.connect(OADB, isolation_level=None) as co:  # autocommit
    co.execute("PRAGMA journal_mode=WAL;")
    for tf, step in TFs.items():
        tbl = f"ohlcv_{tf}"
        ok = co.execute("SELECT name FROM sqlite_master WHERE type='table' AND name=?;", (tbl,)).fetchone()
        if not ok:
            continue

        # supprimer les lignes anciennes
        co.execute(f"""
            DELETE FROM {tbl}
            WHERE ts NOT IN (
                SELECT ts FROM {tbl} AS t2
                WHERE t2.instId={tbl}.instId
                ORDER BY t2.ts DESC
                LIMIT 150
            );
        """)

    # vacuum global (hors transaction)
    co.isolation_level = None
    co.execute("VACUUM;")

    # mise à jour health
    for tf, step in TFs.items():
        tbl = f"ohlcv_{tf}"
        row = co.execute(f"SELECT MAX(ts) FROM {tbl};").fetchone()
        last_ts = row[0] if row and row[0] else 0
        age = max(0, int(time.time()) - int(last_ts))
        n = co.execute(f"SELECT COUNT(*) FROM {tbl};").fetchone()[0]
        co.execute("""REPLACE INTO health(module,rows,last_ts,age_s,status,updated_local)
                      VALUES(?,?,?,?,?,datetime('now','localtime'))""",

#!/usr/bin/env bash
set -euo pipefail
ROOT=/opt/scalp/project
DATA="$ROOT/data"
now=$(date +%s)

fmt_age() {
  local diff=${1:-0}
  [ "$diff" -lt 0 ] && diff=0
  printf "%02dh%02dm%02ds" $((diff/3600)) $(((diff%3600)/60)) $((diff%60))
}

has_table() {
  local db="$1" tbl="$2"
  sqlite3 "$db" "SELECT name FROM sqlite_master WHERE type='table' AND name='$tbl';" | grep -q "^$tbl$"
}

echo "===== HEALTH $(date '+%d/%m %H:%M:%S') ====="

# --- U ---
if [ -f "$DATA/u.db" ]; then
  coins=$(sqlite3 "$DATA/u.db" "SELECT COUNT(*) FROM u_syms;")
  last_ts=$(sqlite3 "$DATA/u.db" "SELECT COALESCE(MAX(updated_ts),0) FROM u_syms;")
  age=$(( now - last_ts ))
  status="✅"; [ "$age" -gt 7200 ] && status="⚠️"
  printf "U       | coins=%-3s | age=%s  | %s\n" "$coins" "$(fmt_age "$age")" "$status"
else
  echo "U       | ❌ no data"
fi

# --- OA (5m/15m/30m) ---
oa_coins=0
if [ -f "$DATA/oa.db" ]; then
  for tf in 5m 15m 30m; do
    tbl="ohlcv_${tf}"
    if has_table "$DATA/oa.db" "$tbl"; then
      coins=$(sqlite3 "$DATA/oa.db" "SELECT COUNT(DISTINCT instId) FROM $tbl;")
      [ "$tf" = "5m" ] && oa_coins="$coins"
      last_ts=$(sqlite3 "$DATA/oa.db" "SELECT COALESCE(MAX(ts),0) FROM $tbl;")
      age=$(( now - last_ts ))
      case "$tf" in
        5m)  thresh=$((600)) ;;
        15m) thresh=$((1800)) ;;
        30m) thresh=$((3600)) ;;
      esac
      status="✅"; [ "$age" -gt "$thresh" ] && status="⚠️"
      printf "OA-%-4s | coins=%-3s | age=%s  | %s\n" "$tf" "$coins" "$(fmt_age "$age")" "$status"
    else
      printf "OA-%-4s | ❌ no data\n" "$tf"
    fi
  done
else
  echo "OA      | ❌ no data"
fi

# --- A ---
if [ -f "$DATA/a.db" ]; then
  src_table=""
  for t in ctx_A v_ctx_latest v_contexts_A_latest; do
    if has_table "$DATA/a.db" "$t"; then src_table="$t"; break; fi
  done
  if [ -n "$src_table" ]; then
    read buy hold sell none total last_ts <<<$(sqlite3 -separator ' ' "$DATA/a.db" "
      SELECT
        IFNULL(SUM(ctx='buy'),0),
        IFNULL(SUM(ctx='hold'),0),
        IFNULL(SUM(ctx='sell'),0),
        IFNULL(SUM(ctx='none'),0),
        IFNULL(COUNT(*),0),
        IFNULL(MAX(updated_ts),0)
      FROM $src_table;
    ")
    : "${buy:=0}" "${hold:=0}" "${sell:=0}" "${none:=0}" "${total:=0}" "${last_ts:=0}"
    age=$(( now - last_ts ))
    match="✅"; [ "$total" -ne "${oa_coins:-0}" ] && match="⚠️"
    printf "A       | buy=%-2s hold=%-2s sell=%-2s none=%-2s total=%-3s | age=%s  | %s\n" \
      "$buy" "$hold" "$sell" "$none" "$total" "$(fmt_age "$age")" "$match"
  else
    echo "A       | ❌ no context table"
  fi
else
  echo "A       | ❌ no data"
fi

# --- OB / T / B / X ---
for m in OB T B X; do
  echo "$(printf '%-7s' "$m")| ❌ no data"
done

echo "==================================="
