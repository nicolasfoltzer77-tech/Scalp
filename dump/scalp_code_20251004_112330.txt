#!/usr/bin/env python3
import ccxt, json, time
from pathlib import Path

DATA = Path("/opt/scalp/data")
CACHE90 = DATA / "top90_active.json"
INACTIVE = DATA / "coins_90d.json"

EXCLUDE_LEV = ("UP","DOWN","BULL","BEAR","3L","3S","5L","5S")
EXCLUDE_STABLES = {"USDT","USDC","FDUSD","TUSD","DAI","EUR","TRY","USD"}

def load_usdt_swaps():
    ex = ccxt.bitget({'options': {'defaultType':'swap'}, 'enableRateLimit': True})
    ex.load_markets()
    mapping = {}
    for sym, m in ex.markets.items():
        if not m.get('swap'): continue
        if m.get('quote') != 'USDT': continue
        base = (m.get('base') or '').upper()
        if base in EXCLUDE_STABLES: continue
        if any(tag in base for tag in EXCLUDE_LEV): continue
        mapping[base] = sym
    return ex, mapping

def has_2_daily_candles_around(ex, market, days=90, tol=3):
    now_ms = int(time.time()*1000)
    day = 24*3600*1000
    since = now_ms - (days + tol)*day
    rows = ex.fetch_ohlcv(market, '1d', since=since, limit=10)
    return len(rows) >= 2

def main():
    DATA.mkdir(parents=True, exist_ok=True)
    ex, mapping = load_usdt_swaps()
    actives, inact = [], []
    for base, mkt in mapping.items():
        ok = False
        try:
            ok = has_2_daily_candles_around(ex, mkt, 90, 3)
        except Exception:
            ok = False
        if ok:
            actives.append(base)
        else:
            inact.append({"symbol": base, "reason": "no_1d_90d_window"})
    actives = sorted(dict.fromkeys(actives))
    CACHE90.write_text(json.dumps(actives, indent=2))
    INACTIVE.write_text(json.dumps(inact, indent=2))
    print(f"ACTIVE={len(actives)} INACTIVE={len(inact)}")
if __name__ == "__main__":
    main()

import ccxt, time
from typing import Dict, Any, Optional
from scalp.utils.env import get
from scalp.utils.log import setup
log = setup("bitget")

def make_client() -> ccxt.bitget:
    params = {
        "apiKey": get("SCALP_API_KEY") or get("BITGET_API_KEY"),
        "secret": get("SCALP_API_SECRET") or get("BITGET_API_SECRET"),
        "password": get("SCALP_API_PASSWORD") or get("BITGET_PASSPHRASE") or get("BITGET_PASSWORD"),
        "enableRateLimit": True,
        "options": {"defaultType": "swap"},
    }
    ex = ccxt.bitget(params)
    if (get("SCALP_ENV","paper").lower() != "live"):
        ex.set_sandbox_mode(True)
    return ex

def _resolve_symbol(ex: ccxt.bitget, hint: Optional[str]) -> str:
    ex.load_markets()
    # id brut -> symbole ccxt
    if hint and hint in getattr(ex, "markets_by_id", {}):
        return ex.markets_by_id[hint]["symbol"]
    # perp usdt
    if "BTC/USDT:USDT" in ex.markets: return "BTC/USDT:USDT"
    # spot fallback
    if "BTC/USDT" in ex.markets: return "BTC/USDT"
    for s in ex.markets.keys():
        if s.startswith("BTC/USDT"): return s
    raise ccxt.BadSymbol("BTC USDT symbol not found on bitget")

def market(hint: Optional[str]="BTCUSDT_UMCBL") -> Dict[str, Any]:
    ex = make_client()
    sym = _resolve_symbol(ex, hint)
    return ex.market(sym)

def last_price(hint: Optional[str]="BTCUSDT_UMCBL") -> float:
    ex = make_client()
    sym = _resolve_symbol(ex, hint)
    t = ex.fetch_ticker(sym)
    px = t.get("last") or t.get("close") or t.get("ask") or t.get("bid")
    if not px: raise RuntimeError("No price for " + sym)
    return float(px)

def fetch_balance_swap() -> Dict[str, Any]:
    ex = make_client()
    try:
        return ex.fetch_balance(params={"productType":"USDT-FUTURES"})
    except Exception:
        return ex.fetch_balance()

def create_order(unified_symbol: str, side: str, type_: str, qty: float,
                 price: Optional[float]=None, reduce_only: bool=False,
                 tif: str="GTC", dry_run: bool=True) -> Dict[str, Any]:
    ex = make_client()
    params = {"reduceOnly": reduce_only, "timeInForce": tif}
    s = side.lower()
    if dry_run:
        return {"dry_run": True, "symbol": unified_symbol, "side": s, "type": type_.lower(),
                "qty": qty, "price": price, "params": params}
    if type_.upper() == "MARKET":
        return ex.create_order(unified_symbol, "market", s, qty, params=params)
    else:
        return ex.create_order(unified_symbol, "limit", s, qty, price, params=params)

def best_qty(notional_usdt: float, px: float, mkt: Dict[str,Any]) -> float:
    prec = (mkt.get("precision") or {}).get("amount", 6)
    min_amt = ((mkt.get("limits") or {}).get("amount") or {}).get("min", 0) or 0
    amt = max(notional_usdt / max(px, 1e-9), min_amt or 0)
    q = round(amt, prec)
    return q if q > 0 else round(min_amt or 0.001, prec)

import ccxt, json
from scalp.utils.db import conn, now_ms
from scalp.utils.log import get
log = get("router")

def resolve_symbol(ex:ccxt.bitget):
    ex.load_markets()
    for s,m in ex.markets.items():
        if m.get("swap") and m.get("base")=="BTC" and m.get("quote")=="USDT":
            return s
    raise RuntimeError("BTC perp USDT introuvable")

def get_px_fallback(symbol_ccxt):
    c = conn()
    r = c.execute("SELECT last FROM ws_ticks WHERE inst_id='BTCUSDT' ORDER BY id DESC LIMIT 1").fetchone()
    if r and r[0]: return float(r[0])
    # fallback OHLCV close
    r = c.execute("SELECT c FROM ohlcv_m1 WHERE symbol=? ORDER BY ts DESC LIMIT 1",(symbol_ccxt,)).fetchone()
    if r: return float(r[0])
    return None

def send_order(side:str, notional_usdt:float=10.0, dry_run:bool=True):
    ex = ccxt.bitget({'enableRateLimit': True, 'options': {'defaultType':'swap'}})
    sym = resolve_symbol(ex)
    px = get_px_fallback(sym)
    if px is None: raise RuntimeError("prix indisponible")
    qty = round(notional_usdt/px, 6)
    res = {"dry_run": dry_run, "symbol": sym, "side": side, "px": px, "qty": qty}
    c = conn()
    c.execute("INSERT INTO orders(ts,symbol,side,px,qty,dry_run,result) VALUES(?,?,?,?,?,?,?)",
              (now_ms(), sym, side, px, qty, 1 if dry_run else 0, json.dumps(res)))
    log.info(f"ORDER {side} {qty}@{px} DRY={dry_run}")
    return res

JOIN top35_mat t USING(symbol)

  ON l.inst_id = REPLACE(SUBSTR(i.symbol,1,INSTR(i.symbol,':')-1),'/','')
ORDER BY t.rank;
"""

INSERT INTO b_preview(symbol, ts, ctx_a, PB, PH, PS, gates,
                      pullback, breakout, meanrev,
                      score, boosted, vetoed,
                      decision, entry, atr1m, sl, tp, qty, note)

ON CONFLICT(symbol) DO UPDATE SET

  decision=excluded.decision, entry=excluded.entry, atr1m=excluded.atr1m,

    gates = "prefilters:OK"

def run(symbol:str, lookback:int=120, atr_n:int=14, k_break:float=1.0):
    c = conn()
    rows = list(c.execute("SELECT ts,o,h,l,c FROM ohlcv_m1 WHERE symbol=? ORDER BY ts DESC LIMIT ?",
                          (symbol, lookback)))
    rows.reverse()
    if len(rows) < max(atr_n+2, 20): 
        log.info("pas assez de barres")
        return "NONE", 0.0, {}
    closes=[r[4] for r in rows]
    highs=[r[2] for r in rows]
    lows=[r[3] for r in rows]
    # ATR
    trs=[max(highs[i]-lows[i], abs(highs[i]-closes[i-1]), abs(lows[i]-closes[i-1])) for i in range(1,len(rows))]
    atr = mean(trs[-atr_n:])
    # Breakout sur plus haut/bas N
    N = 20
    hh = max(highs[-N-1:-1]); ll=min(lows[-N-1:-1]); last=closes[-1]
    side = "NONE"
    if last > hh + k_break*atr: side = "BUY"
    elif last < ll - k_break*atr: side = "SELL"
    # persistance
    meta = {"hh":hh,"ll":ll,"atr":atr,"last":last}
    c.execute("INSERT INTO signals(ts,symbol,side,strength,meta) VALUES(?,?,?,?,?)",
              (now_ms(), symbol, side, 1.0 if side!='NONE' else 0.0, json.dumps(meta)))
    log.info(f"signal {side} last={last} hh={hh} ll={ll} atr={atr}")
    return side, (1.0 if side!='NONE' else 0.0), meta

def top35_symbols(c):
    ts=c.execute("SELECT MAX(ts) FROM tops WHERE name='TOP35'").fetchone()[0]
    return [r[0] for r in c.execute("SELECT symbol FROM tops WHERE ts=? AND name='TOP35' ORDER BY rank",(ts,))] if ts else []

def ema(arr, n):
    k=2/(n+1); e=arr[0]; out=[]
    for v in arr: e = v*k + e*(1-k); out.append(e)
    return out

def rsi(arr, n=14):
    gains=[0]; losses=[0]
    for i in range(1,len(arr)):
        d=arr[i]-arr[i-1]; gains.append(max(d,0)); losses.append(max(-d,0))
    eg, el = ema(gains,n), ema(losses,n)
    out=[]
    for g,l in zip(eg,el):
        rs=g/(l+1e-12); out.append(100 - 100/(1+rs))
    return out

def true_ranges(high, low, close):
    out=[high[0]-low[0]]
    for i in range(1,len(high)):
        out.append(max(high[i]-low[i], abs(high[i]-close[i-1]), abs(low[i]-close[i-1])))
    return out

def adx(high, low, close, n=14):
    plus_dm=[0]; minus_dm=[0]
    for i in range(1,len(high)):
        up=high[i]-high[i-1]; dn=low[i-1]-low[i]
        plus_dm.append(up if up>dn and up>0 else 0)
        minus_dm.append(dn if dn>up and dn>0 else 0)
    tr = true_ranges(high,low,close)
    pdi=[100*a/b if b>0 else 0 for a,b in zip(ema(plus_dm,n), ema(tr,n))]
    mdi=[100*a/b if b>0 else 0 for a,b in zip(ema(minus_dm,n), ema(tr,n))]
    dx=[100*abs(p-m)/(p+m) if (p+m)>0 else 0 for p,m in zip(pdi,mdi)]
    return ema(dx,n)

def obv(close, vol):
    out=[0]
    for i in range(1,len(close)):
        if close[i]>close[i-1]: out.append(out[-1]+vol[i])
        elif close[i]<close[i-1]: out.append(out[-1]-vol[i])
        else: out.append(out[-1])
    return out

def macd_hist(close):
    ema12=ema(close,12); ema26=ema(close,26)
    macd=[a-b for a,b in zip(ema12,ema26)]
    signal=ema(macd,9)
    hist=[m-s for m,s in zip(macd,signal)]
    return ema12, ema26, hist

def softmax_tri(S, tau):
    # Formule fournie: e_pos = exp(S/tau), e_neg = exp(-S/tau), Z = e_pos + 1 + e_neg
    # clamp pour stabilité num.
    x = max(-10.0, min(10.0, S/max(tau,1e-9)))
    e_pos = math.exp(x)
    e_neg = math.exp(-x)
    Z = e_pos + 1.0 + e_neg
    return e_pos/Z, 1.0/Z, e_neg/Z  # pb, ph, ps

def clip(v): return max(-1.0,min(1.0,v))

def compute_tf_features(tf, rows):
    if len(rows) < MIN_BARS: return None
    _,o,h,l,c,v = zip(*rows)  # ts,o,h,l,c,v
    close=list(c); high=list(h); low=list(l); vol=list(v)
    ema12, ema26, hist = macd_hist(close)
    # sigma 100 bars sinon fallback
    sigma100 = (sum((h- sum(hist[-100:])/100.0)**2 for h in hist[-100:]) / max(1,len(hist[-100:])) )**0.5 if len(hist)>=2 else abs(hist[-1]) or 1.0
    rs = rsi(close,14)
    ad = adx(high,low,close,14)
    tr = true_ranges(high,low,close); atr = ema(tr,14)
    atr_pct = [a/max(p,1e-12) for a,p in zip(atr, close)]
    ob = obv(close, vol)
    obv_delta20 = [ob[i]-ob[i-20] if i>=20 else 0 for i in range(len(ob))]
    medabs = median([abs(x) for x in (ob[-200:] if len(ob)>=200 else ob+[0]*(200-len(ob)))]) or 1.0
    return {
        "ema12":ema12[-1],"ema26":ema26[-1],"atr14":atr[-1],
        "macd_hist":hist[-1],"macd_sigma100":sigma100,
        "rsi14":rs[-1],"adx14":ad[-1],
        "obv_delta20":obv_delta20[-1],"obv_medabs200":medabs,
        "atr_pct":atr_pct[-1]
    }

def tf_score(tf, f):
    sEMA = math.tanh( (f["ema12"]-f["ema26"]) / (1.5*max(f["atr14"],1e-9)) )
    sMACD = math.tanh( f["macd_hist"] / max(f["macd_sigma100"],1e-9) )
    sRSI = clip( (f["rsi14"]-50.0)/50.0 )
    sADX = clip( (f["adx14"]-20.0)/20.0 )
    sOBV = math.tanh( f["obv_delta20"] / max(f["obv_medabs200"],1e-9) )
    if f["atr_pct"] < ATR_GATE[tf]:
        sEMA *= 0.5; sMACD *= 0.5
    S = (W_IND["EMA"]*sEMA + W_IND["MACD"]*sMACD + W_IND["RSI"]*sRSI +
         W_IND["ADX"]*sADX + W_IND["OBV"]*sOBV)
    pb, ph, ps = softmax_tri(S, TAU[tf])
    return S, pb, ph, ps, {"sEMA":sEMA,"sMACD":sMACD,"sRSI":sRSI,"sADX":sADX,"sOBV":sOBV}

def thresholds(R):

    high = min(1.0, 0.60 - 0.03*R)
    persist = 1 if R>=8 else 2
    return enter, exit_, low, high, persist

def parse_symbol(sym:str)->tuple[str,str,str]:
    # "BTC/USDT:USDT" -> base, quote, instId "BTCUSDT"
    s=sym.strip()
    if ":" in s: core,_=s.split(":",1)
    else: core=s
    base,quote=core.split("/",1)
    return base,quote,f"{base}{quote}"

def load_input(path=INPUT):
    out=[]
    with open(path,"r",newline="") as f:
        rdr=csv.DictReader(f, delimiter="\t")
        for r in rdr:
            base,quote,inst=parse_symbol(r["symbol"])
            out.append({
              "rank": int(r["rank"]),
              "symbol": r["symbol"],
              "inst": inst,
              "PB": float(r.get("PB",0) or 0),
              "PH": float(r.get("PH",0) or 0),
              "PS": float(r.get("PS",0) or 0),
              "a_ctx": r.get("a_ctx","none"),
              "a_streak": int(r.get("a_streak",0) or 0),
              "a_time": r.get("a_time","--:--:--"),
            })
    return sorted(out, key=lambda x:x["rank"])

def read_ticks(c, insts:list[str], since_ms:int, limit_per=50):
    q=f"""
    SELECT inst_id, ts_local, last, bid, ask
    FROM ws_ticks
    WHERE inst_id IN ({",".join("?"*len(insts))}) AND ts_local>=?
    ORDER BY inst_id, ts_local DESC
    """
    rows=c.execute(q, (*insts, since_ms)).fetchall()
    # compacter par instrument
    by={}
    for inst,ts,last,bid,ask in rows:
        if len(by.setdefault(inst,[]))<limit_per:
            by[inst].append((ts,last,bid,ask))
    return by

    c=conn()
    rows=list(c.execute("SELECT c,h,l FROM ohlcv WHERE tf='5m' AND symbol=? ORDER BY ts DESC LIMIT 200", (sym,)))
    rows.reverse()

    closes=[r[0] for r in rows]; highs=[r[1] for r in rows]; lows=[r[2] for r in rows]

    return side, 1.0 if side!="NONE" else 0.0, {"atr":atr,"last":last,"hh":hh,"ll":ll}

#!/usr/bin/env python3
import time, sqlite3, math
from datetime import datetime

DB="/opt/scalp/data/scalp.db"
TICK_MAX_AGE_S = 15

def now_ms(): return int(time.time()*1000)
def hms(ts): return datetime.fromtimestamp(ts/1000).strftime("%H:%M:%S")

def q(con,s,a=()): return con.execute(s,a)

def get_tick(con, base):
    r=q(con,"SELECT last, bid, ask, ts_local FROM tick_last_mat WHERE inst_id=? ORDER BY ts_local DESC LIMIT 1",(base,)).fetchone()
    if not r: return None
    last,bid,ask,ts=r; return float(last),float(bid or last),float(ask or last),int(ts)

def close_trade(con, live, reason, px):
    ts=now_ms()
    # PnL
    side=live["side"]; qty=live["qty"]; entry=live["entry_px"]
    pnl = (px-entry)*(qty if side=="buy" else -qty)
    pnl_pct = (px-entry)/(entry or 1.0) * (1 if side=="buy" else -1)
    q(con,"INSERT INTO paper_closes(lid,close_ts,close_px,reason,pnl_abs,pnl_pct) VALUES(?,?,?,?,?,?)",
      (live["lid"], ts, px, reason, pnl, pnl_pct))
    q(con,"UPDATE paper_live SET status='CLOSED' WHERE lid=?", (live["lid"],))
    # position nette
    delta_qty = -qty if side=="buy" else qty
    r=q(con,"SELECT qty,avg_px FROM paper_positions WHERE symbol=?", (live["symbol"],)).fetchone()
    if r:
      new_qty=r[0]+delta_qty
      new_avg=None if new_qty==0 else r[1]
      q(con,"UPDATE paper_positions SET qty=?, avg_px=?, last_ts=? WHERE symbol=?", (new_qty,new_avg,ts,live["symbol"]))
    con.commit()

def maybe_trail(live, px, atr):
    if live["trail_atr"] is None or atr is None: return live["sl"]
    moved = (px - live["entry_px"]) if live["side"]=="buy" else (live["entry_px"] - px)
    thresh = live["trail_atr"] * atr
    if moved >= thresh:
        if live["side"]=="buy":
            new_sl = max(live["sl"] or -1e18, px - thresh)
        else:
            new_sl = min(live["sl"] or  1e18, px + thresh)
        return new_sl
    return live["sl"]

def run_loop():
    con=sqlite3.connect(DB)
    con.execute("PRAGMA journal_mode=WAL")
    while True:
        lives = q(con,"SELECT lid,order_id,opened_ts,symbol,side,qty,entry_px,sl,tp,trail_atr,timeout_s,atr1m_last FROM paper_live WHERE status='OPEN'").fetchall()
        for L in lives:
            live=dict(zip(["lid","order_id","opened_ts","symbol","side","qty","entry_px","sl","tp","trail_atr","timeout_s","atr1m_last"],L))
            base=live["symbol"]
            tk=get_tick(con, base)
            if not tk: continue
            last,bid,ask,t_ts = tk
            if now_ms()-t_ts > TICK_MAX_AGE_S*1000: continue

            px = ask if live["side"]=="buy" else bid  # exécution côté marché

            # trailing
            new_sl = maybe_trail(live, last, live["atr1m_last"])
            if new_sl != live["sl"]:
                q(con,"UPDATE paper_live SET sl=? WHERE lid=?", (new_sl, live["lid"]))
                con.commit()
                live["sl"]=new_sl

            # SL / TP
            hit_sl = (last <= live["sl"]) if (live["sl"] is not None and live["side"]=="buy") else (last >= live["sl"] if live["sl"] is not None else False and live["side"]=="sell")
            hit_tp = (last >= live["tp"]) if (live["tp"] is not None and live["side"]=="buy") else (last <= live["tp"] if live["tp"] is not None else False and live["side"]=="sell")

            if hit_sl:
                close_trade(con, live, "SL" if new_sl==live["sl"] else "TRAIL", px); continue
            if hit_tp:
                close_trade(con, live, "TP", px); continue

            # Timeout
            if live["timeout_s"] and now_ms() - live["opened_ts"] > live["timeout_s"]*1000:
                close_trade(con, live, "TO", px); continue

        time.sleep(0.25)

if __name__=="__main__":
    run_loop()

def claim_next(con):

                           notional, margin, leverage, trail_atr, timeout_s, reason

    open_count = q(con,"SELECT COUNT(*) FROM paper_live WHERE status='OPEN'").fetchone()[0]
    if open_count >= MAX_CONCURRENT: return False, "max_concurrent"
    # exposure sur la paire (notional cumulé)
    sym_expo = q(con,"SELECT COALESCE(SUM(notional),0) FROM b_orders WHERE ack=1 AND symbol=?",(symbol,)).fetchone()[0]
    if sym_expo + (notional or 0) > EQUITY_PAPER*MAX_EXPO_PCT: return False, "max_expo"
    # anti-overlap: position nette opposée
    r = q(con,"SELECT qty FROM paper_positions WHERE symbol=?", (symbol,)).fetchone()
    net = r[0] if r else 0.0
    if side=="buy" and net<0: return False,"overlap_short"
    if side=="sell" and net>0: return False,"overlap_long"
    return True, ""

def atr14_1m(con, symbol):
    # calcule ATR(14) 1m depuis table ohlcv
    base = symbol.replace("USDT","/USDT")+":USDT"
    rows = q(con, """
      SELECT high, low, close FROM ohlcv
      WHERE symbol=? AND tf='1m'
      ORDER BY ts DESC LIMIT 30
    """,(base,)).fetchall()
    if len(rows)<15: return None
    rows=list(reversed(rows))
    import math
    atr=0.0
    prev_close=rows[0][2]
    trs=[]
    for h,l,c in rows[1:]:
        tr=max(h-l, abs(h-prev_close), abs(l-prev_close))
        trs.append(tr)
        prev_close=c
    n=14; atr = sum(trs[:n])/n
    for t in trs[n:]:
        atr = (atr*(n-1)+t)/n
    return float(atr)

    new_qty = old_qty + (qty if side=="buy" else -qty)

    else:
        signed = (old_avg or 0.0)*old_qty + (px if side=="buy" else -px)*qty
        new_avg = abs(signed/new_qty)

. "$ROOT/.venv/bin/activate" 2>/dev/null || true
export PYTHONPATH="$ROOT"

  python -u "$ROOT/scalp/strategy/analyse_b_readonly.py" || true
fi

# 2) Compte lignes b_preview
CNT="$(sqlite3 "$DB" 'SELECT COUNT(*) FROM b_preview;')"

if [ "$CNT" -gt 0 ]; then
  echo "--- b_preview (15 last) ---"
  sqlite3 -header -column "$DB" "
  SELECT strftime('%H:%M:%S',ts/1000,'unixepoch','localtime') AS time,
         symbol, ctx_a,
         printf('%.3f',PB) PB, printf('%.3f',PH) PH, printf('%.3f',PS) PS,
         decision, IFNULL(entry,'-') entry
  FROM b_preview
  ORDER BY ts DESC LIMIT 15;"
  # export TSV
  sqlite3 -header -separator $'\t' "$DB" "
  SELECT symbol,
         strftime('%H:%M:%S',ts/1000,'unixepoch','localtime') AS time,
         ctx_a, PB, PH, PS, decision, entry
  FROM b_preview
  ORDER BY ts DESC;" > "$ROOT/data/b_out.tsv"
  echo "--- TSV path: $ROOT/data/b_out.tsv"
else
  echo "--- b_preview EMPTY (CNT=0). Fallback lecture-only ---"
  sqlite3 -header -column "$DB" "
  SELECT t.rank,
         i.symbol,
         printf('%.3f',i.PB) PB, printf('%.3f',i.PH) PH, printf('%.3f',i.PS) PS,
         i.a_ctx,
         datetime(l.ts_local/1000,'unixepoch','localtime') AS tick_time,
         l.last AS tick_last
  FROM input_B_mat i
  JOIN top35_mat t USING(symbol)
  LEFT JOIN tick_last_mat l
    ON l.inst_id = REPLACE(SUBSTR(i.symbol,1,INSTR(i.symbol,':')-1),'/','')
  ORDER BY t.rank LIMIT 15;"
  echo "(A tourné à: $(sqlite3 "$DB" "SELECT MAX(a_time_hms) FROM input_B_mat;"))"
fi

# 3) Récap contextes buy/hold/sell/none sur le TOP35 courant
CTX_LINE="$(sqlite3 "$DB" "

  SELECT COALESCE(a_ctx,'none') AS ctx
  FROM input_B_mat
)
SELECT
  'CTX  buy='||SUM(ctx='buy')||
  '  hold='||SUM(ctx='hold')||
  '  sell='||SUM(ctx='sell')||
  '  none='||SUM(ctx='none')||

FROM src;")"

sqlite3 "$DB" <<'SQL'
PRAGMA journal_mode=WAL;
DELETE FROM top35_mat;
INSERT INTO top35_mat(rank, symbol, ts)
SELECT rank, symbol, ts
FROM tops
WHERE name='TOP35'
  AND ts=(SELECT MAX(ts) FROM tops WHERE name='TOP35')
ORDER BY rank;
DELETE FROM tick_last_mat;
INSERT INTO tick_last_mat(inst_id, ts_local, last, bid, ask)
SELECT w.inst_id, w.ts_local, w.last, w.bid, w.ask
FROM ws_ticks w
JOIN (SELECT inst_id, MAX(ts_local) ts FROM ws_ticks GROUP BY inst_id) mx
  ON mx.inst_id=w.inst_id AND mx.ts=w.ts_local;
DELETE FROM input_B_mat;
INSERT INTO input_B_mat(rank, symbol, PB, PH, PS, a_ctx, a_streak, a_ts, a_time_hms)
SELECT
  t.rank, t.symbol,
  IFNULL(s.PB,0), IFNULL(s.PH,0), IFNULL(s.PS,0),
  IFNULL(c.ctx,'none'), IFNULL(c.streak,0),
  IFNULL(s.ts,0),
  CASE WHEN s.ts>0 THEN strftime('%H:%M:%S', s.ts/1000,'unixepoch','localtime') ELSE '--:--:--' END
FROM top35_mat t
LEFT JOIN a_scores s ON s.symbol=t.symbol
LEFT JOIN ctx_a    c ON c.symbol=t.symbol
ORDER BY t.rank;
SQL
echo "DONE"

#!/usr/bin/env bash
set -euo pipefail
DB="/opt/scalp/data/scalp.db"
OUT="/opt/scalp/data/input_B.tsv"

[ -n "${TS:-}" ] || { echo "NO_TOP35"; exit 0; }
while IFS='|' read -r R S; do
  read -r PB PH PS T <<<"$(sqlite3 -separator ' ' "$DB" "
    SELECT IFNULL(PB,0),IFNULL(PH,0),IFNULL(PS,0),
           IFNULL(strftime('%H:%M:%S', ts/1000, 'unixepoch','localtime'),'--:--:--')
    FROM a_scores WHERE symbol='$S' LIMIT 1;")"
  read -r CTX ST <<<"$(sqlite3 -separator ' ' "$DB" "
    SELECT IFNULL(ctx,'none'), IFNULL(streak,0) FROM ctx_a WHERE symbol='$S' LIMIT 1;")"

done < <(sqlite3 -separator '|' "$DB" "SELECT rank,symbol FROM tops WHERE name='TOP35' AND ts=$TS ORDER BY rank;")
